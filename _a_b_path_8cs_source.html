317  protected override void CalculateStep (long targetTick) {
318  int counter = 0;
319 
320  // Continue to search as long as we haven't encountered an error and we haven't found the target
321  while (CompleteState == PathCompleteState.NotCalculated) {
322  searchedNodes++;
323 
324  // Close the current node, if the current node is the target node then the path is finished
325  if (currentR.flag1) {
326  // We found a target point
327  // Mark that node as the end point
328  CompleteWith(currentR.node);
329  break;
330  }
331 
332  if (currentR.H < partialBestTarget.H) {
333  partialBestTarget = currentR;
334  }
335 
336  // Loop through all walkable neighbours of the node and add them to the open list.
337  currentR.node.Open(this, currentR, pathHandler);
338 
339  // Any nodes left to search?
340  if (pathHandler.heap.isEmpty) {
341  if (calculatePartial && partialBestTarget != null) {
342  CompleteState = PathCompleteState.Partial;
343  Trace(partialBestTarget);
344  } else {
345  FailWithError("Searched whole area but could not find target");
346  }
347  return;
348  }
349 
350  // Select the node with the lowest F score and remove it from the open list
351  currentR = pathHandler.heap.Remove();
352 
353  // Check for time every 500 nodes, roughly every 0.5 ms usually
354  if (counter > 500) {
355  // Have we exceded the maxFrameTime, if so we should wait one frame before continuing the search since we don't want the game to lag
356  if (System.DateTime.UtcNow.Ticks >= targetTick) {
357  // Return instead of yield'ing, a separate function handles the yield (CalculatePaths)
358  return;
359  }
360  counter = 0;
361 
362  // Mostly for development
363  if (searchedNodes > 1000000) {
364  throw new System.Exception("Probable infinite loop. Over 1,000,000 nodes searched");
365  }
366  }
367 
368  counter++;
369  }
370 
371  if (CompleteState == PathCompleteState.Complete) {
372  Trace(currentR);
373  } else if (calculatePartial && partialBestTarget != null) {
374  CompleteState = PathCompleteState.Partial;
375  Trace(partialBestTarget);
376  }
377  }
