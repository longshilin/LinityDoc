418  public void SetPath (Path path) {
419  if (path.PipelineState == PathState.Created) {

531  void FinalizePosition (Vector3 nextPosition) {
532  // Use a local variable, it is significantly faster
533  Vector3 currentPosition = simulatedPosition;
534  bool positionDirty1 = false;
535 
536  if (controller != null && controller.enabled && updatePosition) {
537  // Use CharacterController
538  // The Transform may not be at #position if it was outside the navmesh and had to be moved to the closest valid position
539  tr.position = currentPosition;
540  controller.Move((nextPosition - currentPosition) + accumulatedMovementDelta);
541  // Grab the position after the movement to be able to take physics into account
542  // TODO: Add this into the clampedPosition calculation below to make RVO better respond to physics
543  currentPosition = tr.position;
544  if (controller.isGrounded) verticalVelocity = 0;
545  } else {
546  // Use Transform, Rigidbody, Rigidbody2D or nothing at all (if updatePosition = false)
547  float lastElevation;
548  movementPlane.ToPlane(currentPosition, out lastElevation);
549  currentPosition = nextPosition + accumulatedMovementDelta;
550 
551  // Position the character on the ground
552  if (usingGravity) currentPosition = RaycastPosition(currentPosition, lastElevation);
553  positionDirty1 = true;
554  }
555 
556  // Clamp the position to the navmesh after movement is done
557  bool positionDirty2 = false;
558  currentPosition = ClampToNavmesh(currentPosition, out positionDirty2);
559 
560  // Assign the final position to the character if we haven't already set it (mostly for performance, setting the position can be slow)
561  if ((positionDirty1 || positionDirty2) && updatePosition) {
562  // Note that rigid.MovePosition may or may not move the character immediately.
563  // Check the Unity documentation for the special cases.
564  if (rigid != null) rigid.MovePosition(currentPosition);
565  else if (rigid2D != null) rigid2D.MovePosition(currentPosition);
566  else tr.position = currentPosition;
567  }
568 
569  accumulatedMovementDelta = Vector3.zero;
570  simulatedPosition = currentPosition;
571  UpdateVelocity();
572  }

603  protected Vector3 RaycastPosition (Vector3 position, float lastElevation) {
604  RaycastHit hit;
605  float elevation;
606 
607  movementPlane.ToPlane(position, out elevation);
608  float rayLength = centerOffset + Mathf.Max(0, lastElevation-elevation);
609  Vector3 rayOffset = movementPlane.ToWorld(Vector2.zero.ToPFV2(), rayLength).ToUnityV3();
610 
611  if (Physics.Raycast(position + rayOffset, -rayOffset, out hit, rayLength, groundMask, QueryTriggerInteraction.Ignore)) {
612  // Grounded
613  // Make the vertical velocity fall off exponentially. This is reasonable from a physical standpoint as characters
614  // are not completely stiff and touching the ground will not immediately negate all velocity downwards. The AI will
615  // stop moving completely due to the raycast penetration test but it will still *try* to move downwards. This helps
616  // significantly when moving down along slopes as if the vertical velocity would be set to zero when the character
617  // was grounded it would lead to a kind of 'bouncing' behavior (try it, it's hard to explain). Ideally this should
618  // use a more physically correct formula but this is a good approximation and is much more performant. The constant
619  // '5' in the expression below determines how quickly it converges but high values can lead to too much noise.
620  verticalVelocity *= System.Math.Max(0, 1 - 5 * lastDeltaTime);
621  return hit.point;
622  }
623  return position;
624  }
