376  protected virtual void OnPathComplete (Path _p) {

378 
379  if (p == null) throw new System.Exception("This function only handles ABPaths, do not use special path types");
380 
381  canSearchAgain = true;
382 
383  // Increase the reference count on the path.
384  // This is used for path pooling
385  p.Claim(this);
386 
387  // Path couldn't be calculated of some reason.
388  // More info in p.errorLog (debug string)
389  if (p.error) {
390  p.Release(this);
391  return;
392  }
393 
394  if (interpolatePathSwitches) {
395  ConfigurePathSwitchInterpolation();

398

400  var oldPath = path;
401  path = p;
402  reachedEndOfPath = false;

406  if (path.vectorPath != null && path.vectorPath.Count == 1) {
407  path.vectorPath.Insert(0, GetFeetPosition());

512  Quaternion SimulateRotationTowards (Vector3 direction, float deltaTime) {
513  // Rotate unless we are really close to the target
514  if (direction != Vector3.zero) {
515  Quaternion targetRotation = Quaternion.LookRotation(direction, rotationIn2D ? Vector3.back : Vector3.up);
516  // This causes the character to only rotate around the Z axis
517  if (rotationIn2D) targetRotation *= Quaternion.Euler(90, 0, 0);
518  return Quaternion.Slerp(simulatedRotation, targetRotation, deltaTime * rotationSpeed);
519  }

526  protected virtual Vector3 CalculateNextPosition (out Vector3 direction, float deltaTime) {
527  if (!interpolator.valid) {
528  direction = Vector3.zero;
529  return simulatedPosition;
530  }
531 
532  interpolator.distance += deltaTime * speed;
533 
534  if (interpolator.remainingDistance < 0.0001f && !reachedEndOfPath) {
535  reachedEndOfPath = true;
536  OnTargetReached();
537  }
538 
539  direction = interpolator.tangent;
540  pathSwitchInterpolationTime += deltaTime;
541  var alpha = switchPathInterpolationSpeed * pathSwitchInterpolationTime;
542  if (interpolatePathSwitches && alpha < 1f) {
543  // Find the approximate position we would be at if we
544  // would have continued to follow the previous path
545  Vector3 positionAlongPreviousPath = previousMovementOrigin + Vector3.ClampMagnitude(previousMovementDirection, speed * pathSwitchInterpolationTime);
546 
547  // Interpolate between the position on the current path and the position
548  // we would have had if we would have continued along the previous path.
549  return Vector3.Lerp(positionAlongPreviousPath, interpolator.position, alpha);
550  } else {
551  return interpolator.position;
552  }
553  }
