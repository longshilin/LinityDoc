237  protected override void OnPathComplete (Path newPath) {
238  ABPath p = newPath as ABPath;
239 
240  if (p == null) throw new System.Exception("This function only handles ABPaths, do not use special path types");
241 
242  waitingForPathCalculation = false;
243 
244  // Increase the reference count on the new path.
245  // This is used for object pooling to reduce allocations.
246  p.Claim(this);
247 
248  // Path couldn't be calculated of some reason.
249  // More info in p.errorLog (debug string)
250  if (p.error) {
251  p.Release(this);
252  return;
253  }
254 
255  // Release the previous path.
256  if (path != null) path.Release(this);
257 
258  // Replace the old path
259  path = p;
260 
261  // Make sure the path contains at least 2 points
262  if (path.vectorPath.Count == 1) path.vectorPath.Add(path.vectorPath[0]);
263  interpolator.SetPath(path.vectorPath);
264 
265  var graph = UnityHelper.GetGraph(path.path[0]) as ITransformedGraph;
266  movementPlane = graph != null ? graph.transform : (rotationIn2D ? new GraphTransform(Matrix4x4.TRS(Vector3.zero, Quaternion.Euler(-90, 270, 90), Vector3.one)) : GraphTransform.identityTransform);
267 
268  // Reset some variables
269  reachedEndOfPath = false;
270 
271  // Simulate movement from the point where the path was requested
272  // to where we are right now. This reduces the risk that the agent
273  // gets confused because the first point in the path is far away
274  // from the current position (possibly behind it which could cause
275  // the agent to turn around, and that looks pretty bad).
276  interpolator.MoveToLocallyClosestPoint((GetFeetPosition() + p.originalStartPoint.ToUnityV3()) * 0.5f);
277  interpolator.MoveToLocallyClosestPoint(GetFeetPosition());
278 
279  // Update which point we are moving towards.
280  // Note that we need to do this here because otherwise the remainingDistance field might be incorrect for 1 frame.
281  // (due to interpolator.remainingDistance being incorrect).
282  interpolator.MoveToCircleIntersection2D(position, pickNextWaypointDist, movementPlane);
283 
284  var distanceToEnd = remainingDistance;
285  if (distanceToEnd <= endReachedDistance) {
286  reachedEndOfPath = true;
287  OnTargetReached();
288  }
289  }

292  protected override void MovementUpdateInternal (float deltaTime, out Vector3 nextPosition, out Quaternion nextRotation) {
293  float currentAcceleration = maxAcceleration;
294 
295  // If negative, calculate the acceleration from the max speed
296  if (currentAcceleration < 0) currentAcceleration *= -maxSpeed;
297 
298  if (updatePosition) {
299  // Get our current position. We read from transform.position as few times as possible as it is relatively slow
300  // (at least compared to a local variable)
301  simulatedPosition = tr.position;
302  }
303  if (updateRotation) simulatedRotation = tr.rotation;
304 
305  var currentPosition = simulatedPosition;
306 
307  // Update which point we are moving towards
308  interpolator.MoveToCircleIntersection2D(currentPosition, pickNextWaypointDist, movementPlane);
309  var dir = movementPlane.ToPlane(steeringTarget - currentPosition);
310 
311  // Calculate the distance to the end of the path
312  float distanceToEnd = dir.magnitude + Mathf.Max(0, interpolator.remainingDistance);
313 
314  // Check if we have reached the target
315  var prevTargetReached = reachedEndOfPath;
316  reachedEndOfPath = distanceToEnd <= endReachedDistance && interpolator.valid;
317  if (!prevTargetReached && reachedEndOfPath) OnTargetReached();
318  float slowdown;
319 
320  // Normalized direction of where the agent is looking
321  var forwards = movementPlane.ToPlane(simulatedRotation * (rotationIn2D ? Vector3.up : Vector3.forward));
322 
323  // Check if we have a valid path to follow and some other script has not stopped the character
324  if (interpolator.valid && !isStopped) {
325  // How fast to move depending on the distance to the destination.
326  // Move slower as the character gets closer to the destination.
327  // This is always a value between 0 and 1.
328  slowdown = distanceToEnd < slowdownDistance ? Mathf.Sqrt(distanceToEnd / slowdownDistance) : 1;
329 
330  if (reachedEndOfPath && whenCloseToDestination == CloseToDestinationMode.Stop) {
331  // Slow down as quickly as possible
332  velocity2D -= Vector2.ClampMagnitude(velocity2D, currentAcceleration * deltaTime);
333  } else {
334  velocity2D += MovementUtilities.CalculateAccelerationToReachPoint(dir.ToUnityV2(), dir.normalized*maxSpeed, velocity2D, currentAcceleration, rotationSpeed, maxSpeed, forwards) * deltaTime;
335  }
336  } else {
337  slowdown = 1;
338  // Slow down as quickly as possible
339  velocity2D -= Vector2.ClampMagnitude(velocity2D, currentAcceleration * deltaTime);
340  }
341 
342  velocity2D = MovementUtilities.ClampVelocity(velocity2D, maxSpeed, slowdown, slowWhenNotFacingTarget, forwards.ToUnityV2());
343 
344  ApplyGravity(deltaTime);
345 
346  if (rvoController != null && rvoController.enabled) {
347  // Send a message to the RVOController that we want to move
348  // with this velocity. In the next simulation step, this
349  // velocity will be processed and it will be fed back to the
350  // rvo controller and finally it will be used by this script
351  // when calling the CalculateMovementDelta method below
352 
353  // Make sure that we don't move further than to the end point
354  // of the path. If the RVO simulation FPS is low and we did
355  // not do this, the agent might overshoot the target a lot.
356  var rvoTarget = currentPosition.ToPFV3() + movementPlane.ToWorld(Vector2.ClampMagnitude(velocity2D, distanceToEnd), 0f);
357  rvoController.SetTarget(rvoTarget, velocity2D.magnitude, maxSpeed);
358  }
359 
360  // Set how much the agent wants to move during this frame
361  var delta2D = lastDeltaPosition = CalculateDeltaToMoveThisFrame(movementPlane.ToPlane(currentPosition), distanceToEnd, deltaTime);
362  nextPosition = currentPosition + movementPlane.ToWorld(delta2D.ToPFV2(), verticalVelocity * lastDeltaTime).ToUnityV3();
363  CalculateNextRotation(slowdown, out nextRotation);
364  }

366  protected virtual void CalculateNextRotation (float slowdown, out Quaternion nextRotation) {
367  if (lastDeltaTime > 0.00001f) {
368  Vector2 desiredRotationDirection;
369  if (rvoController != null && rvoController.enabled) {
370  // When using local avoidance, use the actual velocity we are moving with if that velocity
371  // is high enough, otherwise fall back to the velocity that we want to move with (velocity2D).
372  // The local avoidance velocity can be very jittery when the character is close to standing still
373  // as it constantly makes small corrections. We do not want the rotation of the character to be jittery.
374  var actualVelocity = lastDeltaPosition/lastDeltaTime;
375  desiredRotationDirection = Vector2.Lerp(velocity2D, actualVelocity, 4 * actualVelocity.magnitude / (maxSpeed + 0.0001f));
376  } else {
377  desiredRotationDirection = velocity2D;
378  }
379 
380  // Rotate towards the direction we are moving in.
381  // Don't rotate when we are very close to the target.
382  var currentRotationSpeed = rotationSpeed * Mathf.Max(0, (slowdown - 0.3f) / 0.7f);
383  nextRotation = SimulateRotationTowards(desiredRotationDirection, currentRotationSpeed * lastDeltaTime);
384  } else {
385  // TODO: simulatedRotation
386  nextRotation = rotation;
387  }
388  }

391  protected override Vector3 ClampToNavmesh (Vector3 position, out bool positionChanged) {
392  if (constrainInsideGraph) {
393  cachedNNConstraint.tags = seeker.traversableTags;
394  cachedNNConstraint.graphMask = seeker.graphMask;
395  cachedNNConstraint.distanceXZ = true;
396  var clampedPosition = PathFindHelper.GetNearest(position, cachedNNConstraint).position;
397 
398  // We cannot simply check for equality because some precision may be lost
399  // if any coordinate transformations are used.
400  var difference = movementPlane.ToPlane(clampedPosition - position.ToPFV3());
401  float sqrDifference = difference.sqrMagnitude;
402  if (sqrDifference > 0.001f*0.001f) {
403  // The agent was outside the navmesh. Remove that component of the velocity
404  // so that the velocity only goes along the direction of the wall, not into it
405  velocity2D -= difference.ToUnityV2() * Vector2.Dot(difference.ToUnityV2(), velocity2D) / sqrDifference;
406 
407  // Make sure the RVO system knows that there was a collision here
408  // Otherwise other agents may think this agent continued
409  // to move forwards and avoidance quality may suffer
410  if (rvoController != null && rvoController.enabled) {
411  rvoController.SetCollisionNormal(difference.ToUnityV2());
412  }
413  positionChanged = true;
414  // Return the new position, but ignore any changes in the y coordinate from the ClampToNavmesh method as the y coordinates in the navmesh are rarely very accurate
415  return position + movementPlane.ToWorld(difference).ToUnityV3();
416  }
417  }
418 
419  positionChanged = false;
420  return position;
421  }
