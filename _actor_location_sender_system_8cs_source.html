136  private static async ETTask RunTask(this ActorLocationSender self, ActorTask task)
137  {
138  ActorMessageSender actorMessageSender = Game.Scene.GetComponent<ActorMessageSenderComponent>().Get(self.ActorId);
139  IActorResponse response;
140  try
141  {
142  // 这里必须使用不抛异常的rpc，因为服务端handler很可能出现错误，返回一个rpc fail的错误，结果这里抛了异常
143  // 这里抛了异常就会导致队列中的消息无法继续发送，导致整个actorlocationsender堵塞
144  response = await actorMessageSender.CallWithoutException(task.ActorRequest);
145  }
146  catch (Exception e)
147  {
148  self.GetParent<ActorLocationSenderComponent>().Remove(self.Id);
149  return;
150  }
151 
152 
153  // 发送成功
154  switch (response.Error)
155  {
156  case ErrorCode.ERR_NotFoundActor:
157  // 如果没找到Actor,重试
158  ++self.FailTimes;
159 
160  // 失败MaxFailTimes次则清空actor发送队列，返回失败
161  if (self.FailTimes > ActorLocationSender.MaxFailTimes)
162  {
163  // 失败直接删除actorproxy
164  Log.Info($"actor send message fail, actorid: {self.Id}");
165  self.RunError(response.Error);
166  self.GetParent<ActorLocationSenderComponent>().Remove(self.Id);
167  return;
168  }
169 
170  // 等待0.5s再发送
171  await Game.Scene.GetComponent<TimerComponent>().WaitAsync(500);
172  self.ActorId = await Game.Scene.GetComponent<LocationProxyComponent>().Get(self.Id);
173  self.AllowGet();
174  return;
175 
176  case ErrorCode.ERR_ActorNoMailBoxComponent:
177  self.RunError(response.Error);
178  self.GetParent<ActorLocationSenderComponent>().Remove(self.Id);
179  return;
180 
181  default:
182  self.LastRecvTime = TimeHelper.Now();
183  self.FailTimes = 0;
184  self.WaitingTasks.Dequeue();
185 
186  // 如果所有的发送消息都得到了返回，发送任务完成，那么删除这个ActorLocationSender，及时回收发送对象
187  if (self.WaitingTasks.Count == 0)
188  {
189  self.GetParent<ActorLocationSenderComponent>().Remove(self.Id);
190  }
191 
192  if (task.Tcs != null)
193  {
194  IActorLocationResponse actorLocationResponse = response as IActorLocationResponse;
195  if (actorLocationResponse == null)
196  {
197  task.Tcs.SetException(new Exception($"actor location respose is not IActorLocationResponse, but is: {response.GetType().Name}"));
198  }
199  task.Tcs.SetResult(actorLocationResponse);
200  }
201  return;
202  }
