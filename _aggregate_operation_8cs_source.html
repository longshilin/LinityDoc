253  public IAsyncCursor<TResult> Execute(IReadBinding binding, CancellationToken cancellationToken)
254  {
255  Ensure.IsNotNull(binding, nameof(binding));
256  EnsureIsReadOnlyPipeline();
257 
258  using (EventContext.BeginOperation())
259  using (var channelSource = binding.GetReadChannelSource(cancellationToken))
260  using (var channel = channelSource.GetChannel(cancellationToken))
261  using (var channelBinding = new ChannelReadBinding(channelSource.Server, channel, binding.ReadPreference, binding.Session.Fork()))
262  {
263  var operation = CreateOperation(channel, channelBinding);
264  var result = operation.Execute(channelBinding, cancellationToken);
265  return CreateCursor(channelSource, channel, operation.Command, result);
266  }
267  }

301  internal BsonDocument CreateCommand(ConnectionDescription connectionDescription, ICoreSession session)
302  {
303  Feature.ReadConcern.ThrowIfNotSupported(connectionDescription.ServerVersion, _readConcern);
304  Feature.Collation.ThrowIfNotSupported(connectionDescription.ServerVersion, _collation);
305 
306  var readConcern = ReadConcernHelper.GetReadConcernForCommand(session, connectionDescription, _readConcern);
307  var command = new BsonDocument
308  {
309  { "aggregate", _collectionNamespace == null ? (BsonValue)1 : _collectionNamespace.CollectionName },
310  { "pipeline", new BsonArray(_pipeline) },
311  { "allowDiskUse", () => _allowDiskUse.Value, _allowDiskUse.HasValue },
312  { "maxTimeMS", () => MaxTimeHelper.ToMaxTimeMS(_maxTime.Value), _maxTime.HasValue },
313  { "collation", () => _collation.ToBsonDocument(), _collation != null },
314  { "hint", () => _hint, _hint != null },
315  { "comment", () => _comment, _comment != null },
316  { "readConcern", readConcern, readConcern != null }
317  };
318 
319  if (Feature.AggregateCursorResult.IsSupported(connectionDescription.ServerVersion))
320  {
321  var useCursor = _useCursor.GetValueOrDefault(true) || connectionDescription.ServerVersion >= new SemanticVersion(3, 5, 0);
322  if (useCursor)
323  {
324  command["cursor"] = new BsonDocument
325  {
326  { "batchSize", () => _batchSize.Value, _batchSize.HasValue }
327  };
328  }
329  }
330 
331  return command;
332  }

409  public override AggregateResult Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
410  {
411  var reader = context.Reader;
412  AggregateResult result = null;
413  reader.ReadStartDocument();
414  while (reader.ReadBsonType() != 0)
415  {
416  var elementName = reader.ReadName();
417  if (elementName == "cursor")
418  {
419  var cursorDeserializer = new CursorDeserializer(_resultSerializer);
420  result = cursorDeserializer.Deserialize(context);
421  }
422  else if (elementName == "result")
423  {
424  var arraySerializer = new ArraySerializer<TResult>(_resultSerializer);
425  result = new AggregateResult();
426  result.Results = arraySerializer.Deserialize(context);
427  }
428  else
429  {
430  reader.SkipValue();
431  }
432  }
433  reader.ReadEndDocument();
434  return result;
435  }
436  }

447  public override AggregateResult Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
448  {
449  var reader = context.Reader;
450  var result = new AggregateResult();
451  reader.ReadStartDocument();
452  while (reader.ReadBsonType() != 0)
453  {
454  var elementName = reader.ReadName();
455  switch (elementName)
456  {
457  case "id":
458  result.CursorId = new Int64Serializer().Deserialize(context);
459  break;
460 
461  case "ns":
462  var ns = reader.ReadString();
463  result.CollectionNamespace = CollectionNamespace.FromFullName(ns);
464  break;
465 
466  case "firstBatch":
467  var arraySerializer = new ArraySerializer<TResult>(_resultSerializer);
468  result.Results = arraySerializer.Deserialize(context);
469  break;
470 
471  default:
472  reader.SkipValue();
473  break;
474  }
475  }
476  reader.ReadEndDocument();
477  return result;
478  }
