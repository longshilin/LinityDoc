211  internal BsonDocument CreateCommand(ICoreSessionHandle session, ConnectionDescription connectionDescription)
212  {
213  var serverVersion = connectionDescription.ServerVersion;
214  Feature.Collation.ThrowIfNotSupported(serverVersion, _collation);
215 
216  var writeConcern = WriteConcernHelper.GetWriteConcernForCommandThatWrites(session, _writeConcern, serverVersion);
217  return new BsonDocument
218  {
219  { "aggregate", _collectionNamespace.CollectionName },
220  { "pipeline", new BsonArray(_pipeline) },
221  { "allowDiskUse", () => _allowDiskUse.Value, _allowDiskUse.HasValue },
222  { "bypassDocumentValidation", () => _bypassDocumentValidation.Value, _bypassDocumentValidation.HasValue && Feature.BypassDocumentValidation.IsSupported(serverVersion) },
223  { "maxTimeMS", () => MaxTimeHelper.ToMaxTimeMS(_maxTime.Value), _maxTime.HasValue },
224  { "collation", () => _collation.ToBsonDocument(), _collation != null },
225  { "writeConcern", writeConcern, writeConcern != null },
226  { "cursor", new BsonDocument(), serverVersion >= new SemanticVersion(3, 5, 0) },
227  { "hint", () => _hint, _hint != null },
228  { "comment", () => _comment, _comment != null }
229  };
230  }

232  private WriteCommandOperation<BsonDocument> CreateOperation(ICoreSessionHandle session, ConnectionDescription connectionDescription)
233  {

238  private void EnsureIsOutputToCollectionPipeline()
239  {
