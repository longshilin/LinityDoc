67  public AndroidFileSystemStream(string fullPath, FileSystemAccess access, bool createNew)
68  {
69  if (string.IsNullOrEmpty(fullPath))
70  {
71  throw new GameFrameworkException("Full path is invalid.");
72  }
73 
74  if (access != FileSystemAccess.Read)
75  {
76  throw new GameFrameworkException(Utility.Text.Format("'{0}' is not supported in AndroidFileSystemStream.", access.ToString()));
77  }
78 
79  if (createNew)
80  {
81  throw new GameFrameworkException("Create new is not supported in AndroidFileSystemStream.");
82  }
83 
84  int position = fullPath.LastIndexOf(SplitFlag, StringComparison.Ordinal);
85  if (position < 0)
86  {
87  throw new GameFrameworkException("Can not find split flag in full path.");
88  }
89 
90  string fileName = fullPath.Substring(position + SplitFlagLength);
91  m_FileStream = InternalOpen(fileName);
92  if (m_FileStream == null)
93  {
94  throw new GameFrameworkException(Utility.Text.Format("Open file '{0}' from Android asset manager failure.", fullPath));
95  }
96 
97  m_FileStreamRawObject = m_FileStream.GetRawObject();

141  {
142  if (origin == SeekOrigin.End)

246  private int InternalRead(int length, out byte[] result)
247  {
248 #if UNITY_2019_2_OR_NEWER
249 #pragma warning disable CS0618
250 #endif
251  IntPtr resultPtr = AndroidJNI.NewByteArray(length);
252 #if UNITY_2019_2_OR_NEWER
253 #pragma warning restore CS0618
254 #endif
255  int offset = 0;
256  int bytesLeft = length;
257  while (bytesLeft > 0)
258  {
259  s_InternalReadArgs[0] = new jvalue() { l = resultPtr };
260  s_InternalReadArgs[1] = new jvalue() { i = offset };
261  s_InternalReadArgs[2] = new jvalue() { i = bytesLeft };
262  int bytesRead = AndroidJNI.CallIntMethod(m_FileStreamRawObject, s_InternalReadMethodId, s_InternalReadArgs);
263  if (bytesRead <= 0)
264  {
265  break;
266  }
267 
268  offset += bytesRead;
269  bytesLeft -= bytesRead;
270  }
271 
272 #if UNITY_2019_2_OR_NEWER
273 #pragma warning disable CS0618
274 #endif
275  result = AndroidJNI.FromByteArray(resultPtr);
276 #if UNITY_2019_2_OR_NEWER
277 #pragma warning restore CS0618
278 #endif
279  AndroidJNI.DeleteLocalRef(resultPtr);
280  return offset;
281  }
