181  // Creates a directory structure with "provides namespace" attribute.
182  // First, retrieves or creates the directory at relativeBasePath, creating parent
183  // directories if needed. Effectively calls OpenFolder(relativeBasePath).
184  // Then, relative to this directory, creates namespacePath directories with "provides
185  // namespace" attribute set. Fails if the attribute can't be set.
186  public AssetFolder OpenNamespacedFolder(string relativeBasePath, string namespacePath)

190  foreach (var pathItem in pathItems)
191  {
192  folder = folder.OpenFolder(pathItem);
193  folder.providesNamespace = true;
