80  public static DateTime ToDateTimeFromMillisecondsSinceEpoch(long millisecondsSinceEpoch)

83  millisecondsSinceEpoch > BsonConstants.DateTimeMaxValueMillisecondsSinceEpoch)

119  {

121  {
122  throw new ArgumentNullException("bytes");
123  }
124 
125  var length = bytes.Length;
126  var c = new char[length * 2];
127 
128  for (int i = 0, j = 0; i < length; i++)
129  {
130  var b = bytes[i];

133  }

197  public static bool TryParseHexString(string s, out byte[] bytes)
198  {
199  bytes = null;
200 
201  if (s == null)
202  {
203  return false;
204  }
205 
206  var buffer = new byte[(s.Length + 1) / 2];
207 
208  var i = 0;
209  var j = 0;
210 
211  if ((s.Length % 2) == 1)
212  {
213  // if s has an odd length assume an implied leading "0"
214  int y;
215  if (!TryParseHexChar(s[i++], out y))
216  {
217  return false;
218  }
219  buffer[j++] = (byte)y;
220  }
221 
222  while (i < s.Length)
223  {
224  int x, y;
225  if (!TryParseHexChar(s[i++], out x))
226  {
227  return false;
228  }
229  if (!TryParseHexChar(s[i++], out y))
230  {
231  return false;
232  }
233  buffer[j++] = (byte)((x << 4) | y);
234  }
235 
236  bytes = buffer;
237  return true;
238  }

240  // private static methods
241  private static bool TryParseHexChar(char c, out int value)
242  {
243  if (c >= '0' && c <= '9')
244  {
245  value = c - '0';
246  return true;
247  }
248 
249  if (c >= 'a' && c <= 'f')
250  {
251  value = 10 + (c - 'a');
252  return true;
253  }
254 
255  if (c >= 'A' && c <= 'F')
256  {
257  value = 10 + (c - 'A');
258  return true;
259  }
260 
261  value = 0;
262  return false;
263  }
