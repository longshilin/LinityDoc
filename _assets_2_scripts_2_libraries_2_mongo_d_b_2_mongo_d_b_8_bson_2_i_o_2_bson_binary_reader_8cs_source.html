124 #pragma warning disable 618 // about obsolete BsonBinarySubType.OldBinary
125  public override BsonBinaryData ReadBinaryData()
126  {
127  if (Disposed) { ThrowObjectDisposedException(); }
128  VerifyBsonType("ReadBinaryData", BsonType.Binary);
129 
130  int size = ReadSize();
131 
132  var subType = _bsonStream.ReadBinarySubType();
133  if (subType == BsonBinarySubType.OldBinary)
134  {
135  // sub type OldBinary has two sizes (for historical reasons)
136  int size2 = ReadSize();
137  if (size2 != size - 4)
138  {
139  throw new FormatException("Binary sub type OldBinary has inconsistent sizes");
140  }
141  size = size2;
142 
143  if (_settings.FixOldBinarySubTypeOnInput)
144  {
145  subType = BsonBinarySubType.Binary; // replace obsolete OldBinary with new Binary sub type
146  }
147  }
148 
149  var bytes = _bsonStream.ReadBytes(size);
150 
151  GuidRepresentation guidRepresentation;
152  switch (subType)
153  {
154  case BsonBinarySubType.UuidLegacy: guidRepresentation = _settings.GuidRepresentation; break;
155  case BsonBinarySubType.UuidStandard: guidRepresentation = GuidRepresentation.Standard; break;
156  default: guidRepresentation = GuidRepresentation.Unspecified; break;
157  }
158 
159  State = GetNextState();
160  return new BsonBinaryData(bytes, subType, guidRepresentation);
161  }

180  public override BsonType ReadBsonType()
181  {
182  if (Disposed) { ThrowObjectDisposedException(); }
183  if (State == BsonReaderState.Initial || State == BsonReaderState.ScopeDocument)
184  {
185  // there is an implied type of Document for the top level and for scope documents
186  CurrentBsonType = BsonType.Document;
187  State = BsonReaderState.Value;
188  return CurrentBsonType;
189  }
190  if (State != BsonReaderState.Type)
191  {
192  ThrowInvalidState("ReadBsonType", BsonReaderState.Type);
193  }
194 
195  if (_context.ContextType == ContextType.Array)
196  {
197  _context.CurrentArrayIndex++;
198  }
199 
200  try
201  {
202 
203  CurrentBsonType = _bsonStream.ReadBsonType();
204  }
205  catch (FormatException ex)
206  {
207  if (ex.Message.StartsWith("Detected unknown BSON type"))
208  {
209  // insert the element name into the error message
210  var periodIndex = ex.Message.IndexOf('.');
211  var dottedElementName = GenerateDottedElementName();
212  var message = ex.Message.Substring(0, periodIndex) + $" for fieldname \"{dottedElementName}\"" + ex.Message.Substring(periodIndex);
213  throw new FormatException(message);
214  }
215  throw;
216  }
217 
218  if (CurrentBsonType == BsonType.EndOfDocument)
219  {
220  switch (_context.ContextType)
221  {
222  case ContextType.Array:
223  State = BsonReaderState.EndOfArray;
224  return BsonType.EndOfDocument;
225  case ContextType.Document:
226  case ContextType.ScopeDocument:
227  State = BsonReaderState.EndOfDocument;
228  return BsonType.EndOfDocument;
229  default:
230  var message = string.Format("BsonType EndOfDocument is not valid when ContextType is {0}.", _context.ContextType);
231  throw new FormatException(message);
232  }
233  }
234  else
235  {
236  switch (_context.ContextType)
237  {
238  case ContextType.Array:
239  _bsonStream.SkipCString(); // ignore array element names
240  State = BsonReaderState.Value;
241  break;
242  case ContextType.Document:
243  case ContextType.ScopeDocument:
244  State = BsonReaderState.Name;
245  break;
246  default:
247  throw new BsonInternalException("Unexpected ContextType.");
248  }
249 
250  return CurrentBsonType;
251  }
252  }

689  public override void SkipValue()
690  {
691  if (Disposed) { ThrowObjectDisposedException(); }
692  if (State != BsonReaderState.Value)
693  {
694  ThrowInvalidState("SkipValue", BsonReaderState.Value);
695  }
696 
697  int skip;
698  switch (CurrentBsonType)
699  {
700  case BsonType.Array: skip = ReadSize() - 4; break;
701  case BsonType.Binary: skip = ReadSize() + 1; break;
702  case BsonType.Boolean: skip = 1; break;
703  case BsonType.DateTime: skip = 8; break;
704  case BsonType.Document: skip = ReadSize() - 4; break;
705  case BsonType.Decimal128: skip = 16; break;
706  case BsonType.Double: skip = 8; break;
707  case BsonType.Int32: skip = 4; break;
708  case BsonType.Int64: skip = 8; break;
709  case BsonType.JavaScript: skip = ReadSize(); break;
710  case BsonType.JavaScriptWithScope: skip = ReadSize() - 4; break;
711  case BsonType.MaxKey: skip = 0; break;
712  case BsonType.MinKey: skip = 0; break;
713  case BsonType.Null: skip = 0; break;
714  case BsonType.ObjectId: skip = 12; break;
715  case BsonType.RegularExpression: _bsonStream.SkipCString(); _bsonStream.SkipCString(); skip = 0; break;
716  case BsonType.String: skip = ReadSize(); break;
717  case BsonType.Symbol: skip = ReadSize(); break;
718  case BsonType.Timestamp: skip = 8; break;
719  case BsonType.Undefined: skip = 0; break;
720  default: throw new BsonInternalException("Unexpected BsonType.");
721  }
722  _bsonStream.Seek(skip, SeekOrigin.Current);
723 
724  State = BsonReaderState.Type;
725  }

748  {
749  string elementName;

751  {
752  try

754  elementName = _bsonStream.ReadCString(Utf8Encodings.Lenient);
755  }
756  catch

773  private string GenerateDottedElementName(BsonBinaryReaderContext context, string elementName)
774  {
775  if (context.ContextType == ContextType.Document)
776  {
777  return GenerateDottedElementName(context.ParentContext, (context.CurrentElementName ?? "?") + "." + elementName);
778  }
779  else if (context.ContextType == ContextType.Array)
780  {
781  var indexElementName = context.CurrentArrayIndex.ToString(NumberFormatInfo.InvariantInfo);
782  return GenerateDottedElementName(context.ParentContext, indexElementName + "." + elementName);
783  }
784  else if (context.ParentContext != null)
785  {
786  return GenerateDottedElementName(context.ParentContext, "?." + elementName);
787  }
788  else
789  {
790  return elementName;
