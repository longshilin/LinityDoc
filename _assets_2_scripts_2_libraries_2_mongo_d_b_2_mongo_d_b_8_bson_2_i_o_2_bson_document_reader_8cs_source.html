116  public override BsonType ReadBsonType()
117  {
118  if (Disposed) { ThrowObjectDisposedException(); }
119  if (State == BsonReaderState.Initial || State == BsonReaderState.ScopeDocument)
120  {
121  // there is an implied type of Document for the top level and for scope documents
122  CurrentBsonType = BsonType.Document;
123  State = BsonReaderState.Value;
124  return CurrentBsonType;
125  }
126  if (State != BsonReaderState.Type)
127  {
128  ThrowInvalidState("ReadBsonType", BsonReaderState.Type);
129  }
130 
131  switch (_context.ContextType)
132  {
133  case ContextType.Array:
134  if (!_context.TryGetNextValue(out _currentValue))
135  {
136  State = BsonReaderState.EndOfArray;
137  return BsonType.EndOfDocument;
138  }
139  State = BsonReaderState.Value;
140  break;
141  case ContextType.Document:
142  BsonElement currentElement;
143  if (!_context.TryGetNextElement(out currentElement))
144  {
145  State = BsonReaderState.EndOfDocument;
146  return BsonType.EndOfDocument;
147  }
148  CurrentName = currentElement.Name;
149  _currentValue = currentElement.Value;
150  State = BsonReaderState.Name;
151  break;
152  default:
153  throw new BsonInternalException("Invalid ContextType.");
154  }
155 
156  CurrentBsonType = _currentValue.BsonType;
157  return CurrentBsonType;
158  }

220  public override void ReadEndArray()
221  {
222  if (Disposed) { ThrowObjectDisposedException(); }
223  if (_context.ContextType != ContextType.Array)
224  {
225  ThrowInvalidContextType("ReadEndArray", _context.ContextType, ContextType.Array);
226  }
227  if (State == BsonReaderState.Type)
228  {
229  ReadBsonType(); // will set state to EndOfArray if at end of array
230  }
231  if (State != BsonReaderState.EndOfArray)
232  {
233  ThrowInvalidState("ReadEndArray", BsonReaderState.EndOfArray);
234  }
235 
236  _context = _context.PopContext();
237  switch (_context.ContextType)
238  {
239  case ContextType.Array: State = BsonReaderState.Type; break;
240  case ContextType.Document: State = BsonReaderState.Type; break;
241  case ContextType.TopLevel: State = BsonReaderState.Done; break;
242  default: throw new BsonInternalException("Unexpected ContextType.");
243  }
244  }
