300  public virtual void WriteRawBsonArray(IByteBuffer slice)
301  {
302  // overridden in BsonBinaryWriter to write the raw bytes to the stream
303  // for all other streams, deserialize the raw bytes and serialize the resulting array instead
304 
305  using (var chunkSource = new InputBufferChunkSource(BsonChunkPool.Default))
306  using (var buffer = new MultiChunkBuffer(chunkSource))
307  using (var stream = new ByteBufferStream(buffer))
308  {
309  // wrap the array in a fake document so we can deserialize it
310  var documentLength = slice.Length + 8;
311  buffer.EnsureCapacity(documentLength);
312  stream.WriteInt32(documentLength);
313  stream.WriteBsonType(BsonType.Array);
314  stream.WriteByte((byte)'x');
315  stream.WriteByte(0);
316  stream.WriteSlice(slice);
317  stream.WriteByte(0);
318  buffer.MakeReadOnly();
319 
320  stream.Position = 0;
321  using (var reader = new BsonBinaryReader(stream, BsonBinaryReaderSettings.Defaults))
322  {
323  var deserializationContext = BsonDeserializationContext.CreateRoot(reader);
324  reader.ReadStartDocument();
325  reader.ReadName("x");
326  var array = BsonArraySerializer.Instance.Deserialize(deserializationContext);
327  reader.ReadEndDocument();
328 
329  var serializationContext = BsonSerializationContext.CreateRoot(this);
330  BsonArraySerializer.Instance.Serialize(serializationContext, array);
331  }
332  }
333  }

425  protected void ThrowInvalidContextType(
426  string methodName,
427  ContextType actualContextType,
428  params ContextType[] validContextTypes)
429  {
430  var validContextTypesString = string.Join(" or ", validContextTypes.Select(c => c.ToString()).ToArray());

442  protected void ThrowInvalidState(string methodName, params BsonWriterState[] validStates)
443  {
444  string message;
445  if (_state == BsonWriterState.Initial || _state == BsonWriterState.ScopeDocument || _state == BsonWriterState.Done)
446  {
447  if (!methodName.StartsWith("End", StringComparison.Ordinal) && methodName != "WriteName")
448  {
449  var typeName = methodName.Substring(5);
450  if (typeName.StartsWith("Start", StringComparison.Ordinal))
451  {
452  typeName = typeName.Substring(5);
453  }
454  var article = "A";
455  if (new char[] { 'A', 'E', 'I', 'O', 'U' }.Contains(typeName[0]))
456  {
457  article = "An";
458  }
459  message = string.Format(
460  "{0} {1} value cannot be written to the root level of a BSON document.",
461  article, typeName);
462  throw new InvalidOperationException(message);
463  }
464  }
465 
466  var validStatesString = string.Join(" or ", validStates.Select(s => s.ToString()).ToArray());
467  message = string.Format(
468  "{0} can only be called when State is {1}, not when State is {2}",
469  methodName, validStatesString, _state);
470  throw new InvalidOperationException(message);
471  }
