134  {

137  throw new ArgumentNullException("buffer");
138  }

162

277  private int FindNullByte()
278  {
279  var position = _position;
280  while (position < _length)
281  {
282  var segment = _buffer.AccessBackingBytes(position);
283  var endOfSegmentIndex = segment.Offset + segment.Count;
284  for (var index = segment.Offset; index < endOfSegmentIndex; index++)
285  {
286  if (segment.Array[index] == 0)
287  {
288  return position + (index - segment.Offset);
289  }
290  }
291  position += segment.Count;
292  }
293 
294  throw new EndOfStreamException();
295  }

366  public override ArraySegment<byte> ReadCStringBytes()
367  {
368  ThrowIfDisposed();
369  ThrowIfEndOfStream(1);
370 
371  var segment = _buffer.AccessBackingBytes(_position);
372  var index = Array.IndexOf<byte>(segment.Array, 0, segment.Offset, segment.Count);
373  if (index != -1)
374  {
375  var length = index - segment.Offset;
376  _position += length + 1; // advance over the null byte
377  return new ArraySegment<byte>(segment.Array, segment.Offset, length); // without the null byte
378  }
379  else
380  {
381  var nullPosition = FindNullByte();
382  var length = nullPosition - _position;
383  var cstring = ReadBytes(length + 1); // advance over the null byte
384  return new ArraySegment<byte>(cstring, 0, length); // without the null byte
385  }

420  {
421  ThrowIfDisposed();
422  ThrowIfEndOfStream(4);
423 
424  var segment = _buffer.AccessBackingBytes(_position);
425  if (segment.Count >= 4)
426  {
427  _position += 4;

491  public override string ReadString(UTF8Encoding encoding)
492  {
493  if (encoding == null)
494  {
495  throw new ArgumentNullException("encoding");
496  }
497  ThrowIfDisposed();
498 
499  var length = ReadInt32();
500  if (length <= 0)
501  {
502  var message = string.Format("Invalid string length: {0}.", length);
503  throw new FormatException(message);
504  }
505 
506  var segment = _buffer.AccessBackingBytes(_position);
507  if (segment.Count >= length)
508  {
509  ThrowIfEndOfStream(length);
510  if (segment.Array[segment.Offset + length - 1] != 0)
511  {
512  throw new FormatException("String is missing terminating null byte.");
513  }
514  _position += length;
515  return Utf8Helper.DecodeUtf8String(segment.Array, segment.Offset, length - 1, encoding);
516  }
517  else
518  {
519  var bytes = length <= _tempUtf8.Length ? _tempUtf8 : new byte[length];
520  this.ReadBytes(bytes, 0, length);
521  if (bytes[length - 1] != 0)
522  {
523  throw new FormatException("String is missing terminating null byte.");
524  }
525  return Utf8Helper.DecodeUtf8String(bytes, 0, length - 1, encoding);

538  public override void WriteCString(string value)
539  {
540  if (value == null)
541  {
542  throw new ArgumentNullException("value");
543  }
544  ThrowIfDisposed();
545 
546  var maxLength = CStringUtf8Encoding.GetMaxByteCount(value.Length) + 1;
547  PrepareToWrite(maxLength);
548 
549  int actualLength;
550  var segment = _buffer.AccessBackingBytes(_position);
551  if (segment.Count >= maxLength)
552  {
553  actualLength = CStringUtf8Encoding.GetBytes(value, segment.Array, segment.Offset, Utf8Encodings.Strict);
554  segment.Array[segment.Offset + actualLength] = 0;
555  }
556  else
557  {
558  byte[] bytes;
559  if (maxLength <= _tempUtf8.Length)
560  {
561  bytes = _tempUtf8;

572  }

576  }

579  }

684  {

688  PrepareToWrite(maxLength);
689

701  segment.Array[segment.Offset + 4 + actualLength] = 0;
702  }
