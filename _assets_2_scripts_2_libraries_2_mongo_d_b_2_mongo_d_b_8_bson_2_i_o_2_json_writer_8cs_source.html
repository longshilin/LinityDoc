113  public override void WriteBinaryData(BsonBinaryData binaryData)
114  {
115  if (Disposed) { throw new ObjectDisposedException("JsonWriter"); }
116  if (State != BsonWriterState.Value && State != BsonWriterState.Initial)
117  {
118  ThrowInvalidState("WriteBinaryData", BsonWriterState.Value, BsonWriterState.Initial);
119  }
120 
121  var subType = binaryData.SubType;
122  var bytes = binaryData.Bytes;
123  var guidRepresentation = binaryData.GuidRepresentation;
124 
125  WriteNameHelper(Name);
126  switch (Settings.OutputMode)
127  {
128  case JsonOutputMode.Strict:
129  _textWriter.Write("{{ \"$binary\" : \"{0}\", \"$type\" : \"{1}\" }}", Convert.ToBase64String(bytes), ((int)subType).ToString("x2"));
130  break;
131 
132  case JsonOutputMode.Shell:
133  default:
134  switch (subType)
135  {
136  case BsonBinarySubType.UuidLegacy:
137  case BsonBinarySubType.UuidStandard:
138  _textWriter.Write(GuidToString(subType, bytes, guidRepresentation));
139  break;
140 
141  default:
142  _textWriter.Write("new BinData({0}, \"{1}\")", (int)subType, Convert.ToBase64String(bytes));
143  break;
144  }
145  break;
146  }
147 
148  State = GetNextState();
149  }

705  // private methods
706  private string EscapedString(string value)
707  {
708  if (value.All(c => !NeedsEscaping(c)))
709  {
710  return value;
711  }
712 
713  var sb = new StringBuilder(value.Length);
714 
715  foreach (char c in value)
716  {
717  switch (c)
718  {
719  case '"': sb.Append("\\\""); break;
720  case '\\': sb.Append("\\\\"); break;
721  case '\b': sb.Append("\\b"); break;
722  case '\f': sb.Append("\\f"); break;
723  case '\n': sb.Append("\\n"); break;
724  case '\r': sb.Append("\\r"); break;
725  case '\t': sb.Append("\\t"); break;
726  default:
727  switch (CharUnicodeInfo.GetUnicodeCategory(c))
728  {
729  case UnicodeCategory.UppercaseLetter:
730  case UnicodeCategory.LowercaseLetter:
731  case UnicodeCategory.TitlecaseLetter:
732  case UnicodeCategory.OtherLetter:
733  case UnicodeCategory.DecimalDigitNumber:
734  case UnicodeCategory.LetterNumber:
735  case UnicodeCategory.OtherNumber:
736  case UnicodeCategory.SpaceSeparator:
737  case UnicodeCategory.ConnectorPunctuation:
738  case UnicodeCategory.DashPunctuation:
739  case UnicodeCategory.OpenPunctuation:
740  case UnicodeCategory.ClosePunctuation:
741  case UnicodeCategory.InitialQuotePunctuation:
742  case UnicodeCategory.FinalQuotePunctuation:
743  case UnicodeCategory.OtherPunctuation:
744  case UnicodeCategory.MathSymbol:
745  case UnicodeCategory.CurrencySymbol:
746  case UnicodeCategory.ModifierSymbol:
747  case UnicodeCategory.OtherSymbol:
748  sb.Append(c);
749  break;
750  default:
751  sb.AppendFormat("\\u{0:x4}", (int)c);
752  break;
753  }
754  break;
755  }
756  }
757 
758  return sb.ToString();
759  }

773  private string GuidToString(BsonBinarySubType subType, byte[] bytes, GuidRepresentation guidRepresentation)
774  {
775  if (bytes.Length != 16)
776  {
777  var message = string.Format("Length of binary subtype {0} must be 16, not {1}.", subType, bytes.Length);
778  throw new ArgumentException(message);
779  }
780  if (subType == BsonBinarySubType.UuidLegacy && guidRepresentation == GuidRepresentation.Standard)
781  {
782  throw new ArgumentException("GuidRepresentation for binary subtype UuidLegacy must not be Standard.");
783  }
784  if (subType == BsonBinarySubType.UuidStandard && guidRepresentation != GuidRepresentation.Standard)
785  {
786  var message = string.Format("GuidRepresentation for binary subtype UuidStandard must be Standard, not {0}.", guidRepresentation);
787  throw new ArgumentException(message);
788  }
789 
790  if (guidRepresentation == GuidRepresentation.Unspecified)
791  {
792  var s = BsonUtils.ToHexString(bytes);
793  var parts = new string[]
794  {
795  s.Substring(0, 8),
796  s.Substring(8, 4),
797  s.Substring(12, 4),
798  s.Substring(16, 4),
799  s.Substring(20, 12)
800  };
801  return string.Format("HexData({0}, \"{1}\")", (int)subType, string.Join("-", parts));
802  }
803  else
804  {
805  string uuidConstructorName;
806  switch (guidRepresentation)
807  {
808  case GuidRepresentation.CSharpLegacy: uuidConstructorName = "CSUUID"; break;
809  case GuidRepresentation.JavaLegacy: uuidConstructorName = "JUUID"; break;
810  case GuidRepresentation.PythonLegacy: uuidConstructorName = "PYUUID"; break;
811  case GuidRepresentation.Standard: uuidConstructorName = "UUID"; break;
812  default: throw new BsonInternalException("Unexpected GuidRepresentation");
813  }
814  var guid = GuidConverter.FromBytes(bytes, guidRepresentation);
815  return string.Format("{0}(\"{1}\")", uuidConstructorName, guid.ToString());
816  }
817  }

819  private bool NeedsEscaping(char c)
820  {
821  switch (c)
822  {
823  case '"':
824  case '\\':
825  case '\b':
826  case '\f':
827  case '\n':
828  case '\r':
829  case '\t':
830  return true;
831 
832  default:
833  switch (CharUnicodeInfo.GetUnicodeCategory(c))
834  {
835  case UnicodeCategory.UppercaseLetter:
836  case UnicodeCategory.LowercaseLetter:
837  case UnicodeCategory.TitlecaseLetter:
838  case UnicodeCategory.OtherLetter:
839  case UnicodeCategory.DecimalDigitNumber:
840  case UnicodeCategory.LetterNumber:
841  case UnicodeCategory.OtherNumber:
842  case UnicodeCategory.SpaceSeparator:
843  case UnicodeCategory.ConnectorPunctuation:
844  case UnicodeCategory.DashPunctuation:
845  case UnicodeCategory.OpenPunctuation:
846  case UnicodeCategory.ClosePunctuation:
847  case UnicodeCategory.InitialQuotePunctuation:
848  case UnicodeCategory.FinalQuotePunctuation:
849  case UnicodeCategory.OtherPunctuation:
850  case UnicodeCategory.MathSymbol:
851  case UnicodeCategory.CurrencySymbol:

862  private void WriteNameHelper(string name)
863  {
864  switch (_context.ContextType)
865  {
866  case ContextType.Array:
867  // don't write Array element names in Json
868  if (_context.HasElements)
869  {
870  _textWriter.Write(", ");
871  }
872  break;
873  case ContextType.Document:
874  case ContextType.ScopeDocument:
875  if (_context.HasElements)
876  {
877  _textWriter.Write(",");
878  }
879  if (Settings.Indent)
880  {
881  _textWriter.Write(Settings.NewLineChars);
882  _textWriter.Write(_context.Indentation);
883  }
884  else
885  {
886  _textWriter.Write(" ");
887  }
888  WriteQuotedString(name);
889  _textWriter.Write(" : ");
890  break;
891  case ContextType.TopLevel:
892  break;
893  default:
894  throw new BsonInternalException("Invalid ContextType.");
895  }
896
