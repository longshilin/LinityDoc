488  public virtual int CompareTo(BsonArray rhs)
489  {
490  if (rhs == null) { return 1; }
491 
492  // lhs and rhs might be subclasses of BsonArray
493  using (var lhsEnumerator = GetEnumerator())
494  using (var rhsEnumerator = rhs.GetEnumerator())
495  {
496  while (true)
497  {
498  var lhsHasNext = lhsEnumerator.MoveNext();
499  var rhsHasNext = rhsEnumerator.MoveNext();
500  if (!lhsHasNext && !rhsHasNext) { return 0; }
501  if (!lhsHasNext) { return -1; }
502  if (!rhsHasNext) { return 1; }
503 
504  var lhsValue = lhsEnumerator.Current;
505  var rhsValue = rhsEnumerator.Current;
506  var result = lhsValue.CompareTo(rhsValue);
507  if (result != 0) { return result; }
508  }
509  }
510  }
