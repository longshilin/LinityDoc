38  public BsonRegularExpression(string pattern)
39  {
40  if (pattern == null)
41  {
42  throw new ArgumentNullException("pattern");
43  }
44  if (pattern.Length > 0 && pattern[0] == '/')
45  {
46  var index = pattern.LastIndexOf('/');
47  var escaped = pattern.Substring(1, index - 1);
48  var unescaped = (escaped == "(?:)") ? "" : escaped.Replace("\\/", "/");
49  _pattern = unescaped;
50  _options = pattern.Substring(index + 1);
51  }
52  else
53  {
54  _pattern = pattern;
55  _options = "";
56  }
57  }
58 
64  public BsonRegularExpression(string pattern, string options)
65  {
66  if (pattern == null)

68  throw new ArgumentNullException("pattern");
69  }

72  }

78  public BsonRegularExpression(Regex regex)
79  {
80  if (regex == null)
81  {
82  throw new ArgumentNullException("regex");
83  }
84  _pattern = regex.ToString();
85  _options = "";
86  if ((regex.Options & RegexOptions.IgnoreCase) != 0)
87  {
88  _options += "i";
89  }
90  if ((regex.Options & RegexOptions.Multiline) != 0)
91  {
92  _options += "m";
93  }
94  if ((regex.Options & RegexOptions.IgnorePatternWhitespace) != 0)
95  {
96  _options += "x";
97  }
98  if ((regex.Options & RegexOptions.Singleline) != 0)
99  {
100  _options += "s";
101  }
102  }
