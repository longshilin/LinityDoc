1177  {
1178  // optimize away the call to MapToBsonValue for the most common cases
1179  if (value == null)
1180  {
1181  return BsonNull.Value;
1182  }
1183  else if (value is BsonValue)
1184  {
1185  return (BsonValue)value;
1186  }
1187  else if (value is int)
1188  {
1189  return (BsonInt32)(int)value;
1190  }
1191  else if (value is string)
1192  {
1193  return (BsonString)(string)value;
1194  }
1195  else if (value is bool)
1196  {
1197  return (BsonBoolean)((bool)value);
1198  }
1199  else if (value is DateTime)
1200  {
1201  return new BsonDateTime((DateTime)value);
1202  }
1203  else if (value is long)
1204  {
1205  return (BsonInt64)(long)value;
1206  }
1207  else if (value is double)
