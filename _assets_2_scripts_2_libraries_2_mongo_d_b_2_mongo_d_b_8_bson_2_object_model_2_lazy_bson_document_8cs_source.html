180  private IEnumerable<BsonElement> MaterializeThisLevel()
181  {
182  var elements = new List<BsonElement>();
183 
184  using (var stream = new ByteBufferStream(_slice, ownsBuffer: false))
185  using (var bsonReader = new BsonBinaryReader(stream, _readerSettings))
186  {
187  var context = BsonDeserializationContext.CreateRoot(bsonReader);
188 
189  bsonReader.ReadStartDocument();
190  BsonType bsonType;
191  while ((bsonType = bsonReader.ReadBsonType()) != BsonType.EndOfDocument)
192  {
193  var name = bsonReader.ReadName();
194  BsonValue value;
195  switch (bsonType)
196  {
197  case BsonType.Array: value = DeserializeLazyBsonArray(bsonReader); break;
198  case BsonType.Document: value = DeserializeLazyBsonDocument(bsonReader); break;
199  default: value = BsonValueSerializer.Instance.Deserialize(context); break;
200  }
201  elements.Add(new BsonElement(name, value));
202  }
203  bsonReader.ReadEndDocument();
204  }
205 
206  return elements;
207  }
