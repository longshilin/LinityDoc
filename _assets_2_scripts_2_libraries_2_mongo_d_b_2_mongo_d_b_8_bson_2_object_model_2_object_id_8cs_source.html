278  public static byte[] Pack(int timestamp, int machine, short pid, int increment)
279  {
280  if ((machine & 0xff000000) != 0)
281  {
282  throw new ArgumentOutOfRangeException("machine", "The machine value must be between 0 and 16777215 (it must fit in 3 bytes).");
283  }
284  if ((increment & 0xff000000) != 0)
285  {
286  throw new ArgumentOutOfRangeException("increment", "The increment value must be between 0 and 16777215 (it must fit in 3 bytes).");
287  }

289  byte[] bytes = new byte[12];
290  bytes[0] = (byte)(timestamp >> 24);
291  bytes[1] = (byte)(timestamp >> 16);
292  bytes[2] = (byte)(timestamp >> 8);
293  bytes[3] = (byte)(timestamp);
294  bytes[4] = (byte)(machine >> 16);

361  {
