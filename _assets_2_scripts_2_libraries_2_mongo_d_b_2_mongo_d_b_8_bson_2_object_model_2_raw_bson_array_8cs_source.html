530  public BsonArray Materialize(BsonBinaryReaderSettings binaryReaderSettings)
531  {
532  ThrowIfDisposed();
533 
534  // because BsonBinaryReader can only read documents at the top level we have to wrap the RawBsonArray in a document
535  var document = new BsonDocument("array", this);
536  var bytes = document.ToBson();
537 
538  using (var stream = new MemoryStream(bytes))
539  using (var reader = new BsonBinaryReader(stream, binaryReaderSettings))
540  {
541  var context = BsonDeserializationContext.CreateRoot(reader);
542  var materializedDocument = BsonDocumentSerializer.Instance.Deserialize(context);
543  return materializedDocument["array"].AsBsonArray;
544  }
545  }

602  protected virtual void Dispose(bool disposing)
603  {

620  }

657  private BsonValue DeserializeBsonValue(BsonDeserializationContext context)
658  {
659  var bsonReader = context.Reader;
660  switch (bsonReader.GetCurrentBsonType())
661  {
662  case BsonType.Array: return DeserializeRawBsonArray(bsonReader);
663  case BsonType.Document: return DeserializeRawBsonDocument(bsonReader);
664  default: return BsonValueSerializer.Instance.Deserialize(context);
