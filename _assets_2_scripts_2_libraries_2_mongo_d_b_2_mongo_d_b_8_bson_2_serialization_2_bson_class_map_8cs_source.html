337  public static BsonClassMap LookupClassMap(Type classType)
338  {
339  if (classType == null)
340  {
341  throw new ArgumentNullException("classType");
342  }
343 
344  BsonSerializer.ConfigLock.EnterReadLock();
345  try
346  {
347  BsonClassMap classMap;
348  if (__classMaps.TryGetValue(classType, out classMap))
349  {
350  if (classMap.IsFrozen)
351  {
352  return classMap;
353  }
354  }
355  }
356  finally
357  {
358  BsonSerializer.ConfigLock.ExitReadLock();
359  }
360 
361  BsonSerializer.ConfigLock.EnterWriteLock();
362  try
363  {
364  BsonClassMap classMap;
365  if (!__classMaps.TryGetValue(classType, out classMap))
366  {
367  // automatically create a classMap for classType and register it
368  var classMapDefinition = typeof(BsonClassMap<>);
369  var classMapType = classMapDefinition.MakeGenericType(classType);
370  classMap = (BsonClassMap)Activator.CreateInstance(classMapType);
371  classMap.AutoMap();
372  RegisterClassMap(classMap);
373  }
374  return classMap.Freeze();
375  }
376  finally
377  {
378  BsonSerializer.ConfigLock.ExitWriteLock();
379  }
380  }

454  public BsonClassMap Freeze()
455  {
456  BsonSerializer.ConfigLock.EnterReadLock();
457  try
458  {
459  if (_frozen)
460  {
461  return this;
462  }
463  }
464  finally
465  {
466  BsonSerializer.ConfigLock.ExitReadLock();
467  }
468 
469  BsonSerializer.ConfigLock.EnterWriteLock();
470  try
471  {
472  if (!_frozen)
473  {
474  __freezeNestingLevel++;
475  try
476  {
477  var baseType = _classType.GetTypeInfo().BaseType;
478  if (baseType != null)
479  {
480  if (_baseClassMap == null)
481  {
482  _baseClassMap = LookupClassMap(baseType);
483  }
484  _baseClassMap.Freeze();
485  _discriminatorIsRequired |= _baseClassMap._discriminatorIsRequired;
486  _hasRootClass |= (_isRootClass || _baseClassMap.HasRootClass);
487  _allMemberMaps.AddRange(_baseClassMap.AllMemberMaps);
488  if (_baseClassMap.IgnoreExtraElements && _baseClassMap.IgnoreExtraElementsIsInherited)
489  {
490  _ignoreExtraElements = true;
491  _ignoreExtraElementsIsInherited = true;
492  }
493  }
494  _declaredMemberMaps = _declaredMemberMaps.OrderBy(m => m.Order).ToList(); // we're counting on OrderBy being a stable sort
495  _allMemberMaps.AddRange(_declaredMemberMaps);
496 
497  if (_idMemberMap == null)
498  {
499  // see if we can inherit the idMemberMap from our base class
500  if (_baseClassMap != null)
501  {
502  _idMemberMap = _baseClassMap.IdMemberMap;
503  }
504  }
505  else
506  {
507  if (_idMemberMap.ClassMap == this)
508  {
509  // conventions could have set this to an improper value
510  _idMemberMap.SetElementName("_id");
511  }
512  }
513 
514  if (_extraElementsMemberMap == null)
515  {
516  // see if we can inherit the extraElementsMemberMap from our base class
517  if (_baseClassMap != null)
518  {
519  _extraElementsMemberMap = _baseClassMap.ExtraElementsMemberMap;
520  }
521  }
522 
523  _extraElementsMemberIndex = -1;
524  for (int memberIndex = 0; memberIndex < _allMemberMaps.Count; ++memberIndex)
525  {
526  var memberMap = _allMemberMaps[memberIndex];
527  int conflictingMemberIndex;
528  if (!_elementTrie.TryGetValue(memberMap.ElementName, out conflictingMemberIndex))
529  {
530  _elementTrie.Add(memberMap.ElementName, memberIndex);
531  }
532  else
533  {
534  var conflictingMemberMap = _allMemberMaps[conflictingMemberIndex];
535  var fieldOrProperty = (memberMap.MemberInfo is FieldInfo) ? "field" : "property";
536  var conflictingFieldOrProperty = (conflictingMemberMap.MemberInfo is FieldInfo) ? "field" : "property";
537  var conflictingType = conflictingMemberMap.MemberInfo.DeclaringType;
538 
539  string message;
540  if (conflictingType == _classType)
541  {
542  message = string.Format(
543  "The {0} '{1}' of type '{2}' cannot use element name '{3}' because it is already being used by {4} '{5}'.",
544  fieldOrProperty, memberMap.MemberName, _classType.FullName, memberMap.ElementName, conflictingFieldOrProperty, conflictingMemberMap.MemberName);
545  }
546  else
547  {
548  message = string.Format(
549  "The {0} '{1}' of type '{2}' cannot use element name '{3}' because it is already being used by {4} '{5}' of type '{6}'.",
550  fieldOrProperty, memberMap.MemberName, _classType.FullName, memberMap.ElementName, conflictingFieldOrProperty, conflictingMemberMap.MemberName, conflictingType.FullName);
551  }
552  throw new BsonSerializationException(message);
553  }
554  if (memberMap == _extraElementsMemberMap)
555  {
556  _extraElementsMemberIndex = memberIndex;
557  }
558  }
559 
560  // mark this classMap frozen before we start working on knownTypes
561  // because we might get back to this same classMap while processing knownTypes
562  foreach (var creatorMap in _creatorMaps)
563  {
564  creatorMap.Freeze();
565  }
566  foreach (var memberMap in _declaredMemberMaps)
567  {
568  memberMap.Freeze();
569  }
570  _frozen = true;
571 
572  // use a queue to postpone processing of known types until we get back to the first level call to Freeze
573  // this avoids infinite recursion when going back down the inheritance tree while processing known types
574  foreach (var knownType in _knownTypes)
575  {
576  __knownTypesQueue.Enqueue(knownType);
577  }
578 
579  // if we are back to the first level go ahead and process any queued known types
580  if (__freezeNestingLevel == 1)
581  {
582  while (__knownTypesQueue.Count != 0)
583  {
584  var knownType = __knownTypesQueue.Dequeue();
585  LookupClassMap(knownType); // will AutoMap and/or Freeze knownType if necessary
586  }
587  }
588  }
589  finally
590  {
591  __freezeNestingLevel--;
592  }
593  }
594  }
595  finally
596  {
597  BsonSerializer.ConfigLock.ExitWriteLock();
598  }
599  return this;
600  }

880  public BsonMemberMap MapMember(MemberInfo memberInfo)
881  {
882  if (memberInfo == null)
883  {
884  throw new ArgumentNullException("memberInfo");
885  }
886  if (!(memberInfo is FieldInfo) && !(memberInfo is PropertyInfo))
887  {
888  throw new ArgumentException("MemberInfo must be either a FieldInfo or a PropertyInfo.", "memberInfo");
889  }
890  EnsureMemberInfoIsForThisClass(memberInfo);
891 
892  if (_frozen) { ThrowFrozenException(); }
893  var memberMap = _declaredMemberMaps.Find(m => m.MemberInfo == memberInfo);
894  if (memberMap == null)
895  {
896  memberMap = new BsonMemberMap(this, memberInfo);
897  _declaredMemberMaps.Add(memberMap);
898  }
899  return memberMap;
900  }

1006  public void AddKnownType(Type type)
1007  {

1009  {
1010  string message = string.Format("Class {0} cannot be assigned to Class {1}. Ensure that known types are derived from the mapped class.", type.FullName, _classType.FullName);
1011  throw new ArgumentNullException("type", message);
1012  }
1013 
1014  if (_frozen) { ThrowFrozenException(); }

1526  private static MemberInfo GetMemberInfoFromLambda<TMember>(Expression<Func<TClass, TMember>> memberLambda)
1527  {
1528  var body = memberLambda.Body;
1529  MemberExpression memberExpression;
1530  switch (body.NodeType)
1531  {
1532  case ExpressionType.MemberAccess:
1533  memberExpression = (MemberExpression)body;
1534  break;
1535  case ExpressionType.Convert:
1536  var convertExpression = (UnaryExpression)body;
1537  memberExpression = (MemberExpression)convertExpression.Operand;
1538  break;
1539  default:
1540  throw new BsonSerializationException("Invalid lambda expression");
1541  }
1542  var memberInfo = memberExpression.Member;
1543  if (memberInfo is PropertyInfo)
1544  {
1545  if (memberInfo.DeclaringType.GetTypeInfo().IsInterface)
1546  {
1547  memberInfo = FindPropertyImplementation((PropertyInfo)memberInfo, typeof(TClass));
1548  }
1549  }
1550  else if (!(memberInfo is FieldInfo))
1551  {
1552  throw new BsonSerializationException("Invalid lambda expression");
1553  }
1554  return memberInfo;
1555  }

1562  private static PropertyInfo FindPropertyImplementation(PropertyInfo interfacePropertyInfo, Type actualType)
1563  {
1564  var interfaceType = interfacePropertyInfo.DeclaringType;
1565 
1566 #if NETSTANDARD1_5 || NETSTANDARD1_6
1567  var actualTypeInfo = actualType.GetTypeInfo();
1568  var bindingFlags = BindingFlags.Instance | BindingFlags.NonPublic | BindingFlags.Public;
1569  var actualTypePropertyInfos = actualTypeInfo.GetMembers(bindingFlags).OfType<PropertyInfo>();
1570 
1571  var explicitlyImplementedPropertyName = $"{interfacePropertyInfo.DeclaringType.FullName}.{interfacePropertyInfo.Name}".Replace("+", ".");
1572  var explicitlyImplementedPropertyInfo = actualTypePropertyInfos
1573  .Where(p => p.Name == explicitlyImplementedPropertyName)
1574  .SingleOrDefault();
1575  if (explicitlyImplementedPropertyInfo != null)
1576  {
1577  return explicitlyImplementedPropertyInfo;
1578  }
1579 
1580  var implicitlyImplementedPropertyInfo = actualTypePropertyInfos
1581  .Where(p => p.Name == interfacePropertyInfo.Name && p.PropertyType == interfacePropertyInfo.PropertyType)
1582  .SingleOrDefault();
1583  if (implicitlyImplementedPropertyInfo != null)
1584  {
1585  return implicitlyImplementedPropertyInfo;
1586  }
1587 
1588  throw new BsonSerializationException($"Unable to find property info for property: '{interfacePropertyInfo.Name}'.");
1589 #else
1590  // An interface map must be used because because there is no
1591  // other officially documented way to derive the explicitly
1592  // implemented property name.
1593  var interfaceMap = actualType.GetInterfaceMap(interfaceType);
1594 
1595  var interfacePropertyAccessors = GetPropertyAccessors(interfacePropertyInfo);
1596 
1597  var actualPropertyAccessors = interfacePropertyAccessors.Select(interfacePropertyAccessor =>
1598  {
1599  var index = Array.IndexOf<MethodInfo>(interfaceMap.InterfaceMethods, interfacePropertyAccessor);
1600 
1601  return interfaceMap.TargetMethods[index];
1602  });
1603 
1604  // Binding must be done by accessor methods because interface
1605  // maps only map accessor methods and do not map properties.
1606  return actualType.GetTypeInfo().GetProperties(BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic)
1607  .Single(propertyInfo =>
1608  {
1609  // we are looking for a property that implements all the required accessors
1610  var propertyAccessors = GetPropertyAccessors(propertyInfo);
1611  return actualPropertyAccessors.All(x => propertyAccessors.Contains(x));
1612  });
1613 #endif
1614  }
