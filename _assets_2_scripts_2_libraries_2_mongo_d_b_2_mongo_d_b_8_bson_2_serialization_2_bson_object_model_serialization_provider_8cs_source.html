60  public override IBsonSerializer GetSerializer(Type type, IBsonSerializerRegistry serializerRegistry)
61  {
62  if (type == null)
63  {
64  throw new ArgumentNullException("type");
65  }
66  var typeInfo = type.GetTypeInfo();
67  if (typeInfo.IsGenericType && typeInfo.ContainsGenericParameters)
68  {
69  var message = string.Format("Generic type {0} has unassigned type parameters.", BsonUtils.GetFriendlyTypeName(type));
70  throw new ArgumentException(message, "type");
71  }
72 
73  IBsonSerializer serializer;
74  if (__serializers.TryGetValue(type, out serializer))
75  {
76  return serializer;
77  }
