57  public Delegate CompileCreatorDelegate<TClass>(Expression<Func<TClass, TClass>> creatorLambda, out IEnumerable<MemberInfo> arguments)
58  {
59  // transform c => expression (where c is the prototype parameter)
60  // to (p1, p2, ...) => expression' where expression' is expression with every c.X replaced by p#
61 
62  _prototypeParameter = creatorLambda.Parameters[0];
63  _parameters = new Dictionary<MemberInfo, ParameterExpression>();
64  var body = Visit(creatorLambda.Body);
65  var lambda = Expression.Lambda(body, _parameters.Values.ToArray());
66  var @delegate = lambda.Compile();
67 
68  arguments = _parameters.Keys.ToArray();
69  return @delegate;
70  }

77  public Delegate CompileFactoryMethodDelegate(MethodInfo methodInfo)
78  {

81

83  var body = Expression.Call(methodInfo, parameters);
84  var lambda = Expression.Lambda(body, parameters);

104  parameter = Expression.Parameter(node.Type, parameterName);
