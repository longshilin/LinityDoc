56  case ExpressionType.Convert:
57  case ExpressionType.ConvertChecked:
58  case ExpressionType.ArrayLength:

60  case ExpressionType.TypeAs:

64  case ExpressionType.Subtract:
65  case ExpressionType.SubtractChecked:
66  case ExpressionType.Multiply:
67  case ExpressionType.MultiplyChecked:

76  case ExpressionType.GreaterThan:

80  case ExpressionType.Coalesce:

86  case ExpressionType.TypeIs:

96  case ExpressionType.Call:
97  return this.VisitMethodCall((MethodCallExpression)node);
98  case ExpressionType.Lambda:
99  return this.VisitLambda((LambdaExpression)node);
100  case ExpressionType.New:
101  return this.VisitNew((NewExpression)node);
102  case ExpressionType.NewArrayInit:
103  case ExpressionType.NewArrayBounds:
104  return this.VisitNewArray((NewArrayExpression)node);
105  case ExpressionType.Invoke:

109  case ExpressionType.ListInit:

121  protected ReadOnlyCollection<Expression> Visit(ReadOnlyCollection<Expression> nodes)
122  {
123  List<Expression> list = null;
124  for (int i = 0, n = nodes.Count; i < n; i++)
125  {
126  Expression node = this.Visit(nodes[i]);
127  if (list != null)
128  {
129  list.Add(node);
130  }
131  else if (node != nodes[i])
132  {
133  list = new List<Expression>(n);
134  for (int j = 0; j < i; j++)
135  {
136  list.Add(nodes[j]);
137  }
138  list.Add(node);
139  }
140  }
141  if (list != null)
142  {
143  return list.AsReadOnly();
144  }
145  return nodes;
146  }

153  protected virtual Expression VisitBinary(BinaryExpression node)
154  {
155  Expression left = this.Visit(node.Left);
156  Expression right = this.Visit(node.Right);
157  Expression conversion = this.Visit(node.Conversion);
158  if (left != node.Left || right != node.Right || conversion != node.Conversion)
159  {
160  if (node.NodeType == ExpressionType.Coalesce && node.Conversion != null)
161  {
162  return Expression.Coalesce(left, right, conversion as LambdaExpression);
163  }
164  else
165  {
166  return Expression.MakeBinary(node.NodeType, left, right, node.IsLiftedToNull, node.Method);
167  }
168  }
169  return node;
170  }
