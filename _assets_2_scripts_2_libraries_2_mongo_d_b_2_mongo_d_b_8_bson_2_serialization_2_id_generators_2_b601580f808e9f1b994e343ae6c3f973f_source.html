124  public bool IsEmpty(object id)

131  var idBsonBinaryData = (BsonBinaryData)id;

133  if (subType != BsonBinarySubType.UuidLegacy && subType != BsonBinarySubType.UuidStandard)
134  {
135  throw new ArgumentOutOfRangeException("id", "The binary sub type of the id value passed to the BsonBinaryDataGuidGenerator IsEmpty method is not UuidLegacy or UuidStandard.");
