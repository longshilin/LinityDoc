47  public override TBsonValue Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
48  {
49  var bsonReader = context.Reader;
50 
51  var bsonType = bsonReader.GetCurrentBsonType();
52  if (bsonType == BsonType.Document && IsCSharpNullRepresentation(bsonReader))
53  {
54  // if IsCSharpNullRepresentation returns true it will have consumed the document representing C# null
55  return null;
56  }
57 
58  // handle BSON null for backward compatibility with existing data (new data would have _csharpnull)
59  if (bsonType == BsonType.Null && (args.NominalType != typeof(BsonValue) && args.NominalType != typeof(BsonNull)))
60  {
61  bsonReader.ReadNull();
62  return null;
63  }
64 
65  return _wrappedSerializer.Deserialize(context);
66  }

74  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, TBsonValue value)
75  {
76  if (value == null)
77  {
78  var bsonWriter = context.Writer;
79  bsonWriter.WriteStartDocument();
80  bsonWriter.WriteBoolean("_csharpnull", true);
81  bsonWriter.WriteEndDocument();
82  }
83  else

90  private bool IsCSharpNullRepresentation(IBsonReader bsonReader)
91  {
92  var bookmark = bsonReader.GetBookmark();
93  bsonReader.ReadStartDocument();
94  var bsonType = bsonReader.ReadBsonType();
95  if (bsonType == BsonType.Boolean)
96  {
97  var name = bsonReader.ReadName();
98  if (name == "_csharpnull" || name == "$csharpnull")
99  {
100  var value = bsonReader.ReadBoolean();
101  if (value)
102  {
103  bsonType = bsonReader.ReadBsonType();
104  if (bsonType == BsonType.EndOfDocument)
105  {
106  bsonReader.ReadEndDocument();
107  return true;
108  }
109  }
110  }
111  }
112 
113  bsonReader.ReturnToBookmark(bookmark);
114  return false;
115  }
