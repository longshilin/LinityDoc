83  public override byte Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
84  {
85  var bsonReader = context.Reader;
86  byte value;
87  var lostData = false;
88 
89  var bsonType = bsonReader.GetCurrentBsonType();
90  switch (bsonType)
91  {
92  case BsonType.Binary:
93  var bytes = bsonReader.ReadBytes();
94  if (bytes.Length != 1)
95  {
96  throw new FormatException("Binary data for Byte must be exactly one byte long.");
97  }
98  value = bytes[0];
99  break;
100 
101  case BsonType.Int32:
102  var int32Value = bsonReader.ReadInt32();
103  value = (byte)int32Value;
104  lostData = (int)value != int32Value;
105  break;
106 
107  case BsonType.Int64:
108  var int64Value = bsonReader.ReadInt64();
109  value = (byte)int64Value;
110  lostData = (int)value != int64Value;
111  break;
112 
113  case BsonType.String:
114  var s = bsonReader.ReadString();
115  if (s.Length == 1)
116  {
117  s = "0" + s;
118  }
119  value = byte.Parse(s, NumberStyles.HexNumber);
120  break;
121 
122  default:
123  throw CreateCannotDeserializeFromBsonTypeException(bsonType);
124  }
125 
126  if (lostData)
127  {
128  var message = string.Format("Data loss occurred when trying to convert from {0} to Byte.", bsonType);
129  throw new FormatException(message);
130  }
131 
132  return value;
133  }
