196  private TDictionary DeserializeArrayRepresentation(BsonDeserializationContext context)
197  {
198  var dictionary = CreateInstance();
199 
200  var bsonReader = context.Reader;
201  bsonReader.ReadStartArray();
202  while (bsonReader.ReadBsonType() != BsonType.EndOfDocument)
203  {
204  object key;
205  object value;
206 
207  var bsonType = bsonReader.GetCurrentBsonType();
208  switch (bsonType)
209  {
210  case BsonType.Array:
211  bsonReader.ReadStartArray();
212  key = _keySerializer.Deserialize(context);
213  value = _valueSerializer.Deserialize(context);
214  bsonReader.ReadEndArray();
215  break;
216 
217  case BsonType.Document:
218  key = null;
219  value = null;
220  _helper.DeserializeMembers(context, (elementName, flag) =>
221  {
222  switch (flag)
223  {
224  case Flags.Key: key = _keySerializer.Deserialize(context); break;
225  case Flags.Value: value = _valueSerializer.Deserialize(context); break;
226  }
227  });
228  break;
229 
230  default:
231  throw CreateCannotDeserializeFromBsonTypeException(bsonType);
232  }
233 
234  dictionary.Add(key, value);
235  }
236  bsonReader.ReadEndArray();
237 
238  return dictionary;
239  }

241  private TDictionary DeserializeDocumentRepresentation(BsonDeserializationContext context)
242  {
243  var dictionary = CreateInstance();
244  var bsonReader = context.Reader;
245  bsonReader.ReadStartDocument();
246  while (bsonReader.ReadBsonType() != BsonType.EndOfDocument)
247  {
248  var key = DeserializeKeyString(bsonReader.ReadName());
249  var value = _valueSerializer.Deserialize(context);
250  dictionary.Add(key, value);
251  }
252  bsonReader.ReadEndDocument();
253  return dictionary;

313  {
314  var keyDocument = new BsonDocument();

316  {
317  var context = BsonSerializationContext.CreateRoot(keyWriter);
318  keyWriter.WriteStartDocument();
319  keyWriter.WriteName("k");
320  _keySerializer.Serialize(context, key);
321  keyWriter.WriteEndDocument();

326  {
327  throw new BsonSerializationException("When using DictionaryRepresentation.Document key values must serialize as strings.");
328  }
329 
330  return (string)keyValue;
