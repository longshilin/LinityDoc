74  public override TValue Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
75  {
76  var bsonReader = context.Reader;
77  var nominalType = args.NominalType;
78  var actualType = _discriminatorConvention.GetActualType(bsonReader, nominalType);
79  var serializer = BsonSerializer.LookupSerializer(actualType);
80 
81  TValue value = default(TValue);
82  _helper.DeserializeMembers(context, (elementName, flag) =>
83  {
84  switch (flag)
85  {
86  case Flags.Discriminator:
87  bsonReader.SkipValue();
88  break;
89  case Flags.Value:
90  var valueDeserializationArgs = new BsonDeserializationArgs { NominalType = actualType };
91  value = (TValue)serializer.Deserialize(context, valueDeserializationArgs);
92  break;
93  }
94  });
95 
96  return value;
97  }

104  public bool IsPositionedAtDiscriminatedWrapper(BsonDeserializationContext context)
105  {
106  var bsonReader = context.Reader;
107  var bookmark = bsonReader.GetBookmark();
108 
109  try
110  {
111  if (bsonReader.GetCurrentBsonType() != BsonType.Document) { return false; }
112  var foundFields = _isPositionedHelper.DeserializeMembers(context, (elementName, flag) =>
113  {
114  context.Reader.SkipValue();
115  });
116  return foundFields == (Flags.Discriminator | Flags.Value);
117  }
118  finally
119  {
120  bsonReader.ReturnToBookmark(bookmark);
121  }
122  }

130  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, TValue value)
131  {
132  var bsonWriter = context.Writer;
133  var nominalType = args.NominalType;
134  var actualType = value.GetType();
135  var discriminator = _discriminatorConvention.GetDiscriminator(nominalType, actualType);
136 
137  bsonWriter.WriteStartDocument();
138  bsonWriter.WriteName(_discriminatorConvention.ElementName);
139  BsonValueSerializer.Instance.Serialize(context, discriminator);
140  bsonWriter.WriteName("_v");
141  args.NominalType = actualType;
142  _wrappedSerializer.Serialize(context, args, value);
143  bsonWriter.WriteEndDocument();
144  }
