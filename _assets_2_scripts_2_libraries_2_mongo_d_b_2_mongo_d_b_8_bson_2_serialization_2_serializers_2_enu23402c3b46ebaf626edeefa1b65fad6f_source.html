153  protected override TValue FinalizeResult(object accumulator)
154  {
155  // find and call a constructor that we can pass the accumulator to
156  var accumulatorType = accumulator.GetType();
157  foreach (var constructorInfo in typeof(TValue).GetTypeInfo().GetConstructors())
158  {
159  var parameterInfos = constructorInfo.GetParameters();
160  if (parameterInfos.Length == 1 && parameterInfos[0].ParameterType.GetTypeInfo().IsAssignableFrom(accumulatorType))
161  {
162  return (TValue)constructorInfo.Invoke(new object[] { accumulator });
163  }
164  }
165 
166  // otherwise try to find a no-argument constructor and an Add method
167  var valueTypeInfo = typeof(TValue).GetTypeInfo();
168  var noArgumentConstructorInfo = valueTypeInfo.GetConstructor(new Type[] { });
169  var addMethodInfo = typeof(TValue).GetTypeInfo().GetMethod("Add", new Type[] { typeof(TItem) });
170  if (noArgumentConstructorInfo != null && addMethodInfo != null)
171  {
172  var value = (TValue)noArgumentConstructorInfo.Invoke(new Type[] { });
173  foreach (var item in (IEnumerable<TItem>)accumulator)
174  {
175  addMethodInfo.Invoke(value, new object[] { item });
176  }
177  return value;
178  }
179 
180  var message = string.Format("Type '{0}' does not have a suitable constructor or Add method.", typeof(TValue).FullName);
181  throw new BsonSerializationException(message);
182  }
