239  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, TInterface value)
240  {
241  var bsonWriter = context.Writer;
242 
243  if (value == null)
244  {
245  bsonWriter.WriteNull();
246  }
247  else
248  {
249  var actualType = value.GetType();
250  if (actualType == typeof(TImplementation))
251  {
252  _lazyImplementationSerializer.Value.Serialize(context, (TImplementation)value);
253  }
254  else
255  {
256  var serializer = BsonSerializer.LookupSerializer(actualType);
257  serializer.Serialize(context, value);
258  }
259  }
