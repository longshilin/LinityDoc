107  public override ushort Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
108  {

110 
111  var bsonType = bsonReader.GetCurrentBsonType();
112  switch (bsonType)
113  {
114  case BsonType.Decimal128:
115  return _converter.ToUInt16(bsonReader.ReadDecimal128());
116 
117  case BsonType.Double:
118  return _converter.ToUInt16(bsonReader.ReadDouble());
119 
120  case BsonType.Int32:
121  return _converter.ToUInt16(bsonReader.ReadInt32());
122 
123  case BsonType.Int64:
124  return _converter.ToUInt16(bsonReader.ReadInt64());
125 
126  case BsonType.String:

129  default:

141  {

147  bsonWriter.WriteDecimal128(_converter.ToDecimal128(value));

150  case BsonType.Double:
151  bsonWriter.WriteDouble(_converter.ToDouble(value));
152  break;
153 
154  case BsonType.Int32:
155  bsonWriter.WriteInt32(_converter.ToInt32(value));
156  break;
157 
158  case BsonType.Int64:
159  bsonWriter.WriteInt64(_converter.ToInt64(value));
160  break;
161 
162  case BsonType.String:
163  bsonWriter.WriteString(JsonConvert.ToString(value));
164  break;
165 
166  default:

168  throw new BsonSerializationException(message);
