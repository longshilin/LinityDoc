540  void RecalculateDebugLimits () {
541  debugFloor = float.PositiveInfinity;
542  debugRoof = float.NegativeInfinity;
543 
544  bool ignoreSearchTree = !showSearchTree || debugPathData == null;
545  for (int i = 0; i < graphs.Length; i++) {
546  if (graphs[i] != null && graphs[i].drawGizmos) {
547  graphs[i].GetNodes(node => {
548  if (ignoreSearchTree || Pathfinding.Util.GraphGizmoHelper.InSearchTree(node, debugPathData, debugPathID)) {
549  if (debugMode == GraphDebugMode.Penalty) {
550  debugFloor = Mathf.Min(debugFloor, node.Penalty);
551  debugRoof = Mathf.Max(debugRoof, node.Penalty);
552  } else if (debugPathData != null) {
553  var rnode = debugPathData.GetPathNode(node);
554  switch (debugMode) {
555  case GraphDebugMode.F:
556  debugFloor = Mathf.Min(debugFloor, rnode.F);
557  debugRoof = Mathf.Max(debugRoof, rnode.F);
558  break;
559  case GraphDebugMode.G:
560  debugFloor = Mathf.Min(debugFloor, rnode.G);
561  debugRoof = Mathf.Max(debugRoof, rnode.G);
562  break;
563  case GraphDebugMode.H:
564  debugFloor = Mathf.Min(debugFloor, rnode.H);
565  debugRoof = Mathf.Max(debugRoof, rnode.H);
566  break;
567  }
568  }
569  }
570  });
571  }
572  }
573 
574  if (float.IsInfinity(debugFloor)) {
575  debugFloor = 0;
576  debugRoof = 1;
577  }
578 
579  // Make sure they are not identical, that will cause the color interpolation to fail
580  if (debugRoof-debugFloor < 1) debugRoof += 1;
581  }

586  private void OnDrawGizmos () {
587  // Make sure the singleton pattern holds
588  // Might not hold if the Awake method
589  // has not been called yet
590  if (active == null) active = this;
591 
592  if (active != this || graphs == null) {
593  return;
594  }
595 
596  // In Unity one can select objects in the scene view by simply clicking on them with the mouse.
597  // Graph gizmos interfere with this however. If we would draw a mesh here the user would
598  // not be able to select whatever was behind it because the gizmos would block them.
599  // (presumably Unity cannot associate the gizmos with the AstarPath component because we are using
600  // Graphics.DrawMeshNow to draw most gizmos). It turns out that when scene picking happens
601  // then Event.current.type will be 'mouseUp'. We will therefore ignore all events which are
602  // not repaint events to make sure that the gizmos do not interfere with any kind of scene picking.
603  // This will not have any visual impact as only repaint events will result in any changes on the screen.
604  // From testing it seems the only events that can happen during OnDrawGizmos are the mouseUp and repaint events.
605  if (Event.current.type != EventType.Repaint) return;
606 
607  AstarProfiler.StartProfile("OnDrawGizmos");
608 
609  if (workItems.workItemsInProgress || isScanning) {
610  // If updating graphs, graph info might not be valid right now
611  // so just draw the same thing as last frame
612  gizmos.DrawExisting();
613  } else {
614  if (showNavGraphs && !manualDebugFloorRoof) {
615  RecalculateDebugLimits();
616  }
617 
618  Profiler.BeginSample("Graph.OnDrawGizmos");
619  // Loop through all graphs and draw their gizmos
620  for (int i = 0; i < graphs.Length; i++) {
621  if (graphs[i] != null && graphs[i].drawGizmos)
622  ((NavmeshBase)(graphs[i])).OnDrawGizmos(gizmos, showNavGraphs);
623  }
624  Profiler.EndSample();
625 
626  if (showNavGraphs) {
627  euclideanEmbedding.OnDrawGizmos();
628  }
629  }

631  gizmos.FinalizeDraw();

878  public static int CalculateThreadCount (ThreadCount count) {
879 #if UNITY_WEBGL
880  return 0;
881 #else
882  if (count == ThreadCount.AutomaticLowLoad || count == ThreadCount.AutomaticHighLoad) {
883 #if ASTARDEBUG
884  Debug.Log(SystemInfo.systemMemorySize + " " + SystemInfo.processorCount + " " + SystemInfo.processorType);
885 #endif
886 
887  int logicalCores = Mathf.Max(1, SystemInfo.processorCount);
888  int memory = SystemInfo.systemMemorySize;
889 
890  if (memory <= 0) {
891  Debug.LogError("Machine reporting that is has <= 0 bytes of RAM. This is definitely not true, assuming 1 GiB");
892  memory = 1024;
893  }
894 
895  if (logicalCores <= 1) return 0;
896 
897  if (memory <= 512) return 0;
898 
899  if (count == ThreadCount.AutomaticHighLoad) {
900  if (memory <= 1024) logicalCores = System.Math.Min(logicalCores, 2);
901  } else {
902  //Always run at at most processorCount-1 threads (one core reserved for unity thread).
903  // Many computers use hyperthreading, so dividing by two is used to remove the hyperthreading cores, pathfinding
904  // doesn't scale well past the number of physical cores anyway
905  logicalCores /= 2;
906  logicalCores = Mathf.Max(1, logicalCores);
907 
908  if (memory <= 1024) logicalCores = System.Math.Min(logicalCores, 2);
909 
910  logicalCores = System.Math.Min(logicalCores, 6);
911  }
912 
913  return logicalCores;
914  } else {
915  int val = (int)count;
916  return val;
917  }
918 #endif
919  }
