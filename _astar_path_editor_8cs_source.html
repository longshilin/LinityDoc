594  void DrawGraphInfoArea (GraphEditor graphEditor) {
595  graphEditor.infoFadeArea.Begin();
596 
597  if (graphEditor.infoFadeArea.BeginFade()) {
598  bool anyNodesNull = false;
599  int total = 0;
600  int numWalkable = 0;
601 
602  // Calculate number of nodes in the graph
603  KeyValuePair<float, KeyValuePair<int, int> > pair;
604  graphNodeCounts = graphNodeCounts ?? new Dictionary<NavGraph, KeyValuePair<float, KeyValuePair<int, int> > >();
605 
606  if (!graphNodeCounts.TryGetValue(graphEditor.target, out pair) || (Time.realtimeSinceStartup-pair.Key) > 2) {
607  graphEditor.target.GetNodes(node => {
608  if (node == null) {
609  anyNodesNull = true;
610  } else {
611  total++;
612  if (node.Walkable) numWalkable++;
613  }
614  });
615  pair = new KeyValuePair<float, KeyValuePair<int, int> >(Time.realtimeSinceStartup, new KeyValuePair<int, int>(total, numWalkable));
616  graphNodeCounts[graphEditor.target] = pair;
617  }
618 
619  total = pair.Value.Key;
620  numWalkable = pair.Value.Value;
621 
622 
623  EditorGUI.indentLevel++;
624 
625  if (anyNodesNull) {
626  Debug.LogError("Some nodes in the graph are null. Please report this error.");
627  }
628 
629  EditorGUILayout.LabelField("Nodes", total.ToString());
630  EditorGUILayout.LabelField("Walkable", numWalkable.ToString());
631  EditorGUILayout.LabelField("Unwalkable", (total-numWalkable).ToString());
632  if (total == 0) EditorGUILayout.HelpBox("The number of nodes in the graph is zero. The graph might not be scanned", MessageType.Info);
633 
634  EditorGUI.indentLevel--;
635  }
