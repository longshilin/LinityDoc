150  static bool CheckForUpdates () {
151  if (updateCheckDownload != null && updateCheckDownload.isDone) {
152  if (!string.IsNullOrEmpty(updateCheckDownload.error)) {
153  Debug.LogWarning("There was an error checking for updates to the A* Pathfinding Project\n" +
154  "The error might disappear if you switch build target from Webplayer to Standalone because of the webplayer security emulation\nError: " +
155  updateCheckDownload.error);
156  updateCheckDownload = null;
157  return false;
158  }
159  UpdateCheckCompleted(updateCheckDownload.downloadHandler.text);
160  updateCheckDownload = null;
161  }
162 
163  // Check if it is time to check for updates
164  // Check for updates a bit earlier if we are in play mode or have the AstarPath object in the scene
165  // as then the collected statistics will be a bit more accurate
166  var offsetMinutes = (Application.isPlaying && Time.time > 60) || AstarPath.active != null ? -20 : 20;
167  var minutesUntilUpdate = lastUpdateCheck.AddDays(updateCheckRate).AddMinutes(offsetMinutes).Subtract(System.DateTime.UtcNow).TotalMinutes;
168  if (minutesUntilUpdate < 0) {
169  DownloadVersionInfo();
170  }
171 
172  return updateCheckDownload != null || minutesUntilUpdate < 10;

217  static void ParseServerMessage (string result) {
218  if (string.IsNullOrEmpty(result)) {
219  return;
220  }
221 
222  hasParsedServerMessage = true;
223 
224  #if ASTARDEBUG
225  Debug.Log("Result from update check:\n"+result);
226  #endif
227 
228  string[] splits = result.Split('|');
229  latestVersionDescription = splits.Length > 1 ? splits[1] : "";
230 
231  if (splits.Length > 4) {
232  // First 4 are just compatibility fields
233  var fields = splits.Skip(4).ToArray();
234 
235  // Take all pairs of fields
236  for (int i = 0; i < (fields.Length/2)*2; i += 2) {
237  string key = fields[i];
238  string val = fields[i+1];
239  astarServerData[key] = val;
240  }
241  }
242

244  latestVersion = new System.Version(astarServerData["VERSION:branch"]);
245  } catch (System.Exception ex) {
246  Debug.LogWarning("Could not parse version\n"+ex);
247  }
