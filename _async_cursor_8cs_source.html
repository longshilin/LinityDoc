521  private void SaveBatch(CursorBatch<TDocument> batch)
522  {
523  var documents = batch.Documents;
524 
525  _count += documents.Count;
526  if (_limit > 0 && _count > _limit.Value)
527  {
528  var remove = _count - _limit.Value;
529  var take = documents.Count - remove;
530  documents = documents.Take(take).ToList();
531  _count = _limit.Value;
532  }
533 
534  _currentBatch = documents;
535  _cursorId = batch.CursorId;
536 
537  DisposeChannelSourceIfNoLongerNeeded();
538  }

548  private void ThrowIfKillCursorsCommandFailed(BsonDocument commandResult, ConnectionId connectionId)
549  {
550  if (!commandResult.GetValue("ok", false).ToBoolean())
551  {
552  throw new MongoCommandException(connectionId, "Kill cursors command returned an error.", commandResult);
553  }
554  else
555  {
556  var notFoundCursors = commandResult["cursorsNotFound"].AsBsonArray;
557  if (notFoundCursors.Count > 0)
558  {
559  throw new MongoCursorNotFoundException(connectionId, _cursorId, commandResult);
560  }
561 
562  var killedCursors = commandResult["cursorsKilled"].AsBsonArray.Select(c => c.ToInt64());
563  if (!killedCursors.Contains(_cursorId))
564  {
565  throw new MongoCommandException(connectionId, "Kill cursors command failed.", commandResult);
566  }
567  }

570  private bool TryMoveNext(out bool hasMore)
571  {
572  hasMore = false;
573 
574  if (_firstBatch != null)
575  {
576  _currentBatch = _firstBatch;
577  _firstBatch = null;
578  hasMore = true;
579  return true;
580  }
581 
582  if (_currentBatch == null)
583  {
584  return true;
585  }
586 
587  if (_cursorId == 0 || (_limit > 0 && _count == _limit.Value))
588  {
589  _currentBatch = null;
590  return true;
591  }
592 
593  return false;
594  }
