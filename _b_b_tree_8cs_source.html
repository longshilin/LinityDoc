69  public void RebuildFrom (TriangleMeshNode[] nodes) {
70  Clear();
71 
72  if (nodes.Length == 0) return;
73 
74  // We will use approximately 2N tree nodes
75  EnsureCapacity(Mathf.CeilToInt(nodes.Length * 2.1f));
76  // We will use approximately N node references
77  EnsureNodeCapacity(Mathf.CeilToInt(nodes.Length * 1.1f));
78 
79  // This will store the order of the nodes while the tree is being built
80  // It turns out that it is a lot faster to do this than to actually modify
81  // the nodes and nodeBounds arrays (presumably since that involves shuffling
82  // around 20 bytes of memory (sizeof(pointer) + sizeof(IntRect)) per node
83  // instead of 4 bytes (sizeof(int)).
84  // It also means we don't have to make a copy of the nodes array since
85  // we do not modify it
86  var permutation = ArrayPool<int>.Claim(nodes.Length);
87  for (int i = 0; i < nodes.Length; i++) {
88  permutation[i] = i;
89  }
90 
91  // Precalculate the bounds of the nodes in XZ space.
92  // It turns out that calculating the bounds is a bottleneck and precalculating
93  // the bounds makes it around 3 times faster to build a tree
94  var nodeBounds = ArrayPool<IntRect>.Claim(nodes.Length);
95  for (int i = 0; i < nodes.Length; i++) {
96  Int3 v0, v1, v2;
97  nodes[i].GetVertices(out v0, out v1, out v2);
98 
99  var rect = new IntRect(v0.x, v0.z, v0.x, v0.z);
100  rect = rect.ExpandToContain(v1.x, v1.z);
101  rect = rect.ExpandToContain(v2.x, v2.z);
102  nodeBounds[i] = rect;
103  }
104 
105  RebuildFromInternal(nodes, permutation, nodeBounds, 0, nodes.Length, false);
106 
107  ArrayPool<int>.Release(ref permutation);
108  ArrayPool<IntRect>.Release(ref nodeBounds);
109  }

111  static int SplitByX (TriangleMeshNode[] nodes, int[] permutation, int from, int to, int divider) {
112  int mx = to;
113 
114  for (int i = from; i < mx; i++) {
115  if (nodes[permutation[i]].position.x > divider) {
116  mx--;
117  // Swap items i and mx
118  var tmp = permutation[mx];
119  permutation[mx] = permutation[i];
120  permutation[i] = tmp;
121  i--;
122  }
123  }
124  return mx;
125  }

143  int RebuildFromInternal (TriangleMeshNode[] nodes, int[] permutation, IntRect[] nodeBounds, int from, int to, bool odd) {
144  var rect = NodeBounds(permutation, nodeBounds, from, to);
145  int box = GetBox(rect);
146 
147  if (to - from <= MaximumLeafSize) {
148  var nodeOffset = tree[box].nodeOffset = leafNodes*MaximumLeafSize;
149  EnsureNodeCapacity(nodeOffset + MaximumLeafSize);
150  leafNodes++;
151  // Assign all nodes to the array. Note that we also need clear unused slots as the array from the pool may contain any information
152  for (int i = 0; i < MaximumLeafSize; i++) {
153  nodeLookup[nodeOffset + i] = i < to - from ? nodes[permutation[from + i]] : null;
154  }
155  return box;
156  }
157 
158  int splitIndex;
159  if (odd) {
160  // X
161  int divider = (rect.xmin + rect.xmax)/2;
162  splitIndex = SplitByX(nodes, permutation, from, to, divider);
163  } else {
164  // Y/Z
165  int divider = (rect.ymin + rect.ymax)/2;
166  splitIndex = SplitByZ(nodes, permutation, from, to, divider);
167  }
168 
169  if (splitIndex == from || splitIndex == to) {
170  // All nodes were on one side of the divider
171  // Try to split along the other axis
172 
173  if (!odd) {
174  // X
175  int divider = (rect.xmin + rect.xmax)/2;
176  splitIndex = SplitByX(nodes, permutation, from, to, divider);
177  } else {
178  // Y/Z
179  int divider = (rect.ymin + rect.ymax)/2;
180  splitIndex = SplitByZ(nodes, permutation, from, to, divider);
181  }
182 
183  if (splitIndex == from || splitIndex == to) {
184  // All nodes were on one side of the divider
185  // Just pick one half
186  splitIndex = (from+to)/2;
187  }
188  }
189 
190  tree[box].left = RebuildFromInternal(nodes, permutation, nodeBounds, from, splitIndex, !odd);
191  tree[box].right = RebuildFromInternal(nodes, permutation, nodeBounds, splitIndex, to, !odd);
192 
193  return box;
194  }

197  static IntRect NodeBounds (int[] permutation, IntRect[] nodeBounds, int from, int to) {
198  var rect = nodeBounds[permutation[from]];
199 
200  for (int j = from + 1; j < to; j++) {
201  var otherRect = nodeBounds[permutation[j]];
202 
203  // Equivalent to rect = IntRect.Union(rect, otherRect)
204  // but manually inlining is approximately
205  // 25% faster when building an entire tree.
206  // This code is hot when using navmesh cutting.
207  rect.xmin = Math.Min(rect.xmin, otherRect.xmin);
208  rect.ymin = Math.Min(rect.ymin, otherRect.ymin);
209  rect.xmax = Math.Max(rect.xmax, otherRect.xmax);
210  rect.ymax = Math.Max(rect.ymax, otherRect.ymax);
211  }
212 
213  return rect;
214  }

270  void SearchBoxClosestXZ (int boxi, Vector3 p, ref float closestSqrDist, NNConstraint constraint, ref NNInfoInternal nnInfo) {
271  BBTreeBox box = tree[boxi];
272 
273  if (box.IsLeaf) {
274  var nodes = nodeLookup;
275  for (int i = 0; i < MaximumLeafSize && nodes[box.nodeOffset+i] != null; i++) {
276  var node = nodes[box.nodeOffset+i];
277  // Update the NNInfo
278 #if !SERVER
279  DrawDebugNode(node, 0.2f, UnityEngine.Color.red);
280 #endif
281 
282  if (constraint == null || constraint.Suitable(node)) {
283  Vector3 closest = node.ClosestPointOnNodeXZ(p);
284  // XZ squared distance
285  float dist = (closest.x-p.x)*(closest.x-p.x)+(closest.z-p.z)*(closest.z-p.z);
286 
287  // There's a theoretical case when the closest point is on the edge of a node which may cause the
288  // closest point's xz coordinates to not line up perfectly with p's xz coordinates even though they should
289  // (because floating point errors are annoying). So use a tiny margin to cover most of those cases.
290  const float fuzziness = 0.000001f;
291  if (nnInfo.constrainedNode == null || dist < closestSqrDist - fuzziness || (dist <= closestSqrDist + fuzziness && Mathf.Abs(closest.y - p.y) < Mathf.Abs(nnInfo.constClampedPosition.y - p.y))) {
292  nnInfo.constrainedNode = node;
293  nnInfo.constClampedPosition = closest;
294  closestSqrDist = dist;
295  }
296  }
297  }
298  } else {
299 #if !SERVER
300  DrawDebugRect(box.rect);
301 #endif
302 
303  int first = box.left, second = box.right;
304  float firstDist, secondDist;
305  GetOrderedChildren(ref first, ref second, out firstDist, out secondDist, p);
306 
307  // Search children (closest box first to improve performance)
308  if (firstDist <= closestSqrDist) {
309  SearchBoxClosestXZ(first, p, ref closestSqrDist, constraint, ref nnInfo);
310  }
311 
312  if (secondDist <= closestSqrDist) {
313  SearchBoxClosestXZ(second, p, ref closestSqrDist, constraint, ref nnInfo);
314  }
315  }

481  static float SquaredRectPointDistance (IntRect r, Vector3 p) {
482  Vector3 po = p;
483 
484  p.x = Math.Max(p.x, r.xmin*Int3.PrecisionFactor);
485  p.x = Math.Min(p.x, r.xmax*Int3.PrecisionFactor);
486  p.z = Math.Max(p.z, r.ymin*Int3.PrecisionFactor);
487  p.z = Math.Min(p.z, r.ymax*Int3.PrecisionFactor);
488 
489  // XZ squared magnitude comparison
490  return (p.x-po.x)*(p.x-po.x) + (p.z-po.z)*(p.z-po.z);
491  }
