288  private void InitVersionHelper()
289  {
290  if (string.IsNullOrEmpty(m_VersionHelperTypeName))
291  {
292  return;
293  }
294 
295  Type versionHelperType = Utility.Assembly.GetType(m_VersionHelperTypeName);
296  if (versionHelperType == null)
297  {
298  throw new GameFrameworkException(Utility.Text.Format("Can not find version helper type '{0}'.", m_VersionHelperTypeName));
299  }
300 
301  GameFramework.Version.IVersionHelper versionHelper = (GameFramework.Version.IVersionHelper)Activator.CreateInstance(versionHelperType);
302  if (versionHelper == null)
303  {
304  throw new GameFrameworkException(Utility.Text.Format("Can not create version helper instance '{0}'.", m_VersionHelperTypeName));
