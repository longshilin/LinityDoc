262  private float GetGameSpeed(int selectedGameSpeed)
263  {
264  if (selectedGameSpeed < 0)
265  {
266  return GameSpeed[0];
267  }
268 
269  if (selectedGameSpeed >= GameSpeed.Length)
270  {
271  return GameSpeed[GameSpeed.Length - 1];
272  }
273 
274  return GameSpeed[selectedGameSpeed];
275  }
