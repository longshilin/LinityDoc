29  public static bool UpdatableVersionListSerializeCallback_V0(Stream stream, UpdatableVersionList versionList)
30  {
31  if (!versionList.IsValid)
32  {
33  return false;
34  }
35 
36  Utility.Random.GetRandomBytes(s_CachedHashBytes);
37  using (BinaryWriter binaryWriter = new BinaryWriter(stream, Encoding.UTF8))
38  {
39  binaryWriter.Write(s_CachedHashBytes);
40  binaryWriter.WriteEncryptedString(versionList.ApplicableGameVersion, s_CachedHashBytes);
41  binaryWriter.Write(versionList.InternalResourceVersion);
42  UpdatableVersionList.Asset[] assets = versionList.GetAssets();
43  binaryWriter.Write(assets.Length);
44  UpdatableVersionList.Resource[] resources = versionList.GetResources();
45  binaryWriter.Write(resources.Length);
46  foreach (UpdatableVersionList.Resource resource in resources)
47  {
48  binaryWriter.WriteEncryptedString(resource.Name, s_CachedHashBytes);
49  binaryWriter.WriteEncryptedString(resource.Variant, s_CachedHashBytes);
50  binaryWriter.Write(resource.LoadType);
51  binaryWriter.Write(resource.Length);
52  binaryWriter.Write(resource.HashCode);
53  binaryWriter.Write(resource.CompressedLength);
54  binaryWriter.Write(resource.CompressedHashCode);
55  int[] assetIndexes = resource.GetAssetIndexes();
56  binaryWriter.Write(assetIndexes.Length);
57  byte[] hashBytes = new byte[CachedHashBytesLength];
58  foreach (int assetIndex in assetIndexes)
59  {
60  Utility.Converter.GetBytes(resource.HashCode, hashBytes);
61  UpdatableVersionList.Asset asset = assets[assetIndex];
62  binaryWriter.WriteEncryptedString(asset.Name, hashBytes);
63  int[] dependencyAssetIndexes = asset.GetDependencyAssetIndexes();
64  binaryWriter.Write(dependencyAssetIndexes.Length);
65  foreach (int dependencyAssetIndex in dependencyAssetIndexes)
66  {
67  binaryWriter.WriteEncryptedString(assets[dependencyAssetIndex].Name, hashBytes);
68  }
69  }
70  }
71 
72  UpdatableVersionList.ResourceGroup[] resourceGroups = versionList.GetResourceGroups();
73  binaryWriter.Write(resourceGroups.Length);
74  foreach (UpdatableVersionList.ResourceGroup resourceGroup in resourceGroups)
75  {
76  binaryWriter.WriteEncryptedString(resourceGroup.Name, s_CachedHashBytes);
77  int[] resourceIndexes = resourceGroup.GetResourceIndexes();
78  binaryWriter.Write(resourceIndexes.Length);
79  foreach (ushort resourceIndex in resourceIndexes)
80  {
81  binaryWriter.Write(resourceIndex);
82  }
83  }
84  }
85 
86  Array.Clear(s_CachedHashBytes, 0, CachedHashBytesLength);
87  return true;
88  }
