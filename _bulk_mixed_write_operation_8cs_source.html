211  public BulkWriteOperationResult Execute(IWriteBinding binding, CancellationToken cancellationToken)
212  {
213  using (EventContext.BeginOperation())
214  using (var context = RetryableWriteContext.Create(binding, _retryRequested, cancellationToken))
215  {
216  EnsureCollationIsSupportedIfAnyRequestHasCollation(context);
217  context.DisableRetriesIfAnyWriteRequestIsNotRetryable(_requests);
218  var helper = new BatchHelper(_requests, _isOrdered, _writeConcern);
219  foreach (var batch in helper.GetBatches())
220  {
221  batch.Result = ExecuteBatch(context, batch, cancellationToken);
222  }
223  return helper.GetFinalResultOrThrow(context.Channel.ConnectionDescription.ConnectionId);
224  }
225  }

297  private void EnsureCollationIsSupportedIfAnyRequestHasCollation(RetryableWriteContext context)
298  {

375  private class BatchHelper
376  {
377  // private fields
378  private readonly List<BulkWriteBatchResult> _batchResults = new List<BulkWriteBatchResult>();
379  private bool _hasWriteErrors;
380  private readonly bool _isOrdered;
381  private List<WriteRequestWithIndex> _unprocessed;
382  private WriteConcern _writeConcern;
383 
384  // constructors
385  public BatchHelper(IEnumerable<WriteRequest> requests, bool ordered, WriteConcern writeConcern)
386  {
387  Ensure.IsNotNull(requests, nameof(requests));
388  _isOrdered = ordered;
389  _writeConcern = writeConcern;
390 
391  _unprocessed = new List<WriteRequestWithIndex>();
392  var index = 0;
393  foreach (var request in requests)
394  {
395  _unprocessed.Add(new WriteRequestWithIndex { Index = index++, Request = request });
396  }
397  }
398 
399  // public methods
400  public IEnumerable<Batch> GetBatches()
401  {
402  if (_unprocessed.Count == 0)
403  {
404  throw new InvalidOperationException("Bulk write operation is empty.");
405  }
406 
407  while (_unprocessed.Count > 0 && ShouldContinue())
408  {
409  var batch = GetNextBatch();
410 
411  yield return batch;
412 
413  _batchResults.Add(batch.Result);
414  _hasWriteErrors |= batch.Result.HasWriteErrors;
415  }
416  }
417 
418  public BulkWriteOperationResult GetFinalResultOrThrow(ConnectionId connectionId)
419  {
420  var combiner = new BulkWriteBatchResultCombiner(_batchResults, _writeConcern.IsAcknowledged);
421  return combiner.CreateResultOrThrowIfHasErrors(connectionId, _unprocessed.Select(r => r.Request).ToList());
422  }
423 
424  // private methods
425  private Batch GetNextBatch()
426  {
427  var batchType = _unprocessed[0].Request.RequestType;
428 
429  List<WriteRequest> requests;
430  IndexMap indexMap;
431  if (_isOrdered)
432  {
433  var index = _unprocessed.FindIndex(r => r.Request.RequestType != batchType);
434  var count = index == -1 ? _unprocessed.Count : index;
435  requests = _unprocessed.Take(count).Select(r => r.Request).ToList();
436  indexMap = new IndexMap.RangeBased(0, _unprocessed[0].Index, count);
437  _unprocessed.RemoveRange(0, count);
438  }
439  else
440  {
441  var matching = _unprocessed.Where(r => r.Request.RequestType == batchType).ToList();
442  requests = matching.Select(r => r.Request).ToList();
443  indexMap = new IndexMap.DictionaryBased();
444  for (var i = 0; i < matching.Count; i++)
445  {
446  indexMap.Add(i, matching[i].Index);
447  }
448  _unprocessed = _unprocessed.Where(r => r.Request.RequestType != batchType).ToList();
449  }
450 
451  var writeConcern = _writeConcern;
452  if (!writeConcern.IsAcknowledged && _isOrdered && _unprocessed.Count > 0)
453  {
454  writeConcern = WriteConcern.W1; // explicitly do not use the server's default
455  }
456 
457  return new Batch
458  {
459  BatchType = batchType,
460  Requests = requests,
461  IndexMap = indexMap,
462  WriteConcern = writeConcern
463  };
464  }
465 
466  private bool ShouldContinue()
467  {
468  return !_hasWriteErrors || !_isOrdered;
469  }
