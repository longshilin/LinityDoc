132  internal static BulkWriteResult<TDocument> FromCore(Core.Operations.BulkWriteOperationResult result)
133  {
134  if (result.IsAcknowledged)
135  {
136  return new Acknowledged(
137  result.RequestCount,
138  result.MatchedCount,
139  result.DeletedCount,
140  result.InsertedCount,
141  result.IsModifiedCountAvailable ? (long?)result.ModifiedCount : null,
142  result.ProcessedRequests.Select(r => WriteModel<TDocument>.FromCore(r)),
143  result.Upserts.Select(u => BulkWriteUpsert.FromCore(u)));
144  }
145 
146  return new Unacknowledged(
147  result.RequestCount,
148  result.ProcessedRequests.Select(r => WriteModel<TDocument>.FromCore(r)));
149  }
