284  public static bool operator ==(ByteString lhs, ByteString rhs)
285  {
286  if (ReferenceEquals(lhs, rhs))
287  {
288  return true;
289  }
290  if (ReferenceEquals(lhs, null) || ReferenceEquals(rhs, null))
291  {
292  return false;
293  }
294  if (lhs.bytes.Length != rhs.bytes.Length)
295  {
296  return false;
297  }
298  for (int i = 0; i < lhs.Length; i++)
299  {
300  if (rhs.bytes[i] != lhs.bytes[i])
301  {
302  return false;
303  }
304  }
305  return true;
306  }
