259  public IAsyncCursor<TResult> Execute(IReadBinding binding, CancellationToken cancellationToken)
260  {
261  var bindingHandle = binding as IReadBindingHandle;
262  if (bindingHandle == null)
263  {
264  throw new ArgumentException("The binding value passed to ChangeStreamOperation.Execute must implement IReadBindingHandle.", nameof(binding));
265  }
266 
267  IAsyncCursor<RawBsonDocument> cursor;
268  using (var channelSource = binding.GetReadChannelSource(cancellationToken))
269  using (var channel = channelSource.GetChannel(cancellationToken))
270  using (var channelBinding = new ChannelReadBinding(channelSource.Server, channel, binding.ReadPreference, binding.Session.Fork()))
271  {
272  cursor = Resume(channelBinding, cancellationToken);
273  if (_startAtOperationTime == null && _resumeAfter == null)
274  {
275  var maxWireVersion = channel.ConnectionDescription.IsMasterResult.MaxWireVersion;
276  if (maxWireVersion >= 7)
277  {
278  _startAtOperationTime = binding.Session.OperationTime;
279  }
280  }
281  }
282 
283  return new ChangeStreamCursor<TResult>(cursor, _resultSerializer, bindingHandle.Fork(), this);
284  }

329  private AggregateOperation<RawBsonDocument> CreateAggregateOperation()
330  {
331  var changeStreamStage = CreateChangeStreamStage();
332  var combinedPipeline = CreateCombinedPipeline(changeStreamStage);
333 
334  AggregateOperation<RawBsonDocument> operation;
335  if (_collectionNamespace != null)
336  {
337  operation = new AggregateOperation<RawBsonDocument>(_collectionNamespace, combinedPipeline, RawBsonDocumentSerializer.Instance, _messageEncoderSettings);
338  }
339  else
340  {
341  var databaseNamespace = _databaseNamespace ?? DatabaseNamespace.Admin;
342  operation = new AggregateOperation<RawBsonDocument>(databaseNamespace, combinedPipeline, RawBsonDocumentSerializer.Instance, _messageEncoderSettings);
343  }
344 
345  operation.BatchSize = _batchSize;
346  operation.Collation = _collation;
347  operation.MaxAwaitTime = _maxAwaitTime;
348  operation.ReadConcern = _readConcern;
349 
350  return operation;
351  }

353  private BsonDocument CreateChangeStreamStage()
354  {
355  var changeStreamOptions = new BsonDocument
356  {
357  { "fullDocument", ToString(_fullDocument) },
358  { "allChangesForCluster", true, _collectionNamespace == null && _databaseNamespace == null },
359  { "startAtOperationTime", _startAtOperationTime, _startAtOperationTime != null },
360  { "resumeAfter", _resumeAfter, _resumeAfter != null }
361  };
362  return new BsonDocument("$changeStream", changeStreamOptions);
363  }

365  private List<BsonDocument> CreateCombinedPipeline(BsonDocument changeStreamStage)
366  {

368  combinedPipeline.Add(changeStreamStage);
369  combinedPipeline.AddRange(_pipeline);
370  return combinedPipeline;
