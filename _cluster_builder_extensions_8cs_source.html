59  public static ClusterBuilder ConfigureWithConnectionString(this ClusterBuilder builder, ConnectionString connectionString)
60  {
61  Ensure.IsNotNull(builder, nameof(builder));
62  Ensure.IsNotNull(connectionString, nameof(connectionString));
63 
64  connectionString = connectionString.Resolve();
65 
66  // TCP
67  if (connectionString.ConnectTimeout != null)
68  {
69  builder = builder.ConfigureTcp(s => s.With(connectTimeout: connectionString.ConnectTimeout.Value));
70  }
71  if (connectionString.HeartbeatInterval.HasValue)
72  {
73  builder = builder.ConfigureServer(s => s.With(heartbeatInterval: connectionString.HeartbeatInterval.Value));
74  }
75  if (connectionString.HeartbeatTimeout.HasValue)
76  {
77  builder = builder.ConfigureServer(s => s.With(heartbeatTimeout: connectionString.HeartbeatTimeout.Value));
78  }
79  if (connectionString.Ipv6.HasValue && connectionString.Ipv6.Value)
80  {
81  builder = builder.ConfigureTcp(s => s.With(addressFamily: AddressFamily.InterNetworkV6));
82  }
83 
84  if (connectionString.SocketTimeout != null)
85  {
86  builder = builder.ConfigureTcp(s => s.With(
87  readTimeout: connectionString.SocketTimeout.Value,
88  writeTimeout: connectionString.SocketTimeout.Value));
89  }
90 
91  if (connectionString.Ssl != null)
92  {
93  builder = builder.ConfigureSsl(ssl =>
94  {
95  if (!connectionString.SslVerifyCertificate.GetValueOrDefault(true))
96  {
97  ssl = ssl.With(
98  serverCertificateValidationCallback: new RemoteCertificateValidationCallback(AcceptAnySslCertificate));
99  }
100 
101  return ssl;
102  });
103  }
104 
105  // Connection
106  if (connectionString.Username != null)
107  {
108  var authenticator = CreateAuthenticator(connectionString);
109  builder = builder.ConfigureConnection(s => s.With(authenticators: new[] { authenticator }));
110  }
111  if (connectionString.ApplicationName != null)
112  {
113  builder = builder.ConfigureConnection(s => s.With(applicationName: connectionString.ApplicationName));
114  }
115  if (connectionString.MaxIdleTime != null)
116  {
117  builder = builder.ConfigureConnection(s => s.With(maxIdleTime: connectionString.MaxIdleTime.Value));
118  }
119  if (connectionString.MaxLifeTime != null)
120  {
121  builder = builder.ConfigureConnection(s => s.With(maxLifeTime: connectionString.MaxLifeTime.Value));
122  }
123 
124  // Connection Pool
125  if (connectionString.MaxPoolSize != null)
126  {
127  builder = builder.ConfigureConnectionPool(s => s.With(maxConnections: connectionString.MaxPoolSize.Value));
128  }
129  if (connectionString.MinPoolSize != null)
130  {
131  builder = builder.ConfigureConnectionPool(s => s.With(minConnections: connectionString.MinPoolSize.Value));
132  }
133  if (connectionString.WaitQueueSize != null)
134  {
135  builder = builder.ConfigureConnectionPool(s => s.With(waitQueueSize: connectionString.WaitQueueSize.Value));
136  }
137  else if (connectionString.WaitQueueMultiple != null)
138  {
139  var maxConnections = connectionString.MaxPoolSize ?? new ConnectionPoolSettings().MaxConnections;
140  var waitQueueSize = (int)Math.Round(maxConnections * connectionString.WaitQueueMultiple.Value);
141  builder = builder.ConfigureConnectionPool(s => s.With(waitQueueSize: waitQueueSize));
142  }
143  if (connectionString.WaitQueueTimeout != null)
144  {
145  builder = builder.ConfigureConnectionPool(s => s.With(waitQueueTimeout: connectionString.WaitQueueTimeout.Value));
146  }
147 
148  // Server
149 
150  // Cluster
151  builder = builder.ConfigureCluster(s => s.With(connectionMode: connectionString.Connect));
152  if (connectionString.Hosts.Count > 0)
153  {
154  builder = builder.ConfigureCluster(s => s.With(endPoints: Optional.Enumerable(connectionString.Hosts)));
155  }
156  if (connectionString.ReplicaSet != null)
157  {
158  builder = builder.ConfigureCluster(s => s.With(
159  connectionMode: ClusterConnectionMode.ReplicaSet,
160  replicaSetName: connectionString.ReplicaSet));
161  }
162  if (connectionString.ServerSelectionTimeout != null)
163  {
164  builder = builder.ConfigureCluster(s => s.With(serverSelectionTimeout: connectionString.ServerSelectionTimeout.Value));
165  }
166 
167  return builder;
168  }

180  private static IAuthenticator CreateAuthenticator(ConnectionString connectionString)
181  {
182  if (connectionString.Password != null)
183  {
184  var defaultSource = GetDefaultSource(connectionString);
185 
186  var credential = new UsernamePasswordCredential(
187  connectionString.AuthSource ?? connectionString.DatabaseName ?? defaultSource,
188  connectionString.Username,
189  connectionString.Password);
190 
191  if (connectionString.AuthMechanism == null)
192  {
193  return new DefaultAuthenticator(credential);
194  }
195 #pragma warning disable 618
196  else if (connectionString.AuthMechanism == MongoDBCRAuthenticator.MechanismName)
197  {
198  return new MongoDBCRAuthenticator(credential);
199 #pragma warning restore 618
200  }
201  else if (connectionString.AuthMechanism == ScramSha1Authenticator.MechanismName)
202  {
203  return new ScramSha1Authenticator(credential);
204  }
205  else if (connectionString.AuthMechanism == ScramSha256Authenticator.MechanismName)
206  {
207  return new ScramSha256Authenticator(credential);
208  }
209  else if (connectionString.AuthMechanism == PlainAuthenticator.MechanismName)
210  {
211  return new PlainAuthenticator(credential);
212  }
213  else if (connectionString.AuthMechanism == GssapiAuthenticator.MechanismName)
214  {
215  return new GssapiAuthenticator(credential, connectionString.AuthMechanismProperties);
216  }
217  }
218  else
219  {
220  if (connectionString.AuthMechanism == MongoDBX509Authenticator.MechanismName)
221  {
222  return new MongoDBX509Authenticator(connectionString.Username);
223  }
224  else if (connectionString.AuthMechanism == GssapiAuthenticator.MechanismName)
225  {
226  return new GssapiAuthenticator(connectionString.Username, connectionString.AuthMechanismProperties);
227  }
228  }
229 
230  throw new NotSupportedException("Unable to create an authenticator.");
231  }
232 
233  private static string GetDefaultSource(ConnectionString connectionString)
234  {
235  if (connectionString.AuthMechanism != null && connectionString.AuthMechanism.Equals("GSSAPI", StringComparison.OrdinalIgnoreCase))
236  {
237  return "$external";
238  }
239 
240  return "admin";
241  }
