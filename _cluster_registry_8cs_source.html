132  private SslStreamSettings ConfigureSsl(SslStreamSettings settings, ClusterKey clusterKey)
133  {

136  var sslSettings = clusterKey.SslSettings ?? new SslSettings();
137 
138  var validationCallback = sslSettings.ServerCertificateValidationCallback;
139  if (validationCallback == null && !clusterKey.VerifySslCertificate)
140  {
141  validationCallback = AcceptAnySslCertificate;

143 
144  return settings.With(
145  clientCertificates: Optional.Enumerable(sslSettings.ClientCertificates ?? Enumerable.Empty<X509Certificate>()),
146  checkCertificateRevocation: sslSettings.CheckCertificateRevocation,
147  clientCertificateSelectionCallback: sslSettings.ClientCertificateSelectionCallback,
148  enabledProtocols: sslSettings.EnabledSslProtocols,
149  serverCertificateValidationCallback: validationCallback);

170  internal ICluster GetOrCreateCluster(ClusterKey clusterKey)
171  {
172  lock (_lock)
173  {
174  ICluster cluster;
175  if (!_registry.TryGetValue(clusterKey, out cluster))
176  {

198  public void UnregisterAndDisposeCluster(ICluster cluster)
199  {
200  Ensure.IsNotNull(cluster, nameof(cluster));
201 
202  lock (_lock)
203  {
204  ClusterKey clusterKey = null;
205  foreach (var keyValuePair in _registry)
206  {
207  if (object.ReferenceEquals(keyValuePair.Value, cluster))
208  {
209  clusterKey = keyValuePair.Key;
210  break;
211  }
212  }
213 
214  if (clusterKey != null)
215  {
216  _registry.Remove(clusterKey);
217  }
218  }
219 
220  cluster.Dispose();
221  }
