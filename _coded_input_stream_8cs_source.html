328  public uint PeekTag()
329  {
330  if (hasNextTag)
331  {
332  return nextTag;
333  }
334 
335  uint savedLast = lastTag;
336  nextTag = ReadTag();
337  hasNextTag = true;
338  lastTag = savedLast; // Undo the side effect of ReadTag
339  return nextTag;
340  }
341 
351  public uint ReadTag()
352  {
353  if (hasNextTag)
354  {
355  lastTag = nextTag;
356  hasNextTag = false;
357  return lastTag;
358  }
359 
360  // Optimize for the incredibly common case of having at least two bytes left in the buffer,
361  // and those two bytes being enough to get the tag. This will be true for fields up to 4095.
362  if (bufferPos + 2 <= bufferSize)
363  {
364  int tmp = buffer[bufferPos++];
365  if (tmp < 128)
366  {
367  lastTag = (uint)tmp;
368  }
369  else
370  {
371  int result = tmp & 0x7f;
372  if ((tmp = buffer[bufferPos++]) < 128)
373  {
374  result |= tmp << 7;
375  lastTag = (uint) result;
376  }
377  else
378  {
379  // Nope, rewind and go the potentially slow route.
380  bufferPos -= 2;
381  lastTag = ReadRawVarint32();
382  }
383  }
384  }
385  else
386  {
387  if (IsAtEnd)
388  {
389  lastTag = 0;
390  return 0; // This is the only case in which we return 0.
391  }
392 
393  lastTag = ReadRawVarint32();
394  }
395  if (lastTag == 0)
396  {
397  // If we actually read zero, that's not a valid tag.
398  throw InvalidProtocolBufferException.InvalidTag();
399  }
400  return lastTag;
401  }

416  public void SkipLastField()
417  {
418  if (lastTag == 0)
419  {
420  throw new InvalidOperationException("SkipLastField cannot be called at the end of a stream");
421  }
422  switch (WireFormat.GetTagWireType(lastTag))
423  {
424  case WireFormat.WireType.StartGroup:
425  SkipGroup(lastTag);
426  break;
427  case WireFormat.WireType.EndGroup:
428  throw new InvalidProtocolBufferException("SkipLastField called on an end-group tag, indicating that the corresponding start-group was missing");
429  case WireFormat.WireType.Fixed32:
430  ReadFixed32();
431  break;
432  case WireFormat.WireType.Fixed64:
433  ReadFixed64();
434  break;
435  case WireFormat.WireType.LengthDelimited:
436  var length = ReadLength();
437  SkipRawBytes(length);
438  break;
439  case WireFormat.WireType.Varint:
440  ReadRawVarint32();
441  break;
442  }
443  }

445  private void SkipGroup(uint startGroupTag)
446  {
447  // Note: Currently we expect this to be the way that groups are read. We could put the recursion
448  // depth changes into the ReadTag method instead, potentially...
449  recursionDepth++;
450  if (recursionDepth >= recursionLimit)
451  {
452  throw InvalidProtocolBufferException.RecursionLimitExceeded();
453  }
454  uint tag;
455  while (true)
456  {
457  tag = ReadTag();
458  if (tag == 0)
459  {
460  throw InvalidProtocolBufferException.TruncatedMessage();
461  }
462  // Can't call SkipLastField for this case- that would throw.
463  if (WireFormat.GetTagWireType(tag) == WireFormat.WireType.EndGroup)
464  {
465  break;
466  }
467  // This recursion will allow us to handle nested groups.
468  SkipLastField();
469  }
470  int startField = WireFormat.GetTagFieldNumber(startGroupTag);
471  int endField = WireFormat.GetTagFieldNumber(tag);
472  if (startField != endField)
473  {
474  throw new InvalidProtocolBufferException("Mismatched end-group tag. Started with field " + startField + "; ended with field " + endField);
475  }
476  recursionDepth--;
477  }

583  public void ReadMessage(IMessage builder)
584  {
585  int length = ReadLength();
586  if (recursionDepth >= recursionLimit)
587  {
588  throw InvalidProtocolBufferException.RecursionLimitExceeded();
589  }
590  int oldLimit = PushLimit(length);
591  ++recursionDepth;
592  builder.MergeFrom(this);
593  CheckReadEndOfStreamTag();
594  // Check that we've read exactly as much data as expected.
595  if (!ReachedLimit)
596  {
597  throw InvalidProtocolBufferException.TruncatedMessage();
598  }
599  --recursionDepth;
600  PopLimit(oldLimit);
601  }

708  private uint SlowReadRawVarint32()
709  {
710  int tmp = ReadRawByte();
711  if (tmp < 128)
712  {
713  return (uint) tmp;
714  }
715  int result = tmp & 0x7f;
716  if ((tmp = ReadRawByte()) < 128)
717  {
718  result |= tmp << 7;
719  }
720  else
721  {
722  result |= (tmp & 0x7f) << 7;
723  if ((tmp = ReadRawByte()) < 128)
724  {
725  result |= tmp << 14;
726  }
727  else
728  {
729  result |= (tmp & 0x7f) << 14;
730  if ((tmp = ReadRawByte()) < 128)
731  {
732  result |= tmp << 21;
733  }
734  else
735  {
736  result |= (tmp & 0x7f) << 21;
737  result |= (tmp = ReadRawByte()) << 28;
738  if (tmp >= 128)
739  {
740  // Discard upper 32 bits.
741  for (int i = 0; i < 5; i++)
742  {
743  if (ReadRawByte() < 128)
744  {
745  return (uint) result;
746  }
747  }
748  throw InvalidProtocolBufferException.MalformedVarint();
749  }
750  }
751  }
752  }
753  return (uint) result;
754  }

762  internal uint ReadRawVarint32()
763  {
764  if (bufferPos + 5 > bufferSize)
765  {
766  return SlowReadRawVarint32();
767  }
768 
769  int tmp = buffer[bufferPos++];
770  if (tmp < 128)
771  {
772  return (uint) tmp;
773  }
774  int result = tmp & 0x7f;
775  if ((tmp = buffer[bufferPos++]) < 128)
776  {
777  result |= tmp << 7;
778  }
779  else
780  {
781  result |= (tmp & 0x7f) << 7;
782  if ((tmp = buffer[bufferPos++]) < 128)
783  {
784  result |= tmp << 14;
785  }
786  else
787  {
788  result |= (tmp & 0x7f) << 14;
789  if ((tmp = buffer[bufferPos++]) < 128)
790  {
791  result |= tmp << 21;
792  }
793  else
794  {
795  result |= (tmp & 0x7f) << 21;
796  result |= (tmp = buffer[bufferPos++]) << 28;
797  if (tmp >= 128)
798  {
799  // Discard upper 32 bits.
800  // Note that this has to use ReadRawByte() as we only ensure we've
801  // got at least 5 bytes at the start of the method. This lets us
802  // use the fast path in more cases, and we rarely hit this section of code.
803  for (int i = 0; i < 5; i++)
804  {
805  if (ReadRawByte() < 128)
806  {
807  return (uint) result;
808  }
809  }
810  throw InvalidProtocolBufferException.MalformedVarint();
811  }
812  }
813  }
814  }
815  return (uint) result;
816  }

827  internal static uint ReadRawVarint32(Stream input)
828  {
829  int result = 0;
830  int offset = 0;
831  for (; offset < 32; offset += 7)
832  {
833  int b = input.ReadByte();
834  if (b == -1)
835  {
836  throw InvalidProtocolBufferException.TruncatedMessage();
837  }
838  result |= (b & 0x7f) << offset;
839  if ((b & 0x80) == 0)
840  {
841  return (uint) result;
842  }
843  }
844  // Keep reading up to 64 bits.
845  for (; offset < 64; offset += 7)
846  {
847  int b = input.ReadByte();
848  if (b == -1)
849  {
850  throw InvalidProtocolBufferException.TruncatedMessage();
851  }
852  if ((b & 0x80) == 0)
853  {
854  return (uint) result;
855  }
856  }
857  throw InvalidProtocolBufferException.MalformedVarint();
858  }

863  internal ulong ReadRawVarint64()
864  {
865  int shift = 0;
866  ulong result = 0;
867  while (shift < 64)
868  {
869  byte b = ReadRawByte();
870  result |= (ulong) (b & 0x7F) << shift;
871  if ((b & 0x80) == 0)
872  {
873  return result;
874  }
875  shift += 7;
876  }
877  throw InvalidProtocolBufferException.MalformedVarint();
878  }

896  {

946  internal int PushLimit(int byteLimit)
947  {
948  if (byteLimit < 0)
949  {
950  throw InvalidProtocolBufferException.NegativeSize();
951  }
952  byteLimit += totalBytesRetired + bufferPos;
953  int oldLimit = currentLimit;
954  if (byteLimit > oldLimit)
955  {
956  throw InvalidProtocolBufferException.TruncatedMessage();
957  }
958  currentLimit = byteLimit;
959 
960  RecomputeBufferSizeAfterLimit();
961 
962  return oldLimit;
963  }

965  private void RecomputeBufferSizeAfterLimit()
966  {
967  bufferSize += bufferSizeAfterLimit;
968  int bufferEnd = totalBytesRetired + bufferSize;
969  if (bufferEnd > currentLimit)
970  {
971  // Limit is in current buffer.
972  bufferSizeAfterLimit = bufferEnd - currentLimit;
973  bufferSize -= bufferSizeAfterLimit;
974  }
975  else
976  {
977  bufferSizeAfterLimit = 0;
978  }
979  }

984  internal void PopLimit(int oldLimit)
985  {

988  }

1026  private bool RefillBuffer(bool mustSucceed)
1027  {
1028  if (bufferPos < bufferSize)
1029  {
1030  throw new InvalidOperationException("RefillBuffer() called when buffer wasn't empty.");
1031  }
1032 
1033  if (totalBytesRetired + bufferSize == currentLimit)
1034  {
1035  // Oops, we hit a limit.
1036  if (mustSucceed)
1037  {
1038  throw InvalidProtocolBufferException.TruncatedMessage();
1039  }
1040  else
1041  {
1042  return false;
1043  }
1044  }
1045 
1046  totalBytesRetired += bufferSize;
1047 
1048  bufferPos = 0;
1049  bufferSize = (input == null) ? 0 : input.Read(buffer, 0, buffer.Length);
1050  if (bufferSize < 0)
1051  {
1052  throw new InvalidOperationException("Stream.Read returned a negative count");
1053  }
1054  if (bufferSize == 0)
1055  {
1056  if (mustSucceed)
1057  {
1058  throw InvalidProtocolBufferException.TruncatedMessage();
1059  }
1060  else
1061  {
1062  return false;
1063  }
1064  }
1065  else
1066  {
1067  RecomputeBufferSizeAfterLimit();
1068  int totalBytesRead =
1069  totalBytesRetired + bufferSize + bufferSizeAfterLimit;
1070  if (totalBytesRead > sizeLimit || totalBytesRead < 0)
1071  {
1072  throw InvalidProtocolBufferException.SizeLimitExceeded();
1073  }
1074  return true;
1075  }
1076  }

1099  internal byte[] ReadRawBytes(int size)
1100  {
1101  if (size < 0)
1102  {
1103  throw InvalidProtocolBufferException.NegativeSize();
1104  }
1105 
1106  if (totalBytesRetired + bufferPos + size > currentLimit)
1107  {
1108  // Read to the end of the stream (up to the current limit) anyway.
1109  SkipRawBytes(currentLimit - totalBytesRetired - bufferPos);
1110  // Then fail.
1111  throw InvalidProtocolBufferException.TruncatedMessage();
1112  }
1113 
1114  if (size <= bufferSize - bufferPos)
1115  {
1116  // We have all the bytes we need already.
1117  byte[] bytes = new byte[size];
1118  ByteArray.Copy(buffer, bufferPos, bytes, 0, size);
1119  bufferPos += size;
1120  return bytes;
1121  }
1122  else if (size < buffer.Length)
1123  {
1124  // Reading more bytes than are in the buffer, but not an excessive number
1125  // of bytes. We can safely allocate the resulting array ahead of time.
1126 
1127  // First copy what we have.
1128  byte[] bytes = new byte[size];
1129  int pos = bufferSize - bufferPos;
1130  ByteArray.Copy(buffer, bufferPos, bytes, 0, pos);
1131  bufferPos = bufferSize;
1132 
1133  // We want to use RefillBuffer() and then copy from the buffer into our
1134  // byte array rather than reading directly into our byte array because
1135  // the input may be unbuffered.
1136  RefillBuffer(true);
1137 
1138  while (size - pos > bufferSize)
1139  {
1140  Buffer.BlockCopy(buffer, 0, bytes, pos, bufferSize);
1141  pos += bufferSize;
1142  bufferPos = bufferSize;
