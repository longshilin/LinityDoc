98  public void BeforeSending(IEnumerable<RequestMessage> messages, ConnectionId connectionId, IByteBuffer buffer, MessageEncoderSettings encoderSettings, Stopwatch stopwatch)
99  {
100  using (var stream = new ByteBufferStream(buffer, ownsBuffer: false))
101  {
102  var messageQueue = new Queue<RequestMessage>(messages);
103 
104  while (messageQueue.Count > 0)
105  {
106  ProcessRequestMessages(messageQueue, connectionId, stream, encoderSettings, stopwatch);
107  }
108  }
109  }

111  public void AfterSending(IEnumerable<RequestMessage> messages, ConnectionId connectionId)
112  {
113  foreach (var message in messages)
114  {
115  CommandState state;
116  if (_state.TryGetValue(message.RequestId, out state) &&
117  state.ExpectedResponseType == ExpectedResponseType.None)
118  {
119  state.Stopwatch.Stop();
120  if (_succeededEvent != null)
121  {
122  var @event = new CommandSucceededEvent(
123  state.CommandName,
124  state.NoResponseResponse ?? new BsonDocument("ok", 1),
125  state.OperationId,
126  message.RequestId,
127  connectionId,
128  state.Stopwatch.Elapsed);
129 
130  _succeededEvent(@event);
131  }
132 
133  _state.TryRemove(message.RequestId, out state);
134  }
135  }
136  }

230  private void ProcessRequestMessages(Queue<RequestMessage> messageQueue, ConnectionId connectionId, Stream stream, MessageEncoderSettings encoderSettings, Stopwatch stopwatch)
231  {
232  var message = messageQueue.Dequeue();
233  switch (message.MessageType)
234  {
235  case MongoDBMessageType.Command:
236  ProcessCommandRequestMessage((CommandRequestMessage)message, messageQueue, connectionId, new CommandMessageBinaryEncoder(stream, encoderSettings), stopwatch);
237  break;
238  case MongoDBMessageType.Delete:
239  ProcessDeleteMessage((DeleteMessage)message, messageQueue, connectionId, new DeleteMessageBinaryEncoder(stream, encoderSettings), stopwatch);
240  break;
241  case MongoDBMessageType.GetMore:
242  ProcessGetMoreMessage((GetMoreMessage)message, connectionId, stopwatch);
243  break;
244  case MongoDBMessageType.Insert:
245  ProcessInsertMessage(message, messageQueue, connectionId, new InsertMessageBinaryEncoder<RawBsonDocument>(stream, encoderSettings, RawBsonDocumentSerializer.Instance), stopwatch);
246  break;
247  case MongoDBMessageType.KillCursors:
248  ProcessKillCursorsMessages((KillCursorsMessage)message, connectionId, stopwatch);
249  break;
250  case MongoDBMessageType.Query:
251  ProcessQueryMessage((QueryMessage)message, connectionId, new QueryMessageBinaryEncoder(stream, encoderSettings), stopwatch);
252  break;
253  case MongoDBMessageType.Update:
254  ProcessUpdateMessage((UpdateMessage)message, messageQueue, connectionId, new UpdateMessageBinaryEncoder(stream, encoderSettings), stopwatch);
255  break;
256  default:
257  throw new MongoInternalException("Invalid message type.");
258  }
259  }

261  private void ProcessCommandRequestMessage(CommandRequestMessage originalMessage, Queue<RequestMessage> messageQueue, ConnectionId connectionId, CommandMessageBinaryEncoder encoder, Stopwatch stopwatch)
262  {
263  var requestId = originalMessage.RequestId;
264  var operationId = EventContext.OperationId;
265 
266  var decodedMessage = encoder.ReadMessage();
267  using (new CommandMessageDisposer(decodedMessage))
268  {
269  var type0Section = decodedMessage.Sections.OfType<Type0CommandMessageSection>().Single();
270  var command = (BsonDocument)type0Section.Document;
271  var type1Sections = decodedMessage.Sections.OfType<Type1CommandMessageSection>().ToList();
272  if (type1Sections.Count > 0)
273  {
274  command = new BsonDocument(command); // materialize the top level of the command RawBsonDocument
275  foreach (var type1Section in type1Sections)
276  {
277  var name = type1Section.Identifier;
278  var items = new BsonArray(type1Section.Documents.GetBatchItems().Cast<RawBsonDocument>());
279  command[name] = items;
280  }
281  }
282  var commandName = command.GetElement(0).Name;
283  var databaseName = command["$db"].AsString;
284  var databaseNamespace = new DatabaseNamespace(databaseName);
285  if (__securitySensitiveCommands.Contains(commandName))
286  {
287  command = new BsonDocument();
288  }
289 
290  if (_startedEvent != null)
291  {
292 
293  var @event = new CommandStartedEvent(
294  commandName,
295  command,
296  databaseNamespace,
297  operationId,
298  requestId,
299  connectionId);
300 
301  _startedEvent(@event);
302  }
303 
304  if (_shouldTrackState)
305  {
306  _state.TryAdd(requestId, new CommandState
307  {
308  CommandName = commandName,
309  OperationId = operationId,
310  Stopwatch = stopwatch,
311  QueryNamespace = new CollectionNamespace(databaseNamespace, "$cmd"),
312  ExpectedResponseType = decodedMessage.MoreToCome ? ExpectedResponseType.None : ExpectedResponseType.Command
313  });
314  }
315  }
316  }

318  private void ProcessCommandResponseMessage(CommandState state, CommandResponseMessage message, IByteBuffer buffer, ConnectionId connectionId, MessageEncoderSettings encoderSettings)
319  {
320  var wrappedMessage = message.WrappedMessage;
321  var type0Section = wrappedMessage.Sections.OfType<Type0CommandMessageSection<RawBsonDocument>>().Single();
322  var reply = (BsonDocument)type0Section.Document;
323 
324  BsonValue ok;
325  if (!reply.TryGetValue("ok", out ok))
326  {
327  // this is a degenerate case with the server and
328  // we don't really know what to do here...
329  return;
330  }
331 
332  if (__securitySensitiveCommands.Contains(state.CommandName))
333  {
334  reply = new BsonDocument();
335  }
336 
337  if (ok.ToBoolean())
338  {
339  if (_succeededEvent != null)
340  {
341  _succeededEvent(new CommandSucceededEvent(
342  state.CommandName,
343  reply,
344  state.OperationId,
345  message.ResponseTo,
346  connectionId,
347  state.Stopwatch.Elapsed));
348  }
349  }
350  else
351  {
352  if (_failedEvent != null)
353  {
354  _failedEvent(new CommandFailedEvent(
355  state.CommandName,
356  new MongoCommandException(
357  connectionId,
358  string.Format("{0} command failed", state.CommandName),
359  null,
360  reply),
361  state.OperationId,
362  message.ResponseTo,
363  connectionId,
364  state.Stopwatch.Elapsed));
365  }
366  }

369  private void ProcessDeleteMessage(DeleteMessage originalMessage, Queue<RequestMessage> messageQueue, ConnectionId connectionId, DeleteMessageBinaryEncoder encoder, Stopwatch stopwatch)
370  {
371  var commandName = "delete";
372  var operationId = EventContext.OperationId;
373  int requestId = originalMessage.RequestId;
374  var expectedResponseType = ExpectedResponseType.None;
375  int gleRequestId;
376  WriteConcern writeConcern;
377  if (TryGetWriteConcernFromGLE(messageQueue, out gleRequestId, out writeConcern))
378  {
379  requestId = gleRequestId;
380  expectedResponseType = ExpectedResponseType.GLE;
381  }
382 
383  if (_startedEvent != null)
384  {
385  var decodedMessage = encoder.ReadMessage(RawBsonDocumentSerializer.Instance);
386  try
387  {
388  var entry = new BsonDocument
389  {
390  { "q", decodedMessage.Query },
391  { "limit", decodedMessage.IsMulti ? 0 : 1 }
392  };
393  var command = new BsonDocument
394  {
395  { commandName, decodedMessage.CollectionNamespace.CollectionName },
396  { "deletes", new BsonArray(new [] { entry }) }
397  };
398 
399  if (writeConcern == null)
400  {
401  command["writeConcern"] = WriteConcern.Unacknowledged.ToBsonDocument();
402  }
403  else if (!writeConcern.IsServerDefault)
404  {
405  command["writeConcern"] = writeConcern.ToBsonDocument();
406  }
407 
408  var @event = new CommandStartedEvent(
409  commandName,
410  command,
411  decodedMessage.CollectionNamespace.DatabaseNamespace,
412  operationId,
413  requestId,
414  connectionId);
415 
416  _startedEvent(@event);
417  }
418  finally
419  {
420  var disposable = decodedMessage.Query as IDisposable;
421  if (disposable != null)
422  {
423  disposable.Dispose();
424  }
425  }
426  }
427 
428  if (_shouldTrackState)
429  {
430  _state.TryAdd(requestId, new CommandState
431  {
432  CommandName = commandName,
433  OperationId = operationId,
434  Stopwatch = stopwatch,
435  ExpectedResponseType = expectedResponseType
436  });
437  }

477  private void ProcessInsertMessage(RequestMessage message, Queue<RequestMessage> messageQueue, ConnectionId connectionId, InsertMessageBinaryEncoder<RawBsonDocument> encoder, Stopwatch stopwatch)
478  {
479  var commandName = "insert";
480  var operationId = EventContext.OperationId;
481  var requestId = message.RequestId;
482  var expectedResponseType = ExpectedResponseType.None;
483  int numberOfDocuments = 0;
484  int gleRequestId;
485  WriteConcern writeConcern;
486  if (TryGetWriteConcernFromGLE(messageQueue, out gleRequestId, out writeConcern))
487  {
488  requestId = gleRequestId;
489  expectedResponseType = ExpectedResponseType.GLE;
490  }
491 
492  if (_startedEvent != null)
493  {
494  // InsertMessage is generic, and we don't know the generic type...
495  // Plus, for this we really want BsonDocuments, not whatever the generic type is.
496  var decodedMessage = encoder.ReadMessage();
497 
498  var documents = decodedMessage.DocumentSource.GetBatchItems();
499  numberOfDocuments = documents.Count;
500  try
501  {
502  var command = new BsonDocument
503  {
504  { commandName, decodedMessage.CollectionNamespace.CollectionName },
505  { "documents", new BsonArray(documents) },
506  { "ordered", !decodedMessage.ContinueOnError }
507  };
508 
509  if (writeConcern == null)
510  {
511  command["writeConcern"] = WriteConcern.Unacknowledged.ToBsonDocument();
512  }
513  else if (!writeConcern.IsServerDefault)
514  {
515  command["writeConcern"] = writeConcern.ToBsonDocument();
516  }
517 
518  var @event = new CommandStartedEvent(
519  commandName,
520  command,
521  decodedMessage.CollectionNamespace.DatabaseNamespace,
522  operationId,
523  requestId,
524  connectionId);
525 
526  _startedEvent(@event);
527  }
528  finally
529  {
530  foreach (var document in documents)
531  {
532  document.Dispose();
533  }
534  }
535  }
536 
537  if (_shouldTrackState)
538  {
539  _state.TryAdd(requestId, new CommandState
540  {
541  CommandName = commandName,
542  OperationId = operationId,
543  Stopwatch = stopwatch,
544  ExpectedResponseType = expectedResponseType,
545  NumberOfInsertedDocuments = numberOfDocuments
546  });
547  }
548  }

592  private void ProcessQueryMessage(QueryMessage originalMessage, ConnectionId connectionId, QueryMessageBinaryEncoder encoder, Stopwatch stopwatch)
593  {
594  var requestId = originalMessage.RequestId;
595  var operationId = EventContext.OperationId;
596 
597  var decodedMessage = encoder.ReadMessage(RawBsonDocumentSerializer.Instance);
598  try
599  {
600  var isCommand = IsCommand(decodedMessage.CollectionNamespace);
601  string commandName;
602  BsonDocument command;
603  if (isCommand)
604  {
605  command = decodedMessage.Query;
606  var firstElement = command.GetElement(0);
607  commandName = firstElement.Name;
608  if (__securitySensitiveCommands.Contains(commandName))
609  {
610  command = new BsonDocument();
611  }
612  }
613  else
614  {
615  commandName = "find";
616  command = BuildFindCommandFromQuery(decodedMessage);
617  if (decodedMessage.Query.GetValue("$explain", false).ToBoolean())
618  {
619  commandName = "explain";
620  command = new BsonDocument("explain", command);
621  }
622  }
623 
624  if (_startedEvent != null)
625  {
626  var @event = new CommandStartedEvent(
627  commandName,
628  command,
629  decodedMessage.CollectionNamespace.DatabaseNamespace,
630  operationId,
631  requestId,
632  connectionId);
633 
634  _startedEvent(@event);
635  }
636 
637  if (_shouldTrackState)
638  {
639  _state.TryAdd(requestId, new CommandState
640  {
641  CommandName = commandName,
642  OperationId = operationId,
643  Stopwatch = stopwatch,
644  QueryNamespace = decodedMessage.CollectionNamespace,
645  ExpectedResponseType = isCommand ? ExpectedResponseType.Command : ExpectedResponseType.Query
646  });
647  }
648  }
649  finally
650  {
651  var disposable = decodedMessage.Query as IDisposable;
652  if (disposable != null)
653  {
654  disposable.Dispose();
655  }
656  disposable = decodedMessage.Fields as IDisposable;
657  if (disposable != null)
658  {
659  disposable.Dispose();
660  }
661  }
662  }

664  private void ProcessReplyMessage(CommandState state, ResponseMessage message, IByteBuffer buffer, ConnectionId connectionId, MessageEncoderSettings encoderSettings)
665  {
666  state.Stopwatch.Stop();
667  bool disposeOfDocuments = false;
668  var replyMessage = message as ReplyMessage<RawBsonDocument>;
669  if (replyMessage == null)
670  {
671  // ReplyMessage is generic, which means that we can't use it here, so, we need to use a different one...
672  using (var stream = new ByteBufferStream(buffer, ownsBuffer: false))
673  {
674  var encoderFactory = new BinaryMessageEncoderFactory(stream, encoderSettings);
675  replyMessage = (ReplyMessage<RawBsonDocument>)encoderFactory
676  .GetReplyMessageEncoder(RawBsonDocumentSerializer.Instance)
677  .ReadMessage();
678  disposeOfDocuments = true;
679  }
680  }
681 
682  try
683  {
684  if (replyMessage.CursorNotFound ||
685  replyMessage.QueryFailure ||
686  (state.ExpectedResponseType != ExpectedResponseType.Query && replyMessage.Documents.Count == 0))
687  {
688  var queryFailureDocument = replyMessage.QueryFailureDocument;
689  if (__securitySensitiveCommands.Contains(state.CommandName))
690  {
691  queryFailureDocument = new BsonDocument();
692  }
693  if (_failedEvent != null)
694  {
695  _failedEvent(new CommandFailedEvent(
696  state.CommandName,
697  new MongoCommandException(
698  connectionId,
699  string.Format("{0} command failed", state.CommandName),
700  null,
701  queryFailureDocument),
702  state.OperationId,
703  replyMessage.ResponseTo,
704  connectionId,
705  state.Stopwatch.Elapsed));
706  }
707  }
708  else
709  {
710  switch (state.ExpectedResponseType)
711  {
712  case ExpectedResponseType.Command:
713  ProcessCommandReplyMessage(state, replyMessage, connectionId);
714  break;
715  case ExpectedResponseType.GLE:
716  ProcessGLEReplyMessage(state, replyMessage, connectionId);
717  break;
718  case ExpectedResponseType.Query:
719  ProcessQueryReplyMessage(state, replyMessage, connectionId);
720  break;
721  }
722  }
723  }
724  finally
725  {
726  if (disposeOfDocuments && replyMessage.Documents != null)
727  {
728  replyMessage.Documents.ForEach(d => d.Dispose());
729  }
730  }
731  }

779  {

781  BsonValue ok;

784  // this is a degenerate case with the server and

906  private void ProcessUpdateMessage(UpdateMessage originalMessage, Queue<RequestMessage> messageQueue, ConnectionId connectionId, UpdateMessageBinaryEncoder encoder, Stopwatch stopwatch)
907  {
908  var commandName = "update";
909  int requestId = originalMessage.RequestId;
910  var operationId = EventContext.OperationId;
911  var expectedResponseType = ExpectedResponseType.None;
912  BsonValue upsertedId = null;
913  int gleRequestId;
914  WriteConcern writeConcern;
915  if (TryGetWriteConcernFromGLE(messageQueue, out gleRequestId, out writeConcern))

919  }
920

922  {

924  try
925  {
926  if (_shouldTrackState)
927  {
928  // GLE result on older versions of the server didn't return
929  // the upserted id when it wasn't an object id, so we'll
930  // attempt to get it from the messages.
931  if (!decodedMessage.Update.TryGetValue("_id", out upsertedId))
932  {
933  decodedMessage.Query.TryGetValue("_id", out upsertedId);
934  }
935  }
936

938  {
939  { "q", decodedMessage.Query },
940  { "u", decodedMessage.Update },
941  { "upsert", decodedMessage.IsUpsert },
942  { "multi", decodedMessage.IsMulti }

951  {
952  command["writeConcern"] = WriteConcern.Unacknowledged.ToBsonDocument();
953  }
954  else if (!writeConcern.IsServerDefault)
955  {
956  command["writeConcern"] = writeConcern.ToBsonDocument();
957  }
958

960  commandName,
961  command,
962  decodedMessage.CollectionNamespace.DatabaseNamespace,
963  operationId,
964  requestId,
965  connectionId);
966 
967  _startedEvent(@event);

970  {

972  if (disposable != null)
973  {
974  disposable.Dispose();
975  }
976  disposable = decodedMessage.Update as IDisposable;
977  if (disposable != null)

979  disposable.Dispose();

981  }
982  }

997  private BsonDocument BuildFindCommandFromQuery(QueryMessage message)
998  {
999  var batchSize = EventContext.FindOperationBatchSize ?? 0;
1000  var limit = EventContext.FindOperationLimit ?? 0;
1001  var command = new BsonDocument
1002  {
1003  { "find", message.CollectionNamespace.CollectionName },
1004  { "projection", message.Fields, message.Fields != null },
1005  { "skip", message.Skip, message.Skip > 0 },
1006  { "batchSize", batchSize, batchSize > 0 },
1007  { "limit", limit, limit != 0 },
1008  { "awaitData", message.AwaitData, message.AwaitData },
1009  { "noCursorTimeout", message.NoCursorTimeout, message.NoCursorTimeout },
1010  { "allowPartialResults", message.PartialOk, message.PartialOk },
1011  { "tailable", message.TailableCursor, message.TailableCursor },
1012  { "oplogReplay", message.OplogReplay, message.OplogReplay }
1013  };
1014 
1015  var query = message.Query;
1016  if (query.ElementCount == 1 && !query.Contains("$query"))
1017  {
1018  command["filter"] = query;
1019  }
1020  else
1021  {
1022  foreach (var element in query)
1023  {
1024  switch (element.Name)
1025  {
1026  case "$query":
1027  command["filter"] = element.Value;
1028  break;
1029  case "$orderby":
1030  command["sort"] = element.Value;
1031  break;
1032  case "$showDiskLoc":
1033  command["showRecordId"] = element.Value;
1034  break;
1035  case "$explain":
1036  // explain is special and gets handled elsewhere
1037  break;
1038  default:
1039  if (element.Name.StartsWith("$"))
1040  {
1041  command[element.Name.Substring(1)] = element.Value;
1042  }
1043  else
1044  {
1045  // theoretically, this should never happen and is illegal.
1046  // however, we'll push this up anyways and a command-failure
1047  // event will likely get thrown.
1048  command[element.Name] = element.Value;
1049  }
1050  break;
1051  }
1052  }
1053  }
1054 
1055  return command;
1056  }

1058  private bool TryGetWriteConcernFromGLE(Queue<RequestMessage> messageQueue, out int requestId, out WriteConcern writeConcern)
1059  {
1060  requestId = -1;
1061  writeConcern = null;
1062  if (messageQueue.Count == 0)
1063  {
1064  return false;
1065  }
1066 
1067  var message = messageQueue.Peek();
1068  if (message.MessageType != MongoDBMessageType.Query)
1069  {
1070  return false;
1071  }
1072 
1073  var queryMessage = (QueryMessage)message;
1074 
1075  if (!IsCommand(queryMessage.CollectionNamespace))
1076  {
1077  return false;
1078  }
1079 
1080  var query = queryMessage.Query;
1081  var firstElement = query.GetElement(0);
1082  if (firstElement.Name != "getLastError")
1083  {
1084  return false;
1085  }
1086 
1087  messageQueue.Dequeue(); // consume it so that we don't process it later...
1088  requestId = queryMessage.RequestId;
1089 
1090  writeConcern = WriteConcern.FromBsonDocument(query);
1091  return true;
1092  }
