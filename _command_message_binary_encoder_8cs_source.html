52  public CommandMessage ReadMessage()
53  {
54  var reader = CreateBinaryReader();
55  var stream = reader.BsonStream;
56  var messageStartPosition = stream.Position;
57 
58  var messageLength = stream.ReadInt32();
59  EnsureMessageLengthIsValid(messageLength);
60  var messageEndPosition = messageStartPosition + messageLength;
61 
62  var requestId = stream.ReadInt32();
63  var responseTo = stream.ReadInt32();
64  var opcode = (Opcode)stream.ReadInt32();
65  EnsureOpcodeIsValid(opcode);
66  var flags = (OpMsgFlags)stream.ReadInt32();
67  EnsureFlagsAreValid(flags);
68  var moreToCome = (flags & OpMsgFlags.MoreToCome) != 0;
69  var sections = ReadSections(reader, messageEndPosition);
70  EnsureExactlyOneType0SectionIsPresent(sections);
71  EnsureMessageEndedAtEndPosition(stream, messageEndPosition);
72 
73  return new CommandMessage(requestId, responseTo, sections, moreToCome);
74  }

80  public void WriteMessage(CommandMessage message)
81  {

83 
84  var writer = CreateBinaryWriter();

87 
88  stream.WriteInt32(0); // messageLength

94  stream.BackpatchSize(messageStartPosition);
95

287  private void WriteType1Section(BsonBinaryWriter writer, Type1CommandMessageSection section, long messageStartPosition)
288  {
289  var stream = writer.BsonStream;
290  var serializer = section.DocumentSerializer;
291  var context = BsonSerializationContext.CreateRoot(writer);
292  var maxMessageSize = MaxMessageSize;
293 
294  var payloadStartPosition = stream.Position;
295  stream.WriteInt32(0); // size
296  stream.WriteCString(section.Identifier);
297 
298  var batch = section.Documents;
299  var maxDocumentSize = section.MaxDocumentSize ?? writer.Settings.MaxDocumentSize;
300  writer.PushSettings(s => ((BsonBinaryWriterSettings)s).MaxDocumentSize = maxDocumentSize);
301  writer.PushElementNameValidator(section.ElementNameValidator);
302  try
303  {
304  var maxBatchCount = Math.Min(batch.Count, section.MaxBatchCount ?? int.MaxValue);
305  var processedCount = maxBatchCount;
306  for (var i = 0; i < maxBatchCount; i++)
307  {
308  var documentStartPosition = stream.Position;
309  var document = batch.Items[batch.Offset + i];
310  serializer.Serialize(context, document);
311 
312  var messageSize = stream.Position - messageStartPosition;
313  if (messageSize > maxMessageSize && batch.CanBeSplit)
314  {
315  stream.Position = documentStartPosition;
316  stream.SetLength(documentStartPosition);
317  processedCount = i;
318  break;
319  }
320  }
321  batch.SetProcessedCount(processedCount);
322  }
323  finally
324  {
325  writer.PopElementNameValidator();
326  writer.PopSettings();
327  }
328  stream.BackpatchSize(payloadStartPosition);
329  }

383  private long FindType0SectionPosition()
384  {
385  _stream.Position = _messageStartPosition;
386  var messageLength = _stream.ReadInt32();
387 
388  _stream.Position = _messageStartPosition + 20;
389  var messageEndPosition = _messageStartPosition + messageLength;
390  while (_stream.Position < messageEndPosition)
391  {
392  var sectionPosition = _stream.Position;
393 
394  var payloadType = (PayloadType)_stream.ReadByte();
395  if (payloadType == PayloadType.Type0)
396  {
397  return sectionPosition;
398  }
399  else if (payloadType != PayloadType.Type1)
400  {
401  throw new FormatException($"Invalid payload type: {payloadType}.");
402  }
403 
404  var sectionSize = _stream.ReadInt32();

406  }

411  private long FindWPosition()
412  {
413  var type0SectionPosition = FindType0SectionPosition();
414 
415  _stream.Position = type0SectionPosition + 1;
416  using (var reader = new BsonBinaryReader(_stream))
417  {
418  reader.ReadStartDocument();
419  while (reader.ReadBsonType() != 0)
420  {
421  if (reader.ReadName() == "writeConcern")
422  {
423  reader.ReadStartDocument();
424  while (reader.ReadBsonType() != 0)
425  {
426  if (reader.ReadName() == "w")
427  {
428  if (reader.CurrentBsonType == BsonType.Int32)
429  {
430  return _stream.Position;
431  }
432  goto notFound;
433  }
434 
435  reader.SkipValue();
436  }
437  goto notFound;
438  }
439 
440  reader.SkipValue();
441  }
442  }
443
