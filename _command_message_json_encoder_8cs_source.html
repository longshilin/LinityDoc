51  public CommandMessage ReadMessage()
52  {
53  var reader = CreateJsonReader();
54  var context = BsonDeserializationContext.CreateRoot(reader);
55  var messageDocument = BsonDocumentSerializer.Instance.Deserialize(context);
56 
57  var opcode = messageDocument["opcode"].AsString;
58  if (opcode != "opmsg")
59  {
60  throw new FormatException($"Command message invalid opcode: \"{opcode}\".");
61  }
62  var requestId = messageDocument["requestId"].ToInt32();
63  var responseTo = messageDocument["responseTo"].ToInt32();
64  var moreToCome = messageDocument.GetValue("moreToCome", false).ToBoolean();
65  var sections = ReadSections(messageDocument["sections"].AsBsonArray.Cast<BsonDocument>());
66 
67  return new CommandMessage(requestId, responseTo, sections, moreToCome);
68  }

105  private CommandMessageSection ReadSection(BsonDocument sectionDocument)
106  {
107  var payloadType = sectionDocument["payloadType"].ToInt32();
108  switch (payloadType)
109  {
110  case 0:
111  return ReadType0Section(sectionDocument);
112 
113  case 1:
114  return ReadType1Section(sectionDocument);
115 
116  default:
117  throw new FormatException($"Command message invalid payload type: {payloadType}.");
118  }

138  private CommandMessageSection ReadType1Section(BsonDocument sectionDocument)
139  {
140  var identifier = sectionDocument["identifier"].AsString;
141  var documents = sectionDocument["documents"].AsBsonArray.Cast<BsonDocument>().ToList();
142  var batch = new BatchableSource<BsonDocument>(documents, canBeSplit: false);
143  return new Type1CommandMessageSection<BsonDocument>(identifier, batch, BsonDocumentSerializer.Instance, NoOpElementNameValidator.Instance, null, null);
144  }

146  private void WriteSection(IBsonWriter writer, CommandMessageSection section)
147  {
148  writer.WriteStartDocument();
149  writer.WriteInt32("payloadType", (int)section.PayloadType);
150 
151  switch (section.PayloadType)
152  {
153  case PayloadType.Type0:
154  WriteType0Section(writer, (Type0CommandMessageSection)section);
155  break;
156 
157  case PayloadType.Type1:
158  WriteType1Section(writer, (Type1CommandMessageSection)section);
159  break;
160 
161  default:
162  throw new ArgumentException($"Invalid payload type: {section.PayloadType}.");
163  }
164 
165  writer.WriteEndDocument();
166  }
167 
168  private void WriteSections(IBsonWriter writer, IEnumerable<CommandMessageSection> sections)
169  {
170  writer.WriteStartArray();
171  foreach (var section in sections)
172  {
173  WriteSection(writer, section);
174  }
175  writer.WriteEndArray();
176  }

186  private void WriteType1Section(IBsonWriter writer, Type1CommandMessageSection section)
187  {
188  writer.WriteString("identifier", section.Identifier);
189  writer.WriteName("documents");
190  writer.WriteStartArray();
191  var batch = section.Documents;
192  var serializer = section.DocumentSerializer;
193  var context = BsonSerializationContext.CreateRoot(writer);
194  for (var i = 0; i < batch.Count; i++)
195  {
196  var document = batch.Items[batch.Offset + i];
197  serializer.Serialize(context, document);
198  }
199  writer.WriteEndArray();
200  }
