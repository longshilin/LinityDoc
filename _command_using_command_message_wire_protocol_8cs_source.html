82  // public methods
83  public TCommandResult Execute(IConnection connection, CancellationToken cancellationToken)
84  {
85  try
86  {
87  var message = CreateCommandMessage(connection.Description);
88 
89  try
90  {
91  connection.SendMessage(message, _messageEncoderSettings, cancellationToken);
92  }
93  finally
94  {
95  MessageWasProbablySent(message);
96  }
97 
98  if (message.WrappedMessage.ResponseExpected)
99  {
100  var encoderSelector = new CommandResponseMessageEncoderSelector();
101  var response = (CommandResponseMessage)connection.ReceiveMessage(message.RequestId, encoderSelector, _messageEncoderSettings, cancellationToken);
102  return ProcessResponse(connection.ConnectionId, response.WrappedMessage);
103  }
104  else
105  {
106  return default(TCommandResult);
107  }
108  }
109  catch (MongoException exception) when (ShouldAddTransientTransactionError(exception))
110  {
111  exception.AddErrorLabel("TransientTransactionError");
112  throw;
113  }
114  }

150  private CommandRequestMessage CreateCommandMessage(ConnectionDescription connectionDescription)
151  {
152  var requestId = RequestMessage.GetNextRequestId();
153  var responseTo = 0;
154  var sections = CreateSections(connectionDescription);
155  var moreToCome = _responseHandling == CommandResponseHandling.NoResponseExpected;
156  var wrappedMessage = new CommandMessage(requestId, responseTo, sections, moreToCome)
157  {
158  PostWriteAction = _postWriteAction
159  };
160  var shouldBeSent = (Func<bool>)(() => true);
161 
162  return new CommandRequestMessage(wrappedMessage, shouldBeSent);
163  }

178  private Type0CommandMessageSection<BsonDocument> CreateType0Section(ConnectionDescription connectionDescription)
179  {
180  var extraElements = new List<BsonElement>();
181 
182  var dbElement = new BsonElement("$db", _databaseNamespace.DatabaseName);
183  extraElements.Add(dbElement);
184 
185  if (_readPreference != null && _readPreference != ReadPreference.Primary)
186  {
187  var readPreferenceDocument = QueryHelper.CreateReadPreferenceDocument(_readPreference);
188  var readPreferenceElement = new BsonElement("$readPreference", readPreferenceDocument);
189  extraElements.Add(readPreferenceElement);
190  }
191 
192  if (_session.Id != null)
193  {
194  var lsidElement = new BsonElement("lsid", _session.Id);
195  extraElements.Add(lsidElement);
196  }
197 
198  if (_session.ClusterTime != null)
199  {
200  var clusterTimeElement = new BsonElement("$clusterTime", _session.ClusterTime);
201  extraElements.Add(clusterTimeElement);
202  }
203  Action<BsonWriterSettings> writerSettingsConfigurator = s => s.GuidRepresentation = GuidRepresentation.Unspecified;
204 
205  _session.AboutToSendCommand();
206  if (_session.IsInTransaction)
207  {
208  var transaction = _session.CurrentTransaction;
209  extraElements.Add(new BsonElement("txnNumber", transaction.TransactionNumber));
210  if (transaction.State == CoreTransactionState.Starting)
211  {
212  extraElements.Add(new BsonElement("startTransaction", true));
213  var readConcern = ReadConcernHelper.GetReadConcernForFirstCommandInTransaction(_session, connectionDescription);
214  if (readConcern != null)
215  {
216  extraElements.Add(new BsonElement("readConcern", readConcern));
217  }
218  }
219  extraElements.Add(new BsonElement("autocommit", false));
220  }
221 
222  var elementAppendingSerializer = new ElementAppendingSerializer<BsonDocument>(BsonDocumentSerializer.Instance, extraElements, writerSettingsConfigurator);
223  return new Type0CommandMessageSection<BsonDocument>(_command, elementAppendingSerializer);
224  }

240  [System.Diagnostics.CodeAnalysis.SuppressMessage("Microsoft.Usage", "CA2202:Do not dispose objects multiple times")]
241  private TCommandResult ProcessResponse(ConnectionId connectionId, CommandMessage responseMessage)
242  {
243  using (new CommandMessageDisposer(responseMessage))
244  {
245  var rawDocument = responseMessage.Sections.OfType<Type0CommandMessageSection<RawBsonDocument>>().Single().Document;
246 
247  var binaryReaderSettings = new BsonBinaryReaderSettings();
248  if (_messageEncoderSettings != null)
249  {
250  binaryReaderSettings.Encoding = _messageEncoderSettings.GetOrDefault<UTF8Encoding>(MessageEncoderSettingsName.ReadEncoding, Utf8Encodings.Strict);
251  binaryReaderSettings.GuidRepresentation = _messageEncoderSettings.GetOrDefault<GuidRepresentation>(MessageEncoderSettingsName.GuidRepresentation, GuidRepresentation.CSharpLegacy);
252  };
253 
254  BsonValue clusterTime;
255  if (rawDocument.TryGetValue("$clusterTime", out clusterTime))
256  {
257  // note: we are assuming that _session is an instance of ClusterClockAdvancingClusterTime
258  // and that calling _session.AdvanceClusterTime will have the side effect of advancing the cluster's ClusterTime also
259  var materializedClusterTime = ((RawBsonDocument)clusterTime).Materialize(binaryReaderSettings);
260  _session.AdvanceClusterTime(materializedClusterTime);
261  }
262 
263  BsonValue operationTime;
264  if (rawDocument.TryGetValue("operationTime", out operationTime))
265  {
266  _session.AdvanceOperationTime(operationTime.AsBsonTimestamp);
267  }
268 
269  if (!rawDocument.GetValue("ok", false).ToBoolean())
270  {
271  var materializedDocument = rawDocument.Materialize(binaryReaderSettings);
272 
273  var commandName = _command.GetElement(0).Name;
274  if (commandName == "$query")
275  {
276  commandName = _command["$query"].AsBsonDocument.GetElement(0).Name;
277  }
278 
279  var notPrimaryOrNodeIsRecoveringException = ExceptionMapper.MapNotPrimaryOrNodeIsRecovering(connectionId, _command, materializedDocument, "errmsg");
280  if (notPrimaryOrNodeIsRecoveringException != null)
281  {
282  throw notPrimaryOrNodeIsRecoveringException;
283  }
284 
285  var mappedException = ExceptionMapper.Map(connectionId, materializedDocument);
286  if (mappedException != null)
287  {
288  throw mappedException;
289  }
290 
291  string message;
292  BsonValue errmsgBsonValue;
293  if (materializedDocument.TryGetValue("errmsg", out errmsgBsonValue) && errmsgBsonValue.IsString)
294  {
295  var errmsg = errmsgBsonValue.ToString();
296  message = string.Format("Command {0} failed: {1}.", commandName, errmsg);
297  }
298  else
299  {
300  message = string.Format("Command {0} failed.", commandName);
301  }
302 
303  throw new MongoCommandException(connectionId, message, _command, materializedDocument);
304  }
305 
306  if (rawDocument.Contains("writeConcernError"))
307  {
308  var materializedDocument = rawDocument.Materialize(binaryReaderSettings);
309  var writeConcernError = materializedDocument["writeConcernError"].AsBsonDocument;
310  var message = writeConcernError.AsBsonDocument.GetValue("errmsg", null)?.AsString;
311  var writeConcernResult = new WriteConcernResult(materializedDocument);
312  throw new MongoWriteConcernException(connectionId, message, writeConcernResult);
313  }
314 
315  using (var stream = new ByteBufferStream(rawDocument.Slice, ownsBuffer: false))
316  {
317  using (var reader = new BsonBinaryReader(stream, binaryReaderSettings))
318  {
319  var context = BsonDeserializationContext.CreateRoot(reader);
320  return _resultSerializer.Deserialize(context);
321  }
322  }
323  }
324  }

326  private bool ShouldAddTransientTransactionError(MongoException exception)
327  {
328  if (_session.IsInTransaction)
329  {
330  if (exception is MongoConnectionException)
331  {
332  return true;
333  }
334  }
335 
336  return false;
337  }
