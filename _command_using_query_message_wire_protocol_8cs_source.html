85  private QueryMessage CreateMessage(ConnectionDescription connectionDescription, out bool messageContainsSessionId)
86  {
87  var commandWithPayloads = CombineCommandWithPayloads(connectionDescription);
88  var wrappedCommand = WrapCommandForQueryMessage(commandWithPayloads, connectionDescription, out messageContainsSessionId);
89  var slaveOk = _readPreference != null && _readPreference.ReadPreferenceMode != ReadPreferenceMode.Primary;
90 
91  return new QueryMessage(
92  RequestMessage.GetNextRequestId(),
93  _databaseNamespace.CommandCollection,
94  wrappedCommand,
95  null,
96  _commandValidator,
97  0,
98  -1,
99  slaveOk,
100  false,
101  false,
102  false,
103  false,
104  false)
105  {
106  PostWriteAction = _postWriteAction,
107  ResponseHandling = _responseHandling
108  };

111  public TCommandResult Execute(IConnection connection, CancellationToken cancellationToken)
112  {
113  bool messageContainsSessionId;
114  var message = CreateMessage(connection.Description, out messageContainsSessionId);
115  connection.SendMessage(message, _messageEncoderSettings, cancellationToken);
116  if (messageContainsSessionId)
117  {
118  _session.WasUsed();
119  }
120 
121  switch (message.ResponseHandling)
122  {
123  case CommandResponseHandling.Ignore:
124  IgnoreResponse(connection, message, cancellationToken);
125  return default(TCommandResult);
126  default:
127  var encoderSelector = new ReplyMessageEncoderSelector<RawBsonDocument>(RawBsonDocumentSerializer.Instance);
128  var reply = connection.ReceiveMessage(message.RequestId, encoderSelector, _messageEncoderSettings, cancellationToken);
129  return ProcessReply(connection.ConnectionId, (ReplyMessage<RawBsonDocument>)reply);
130  }
131  }

144  {

156  private BsonDocument CombineCommandWithPayloads(ConnectionDescription connectionDescription)
157  {
158  if (_commandPayloads == null || _commandPayloads.Count == 0)
159  {
160  return _command;
161  }
162 
163  var extraElements = new List<BsonElement>();
164  foreach (var type1Section in _commandPayloads)
165  {
166  var name = type1Section.Identifier;
167  var value = CreatePayloadArray(type1Section, connectionDescription);
168  var element = new BsonElement(name, value);
169  extraElements.Add(element);
170  }
171 
172  var payloadAppendingSerializer = new ElementAppendingSerializer<BsonDocument>(BsonDocumentSerializer.Instance, extraElements);
173  return new BsonDocumentWrapper(_command, payloadAppendingSerializer);
174  }

176  private BsonArray CreatePayloadArray(Type1CommandMessageSection payload, ConnectionDescription connectionDescription)
177  {
178  IBsonSerializer payloadSerializer;
179  if (payload.Documents.CanBeSplit)
180  {
181  payloadSerializer = CreateSizeLimitingPayloadSerializer(payload, connectionDescription);
182  }
183  else
184  {
185  payloadSerializer = CreateFixedCountPayloadSerializer(payload);
186  }
187 
188  var documents = new BsonDocumentWrapper(payload.Documents, payloadSerializer);
189  return new BsonArray { documents };
190  }

192  private IBsonSerializer CreateFixedCountPayloadSerializer(Type1CommandMessageSection payload)
193  {
194  var documentType = payload.DocumentType;
195  var serializerType = typeof(FixedCountBatchableSourceSerializer<>).MakeGenericType(documentType);
196  var itemSerializerType = typeof(IBsonSerializer<>).MakeGenericType(documentType);
197  var constructorParameterTypes = new[] { itemSerializerType, typeof(IElementNameValidator), typeof(int) };
198  var constructorInfo = serializerType.GetTypeInfo().GetConstructor(constructorParameterTypes);
199  var itemSerializer = payload.DocumentSerializer;
200  var itemElementNameValidator = payload.ElementNameValidator;
201  var count = payload.Documents.Count;
202  return (IBsonSerializer)constructorInfo.Invoke(new object[] { itemSerializer, itemElementNameValidator, count });
203  }

205  private IBsonSerializer CreateSizeLimitingPayloadSerializer(Type1CommandMessageSection payload, ConnectionDescription connectionDescription)
206  {
207  var documentType = payload.DocumentType;
208  var serializerType = typeof(SizeLimitingBatchableSourceSerializer<>).MakeGenericType(documentType);
209  var itemSerializerType = typeof(IBsonSerializer<>).MakeGenericType(documentType);
210  var constructorParameterTypes = new[] { itemSerializerType, typeof(IElementNameValidator), typeof(int), typeof(int), typeof(int) };
211  var constructorInfo = serializerType.GetTypeInfo().GetConstructor(constructorParameterTypes);
212  var itemSerializer = payload.DocumentSerializer;
213  var itemElementNameValidator = payload.ElementNameValidator;
214  var maxBatchCount = Math.Min(payload.MaxBatchCount ?? int.MaxValue, connectionDescription.MaxBatchCount);
215  var maxItemSize = payload.MaxDocumentSize ?? connectionDescription.MaxDocumentSize;
216  var maxBatchSize = connectionDescription.MaxDocumentSize;
217  return (IBsonSerializer)constructorInfo.Invoke(new object[] { itemSerializer, itemElementNameValidator, maxBatchCount, maxItemSize, maxBatchSize });
218  }

226  [System.Diagnostics.CodeAnalysis.SuppressMessage("Microsoft.Usage", "CA2202:Do not dispose objects multiple times")]
227  private TCommandResult ProcessReply(ConnectionId connectionId, ReplyMessage<RawBsonDocument> reply)
228  {
229  if (reply.NumberReturned == 0)
230  {
231  throw new MongoCommandException(connectionId, "Command returned no documents.", _command);
232  }
233  if (reply.NumberReturned > 1)
234  {
235  throw new MongoCommandException(connectionId, "Command returned multiple documents.", _command);
236  }
237  if (reply.QueryFailure)
238  {
239  var failureDocument = reply.QueryFailureDocument;
240  throw ExceptionMapper.Map(connectionId, failureDocument) ?? new MongoCommandException(connectionId, "Command failed.", _command, failureDocument);
241  }
242 
243  using (var rawDocument = reply.Documents[0])
244  {
245  var binaryReaderSettings = new BsonBinaryReaderSettings();
246  if (_messageEncoderSettings != null)
247  {
248  binaryReaderSettings.Encoding = _messageEncoderSettings.GetOrDefault<UTF8Encoding>(MessageEncoderSettingsName.ReadEncoding, Utf8Encodings.Strict);
249  binaryReaderSettings.GuidRepresentation = _messageEncoderSettings.GetOrDefault<GuidRepresentation>(MessageEncoderSettingsName.GuidRepresentation, GuidRepresentation.CSharpLegacy);
250  };
251 
252  BsonValue clusterTime;
253  if (rawDocument.TryGetValue("$clusterTime", out clusterTime))
254  {
255  // note: we are assuming that _session is an instance of ClusterClockAdvancingClusterTime
256  // and that calling _session.AdvanceClusterTime will have the side effect of advancing the cluster's ClusterTime also
257  var materializedClusterTime = ((RawBsonDocument)clusterTime).Materialize(binaryReaderSettings);
258  _session.AdvanceClusterTime(materializedClusterTime);
259  }
260  BsonValue operationTime;
261  if (rawDocument.TryGetValue("operationTime", out operationTime))
262  {
263  _session.AdvanceOperationTime(operationTime.AsBsonTimestamp);
264  }
265 
266  if (!rawDocument.GetValue("ok", false).ToBoolean())
267  {
268  var materializedDocument = rawDocument.Materialize(binaryReaderSettings);
269 
270  var commandName = _command.GetElement(0).Name;
271  if (commandName == "$query")
272  {
273  commandName = _command["$query"].AsBsonDocument.GetElement(0).Name;
274  }
275 
276  var notPrimaryOrNodeIsRecoveringException = ExceptionMapper.MapNotPrimaryOrNodeIsRecovering(connectionId, _command, materializedDocument, "errmsg");
277  if (notPrimaryOrNodeIsRecoveringException != null)
278  {
279  throw notPrimaryOrNodeIsRecoveringException;
280  }
281 
282  string message;
283  BsonValue errmsgBsonValue;
284  if (materializedDocument.TryGetValue("errmsg", out errmsgBsonValue) && errmsgBsonValue.IsString)
285  {
286  var errmsg = errmsgBsonValue.ToString();
287  message = string.Format("Command {0} failed: {1}.", commandName, errmsg);
288  }
289  else
290  {
291  message = string.Format("Command {0} failed.", commandName);
292  }
293 
294  var mappedException = ExceptionMapper.Map(connectionId, materializedDocument);
295  if (mappedException != null)
296  {
297  throw mappedException;
298  }
299 
300  throw new MongoCommandException(connectionId, message, _command, materializedDocument);
301  }
302 
303  if (rawDocument.Contains("writeConcernError"))
304  {
305  var materializedDocument = rawDocument.Materialize(binaryReaderSettings);
306  var writeConcernError = materializedDocument["writeConcernError"].AsBsonDocument;
307  var message = writeConcernError.AsBsonDocument.GetValue("errmsg", null)?.AsString;
308  var writeConcernResult = new WriteConcernResult(materializedDocument);
309  throw new MongoWriteConcernException(connectionId, message, writeConcernResult);
310  }
311 
312  using (var stream = new ByteBufferStream(rawDocument.Slice, ownsBuffer: false))
313  {
314  var encoderFactory = new BinaryMessageEncoderFactory(stream, _messageEncoderSettings);
315  var encoder = (ReplyMessageBinaryEncoder<TCommandResult>)encoderFactory.GetReplyMessageEncoder<TCommandResult>(_resultSerializer);
316  using (var reader = encoder.CreateBinaryReader())
317  {
318  var context = BsonDeserializationContext.CreateRoot(reader);
319  return _resultSerializer.Deserialize(context);
320  }
321  }
322  }
323  }

325  private BsonDocument WrapCommandForQueryMessage(BsonDocument command, ConnectionDescription connectionDescription, out bool messageContainsSessionId)
326  {
327  messageContainsSessionId = false;
328  var extraElements = new List<BsonElement>();
329  if (_session.Id != null)
330  {
331  var areSessionsSupported = connectionDescription.IsMasterResult.LogicalSessionTimeout.HasValue;
332  if (areSessionsSupported)
333  {
334  var lsid = new BsonElement("lsid", _session.Id);
335  extraElements.Add(lsid);
336  messageContainsSessionId = true;
337  }
338  else
339  {
340  if (!_session.IsImplicit)
341  {
342  throw new MongoClientException("Sessions are not supported.");
343  }
344  }
345  }
346  if (_session.ClusterTime != null)
347  {
348  var clusterTime = new BsonElement("$clusterTime", _session.ClusterTime);
349  extraElements.Add(clusterTime);
350  }
351  Action<BsonWriterSettings> writerSettingsConfigurator = s => s.GuidRepresentation = GuidRepresentation.Unspecified;
352  var appendExtraElementsSerializer = new ElementAppendingSerializer<BsonDocument>(BsonDocumentSerializer.Instance, extraElements, writerSettingsConfigurator);
353  var commandWithExtraElements = new BsonDocumentWrapper(command, appendExtraElementsSerializer);
354 
355  BsonDocument readPreferenceDocument = null;
356  if (connectionDescription != null)
357  {
358  var serverType = connectionDescription.IsMasterResult.ServerType;
359  readPreferenceDocument = QueryHelper.CreateReadPreferenceDocument(serverType, _readPreference);
360  }
361 
362  var wrappedCommand = new BsonDocument
363  {
364  { "$query", commandWithExtraElements },
365  { "$readPreference", readPreferenceDocument, readPreferenceDocument != null }
366  };
367  if (_additionalOptions != null)
368  {
369  wrappedCommand.Merge(_additionalOptions, overwriteExistingElements: false);
370  }
371 
372  if (wrappedCommand.ElementCount == 1)
373  {
374  return wrappedCommand["$query"].AsBsonDocument;
375  }
376  else
377  {
378  return wrappedCommand;
379  }
380  }

391 
392  public override IgnoredReply Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
393  {
394  context.Reader.ReadStartDocument();
395  while (context.Reader.ReadBsonType() != BsonType.EndOfDocument)
396  {
397  context.Reader.SkipName();
398  context.Reader.SkipValue();
399  }
400  context.Reader.ReadEndDocument();
401  return IgnoredReply.Instance;
402  }
