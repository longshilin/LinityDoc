193  public void AutoBindComponent()
194  {
195  m_BindDatas.ClearArray();
196  Transform[] childs = m_Target.gameObject.GetComponentsInChildren<Transform>();
197 
198  foreach (Transform child in childs)
199  {
200  m_TempFiledNames.Clear();
201  m_TempComponentTypeNames.Clear();
202 
203  if (IsValidBind(child, m_TempFiledNames, m_TempComponentTypeNames))
204  {
205  for (int i = 0; i < m_TempFiledNames.Count; i++)
206  {
207  var tempComponentTypeName = m_TempComponentTypeNames[i];
208  if (tempComponentTypeName.Equals("GameObject"))
209  {
210  AddBindData(m_TempFiledNames[i], child);
211  }
212  else
213  {
214  Component com = child.GetComponent(tempComponentTypeName);
215  if (com == null)
216  {
217  Debug.LogError($"{child.name}上不存在{tempComponentTypeName}的组件");
218  }
219  else
220  {
221  AddBindData(m_TempFiledNames[i], child.GetComponent(m_TempComponentTypeNames[i]));
222  }
223  }
224  }

245  private bool IsValidBind(Transform target, List<string> filedNames, List<string> componentTypeNames)
246  {
247  string[] strArray = target.name.Split('_');
248 
249  if (strArray.Length == 1)
250  {
251  return false;
252  }
253 
254  string filedName = strArray[strArray.Length - 1];
255 
256  for (int i = 0; i < strArray.Length - 1; i++)
257  {
258  string str = strArray[i];
259  string comName;
260  if (m_PrefixesDict.TryGetValue(str, out comName))
261  {
262  filedNames.Add($"{str}_{filedName}");
263  componentTypeNames.Add(comName);
264  }
265  else
266  {
267  Debug.LogError($"{target.name}的命名中{str}不存在对应的组件类型，绑定失败");
268  return false;
269  }
270  }
271 
272  return true;
