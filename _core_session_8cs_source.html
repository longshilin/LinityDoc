118  public void AbortTransaction(CancellationToken cancellationToken = default(CancellationToken))
119  {
120  EnsureAbortTransactionCanBeCalled(nameof(AbortTransaction));
121 
122  try
123  {
124  if (_currentTransaction.IsEmpty)
125  {
126  return;
127  }
128 
129  try
130  {
131  var firstAttempt = CreateAbortTransactionOperation();
132  ExecuteEndTransactionOnPrimary(firstAttempt, cancellationToken);
133  return;
134  }
135  catch (Exception exception) when (ShouldRetryEndTransactionException(exception))
136  {
137  // ignore exception and retry

140  {

142  }
143 
144  try

204  public void AboutToSendCommand()
205  {
206  if (_currentTransaction != null)
207  {
208  switch (_currentTransaction.State)
209  {
210  case CoreTransactionState.Starting: // Starting changes to InProgress after the message is sent to the server
211  case CoreTransactionState.InProgress:
212  return;
213 
214  case CoreTransactionState.Aborted:
215  _currentTransaction = null;
216  break;
217 
218  case CoreTransactionState.Committed:
219  // don't set to null when retrying a commit
220  if (!_isCommitTransactionInProgress)
221  {
222  _currentTransaction = null;
223  }
224  return;
225 
226  default:
227  throw new Exception($"Unexpected transaction state: {_currentTransaction.State}.");
228  }

378  private void EnsureAbortTransactionCanBeCalled(string methodName)
379  {
380  if (_currentTransaction == null)
381  {
382  throw new InvalidOperationException($"{methodName} cannot be called when no transaction started.");
383  }
384 
385  switch (_currentTransaction.State)
386  {
387  case CoreTransactionState.Starting:
388  case CoreTransactionState.InProgress:
389  return;
390 
391  case CoreTransactionState.Aborted:
392  throw new InvalidOperationException($"Cannot call {methodName} twice.");
393 
394  case CoreTransactionState.Committed:
395  throw new InvalidOperationException($"Cannot call {methodName} after calling CommitTransaction.");
396 
397  default:
398  throw new Exception($"{methodName} called in unexpected transaction state: {_currentTransaction.State}.");
399  }
400  }
