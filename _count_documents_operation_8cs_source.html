207  private List<BsonDocument> CreatePipeline()
208  {

210  pipeline.Add(new BsonDocument("$match", _filter ?? new BsonDocument()));
211  if (_skip.HasValue)

214  }
215  if (_limit.HasValue)

217  pipeline.Add(new BsonDocument("$limit", _limit.Value));
218  }
219  pipeline.Add(new BsonDocument("$group", new BsonDocument { { "_id", BsonNull.Value }, { "n", new BsonDocument("$sum", 1) } }));
220  return pipeline;
