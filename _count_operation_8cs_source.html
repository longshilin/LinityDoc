164  internal BsonDocument CreateCommand(ConnectionDescription connectionDescription, ICoreSession session)
165  {
166  Feature.ReadConcern.ThrowIfNotSupported(connectionDescription.ServerVersion, _readConcern);
167  Feature.Collation.ThrowIfNotSupported(connectionDescription.ServerVersion, _collation);
168 
169  var readConcern = ReadConcernHelper.GetReadConcernForCommand(session, connectionDescription, _readConcern);
170  return new BsonDocument
171  {
172  { "count", _collectionNamespace.CollectionName },
173  { "query", _filter, _filter != null },
174  { "limit", () => _limit.Value, _limit.HasValue },
175  { "skip", () => _skip.Value, _skip.HasValue },
176  { "hint", _hint, _hint != null },
177  { "maxTimeMS", () => MaxTimeHelper.ToMaxTimeMS(_maxTime.Value), _maxTime.HasValue },
178  { "collation", () => _collation.ToBsonDocument(), _collation != null },
179  { "readConcern", readConcern, readConcern != null }
180  };
181  }

184  public long Execute(IReadBinding binding, CancellationToken cancellationToken)
185  {
186  Ensure.IsNotNull(binding, nameof(binding));
187  using (var channelSource = binding.GetReadChannelSource(cancellationToken))
188  using (var channel = channelSource.GetChannel(cancellationToken))
189  using (var channelBinding = new ChannelReadBinding(channelSource.Server, channel, binding.ReadPreference, binding.Session.Fork()))
190  {
191  var operation = CreateOperation(channel, channelBinding);
192  var document = operation.Execute(channelBinding, cancellationToken);
193  return document["n"].ToInt64();
194  }
195  }
