243  internal BsonDocument CreateCommand(ICoreSessionHandle session, ConnectionDescription connectionDescription)
244  {
245  var serverVersion = connectionDescription.ServerVersion;
246  Feature.Collation.ThrowIfNotSupported(serverVersion, _collation);
247 
248  var flags = GetFlags();
249  var writeConcern = WriteConcernHelper.GetWriteConcernForCommandThatWrites(session, _writeConcern, serverVersion);
250  return new BsonDocument
251  {
252  { "create", _collectionNamespace.CollectionName },
253  { "capped", () => _capped.Value, _capped.HasValue },
254  { "autoIndexId", () => _autoIndexId.Value, _autoIndexId.HasValue },
255  { "size", () => _maxSize.Value, _maxSize.HasValue },
256  { "max", () => _maxDocuments.Value, _maxDocuments.HasValue },
257  { "flags", () => (int)flags.Value, flags.HasValue },
258  { "storageEngine", () => _storageEngine, _storageEngine != null },
259  { "indexOptionDefaults", _indexOptionDefaults, _indexOptionDefaults != null },
260  { "validator", _validator, _validator != null },
261  { "validationAction", () => _validationAction.Value.ToString().ToLowerInvariant(), _validationAction.HasValue },
262  { "validationLevel", () => _validationLevel.Value.ToString().ToLowerInvariant(), _validationLevel.HasValue },
263  { "collation", () => _collation.ToBsonDocument(), _collation != null },
264  { "writeConcern", writeConcern, writeConcern != null }
265  };
266  }
