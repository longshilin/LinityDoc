324  {

336  { "expireAfterSeconds", () => _expireAfter.Value.TotalSeconds, _expireAfter.HasValue },
337  { "language_override", () => _languageOverride, _languageOverride != null },
338  { "max", () => _max.Value, _max.HasValue },
339  { "min", () => _min.Value, _min.HasValue },
340  { "partialFilterExpression", _partialFilterExpression, _partialFilterExpression != null },
341  { "sparse", () => _sparse.Value, _sparse.HasValue },
342  { "2dsphereIndexVersion", () => _sphereIndexVersion.Value, _sphereIndexVersion.HasValue },
343  { "storageEngine", () => _storageEngine, _storageEngine != null },
344  { "textIndexVersion", () => _textIndexVersion.Value, _textIndexVersion.HasValue },
345  { "unique", () => _unique.Value, _unique.HasValue },
