182  public void ReadData(string dataAssetName, int priority, object userData)
183  {
184  if (m_ResourceManager == null)
185  {
186  throw new GameFrameworkException("You must set resource manager first.");
187  }
188 
189  if (m_DataProviderHelper == null)
190  {
191  throw new GameFrameworkException("You must set data provider helper first.");
192  }
193 
194  HasAssetResult result = m_ResourceManager.HasAsset(dataAssetName);
195  switch (result)
196  {
197  case HasAssetResult.AssetOnDisk:
198  case HasAssetResult.AssetOnFileSystem:
199  m_ResourceManager.LoadAsset(dataAssetName, priority, m_LoadAssetCallbacks, userData);
200  break;
201 
202  case HasAssetResult.BinaryOnDisk:
203  m_ResourceManager.LoadBinary(dataAssetName, m_LoadBinaryCallbacks, userData);
204  break;
205 
206  case HasAssetResult.BinaryOnFileSystem:
207  int dataLength = m_ResourceManager.GetBinaryLength(dataAssetName);
208  EnsureCachedBytesSize(dataLength);
209  if (dataLength != m_ResourceManager.LoadBinaryFromFileSystem(dataAssetName, s_CachedBytes))
210  {
211  throw new GameFrameworkException(Utility.Text.Format("Load binary '{0}' from file system with internal error.", dataAssetName));
212  }
213 
214  try
215  {
216  if (!m_DataProviderHelper.ReadData(m_Owner, dataAssetName, s_CachedBytes, 0, dataLength, userData))
217  {
218  throw new GameFrameworkException(Utility.Text.Format("Load data failure in data provider helper, data asset name '{0}'.", dataAssetName));
219  }
220 
221  if (m_ReadDataSuccessEventHandler != null)
222  {
223  ReadDataSuccessEventArgs loadDataSuccessEventArgs = ReadDataSuccessEventArgs.Create(dataAssetName, 0f, userData);
224  m_ReadDataSuccessEventHandler(this, loadDataSuccessEventArgs);
225  ReferencePool.Release(loadDataSuccessEventArgs);
226  }
227  }
228  catch (Exception exception)
229  {
230  if (m_ReadDataFailureEventHandler != null)
231  {
232  ReadDataFailureEventArgs loadDataFailureEventArgs = ReadDataFailureEventArgs.Create(dataAssetName, exception.ToString(), userData);
233  m_ReadDataFailureEventHandler(this, loadDataFailureEventArgs);
234  ReferencePool.Release(loadDataFailureEventArgs);
235  return;
236  }
237 
238  throw;
239  }
240 
241  break;
242 
243  default:
244  throw new GameFrameworkException(Utility.Text.Format("Data asset '{0}' is '{1}'.", dataAssetName, result.ToString()));
245  }
246  }
