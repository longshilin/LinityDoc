52  private string GetSleepTimeoutDescription(int sleepTimeout)
53  {
54  if (sleepTimeout == SleepTimeout.NeverSleep)

56  return "Never Sleep";
57  }
58 
59  if (sleepTimeout == SleepTimeout.SystemSetting)
