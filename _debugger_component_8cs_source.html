346  private void DrawDebuggerWindowGroup(IDebuggerWindowGroup debuggerWindowGroup)
347  {
348  if (debuggerWindowGroup == null)
349  {
350  return;
351  }
352 
353  List<string> names = new List<string>();
354  string[] debuggerWindowNames = debuggerWindowGroup.GetDebuggerWindowNames();
355  for (int i = 0; i < debuggerWindowNames.Length; i++)
356  {
357  names.Add(Utility.Text.Format("<b>{0}</b>", debuggerWindowNames[i]));
358  }
359 
360  if (debuggerWindowGroup == m_DebuggerManager.DebuggerWindowRoot)
361  {
362  names.Add("<b>Close</b>");
363  }
364 
365  int toolbarIndex = GUILayout.Toolbar(debuggerWindowGroup.SelectedIndex, names.ToArray(), GUILayout.Height(30f), GUILayout.MaxWidth(Screen.width));
366  if (toolbarIndex >= debuggerWindowGroup.DebuggerWindowCount)
367  {
368  m_ShowFullWindow = false;
369  return;
370  }
371 
372  if (debuggerWindowGroup.SelectedWindow == null)
373  {
374  return;
375  }
376 
377  if (debuggerWindowGroup.SelectedIndex != toolbarIndex)
378  {
379  debuggerWindowGroup.SelectedWindow.OnLeave();
380  debuggerWindowGroup.SelectedIndex = toolbarIndex;
381  debuggerWindowGroup.SelectedWindow.OnEnter();
382  }
383 
384  IDebuggerWindowGroup subDebuggerWindowGroup = debuggerWindowGroup.SelectedWindow as IDebuggerWindowGroup;
385  if (subDebuggerWindowGroup != null)
386  {
387  DrawDebuggerWindowGroup(subDebuggerWindowGroup);
388  }
389 
390  debuggerWindowGroup.SelectedWindow.OnDraw();
391  }

393  private void DrawDebuggerWindowIcon(int windowId)
394  {
395  GUI.DragWindow(m_DragRect);
396  GUILayout.Space(5);
397  Color32 color = Color.white;
398  m_ConsoleWindow.RefreshCount();
399  if (m_ConsoleWindow.FatalCount > 0)
400  {
401  color = m_ConsoleWindow.GetLogStringColor(LogType.Exception);
402  }
403  else if (m_ConsoleWindow.ErrorCount > 0)
404  {
405  color = m_ConsoleWindow.GetLogStringColor(LogType.Error);
406  }
407  else if (m_ConsoleWindow.WarningCount > 0)
408  {
409  color = m_ConsoleWindow.GetLogStringColor(LogType.Warning);
410  }
411  else
412  {
413  color = m_ConsoleWindow.GetLogStringColor(LogType.Log);
414  }
415 
416  string title = Utility.Text.Format("<color=#{0}{1}{2}{3}><b>FPS: {4}</b></color>", color.r.ToString("x2"), color.g.ToString("x2"), color.b.ToString("x2"), color.a.ToString("x2"), m_FpsCounter.CurrentFps.ToString("F2"));
417  if (GUILayout.Button(title, GUILayout.Width(100f), GUILayout.Height(40f)))
418  {
419  m_ShowFullWindow = true;
420  }
421  }
