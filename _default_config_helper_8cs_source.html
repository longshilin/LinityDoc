84  public override bool ParseData(IConfigManager configManager, string configString, object userData)
85  {
86  try
87  {
88  int position = 0;
89  string configLineString = null;
90  while ((configLineString = configString.ReadLine(ref position)) != null)
91  {
92  if (configLineString[0] == '#')
93  {
94  continue;
95  }
96 
97  string[] splitedLine = configLineString.Split(ColumnSplitSeparator, StringSplitOptions.None);
98  if (splitedLine.Length != ColumnCount)
99  {
100  Log.Warning("Can not parse config line string '{0}' which column count is invalid.", configLineString);
101  return false;
102  }
103 
104  string configName = splitedLine[1];
105  string configValue = splitedLine[3];
106  if (!configManager.AddConfig(configName, configValue))
107  {
108  Log.Warning("Can not add config with config name '{0}' which may be invalid or duplicate.", configName);
109  return false;
110  }
111  }
112 
113  return true;
114  }
115  catch (Exception exception)
116  {
117  Log.Warning("Can not parse config string with exception '{0}'.", exception.ToString());
118  return false;
119  }
120  }

131  public override bool ParseData(IConfigManager configManager, byte[] configBytes, int startIndex, int length, object userData)
132  {

134  {
135  using (MemoryStream memoryStream = new MemoryStream(configBytes, startIndex, length, false))
136  {
137  using (BinaryReader binaryReader = new BinaryReader(memoryStream, Encoding.UTF8))
138  {
139  while (binaryReader.BaseStream.Position < binaryReader.BaseStream.Length)
140  {
141  string configName = binaryReader.ReadString();
142  string configValue = binaryReader.ReadString();
