244  public override void LoadAsset(object resource, string assetName, Type assetType, bool isScene)
245  {
246  if (m_LoadResourceAgentHelperLoadCompleteEventHandler == null || m_LoadResourceAgentHelperUpdateEventHandler == null || m_LoadResourceAgentHelperErrorEventHandler == null)
247  {
248  Log.Fatal("Load resource agent helper handler is invalid.");
249  return;
250  }
251 
252  AssetBundle assetBundle = resource as AssetBundle;
253  if (assetBundle == null)
254  {
255  LoadResourceAgentHelperErrorEventArgs loadResourceAgentHelperErrorEventArgs = LoadResourceAgentHelperErrorEventArgs.Create(LoadResourceStatus.TypeError, "Can not load asset bundle from loaded resource which is not an asset bundle.");
256  m_LoadResourceAgentHelperErrorEventHandler(this, loadResourceAgentHelperErrorEventArgs);
257  ReferencePool.Release(loadResourceAgentHelperErrorEventArgs);
258  return;
259  }
260 
261  if (string.IsNullOrEmpty(assetName))
262  {
263  LoadResourceAgentHelperErrorEventArgs loadResourceAgentHelperErrorEventArgs = LoadResourceAgentHelperErrorEventArgs.Create(LoadResourceStatus.AssetError, "Can not load asset from asset bundle which child name is invalid.");
264  m_LoadResourceAgentHelperErrorEventHandler(this, loadResourceAgentHelperErrorEventArgs);
265  ReferencePool.Release(loadResourceAgentHelperErrorEventArgs);
266  return;
267  }
268 
269  m_AssetName = assetName;
270  if (isScene)
271  {
272  int sceneNamePositionStart = assetName.LastIndexOf('/');
273  int sceneNamePositionEnd = assetName.LastIndexOf('.');
274  if (sceneNamePositionStart <= 0 || sceneNamePositionEnd <= 0 || sceneNamePositionStart > sceneNamePositionEnd)
275  {
276  LoadResourceAgentHelperErrorEventArgs loadResourceAgentHelperErrorEventArgs = LoadResourceAgentHelperErrorEventArgs.Create(LoadResourceStatus.AssetError, Utility.Text.Format("Scene name '{0}' is invalid.", assetName));
277  m_LoadResourceAgentHelperErrorEventHandler(this, loadResourceAgentHelperErrorEventArgs);
278  ReferencePool.Release(loadResourceAgentHelperErrorEventArgs);
279  return;
280  }
281 
282  string sceneName = assetName.Substring(sceneNamePositionStart + 1, sceneNamePositionEnd - sceneNamePositionStart - 1);
283  m_AsyncOperation = SceneManager.LoadSceneAsync(sceneName, LoadSceneMode.Additive);
284  }
285  else
286  {
287  if (assetType != null)
288  {
289  m_AssetBundleRequest = assetBundle.LoadAssetAsync(assetName, assetType);
290  }
291  else
292  {
293  m_AssetBundleRequest = assetBundle.LoadAssetAsync(assetName);
294  }
295  }
296  }
