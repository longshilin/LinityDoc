141  public override bool ParseData(ILocalizationManager localizationManager, string dictionaryString, object userData)
142  {
143  try
144  {
145  int position = 0;
146  string dictionaryLineString = null;
147  while ((dictionaryLineString = dictionaryString.ReadLine(ref position)) != null)
148  {
149  if (dictionaryLineString[0] == '#')
150  {
151  continue;
152  }
153 
154  string[] splitedLine = dictionaryLineString.Split(ColumnSplitSeparator, StringSplitOptions.None);
155  if (splitedLine.Length != ColumnCount)
156  {
157  Log.Warning("Can not parse dictionary line string '{0}' which column count is invalid.", dictionaryLineString);
158  return false;
159  }
160 
161  string dictionaryKey = splitedLine[1];
162  string dictionaryValue = splitedLine[3];
163  if (!localizationManager.AddRawString(dictionaryKey, dictionaryValue))
164  {
165  Log.Warning("Can not add raw string with dictionary key '{0}' which may be invalid or duplicate.", dictionaryKey);
166  return false;
167  }
168  }
169 
170  return true;
171  }
172  catch (Exception exception)
173  {
174  Log.Warning("Can not parse dictionary string with exception '{0}'.", exception.ToString());
175  return false;
176  }
177  }

188  public override bool ParseData(ILocalizationManager localizationManager, byte[] dictionaryBytes, int startIndex, int length, object userData)
189  {
190  try
191  {
192  using (MemoryStream memoryStream = new MemoryStream(dictionaryBytes, startIndex, length, false))
193  {
194  using (BinaryReader binaryReader = new BinaryReader(memoryStream, Encoding.UTF8))
195  {
196  while (binaryReader.BaseStream.Position < binaryReader.BaseStream.Length)
197  {
198  string dictionaryKey = binaryReader.ReadString();
199  string dictionaryValue = binaryReader.ReadString();
200  if (!localizationManager.AddRawString(dictionaryKey, dictionaryValue))
201  {
202  Log.Warning("Can not add raw string with dictionary key '{0}' which may be invalid or duplicate.", dictionaryKey);
203  return false;
204  }
205  }
206  }
207  }
208 
209  return true;
210  }
211  catch (Exception exception)
212  {
213  Log.Warning("Can not parse dictionary bytes with exception '{0}'.", exception.ToString());
214  return false;
215  }
216  }
