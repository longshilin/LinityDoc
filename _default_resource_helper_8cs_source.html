111  private IEnumerator LoadBytesCo(string fileUri, LoadBytesCallbacks loadBytesCallbacks, object userData)
112  {
113  bool isError = false;
114  byte[] bytes = null;
115  string errorMessage = null;
116  DateTime startTime = DateTime.UtcNow;
117 
118 #if UNITY_5_4_OR_NEWER
119  UnityWebRequest unityWebRequest = UnityWebRequest.Get(fileUri);
120 #if UNITY_2017_2_OR_NEWER
121  yield return unityWebRequest.SendWebRequest();
122 #else
123  yield return unityWebRequest.Send();
124 #endif
125 
126 #if UNITY_2020_2_OR_NEWER
127  isError = unityWebRequest.result != UnityWebRequest.Result.Success;
128 #elif UNITY_2017_1_OR_NEWER
129  isError = unityWebRequest.isNetworkError || unityWebRequest.isHttpError;
130 #else
131  isError = unityWebRequest.isError;
132 #endif
133  bytes = unityWebRequest.downloadHandler.data;
134  errorMessage = isError ? unityWebRequest.error : null;
135  unityWebRequest.Dispose();
136 #else
137  WWW www = new WWW(fileUri);
138  yield return www;
139 
140  isError = !string.IsNullOrEmpty(www.error);
141  bytes = www.bytes;
142  errorMessage = www.error;
143  www.Dispose();
144 #endif
145 
146  if (!isError)
147  {
148  float elapseSeconds = (float)(DateTime.UtcNow - startTime).TotalSeconds;
149  loadBytesCallbacks.LoadBytesSuccessCallback(fileUri, bytes, elapseSeconds, userData);
150  }
151  else if (loadBytesCallbacks.LoadBytesFailureCallback != null)
152  {
153  loadBytesCallbacks.LoadBytesFailureCallback(fileUri, errorMessage, userData);
154  }
155  }
