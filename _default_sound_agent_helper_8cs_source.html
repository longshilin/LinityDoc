420  private IEnumerator FadeToVolume(AudioSource audioSource, float volume, float duration)
421  {
422  float time = 0f;
423  float originalVolume = audioSource.volume;
424  while (time < duration)
425  {
426  time += UnityEngine.Time.deltaTime;
427  audioSource.volume = Mathf.Lerp(originalVolume, volume, time / duration);
428  yield return new WaitForEndOfFrame();
429  }
430 
431  audioSource.volume = volume;
432  }
