92  public void WriteMessage(DeleteMessage message)
93  {

95 
96  var binaryWriter = CreateBinaryWriter();
97  var stream = binaryWriter.BsonStream;
98  var startPosition = stream.Position;
99 
100  stream.WriteInt32(0); // messageSize
101  stream.WriteInt32(message.RequestId);
102  stream.WriteInt32(0); // responseTo
103  stream.WriteInt32((int)Opcode.Delete);
104  stream.WriteInt32(0); // reserved
105  stream.WriteCString(message.CollectionNamespace.FullName);
106  stream.WriteInt32((int)BuildDeleteFlags(message));
107  var context = BsonSerializationContext.CreateRoot(binaryWriter);
108  BsonDocumentSerializer.Instance.Serialize(context, message.Query ?? new BsonDocument());
109  stream.BackpatchSize(startPosition);
