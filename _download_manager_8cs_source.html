394  public int AddDownload(string downloadPath, string downloadUri, string tag, int priority, object userData)
395  {
396  if (string.IsNullOrEmpty(downloadPath))
397  {
398  throw new GameFrameworkException("Download path is invalid.");
399  }
400 
401  if (string.IsNullOrEmpty(downloadUri))
402  {
403  throw new GameFrameworkException("Download uri is invalid.");
404  }
405 
406  if (TotalAgentCount <= 0)
407  {
408  throw new GameFrameworkException("You must add download agent first.");
409  }
410 
411  DownloadTask downloadTask = DownloadTask.Create(downloadPath, downloadUri, tag, priority, m_FlushSize, m_Timeout, userData);
412  m_TaskPool.AddTask(downloadTask);
413  return downloadTask.SerialId;
