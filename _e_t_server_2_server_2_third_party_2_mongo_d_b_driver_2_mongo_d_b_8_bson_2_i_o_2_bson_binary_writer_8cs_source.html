157 #pragma warning disable 618 // about obsolete BsonBinarySubType.OldBinary
162  public override void WriteBinaryData(BsonBinaryData binaryData)
163  {
164  if (Disposed) { throw new ObjectDisposedException("BsonBinaryWriter"); }
165  if (State != BsonWriterState.Value)
166  {
167  ThrowInvalidState("WriteBinaryData", BsonWriterState.Value);
168  }
169 
170  var bytes = binaryData.Bytes;
171  var subType = binaryData.SubType;
172  var guidRepresentation = binaryData.GuidRepresentation;
173  switch (subType)
174  {
175  case BsonBinarySubType.OldBinary:
176  if (Settings.FixOldBinarySubTypeOnOutput)
177  {
178  subType = BsonBinarySubType.Binary; // replace obsolete OldBinary with new Binary sub type
179  }
180  break;
181  case BsonBinarySubType.UuidLegacy:
182  case BsonBinarySubType.UuidStandard:
183  if (Settings.GuidRepresentation != GuidRepresentation.Unspecified)
184  {
185  var expectedSubType = (Settings.GuidRepresentation == GuidRepresentation.Standard) ? BsonBinarySubType.UuidStandard : BsonBinarySubType.UuidLegacy;
186  if (subType != expectedSubType)
187  {
188  var message = string.Format(
189  "The GuidRepresentation for the writer is {0}, which requires the subType argument to be {1}, not {2}.",
190  Settings.GuidRepresentation, expectedSubType, subType);
191  throw new BsonSerializationException(message);
192  }
193  if (guidRepresentation != Settings.GuidRepresentation)
194  {
195  var message = string.Format(
196  "The GuidRepresentation for the writer is {0}, which requires the the guidRepresentation argument to also be {0}, not {1}.",
197  Settings.GuidRepresentation, guidRepresentation);
198  throw new BsonSerializationException(message);
199  }
200  }
201  break;
202  }
203 
204  _bsonStream.WriteBsonType(BsonType.Binary);
205  WriteNameHelper();
206  if (subType == BsonBinarySubType.OldBinary)
207  {
208  // sub type OldBinary has two sizes (for historical reasons)
209  _bsonStream.WriteInt32(bytes.Length + 4);
210  _bsonStream.WriteBinarySubType(subType);
211  _bsonStream.WriteInt32(bytes.Length);
212  }
213  else
214  {
215  _bsonStream.WriteInt32(bytes.Length);
216  _bsonStream.WriteBinarySubType(subType);
217  }
218  _bsonStream.WriteBytes(bytes, 0, bytes.Length);
219 
220  State = GetNextState();
221  }

734  private void BackpatchSize()
735  {
736  var size = _bsonStream.Position - _context.StartPosition;
737  if (size > Settings.MaxDocumentSize)
738  {
739  var message = string.Format("Size {0} is larger than MaxDocumentSize {1}.", size, Settings.MaxDocumentSize);
740  throw new FormatException(message);
741  }
742 
743  var currentPosition = _bsonStream.Position;
744  _bsonStream.Position = _context.StartPosition;
745  _bsonStream.WriteInt32((int)size);
746  _bsonStream.Position = currentPosition;
747  }
