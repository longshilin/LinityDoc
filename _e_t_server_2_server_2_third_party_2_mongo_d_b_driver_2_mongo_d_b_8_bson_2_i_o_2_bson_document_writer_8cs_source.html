417  public override void WriteStartDocument()
418  {
419  if (Disposed) { throw new ObjectDisposedException("BsonDocumentWriter"); }
420  if (State != BsonWriterState.Initial && State != BsonWriterState.Value && State != BsonWriterState.ScopeDocument && State != BsonWriterState.Done)
421  {
422  ThrowInvalidState("WriteStartDocument", BsonWriterState.Initial, BsonWriterState.Value, BsonWriterState.ScopeDocument, BsonWriterState.Done);
423  }
424 
425  base.WriteStartDocument();
426  switch (State)
427  {
428  case BsonWriterState.Initial:
429  case BsonWriterState.Done:
430  _context = new BsonDocumentWriterContext(null, ContextType.Document, _document);
431  break;
432  case BsonWriterState.Value:
433  _context = new BsonDocumentWriterContext(_context, ContextType.Document, new BsonDocument());
434  break;
435  case BsonWriterState.ScopeDocument:
436  _context = new BsonDocumentWriterContext(_context, ContextType.ScopeDocument, new BsonDocument());
437  break;
438  default:
439  throw new BsonInternalException("Unexpected state.");
440  }
441 
442  State = BsonWriterState.Name;
443  }
