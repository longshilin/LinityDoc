262  public virtual IByteBuffer ReadRawBsonArray()
263  {
264  // overridden in BsonBinaryReader to read the raw bytes from the stream
265  // for all other streams, deserialize the array and reserialize it using a BsonBinaryWriter to get the raw bytes
266 
267  var deserializationContext = BsonDeserializationContext.CreateRoot(this);
268  var array = BsonArraySerializer.Instance.Deserialize(deserializationContext);
269 
270  using (var memoryStream = new MemoryStream())
271  using (var bsonWriter = new BsonBinaryWriter(memoryStream, BsonBinaryWriterSettings.Defaults))
272  {
273  var serializationContext = BsonSerializationContext.CreateRoot(bsonWriter);
274  bsonWriter.WriteStartDocument();
275  var startPosition = memoryStream.Position + 3; // just past BsonType, "x" and null byte
276  bsonWriter.WriteName("x");
277  BsonArraySerializer.Instance.Serialize(serializationContext, array);
278  var endPosition = memoryStream.Position;
279  bsonWriter.WriteEndDocument();
280 
281  byte[] memoryStreamBuffer;
282 #if NETSTANDARD1_5 || NETSTANDARD1_6
283  memoryStreamBuffer = memoryStream.ToArray();
284 #else
285  memoryStreamBuffer = memoryStream.GetBuffer();
286 #endif
287  var buffer = new ByteArrayBuffer(memoryStreamBuffer, (int)memoryStream.Length, isReadOnly: true);
288  return new ByteBufferSlice(buffer, (int)startPosition, (int)(endPosition - startPosition));
289  }
290  }

416  protected void VerifyBsonType(string methodName, BsonType requiredBsonType)
417  {
418  if (_state == BsonReaderState.Initial || _state == BsonReaderState.ScopeDocument || _state == BsonReaderState.Type)
419  {
420  ReadBsonType();
421  }
422  if (_state == BsonReaderState.Name)
423  {
424  // ignore name
425  SkipName();
426  }
427  if (_state != BsonReaderState.Value)
428  {
429  ThrowInvalidState(methodName, BsonReaderState.Value);
430  }
431  if (_currentBsonType != requiredBsonType)
432  {
433  var message = string.Format(
434  "{0} can only be called when CurrentBsonType is {1}, not when CurrentBsonType is {2}.",
435  methodName, requiredBsonType, _currentBsonType);
436  throw new InvalidOperationException(message);
