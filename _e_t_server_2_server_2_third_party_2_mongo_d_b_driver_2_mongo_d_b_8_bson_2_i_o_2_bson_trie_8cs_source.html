231  public BsonTrieNode<TValue> GetChild(byte keyByte)
232  {
233  if (_onlyChild != null)
234  {
235  // optimization for nodes that have only one child
236  if (_onlyChild._keyByte == keyByte)
237  {
238  return _onlyChild;
239  }
240  }
241  else if (_children != null)
242  {
243  var index = (uint)((int)keyByte - _minChildKeyByte);
244  if (index < _childrenIndexes.Length)
245  {
246  index = _childrenIndexes[index];
247  if (index < _children.Length)
248  {
249  return _children[index];
250  }
251  }
252  }
253  return null;
254  }

257  internal void AddChild(BsonTrieNode<TValue> child)
258  {
259  if (GetChild(child._keyByte) != null)
260  {
261  throw new ArgumentException("BsonTrieNode already contains a child with the same keyByte.");
262  }
263 
264  if (_children != null)
265  {
266  // add a new child to the existing _children
267  var children = new BsonTrieNode<TValue>[_children.Length + 1];
268  Array.Copy(_children, children, _children.Length);
269  children[children.Length - 1] = child;
270 
271  var childrenIndexes = _childrenIndexes;
272  var minChildKeyByte = _minChildKeyByte;
273  var maxChildKeyByte = _minChildKeyByte + _childrenIndexes.Length - 1;
274 
275  // if new keyByte doesn't fall within existing min/max range expand the range
276  if (child._keyByte < minChildKeyByte)
277  {
278  // grow the indexes on the min side
279  minChildKeyByte = child._keyByte;
280  childrenIndexes = new byte[maxChildKeyByte - minChildKeyByte + 1];
281  var sizeDelta = childrenIndexes.Length - _childrenIndexes.Length;
282  for (var i = 0; i < sizeDelta; i++)
283  {
284  childrenIndexes[i] = 255;
285  }
286  Array.Copy(_childrenIndexes, 0, childrenIndexes, sizeDelta, _childrenIndexes.Length);
287  }
288  else if (child._keyByte > maxChildKeyByte)
289  {
290  // grow the indexes on the max side
291  maxChildKeyByte = child._keyByte;
292  childrenIndexes = new byte[maxChildKeyByte - minChildKeyByte + 1];
293  Array.Copy(_childrenIndexes, 0, childrenIndexes, 0, _childrenIndexes.Length);
294  for (var i = _childrenIndexes.Length; i < childrenIndexes.Length; i++)
295  {
296  childrenIndexes[i] = 255;
297  }
298  }
299  childrenIndexes[child._keyByte - minChildKeyByte] = (byte)(children.Length - 1);
300 
301  _children = children;
302  _childrenIndexes = childrenIndexes;
303  _minChildKeyByte = minChildKeyByte;
304  }
305  else if (_onlyChild != null)
306  {
307  // switch from having an _onlyChild to having two _children
308  var children = new BsonTrieNode<TValue>[2];
309  children[0] = _onlyChild;
310  children[1] = child;
311 
312  var minChildKeyByte = _onlyChild._keyByte;
313  var maxChildKeyByte = child._keyByte;
314  if (minChildKeyByte > maxChildKeyByte)
315  {
316  minChildKeyByte = child._keyByte;
317  maxChildKeyByte = _onlyChild._keyByte;
318  }
319 
320  var childrenIndexes = new byte[maxChildKeyByte - minChildKeyByte + 1];
321  for (var i = 0; i < childrenIndexes.Length; i++)
322  {
323  childrenIndexes[i] = 255;
324  }
325  childrenIndexes[_onlyChild._keyByte - minChildKeyByte] = 0;
326  childrenIndexes[child._keyByte - minChildKeyByte] = 1;
327 
328  _onlyChild = null;
329  _children = children;
330  _childrenIndexes = childrenIndexes;
331  _minChildKeyByte = minChildKeyByte;
332  }
333  else
334  {
335  _onlyChild = child;
336  }
