188  public override BsonType ReadBsonType()
189  {
190  if (Disposed) { ThrowObjectDisposedException(); }
191  if (State == BsonReaderState.Initial || State == BsonReaderState.ScopeDocument)
192  {
193  if (State == BsonReaderState.Initial)
194  {
195  _buffer.ResetBuffer();
196  }
197 
198  // in JSON the top level value can be of any type so fall through
199  State = BsonReaderState.Type;
200  }
201  if (State != BsonReaderState.Type)
202  {
203  ThrowInvalidState("ReadBsonType", BsonReaderState.Type);
204  }
205 
206  if (_context.ContextType == ContextType.Document)
207  {
208  var nameToken = PopToken();
209  switch (nameToken.Type)
210  {
211  case JsonTokenType.String:
212  case JsonTokenType.UnquotedString:
213  CurrentName = nameToken.StringValue;
214  break;
215  case JsonTokenType.EndObject:
216  State = BsonReaderState.EndOfDocument;
217  return BsonType.EndOfDocument;
218  default:
219  var message = string.Format("JSON reader was expecting a name but found '{0}'.", nameToken.Lexeme);
220  throw new FormatException(message);
221  }
222 
223  var colonToken = PopToken();
224  if (colonToken.Type != JsonTokenType.Colon)
225  {
226  var message = string.Format("JSON reader was expecting ':' but found '{0}'.", colonToken.Lexeme);
227  throw new FormatException(message);
228  }
229  }
230 
231  var valueToken = PopToken();
232  if (_context.ContextType == ContextType.Array && valueToken.Type == JsonTokenType.EndArray)
233  {
234  State = BsonReaderState.EndOfArray;
235  return BsonType.EndOfDocument;
236  }
237 
238  var noValueFound = false;
239  switch (valueToken.Type)
240  {
241  case JsonTokenType.BeginArray:
242  CurrentBsonType = BsonType.Array;
243  break;
244  case JsonTokenType.BeginObject:
245  CurrentBsonType = ParseExtendedJson();
246  break;
247  case JsonTokenType.DateTime:
248  CurrentBsonType = BsonType.DateTime;
249  _currentValue = valueToken.DateTimeValue;
250  break;
251  case JsonTokenType.Double:
252  CurrentBsonType = BsonType.Double;
253  _currentValue = valueToken.DoubleValue;
254  break;
255  case JsonTokenType.EndOfFile:
256  CurrentBsonType = BsonType.EndOfDocument;
257  break;
258  case JsonTokenType.Int32:
259  CurrentBsonType = BsonType.Int32;
260  _currentValue = valueToken.Int32Value;
261  break;
262  case JsonTokenType.Int64:
263  CurrentBsonType = BsonType.Int64;
264  _currentValue = valueToken.Int64Value;
265  break;
266  case JsonTokenType.ObjectId:
267  CurrentBsonType = BsonType.ObjectId;
268  _currentValue = valueToken.ObjectIdValue;
269  break;
270  case JsonTokenType.RegularExpression:
271  CurrentBsonType = BsonType.RegularExpression;
272  _currentValue = valueToken.RegularExpressionValue;
273  break;
274  case JsonTokenType.String:
275  CurrentBsonType = BsonType.String;
276  _currentValue = valueToken.StringValue;
277  break;
278  case JsonTokenType.UnquotedString:
279  switch (valueToken.Lexeme)
280  {
281  case "false":
282  case "true":
283  CurrentBsonType = BsonType.Boolean;
284  _currentValue = JsonConvert.ToBoolean(valueToken.Lexeme);
285  break;
286  case "Infinity":
287  CurrentBsonType = BsonType.Double;
288  _currentValue = double.PositiveInfinity;
289  break;
290  case "NaN":
291  CurrentBsonType = BsonType.Double;
292  _currentValue = double.NaN;
293  break;
294  case "null":
295  CurrentBsonType = BsonType.Null;
296  break;
297  case "undefined":
298  CurrentBsonType = BsonType.Undefined;
299  break;
300  case "BinData":
301  CurrentBsonType = BsonType.Binary;
302  _currentValue = ParseBinDataConstructor();
303  break;
304  case "Date":
305  CurrentBsonType = BsonType.String;
306  _currentValue = ParseDateTimeConstructor(false); // withNew = false
307  break;
308  case "HexData":
309  CurrentBsonType = BsonType.Binary;
310  _currentValue = ParseHexDataConstructor();
311  break;
312  case "ISODate":
313  CurrentBsonType = BsonType.DateTime;
314  _currentValue = ParseISODateTimeConstructor();
315  break;
316  case "MaxKey":
317  CurrentBsonType = BsonType.MaxKey;
318  _currentValue = BsonMaxKey.Value;
319  break;
320  case "MinKey":
321  CurrentBsonType = BsonType.MinKey;
322  _currentValue = BsonMinKey.Value;
323  break;
324  case "NumberDecimal":
325  CurrentBsonType = BsonType.Decimal128;
326  _currentValue = ParseNumberDecimalConstructor();
327  break;
328  case "Number":
329  case "NumberInt":
330  CurrentBsonType = BsonType.Int32;
331  _currentValue = ParseNumberConstructor();
332  break;
333  case "NumberLong":
334  CurrentBsonType = BsonType.Int64;
335  _currentValue = ParseNumberLongConstructor();
336  break;
337  case "ObjectId":
338  CurrentBsonType = BsonType.ObjectId;
339  _currentValue = ParseObjectIdConstructor();
340  break;
341  case "RegExp":
342  CurrentBsonType = BsonType.RegularExpression;
343  _currentValue = ParseRegularExpressionConstructor();
344  break;
345  case "Timestamp":
346  CurrentBsonType = BsonType.Timestamp;
347  _currentValue = ParseTimestampConstructor();
348  break;
349  case "UUID":
350  case "GUID":
351  case "CSUUID":
352  case "CSGUID":
353  case "JUUID":
354  case "JGUID":
355  case "PYUUID":
356  case "PYGUID":
357  CurrentBsonType = BsonType.Binary;
358  _currentValue = ParseUUIDConstructor(valueToken.Lexeme);
359  break;
360  case "new":
361  CurrentBsonType = ParseNew(out _currentValue);
362  break;
363  default:
364  noValueFound = true;
365  break;
366  }
367  break;
368  default:
369  noValueFound = true;
370  break;
371  }
372  if (noValueFound)
373  {
374  var message = string.Format("JSON reader was expecting a value but found '{0}'.", valueToken.Lexeme);
375  throw new FormatException(message);
376  }
377  _currentToken = valueToken;
378 
379  if (_context.ContextType == ContextType.Array || _context.ContextType == ContextType.Document)
380  {
381  var commaToken = PopToken();
382  if (commaToken.Type != JsonTokenType.Comma)
383  {
384  PushToken(commaToken);
385  }
386  }
387 
388  switch (_context.ContextType)
389  {
390  case ContextType.Document:
391  case ContextType.ScopeDocument:
392  default:
393  State = BsonReaderState.Name;
394  break;
395  case ContextType.Array:
396  case ContextType.JavaScriptWithScope:
397  case ContextType.TopLevel:
398  State = BsonReaderState.Value;
399  break;
400  }
401  return CurrentBsonType;
402  }

464  {
465  if (Disposed) { ThrowObjectDisposedException(); }
466  if (_context.ContextType != ContextType.Array)

486  }

502  {
503  if (Disposed) { ThrowObjectDisposedException(); }
504  if (_context.ContextType != ContextType.Document && _context.ContextType != ContextType.ScopeDocument)
505  {
506  ThrowInvalidContextType("ReadEndDocument", _context.ContextType, ContextType.Document, ContextType.ScopeDocument);
507  }
508  if (State == BsonReaderState.Type)
509  {
510  ReadBsonType(); // will set state to EndOfDocument if at end of document
511  }
512  if (State != BsonReaderState.EndOfDocument)
513  {
514  ThrowInvalidState("ReadEndDocument", BsonReaderState.EndOfDocument);

524  {

526  case ContextType.Document: State = BsonReaderState.Type; break;
527  case ContextType.TopLevel: State = BsonReaderState.Initial; break;
528  default: throw new BsonInternalException("Unexpected ContextType");
529  }

531  if (_context.ContextType == ContextType.Array || _context.ContextType == ContextType.Document)
532  {
533  var commaToken = PopToken();
534  if (commaToken.Type != JsonTokenType.Comma)
535  {
536  PushToken(commaToken);

810  SkipName();

893  private string FormatJavaScriptDateTimeString(DateTime dateTime)
894  {
895  var utc = BsonUtils.ToUniversalTime(dateTime);
896  var local = BsonUtils.ToLocalTime(utc);
897  var offset = local - utc;
898  var offsetSign = "+";
899  if (offset < TimeSpan.Zero)
