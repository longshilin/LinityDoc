65  {
66  if (chunks == null)
67  {
68  throw new ArgumentNullException("chunks");
69  }

71 
72  var capacity = 0;

74  foreach (var chunk in materializedList)

147  public ArraySegment<byte> AccessBackingBytes(int position)
148  {
149  ThrowIfDisposed();
150  if (position < 0 || position > _length)
151  {
152  throw new ArgumentOutOfRangeException("position");
153  }
154 
155  var chunkIndex = GetChunkIndex(position);
156  if (chunkIndex < _chunks.Count)
157  {
158  var segment = _chunks[chunkIndex].Bytes;
159  var chunkOffset = position - _positions[chunkIndex];
160  var chunkRemaining = segment.Count - chunkOffset;
161  return new ArraySegment<byte>(segment.Array, segment.Offset + chunkOffset, chunkRemaining);
162  }
163  else
164  {
165  if (_chunks.Count > 0)
166  {
167  var segment = _chunks[chunkIndex - 1].Bytes;
168  return new ArraySegment<byte>(segment.Array, segment.Offset + segment.Count, 0);
169  }
170  else
171  {

288  public IByteBuffer GetSlice(int position, int length)
289  {
290  ThrowIfDisposed();
291  if (position < 0 || position > _length)
292  {
293  throw new ArgumentOutOfRangeException("position");
294  }
295  if (length < 0 || position + length > _length)
296  {
297  throw new ArgumentOutOfRangeException("length");
298  }
299  EnsureIsReadOnly();
300 
301  if (length == 0)
302  {
303  return new ByteArrayBuffer(new byte[0]);
304  }
305 
306  var firstChunkIndex = GetChunkIndex(position);
307  var lastChunkIndex = GetChunkIndex(position + length - 1);
308 
309  IByteBuffer forkedBuffer;
310  if (firstChunkIndex == lastChunkIndex)
311  {
312  var forkedChunk = _chunks[firstChunkIndex].Fork();
313  forkedBuffer = new SingleChunkBuffer(forkedChunk, forkedChunk.Bytes.Count, isReadOnly: true);
314  }
315  else
316  {
317  var forkedChunks = _chunks.Skip(firstChunkIndex).Take(lastChunkIndex - firstChunkIndex + 1).Select(c => c.Fork());
318  var forkedBufferLength = _positions[lastChunkIndex + 1] - _positions[firstChunkIndex];
319  forkedBuffer = new MultiChunkBuffer(forkedChunks, forkedBufferLength, isReadOnly: true);
320  }
321 
322  var offset = position - _positions[firstChunkIndex];
323  return new ByteBufferSlice(forkedBuffer, offset, length);
324  }

403  private void ExpandCapacity(int minimumCapacity)
404  {
405  if (_chunkSource == null)
406  {
407  throw new InvalidOperationException("Capacity cannot be expanded because this buffer was created without specifying a chunk source.");
408  }
409 
410  while (_capacity < minimumCapacity)
411  {
412  var chunk = _chunkSource.GetChunk(minimumCapacity);
413  _chunks.Add(chunk);
414  var newCapacity = (long)_capacity + (long)chunk.Bytes.Count;
415  if (newCapacity > int.MaxValue)
416  {
417  throw new InvalidOperationException("Capacity is limited to 2GB.");
418  }
419  _capacity = (int)newCapacity;
420  _positions.Add(_capacity);
421  }

424  private int GetChunkIndex(int position)
425  {
426  // locality of reference means this loop will only execute once most of the time
427  while (true)
428  {
429  if (_chunkIndex + 1 < _positions.Count && position >= _positions[_chunkIndex + 1])
430  {
431  _chunkIndex++;
432  }
433  else if (position < _positions[_chunkIndex])
434  {
435  _chunkIndex--;
436  }
437  else
438  {
439  return _chunkIndex;
440  }
441  }
