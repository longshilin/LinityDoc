255  // public indexers
256  // note: the return type of the indexers is BsonValue and NOT BsonElement so that we can write code like:
257  // BsonDocument car;
258  // car["color"] = "red"; // changes value of existing element or adds new element
259  // note: we are using implicit conversion from string to BsonValue
260  // to convert the returned BsonValue to a .NET type you have two approaches (explicit cast or As method):
261  // string color = (string) car["color"]; // throws exception if value is not a string (returns null if not found)
262  // string color = car["color"].AsString; // throws exception if value is not a string (results in a NullReferenceException if not found)
263  // string color = car["color", "none"].AsString; // throws exception if value is not a string (default to "none" if not found)
264  // the second approach offers a more fluent interface (with fewer parenthesis!)
265  // string name = car["brand"].AsBsonSymbol.Name;
266  // string name = ((BsonSymbol) car["brand"]).Name; // the extra parenthesis are required and harder to read
267  // there are also some conversion methods (and note that ToBoolean uses the JavaScript definition of truthiness)
268  // bool ok = result["ok"].ToBoolean(); // works whether ok is false, true, 0, 0.0, 1, 1.0, "", "xyz", BsonNull.Value, etc...
269  // bool ok = result["ok", false].ToBoolean(); // defaults to false if ok element is not found
270  // int n = result["n"].ToInt32(); // works whether n is Int32, Int64, Double or String (if it can be parsed)
271  // long n = result["n"].ToInt64(); // works whether n is Int32, Int64, Double or String (if it can be parsed)
272  // double d = result["n"].ToDouble(); // works whether d is Int32, Int64, Double or String (if it can be parsed)
273  // to work in terms of BsonElements use Add, GetElement and SetElement
274  // car.Add(new BsonElement("color", "red")); // might throw exception if allowDuplicateNames is false
275  // car.SetElement(new BsonElement("color", "red")); // replaces existing element or adds new element
276  // BsonElement colorElement = car.GetElement("color"); // returns null if element "color" is not found

415  public virtual BsonDocument Add(BsonElement element)
416  {
417  var isDuplicate = IndexOfName(element.Name) != -1;
418  if (isDuplicate && !_allowDuplicateNames)
419  {
420  var message = string.Format("Duplicate element name '{0}'.", element.Name);
421  throw new InvalidOperationException(message);
422  }
423  else
424  {
425  _elements.Add(element);
426  if (!isDuplicate)
427  {
428  if (_indexes == null)
429  {
430  RebuildIndexes();
431  }
432  else
433  {
434  _indexes.Add(element.Name, _elements.Count - 1); // index of the newly added element
435  }
436  }
437  }
438 
439  return this;
440  }

752  public virtual int CompareTo(BsonDocument rhs)
753  {
754  if (rhs == null) { return 1; }
755 
756  // lhs and rhs might be subclasses of BsonDocument
757  using (var lhsEnumerator = Elements.GetEnumerator())
758  using (var rhsEnumerator = rhs.Elements.GetEnumerator())
759  {
760  while (true)
761  {
762  var lhsHasNext = lhsEnumerator.MoveNext();
763  var rhsHasNext = rhsEnumerator.MoveNext();
764  if (!lhsHasNext && !rhsHasNext) { return 0; }
765  if (!lhsHasNext) { return -1; }
766  if (!rhsHasNext) { return 1; }
767 
768  var lhsElement = lhsEnumerator.Current;
769  var rhsElement = rhsEnumerator.Current;
770  var result = lhsElement.Name.CompareTo(rhsElement.Name);
771  if (result != 0) { return result; }
772  result = lhsElement.Value.CompareTo(rhsElement.Value);
773  if (result != 0) { return result; }
774  }
775  }
776  }
