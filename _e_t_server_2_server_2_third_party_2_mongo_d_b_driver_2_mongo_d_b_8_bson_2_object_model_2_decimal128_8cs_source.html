715  public static decimal ToDecimal(Decimal128 d)
716  {
717  if (Flags.IsFirstForm(d._highBits))
718  {
719  if (Decimal128.IsZero(d))
720  {
721  return decimal.Zero;
722  }
723  else if (Decimal128.Compare(d, __minDecimalValue) < 0 || Decimal128.Compare(d, __maxDecimalValue) > 0)
724  {
725  throw new OverflowException("Value is too large or too small to be converted to a Decimal.");
726  }
727 
728  var isNegative = Decimal128.IsNegative(d);
729  var exponent = Decimal128.GetExponent(d);
730  var significand = Decimal128.GetSignificand(d);
731 
732  // decimal significand must fit in 96 bits
733  while ((significand.High >> 32) != 0)
734  {
735  uint remainder; // ignored
736  significand = UInt128.Divide(significand, 10, out remainder);
737  exponent += 1;
738  }
739 
740  // decimal exponents must be between 0 and -28
741  if (exponent > 0)
742  {
743  // bring exponent within range
744  while (exponent > 0)
745  {
746  significand = UInt128.Multiply(significand, (uint)10);
747  exponent -= 1;
748  }
749  }
750  else if (exponent < -28)
751  {
752  // check if exponent is too far out of range to possibly be brought within range
753  if (exponent < -56)
754  {
755  return decimal.Zero;
756  }
757 
758  // bring exponent within range
759  while (exponent < -28)
760  {
761  uint remainder; // ignored
762  significand = UInt128.Divide(significand, (uint)10, out remainder);
763  exponent += 1;
764  }
765 
766  if (significand.Equals(UInt128.Zero))
767  {
768  return decimal.Zero;
769  }
770  }
771 
772  var lo = (int)significand.Low;
773  var mid = (int)(significand.Low >> 32);
774  var hi = (int)significand.High;
775  var scale = (byte)(-exponent);
776 
777  return new decimal(lo, mid, hi, isNegative, scale);
778  }
779  else if (Flags.IsSecondForm(d._highBits))
780  {
781  return Decimal.Zero;
782  }
783  else
784  {
785  throw new OverflowException("Infinity or NaN cannot be converted to Decimal.");
786  }
787  }

1070  public static bool TryParse(string s, out Decimal128 result)
1071  {
1072  if (s == null || s.Length == 0)
1073  {
1074  result = default(Decimal128);
1075  return false;
1076  }
1077 
1078  const string pattern =
1079  @"^(?<sign>[+-])?" +
1080  @"(?<significand>\d+([.]\d*)?|[.]\d+)" +
1081  @"(?<exponent>[eE](?<exponentSign>[+-])?(?<exponentDigits>\d+))?$";
1082 
1083  var match = Regex.Match(s, pattern);
1084  if (!match.Success)
1085  {
1086  if (s.Equals("Inf", StringComparison.OrdinalIgnoreCase) || s.Equals("Infinity", StringComparison.OrdinalIgnoreCase) ||
1087  s.Equals("+Inf", StringComparison.OrdinalIgnoreCase) || s.Equals("+Infinity", StringComparison.OrdinalIgnoreCase))
1088  {
1089  result = Decimal128.PositiveInfinity;
1090  return true;
1091  }
1092 
1093  if (s.Equals("-Inf", StringComparison.OrdinalIgnoreCase) || s.Equals("-Infinity", StringComparison.OrdinalIgnoreCase))
1094  {
1095  result = Decimal128.NegativeInfinity;
1096  return true;
1097  }
1098 
1099  if (s.Equals("NaN", StringComparison.OrdinalIgnoreCase) || s.Equals("-NaN", StringComparison.OrdinalIgnoreCase))
1100  {
1101  result = Decimal128.QNaN;
1102  return true;
1103  }
1104 
1105  result = default(Decimal128);
1106  return false;
1107  }
1108 
1109  var isNegative = match.Groups["sign"].Value == "-";
1110 
1111  var exponent = 0;
1112  if (match.Groups["exponent"].Length != 0)
1113  {
1114  if (!int.TryParse(match.Groups["exponentDigits"].Value, out exponent))
1115  {
1116  result = default(Decimal128);
1117  return false;
1118  }
1119  if (match.Groups["exponentSign"].Value == "-")
1120  {
1121  exponent = -exponent;
1122  }
1123  }
1124 
1125  var significandString = match.Groups["significand"].Value;
1126 
1127  int decimalPointIndex;
1128  if ((decimalPointIndex = significandString.IndexOf('.')) != -1)
1129  {
1130  exponent -= significandString.Length - (decimalPointIndex + 1);
1131  significandString = significandString.Substring(0, decimalPointIndex) + significandString.Substring(decimalPointIndex + 1);
1132  }
1133 
1134  significandString = RemoveLeadingZeroes(significandString);
1135  significandString = ClampOrRound(ref exponent, significandString);
1136 
1137  if (exponent > __exponentMax || exponent < __exponentMin)
1138  {
1139  result = default(Decimal128);
1140  return false;
1141  }
1142  if (significandString.Length > 34)
1143  {
1144  result = default(Decimal128);
1145  return false;
1146  }
1147 
1148  var significand = UInt128.Parse(significandString);
1149 
1150  result = Decimal128.FromComponents(isNegative, (short)exponent, significand);
1151  return true;
1152  }

1154  // private static methods
1155  private static string ClampOrRound(ref int exponent, string significandString)
1156  {
1157  if (exponent > __exponentMax)
1158  {
1159  if (significandString == "0")
1160  {
1161  // since significand is zero simply use the largest possible exponent
1162  exponent = __exponentMax;
1163  }
1164  else
1165  {
1166  // use clamping to bring the exponent into range
1167  var numberOfTrailingZeroesToAdd = exponent - __exponentMax;
1168  var digitsAvailable = 34 - significandString.Length;
1169  if (numberOfTrailingZeroesToAdd <= digitsAvailable)
1170  {
1171  exponent = __exponentMax;
1172  significandString = significandString + new string('0', numberOfTrailingZeroesToAdd);
1173  }
