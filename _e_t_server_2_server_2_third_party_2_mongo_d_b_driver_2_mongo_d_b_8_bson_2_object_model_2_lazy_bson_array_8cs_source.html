169  private IEnumerable<BsonValue> MaterializeThisLevel()
170  {
171  var values = new List<BsonValue>();
172 
173  using (var stream = new ByteBufferStream(_slice, ownsBuffer: false))
174  using (var bsonReader = new BsonBinaryReader(stream, _readerSettings))
175  {
176  var context = BsonDeserializationContext.CreateRoot(bsonReader);
177 
178  bsonReader.ReadStartDocument();
179  BsonType bsonType;
180  while ((bsonType = bsonReader.ReadBsonType()) != BsonType.EndOfDocument)
181  {
182  bsonReader.SkipName();
183  BsonValue value;
184  switch (bsonType)
185  {
186  case BsonType.Array: value = DeserializeLazyBsonArray(bsonReader); break;
187  case BsonType.Document: value = DeserializeLazyBsonDocument(bsonReader); break;
188  default: value = BsonValueSerializer.Instance.Deserialize(context); break;
189  }
190  values.Add(value);
191  }
192  bsonReader.ReadEndDocument();
193  }
194 
195  return values;
196  }
