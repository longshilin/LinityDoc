771  // private methods
772  private void EnsureIsMaterialized()
773  {
774  ThrowIfDisposed();
775  if (!_isMaterialized)
776  {
777  var elements = Materialize();
778  try
779  {
780  _isMaterialized = true;
781  base.AddRange(elements);
782  MaterializeCompleted();
783  }
784  catch
785  {
786  base.Clear();
787  _isMaterialized = false;
788  throw;
789  }
790  }
791  }
