515  // private methods
516  private void EnsureIsMaterialized()
517  {
518  ThrowIfDisposed();
519  if (!_isMaterialized)
520  {
521  var values = Materialize();
522  try
523  {
524  _isMaterialized = true;
525  base.AddRange(values);
526  MaterializeCompleted();
527  }
528  catch
529  {
530  base.Clear();
531  _isMaterialized = false;
532  throw;
533  }
534  }
