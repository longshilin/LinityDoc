87  }
88

92  // for backward compatibility representations of Array and Document are mapped to DictionaryRepresentations if possible
93  var dictionaryRepresentationConfigurable = serializer as IDictionaryRepresentationConfigurable;
94  if (dictionaryRepresentationConfigurable != null)
95  {
96  if (_representation == BsonType.Array || _representation == BsonType.Document)
97  {
98  var dictionaryRepresentation = (_representation == BsonType.Array) ? DictionaryRepresentation.ArrayOfArrays: DictionaryRepresentation.Document;
