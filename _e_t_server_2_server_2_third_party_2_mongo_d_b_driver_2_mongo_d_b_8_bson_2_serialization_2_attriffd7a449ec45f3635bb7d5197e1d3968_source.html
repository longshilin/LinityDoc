73  internal IBsonSerializer CreateSerializer(Type type)
74  {
75  var typeInfo = type.GetTypeInfo();
76  if (typeInfo.ContainsGenericParameters)
77  {
78  var message = "Cannot create a serializer because the type to serialize is an open generic type.";
79  throw new InvalidOperationException(message);
80  }
81 
82  var serializerTypeInfo = _serializerType.GetTypeInfo();
83  if (serializerTypeInfo.ContainsGenericParameters && !typeInfo.IsGenericType)
84  {
85  var message = "Cannot create a serializer because the serializer type is an open generic type and the type to serialize is not generic.";
86  throw new InvalidOperationException(message);
87  }
88 
89  if (serializerTypeInfo.ContainsGenericParameters)
90  {
91  var genericArguments = typeInfo.GetGenericArguments();
92  var closedSerializerType = _serializerType.MakeGenericType(genericArguments);
93  return (IBsonSerializer)Activator.CreateInstance(closedSerializerType);
94  }
95  else
96  {
97  return (IBsonSerializer)Activator.CreateInstance(_serializerType);
98  }
