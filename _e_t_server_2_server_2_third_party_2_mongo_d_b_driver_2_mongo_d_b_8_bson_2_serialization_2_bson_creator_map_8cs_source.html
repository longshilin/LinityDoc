111  public void Freeze()
112  {
113  if (!_isFrozen)
114  {
115  var allMemberMaps = _classMap.AllMemberMaps;
116 
117  var elementNames = new List<string>();
118  var defaultValues = new Dictionary<string, object>();
119 
120  var expectedArgumentsCount = GetExpectedArgumentsCount();
121  if (_arguments != null)
122  {
123  if (_arguments.Count != expectedArgumentsCount)
124  {
125  throw new BsonSerializationException($"Creator map for class {_classMap.ClassType.FullName} has {expectedArgumentsCount} arguments, not {_arguments.Count}.");
126  }
127 
128  foreach (var argument in _arguments)
129  {
130  var memberMap = allMemberMaps.FirstOrDefault(m => IsSameMember(m.MemberInfo, argument));
131  if (memberMap == null)
132  {
133  var message = string.Format("Member '{0}' is not mapped.", argument.Name);
134  throw new BsonSerializationException(message);
135  }
136  elementNames.Add(memberMap.ElementName);
137  if (memberMap.IsDefaultValueSpecified)
138  {
139  defaultValues.Add(memberMap.ElementName, memberMap.DefaultValue);
140  }
141  }
142  }
143  else
144  {
145  if (expectedArgumentsCount != 0)
146  {
147  throw new BsonSerializationException($"Creator map for class {_classMap.ClassType.FullName} has {expectedArgumentsCount} arguments, but none are configured.");
148  }
149  }
150 
151  _elementNames = elementNames;
152  _defaultValues = defaultValues;
153  _isFrozen = true;
154  }
155  }

197  public BsonCreatorMap SetArguments(IEnumerable<string> argumentNames)
198  {
199  if (argumentNames == null)
200  {
201  throw new ArgumentNullException("argumentNames");
202  }
203  if (_isFrozen) { ThrowFrozenException(); }
204 
205  var classTypeInfo = _classMap.ClassType.GetTypeInfo();
206  var arguments = new List<MemberInfo>();
207  foreach (var argumentName in argumentNames)
208  {
209  var bindingFlags = BindingFlags.Instance | BindingFlags.NonPublic | BindingFlags.Public;
210  var memberInfos = classTypeInfo.GetMembers(bindingFlags)
211  .Where(m => m.Name == argumentName && (m is FieldInfo || m is PropertyInfo))
212  .ToArray();
213  if (memberInfos.Length == 0)
214  {
215  var message = string.Format("Class '{0}' does not have a member named '{1}'.", _classMap.ClassType.FullName, argumentName);
216  throw new BsonSerializationException(message);
217  }
218  else if (memberInfos.Length > 1)
219  {
220  var message = string.Format("Class '{0}' has more than one member named '{1}'.", _classMap.ClassType.FullName, argumentName);
221  throw new BsonSerializationException(message);
222  }
223  arguments.Add(memberInfos[0]);
224  }
225 
226  SetArguments(arguments);
227  return this;
228  }

230  // internal methods
231  internal object CreateInstance(Dictionary<string, object> values)
232  {
233  var arguments = new List<object>();
234 
235  // get the values for the arguments to be passed to the creator delegate
236  foreach (var elementName in _elementNames)
237  {
238  object argument;
239  if (values.TryGetValue(elementName, out argument))
240  {
241  values.Remove(elementName);
242  }
243  else if (!_defaultValues.TryGetValue(elementName, out argument))
244  {
245  // shouldn't happen unless there is a bug in ChooseBestCreator
246  throw new BsonInternalException();
247  }
248  arguments.Add(argument);
249  }
250 
251  return _delegate.DynamicInvoke(arguments.ToArray());
252  }

255  private int GetExpectedArgumentsCount()
256  {
257  var constructorInfo = _memberInfo as ConstructorInfo;
258  if (constructorInfo != null)
259  {
260  return constructorInfo.GetParameters().Length;
261  }
262 
263  var methodInfo = _memberInfo as MethodInfo;
264  if (methodInfo != null)
265  {
266  return methodInfo.GetParameters().Length;
267  }
268 
269  var delegateParameters = _delegate.GetMethodInfo().GetParameters();
270  if (delegateParameters.Length == 0)
271  {
272  return 0;
273  }
274  else
275  {
276  // check if delegate is closed over its first parameter
277  if (_delegate.Target != null && _delegate.Target.GetType() == delegateParameters[0].ParameterType)
278  {
279  return delegateParameters.Length - 1;
280  }
281  else
282  {
283  return delegateParameters.Length;
284  }
285  }
