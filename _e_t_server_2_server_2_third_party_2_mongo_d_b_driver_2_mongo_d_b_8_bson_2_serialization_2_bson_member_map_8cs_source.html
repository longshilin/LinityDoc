285  {

503  public bool ShouldSerialize(object obj, object value)
504  {
505  if (_ignoreIfNull)
506  {
507  if (value == null)
508  {
509  return false; // don't serialize null
510  }
511  }
512 
513  if (_ignoreIfDefault)
514  {
515  if (object.Equals(_defaultValue, value))
516  {
517  return false; // don't serialize default value
518  }
519  }
520 
521  if (_shouldSerializeMethod != null && !_shouldSerializeMethod(obj))
522  {
523  // the _shouldSerializeMethod determined that the member shouldn't be serialized
524  return false;
525  }
526

531  private static object GetDefaultValue(Type type)
532  {
533  var typeInfo = type.GetTypeInfo();
534  if (typeInfo.IsEnum)
535  {
536  return Enum.ToObject(type, 0);
537  }
538 
539  switch (Type.GetTypeCode(type))
540  {
541  case TypeCode.Empty:
542 #if NET452
543  case TypeCode.DBNull:
544 #endif
545  case TypeCode.String:
546  break;
547  case TypeCode.Object:
548  if (typeInfo.IsValueType)
549  {
550  return Activator.CreateInstance(type);
551  }
552  break;
553  case TypeCode.Boolean: return false;
554  case TypeCode.Char: return '\0';
555  case TypeCode.SByte: return (sbyte)0;
556  case TypeCode.Byte: return (byte)0;
557  case TypeCode.Int16: return (short)0;
558  case TypeCode.UInt16: return (ushort)0;
559  case TypeCode.Int32: return 0;
560  case TypeCode.UInt32: return 0U;
561  case TypeCode.Int64: return 0L;
562  case TypeCode.UInt64: return 0UL;
563  case TypeCode.Single: return 0F;
564  case TypeCode.Double: return 0D;
565  case TypeCode.Decimal: return 0M;
566  case TypeCode.DateTime: return DateTime.MinValue;
567  }
568  return null;

571  private Action<object, object> GetFieldSetter()
572  {
573  var fieldInfo = (FieldInfo)_memberInfo;
574 
575  if (IsReadOnly)
576  {
577  var message = string.Format(
578  "The field '{0} {1}' of class '{2}' is readonly. To avoid this exception, call IsReadOnly to ensure that setting a value is allowed.",
579  fieldInfo.FieldType.FullName, fieldInfo.Name, fieldInfo.DeclaringType.FullName);
580  throw new BsonSerializationException(message);
581  }
582 
583  var sourceType = fieldInfo.DeclaringType;
584  var method = new DynamicMethod("Set" + fieldInfo.Name, null, new[] { typeof(object), typeof(object) }, true);
585  var gen = method.GetILGenerator();
586 
587  gen.Emit(OpCodes.Ldarg_0);
588  gen.Emit(OpCodes.Castclass, sourceType);
589  gen.Emit(OpCodes.Ldarg_1);
590  gen.Emit(OpCodes.Unbox_Any, fieldInfo.FieldType);
591  gen.Emit(OpCodes.Stfld, fieldInfo);
592  gen.Emit(OpCodes.Ret);
593 
594  return (Action<object, object>)method.CreateDelegate(typeof(Action<object, object>));
595  }

597  private Func<object, object> GetGetter()
598  {
599  var propertyInfo = _memberInfo as PropertyInfo;
600  if (propertyInfo != null)
601  {
602  var getMethodInfo = propertyInfo.GetMethod;
603  if (getMethodInfo == null)
604  {
605  var message = string.Format(
606  "The property '{0} {1}' of class '{2}' has no 'get' accessor.",
607  propertyInfo.PropertyType.FullName, propertyInfo.Name, propertyInfo.DeclaringType.FullName);
608  throw new BsonSerializationException(message);
609  }
610  }
611 
612  // lambdaExpression = (obj) => (object) ((TClass) obj).Member
613  var objParameter = Expression.Parameter(typeof(object), "obj");
614  var lambdaExpression = Expression.Lambda<Func<object, object>>(
615  Expression.Convert(
616  Expression.MakeMemberAccess(
617  Expression.Convert(objParameter, _memberInfo.DeclaringType),
618  _memberInfo
619  ),
620  typeof(object)
621  ),
622  objParameter
623  );
624 
625  return lambdaExpression.Compile();
626  }

628  private Action<object, object> GetPropertySetter()
629  {
630  var propertyInfo = (PropertyInfo)_memberInfo;
631  var setMethodInfo = propertyInfo.SetMethod;
632  if (IsReadOnly)
633  {
634  var message = string.Format(
635  "The property '{0} {1}' of class '{2}' has no 'set' accessor. To avoid this exception, call IsReadOnly to ensure that setting a value is allowed.",
636  propertyInfo.PropertyType.FullName, propertyInfo.Name, propertyInfo.DeclaringType.FullName);
637  throw new BsonSerializationException(message);
638  }
639 
640  // lambdaExpression = (obj, value) => ((TClass) obj).SetMethod((TMember) value)
641  var objParameter = Expression.Parameter(typeof(object), "obj");
642  var valueParameter = Expression.Parameter(typeof(object), "value");
643  var lambdaExpression = Expression.Lambda<Action<object, object>>(
644  Expression.Call(
645  Expression.Convert(objParameter, _memberInfo.DeclaringType),
646  setMethodInfo,
647  Expression.Convert(valueParameter, _memberType)
648  ),
649  objParameter,
650  valueParameter
651  );
652 
653  return lambdaExpression.Compile();
654  }
