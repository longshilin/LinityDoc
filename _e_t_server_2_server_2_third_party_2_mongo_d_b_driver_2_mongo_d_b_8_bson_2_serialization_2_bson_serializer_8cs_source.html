288  public static Type LookupActualType(Type nominalType, BsonValue discriminator)
289  {
290  if (discriminator == null)
291  {
292  return nominalType;
293  }
294 
295  // note: EnsureKnownTypesAreRegistered handles its own locking so call from outside any lock
296  EnsureKnownTypesAreRegistered(nominalType);
297 
298  __configLock.EnterReadLock();
299  try
300  {
301  Type actualType = null;
302 
303  HashSet<Type> hashSet;
304  if (__discriminators.TryGetValue(discriminator, out hashSet))
305  {
306  foreach (var type in hashSet)
307  {
308  if (nominalType.GetTypeInfo().IsAssignableFrom(type))
309  {
310  if (actualType == null)
311  {
312  actualType = type;
313  }
314  else
315  {
316  string message = string.Format("Ambiguous discriminator '{0}'.", discriminator);
317  throw new BsonSerializationException(message);
318  }
319  }
320  }
321  }
322 
323  if (actualType == null && discriminator.IsString)
324  {
325  actualType = TypeNameDiscriminator.GetActualType(discriminator.AsString); // see if it's a Type name
326  }
327 
328  if (actualType == null)
329  {
330  string message = string.Format("Unknown discriminator value '{0}'.", discriminator);
331  throw new BsonSerializationException(message);
332  }
333 
334  if (!nominalType.GetTypeInfo().IsAssignableFrom(actualType))
335  {
336  string message = string.Format(
337  "Actual type {0} is not assignable to expected type {1}.",
338  actualType.FullName, nominalType.FullName);
339  throw new BsonSerializationException(message);
340  }
341 
342  return actualType;
343  }
344  finally
345  {
346  __configLock.ExitReadLock();
347  }
348  }

355  public static IDiscriminatorConvention LookupDiscriminatorConvention(Type type)
356  {
357  var typeInfo = type.GetTypeInfo();
358 
359  __configLock.EnterReadLock();
360  try
361  {
362  IDiscriminatorConvention convention;
363  if (__discriminatorConventions.TryGetValue(type, out convention))
364  {
365  return convention;
366  }
367  }
368  finally
369  {
370  __configLock.ExitReadLock();
371  }
372 
373  __configLock.EnterWriteLock();
374  try
375  {
376  IDiscriminatorConvention convention;
377  if (!__discriminatorConventions.TryGetValue(type, out convention))
378  {
379  if (type == typeof(object))
380  {
381  // if there is no convention registered for object register the default one
382  convention = new ObjectDiscriminatorConvention("_t");
383  RegisterDiscriminatorConvention(typeof(object), convention);
384  }
385  else if (typeInfo.IsInterface)
386  {
387  // TODO: should convention for interfaces be inherited from parent interfaces?
388  convention = LookupDiscriminatorConvention(typeof(object));
389  RegisterDiscriminatorConvention(type, convention);
390  }
391  else
392  {
393  // inherit the discriminator convention from the closest parent (that isn't object) that has one
394  // otherwise default to the standard hierarchical convention
395  Type parentType = type.GetTypeInfo().BaseType;
396  while (convention == null)
397  {
398  if (parentType == typeof(object))
399  {
400  convention = StandardDiscriminatorConvention.Hierarchical;
401  break;
402  }
403  if (__discriminatorConventions.TryGetValue(parentType, out convention))
404  {
405  break;
406  }
407  parentType = parentType.GetTypeInfo().BaseType;
408  }
409 
410  // register this convention for all types between this and the parent type where we found the convention
411  var unregisteredType = type;
412  while (unregisteredType != parentType)
413  {
414  RegisterDiscriminatorConvention(unregisteredType, convention);
415  unregisteredType = unregisteredType.GetTypeInfo().BaseType;
416  }
417  }
418  }
419 
420  return convention;
421  }
422  finally
423  {
424  __configLock.ExitWriteLock();
425  }
426  }

433  public static IIdGenerator LookupIdGenerator(Type type)
434  {
435  var typeInfo = type.GetTypeInfo();
436  __configLock.EnterReadLock();
437  try
438  {
439  IIdGenerator idGenerator;
440  if (__idGenerators.TryGetValue(type, out idGenerator))
441  {
442  return idGenerator;
443  }
444  }
445  finally
446  {
447  __configLock.ExitReadLock();
448  }
449 
450  __configLock.EnterWriteLock();
451  try
452  {
453  IIdGenerator idGenerator;
454  if (!__idGenerators.TryGetValue(type, out idGenerator))
455  {
456  if (typeInfo.IsValueType && __useZeroIdChecker)
457  {
458  var iEquatableDefinition = typeof(IEquatable<>);
459  var iEquatableType = iEquatableDefinition.MakeGenericType(type);
460  if (iEquatableType.GetTypeInfo().IsAssignableFrom(type))
461  {
462  var zeroIdCheckerDefinition = typeof(ZeroIdChecker<>);
463  var zeroIdCheckerType = zeroIdCheckerDefinition.MakeGenericType(type);
464  idGenerator = (IIdGenerator)Activator.CreateInstance(zeroIdCheckerType);
465  }
466  }
467  else if (__useNullIdChecker)
468  {
469  idGenerator = NullIdChecker.Instance;
470  }
471  else
472  {
473  idGenerator = null;
474  }
475 
476  __idGenerators[type] = idGenerator; // remember it even if it's null
477  }
478 
479  return idGenerator;
480  }
481  finally
482  {
483  __configLock.ExitWriteLock();
484  }
485  }

512  public static void RegisterDiscriminator(Type type, BsonValue discriminator)
513  {
514  var typeInfo = type.GetTypeInfo();
515  if (typeInfo.IsInterface)
516  {
517  var message = string.Format("Discriminators can only be registered for classes, not for interface {0}.", type.FullName);
518  throw new BsonSerializationException(message);
519  }
520 
521  __configLock.EnterWriteLock();
522  try
523  {
524  HashSet<Type> hashSet;
525  if (!__discriminators.TryGetValue(discriminator, out hashSet))
526  {
527  hashSet = new HashSet<Type>();
528  __discriminators.Add(discriminator, hashSet);
529  }
530 
531  if (!hashSet.Contains(type))
532  {
533  hashSet.Add(type);
534 
535  // mark all base types as discriminated (so we know that it's worth reading a discriminator)
536  for (var baseType = type.GetTypeInfo().BaseType; baseType != null; baseType = baseType.GetTypeInfo().BaseType)
537  {
538  __discriminatedTypes.Add(baseType);
539  }
540  }
541  }
542  finally
543  {
544  __configLock.ExitWriteLock();
545  }
546  }

672  // internal static methods
673  internal static void EnsureKnownTypesAreRegistered(Type nominalType)
674  {
675  __configLock.EnterReadLock();
676  try
677  {
678  if (__typesWithRegisteredKnownTypes.Contains(nominalType))
679  {
680  return;
681  }
682  }
683  finally
684  {
685  __configLock.ExitReadLock();
686  }
687 
688  __configLock.EnterWriteLock();
689  try
690  {
691  if (!__typesWithRegisteredKnownTypes.Contains(nominalType))
692  {
693  // only call LookupClassMap for classes with a BsonKnownTypesAttribute
694 #if NET452
695  var knownTypesAttribute = nominalType.GetTypeInfo().GetCustomAttributes(typeof(BsonKnownTypesAttribute), false);
696 #else
697  var knownTypesAttribute = nominalType.GetTypeInfo().GetCustomAttributes(typeof(BsonKnownTypesAttribute), false).ToArray();
698 #endif
699  if (knownTypesAttribute != null && knownTypesAttribute.Length > 0)
700  {
701  // try and force a scan of the known types
702  LookupSerializer(nominalType);
703  }
704 
705  __typesWithRegisteredKnownTypes.Add(nominalType);
706  }
707  }
708  finally
709  {
710  __configLock.ExitWriteLock();
711  }
