82  public void RegisterSerializer(Type type, IBsonSerializer serializer)
83  {
84  if (type == null)

92  var typeInfo = type.GetTypeInfo();
93  if (typeof(BsonValue).GetTypeInfo().IsAssignableFrom(type))
94  {
95  var message = string.Format("A serializer cannot be registered for type {0} because it is a subclass of BsonValue.", BsonUtils.GetFriendlyTypeName(type));
96  throw new BsonSerializationException(message);
97  }
98  if (typeInfo.IsGenericType && typeInfo.ContainsGenericParameters)
99  {
100  var message = string.Format("Generic type {0} has unassigned type parameters.", BsonUtils.GetFriendlyTypeName(type));
101  throw new ArgumentException(message, "type");
102  }
103 
104  if (!_cache.TryAdd(type, serializer))
105  {
106  var message = string.Format("There is already a serializer registered for type {0}.", BsonUtils.GetFriendlyTypeName(type));
107  throw new BsonSerializationException(message);

127  private IBsonSerializer CreateSerializer(Type type)
128  {
129  foreach (var serializationProvider in _serializationProviders)
130  {
131  IBsonSerializer serializer;
132 
133  var registryAwareSerializationProvider = serializationProvider as IRegistryAwareBsonSerializationProvider;
134  if (registryAwareSerializationProvider != null)
135  {
136  serializer = registryAwareSerializationProvider.GetSerializer(type, this);
137  }
138  else
139  {
140  serializer = serializationProvider.GetSerializer(type);
141  }
142 
143  if (serializer != null)
144  {
145  return serializer;
146  }
147  }
148 
149  var message = string.Format("No serializer found for type {0}.", type.FullName);
150  throw new BsonSerializationException(message);
151  }
