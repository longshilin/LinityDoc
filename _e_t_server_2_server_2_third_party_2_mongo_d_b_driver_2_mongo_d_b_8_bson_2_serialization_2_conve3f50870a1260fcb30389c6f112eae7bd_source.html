86  public Type GetActualType(IBsonReader bsonReader, Type nominalType)
87  {
88  // the BsonReader is sitting at the value whose actual type needs to be found
89  var bsonType = bsonReader.GetCurrentBsonType();
90  if (bsonType == BsonType.Document)
91  {
92  // ensure KnownTypes of nominalType are registered (so IsTypeDiscriminated returns correct answer)
93  BsonSerializer.EnsureKnownTypesAreRegistered(nominalType);
94 
95  // we can skip looking for a discriminator if nominalType has no discriminated sub types
96  if (BsonSerializer.IsTypeDiscriminated(nominalType))
97  {
98  var bookmark = bsonReader.GetBookmark();
99  bsonReader.ReadStartDocument();
100  var actualType = nominalType;
101  if (bsonReader.FindElement(_elementName))
102  {
103  var context = BsonDeserializationContext.CreateRoot(bsonReader);
104  var discriminator = BsonValueSerializer.Instance.Deserialize(context);
105  if (discriminator.IsBsonArray)
106  {
107  discriminator = discriminator.AsBsonArray.Last(); // last item is leaf class discriminator
108  }
109  actualType = BsonSerializer.LookupActualType(nominalType, discriminator);
110  }
111  bsonReader.ReturnToBookmark(bookmark);
112  return actualType;
113  }
114  }
115 
116  return nominalType;
