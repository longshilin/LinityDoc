32  public void Apply(BsonCreatorMap creatorMap)
33  {
34  if (creatorMap.Arguments == null)
35  {
36  if (creatorMap.MemberInfo != null)
37  {
38  var parameters = GetParameters(creatorMap.MemberInfo);
39  if (parameters != null)
40  {
41  var arguments = new List<MemberInfo>();
42 
43  foreach (var parameter in parameters)
44  {
45  var argument = FindMatchingArgument(creatorMap.ClassMap.ClassType, parameter);
46  if (argument == null)
47  {
48  return;
49  }
50  arguments.Add(argument);
51  }
52 
53  creatorMap.SetArguments(arguments);
54  }
55  }
56  }
57  }

60  private MemberInfo FindMatchingArgument(Type classType, ParameterInfo parameter)
61  {
62  MemberInfo argument;
63  if ((argument = Match(classType, MemberTypes.Property, BindingFlags.Public, parameter)) != null)
64  {
65  return argument;
66  }
67  if ((argument = Match(classType, MemberTypes.Field, BindingFlags.Public, parameter)) != null)
68  {
69  return argument;
70  }
71  if ((argument = Match(classType, MemberTypes.Property, BindingFlags.NonPublic, parameter)) != null)
72  {
73  return argument;
74  }
75  if ((argument = Match(classType, MemberTypes.Field, BindingFlags.NonPublic, parameter)) != null)
76  {
77  return argument;
78  }
79  return null;
80  }

82  private Type GetMemberType(MemberInfo memberInfo)
83  {
84  var fieldInfo = memberInfo as FieldInfo;
85  if (fieldInfo != null)
86  {
87  return fieldInfo.FieldType;
88  }
89 
90  var propertyInfo = memberInfo as PropertyInfo;
91  if (propertyInfo != null)
92  {
93  return propertyInfo.PropertyType;
94  }
95 
96  // should never happen
97  throw new BsonInternalException();
98  }

100  private IEnumerable<ParameterInfo> GetParameters(MemberInfo memberInfo)
101  {
102  var constructorInfo = memberInfo as ConstructorInfo;
103  if (constructorInfo != null)
104  {
105  return constructorInfo.GetParameters();
106  }
107 
108  var methodInfo = memberInfo as MethodInfo;
109  if (methodInfo != null)
110  {
111  return methodInfo.GetParameters();
112  }
113 
114  return null;
115  }

117  private MemberInfo Match(Type classType, MemberTypes memberType, BindingFlags visibility, ParameterInfo parameter)
118  {
119  var classTypeInfo = classType.GetTypeInfo();
120  var bindingAttr = BindingFlags.IgnoreCase | BindingFlags.Instance;
121  var memberInfos = classTypeInfo.GetMember(parameter.Name, memberType, bindingAttr | visibility);
122  if (memberInfos.Length == 1 && GetMemberType(memberInfos[0]) == parameter.ParameterType)
123  {
124  return memberInfos[0];
125  }
126  return null;
127  }
