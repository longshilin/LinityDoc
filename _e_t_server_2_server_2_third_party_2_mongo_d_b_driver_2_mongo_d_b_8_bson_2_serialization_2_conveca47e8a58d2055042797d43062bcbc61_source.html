78  public Type GetActualType(IBsonReader bsonReader, Type nominalType)
79  {
80  // the BsonReader is sitting at the value whose actual type needs to be found
81  var bsonType = bsonReader.GetCurrentBsonType();
82  if (bsonReader.State == BsonReaderState.Value)
83  {
84  Type primitiveType = null;
85  switch (bsonType)
86  {
87  case BsonType.Boolean: primitiveType = typeof(bool); break;
88  case BsonType.Binary:
89  var bookmark = bsonReader.GetBookmark();
90  var binaryData = bsonReader.ReadBinaryData();
91  var subType = binaryData.SubType;
92  if (subType == BsonBinarySubType.UuidStandard || subType == BsonBinarySubType.UuidLegacy)
93  {
94  primitiveType = typeof(Guid);
95  }
96  bsonReader.ReturnToBookmark(bookmark);
97  break;
98  case BsonType.DateTime: primitiveType = typeof(DateTime); break;
99  case BsonType.Decimal128: primitiveType = typeof(Decimal128); break;
100  case BsonType.Double: primitiveType = typeof(double); break;
101  case BsonType.Int32: primitiveType = typeof(int); break;
102  case BsonType.Int64: primitiveType = typeof(long); break;
103  case BsonType.ObjectId: primitiveType = typeof(ObjectId); break;
104  case BsonType.String: primitiveType = typeof(string); break;
105  }
106 
107  // Type.IsAssignableFrom is extremely expensive, always perform a direct type check before calling Type.IsAssignableFrom
108  if (primitiveType != null && (primitiveType == nominalType || nominalType.GetTypeInfo().IsAssignableFrom(primitiveType)))
109  {
110  return primitiveType;
111  }
112  }
113 
114  if (bsonType == BsonType.Document)
115  {
116  var bookmark = bsonReader.GetBookmark();
117  bsonReader.ReadStartDocument();
118  var actualType = nominalType;
119  if (bsonReader.FindElement(_elementName))
120  {
121  var context = BsonDeserializationContext.CreateRoot(bsonReader);
122  var discriminator = BsonValueSerializer.Instance.Deserialize(context);
123  if (discriminator.IsBsonArray)
124  {
125  discriminator = discriminator.AsBsonArray.Last(); // last item is leaf class discriminator
126  }
127  actualType = BsonSerializer.LookupActualType(nominalType, discriminator);
128  }
129  bsonReader.ReturnToBookmark(bookmark);
130  return actualType;
131  }
132 
133  return nominalType;
134  }
