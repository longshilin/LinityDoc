52  {
53  var memberType = memberMap.MemberType;
54  var memberTypeInfo = memberType.GetTypeInfo();
55 
56  if (memberTypeInfo.IsEnum)
57  {
58  var serializer = memberMap.GetSerializer();
59  var representationConfigurableSerializer = serializer as IRepresentationConfigurable;
60  if (representationConfigurableSerializer != null)
61  {

63  memberMap.SetSerializer(reconfiguredSerializer);
64  }
65  return;
66  }
