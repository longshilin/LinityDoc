58  public ReadWriteMemberFinderConvention(MemberTypes memberTypes, BindingFlags bindingFlags)
59  {

62  }

64  // public methods
69  public void Apply(BsonClassMap classMap)
70  {
71  // order is important for backwards compatibility and GetMembers changes the order of finding things.
72  // hence, we'll check member types explicitly instead of letting GetMembers handle it.
73 
74  if ((_memberTypes & MemberTypes.Field) == MemberTypes.Field)
75  {
76  var fields = classMap.ClassType.GetTypeInfo().GetFields(_bindingFlags);
77  foreach (var field in fields)
78  {
79  MapField(classMap, field);
80  }
81  }
82 
83  if ((_memberTypes & MemberTypes.Property) == MemberTypes.Property)
84  {
85  var properties = classMap.ClassType.GetTypeInfo().GetProperties(_bindingFlags);
86  foreach (var property in properties)
87  {
88  MapProperty(classMap, property);
89  }
90  }
91  }
92 
93  // private methods
94  private void MapField(BsonClassMap classMap, FieldInfo fieldInfo)
95  {
96  if (fieldInfo.IsInitOnly || fieldInfo.IsLiteral)
97  {
98  // we can't write
99  return;
100  }
101 
102  classMap.MapMember(fieldInfo);
103  }
104 
105  private void MapProperty(BsonClassMap classMap, PropertyInfo propertyInfo)
106  {
107  if (!propertyInfo.CanRead || !propertyInfo.CanWrite)
108  {
109  // only read and write properties are mapped
110  return;
111  }
112 
113  // skip indexers
114  if (propertyInfo.GetIndexParameters().Length != 0)
115  {
116  return;
117  }
118 
119  // skip overridden properties (they are already included by the base class)
120  var getMethodInfo = propertyInfo.GetMethod;
121  if (getMethodInfo.IsVirtual && getMethodInfo.GetBaseDefinition().DeclaringType != classMap.ClassType)
122  {
123  return;
124  }
125 
126  classMap.MapMember(propertyInfo);
127  }
128  }
