77  public override IBsonSerializer GetSerializer(Type type, IBsonSerializerRegistry serializerRegistry)
78  {
79  if (type == null)
80  {
81  throw new ArgumentNullException("type");
82  }
83  var typeInfo = type.GetTypeInfo();
84  if (typeInfo.IsGenericType && typeInfo.ContainsGenericParameters)
85  {
86  var message = string.Format("Generic type {0} has unassigned type parameters.", BsonUtils.GetFriendlyTypeName(type));
87  throw new ArgumentException(message, "type");
88  }
89 
90  Type serializerType;
91  if (__serializersTypes.TryGetValue(type, out serializerType))
92  {
93  return CreateSerializer(serializerType, serializerRegistry);
94  }
95 
96  if (typeInfo.IsGenericType && !typeInfo.ContainsGenericParameters)
97  {
98  Type serializerTypeDefinition;
99  if (__serializersTypes.TryGetValue(type.GetGenericTypeDefinition(), out serializerTypeDefinition))
100  {
101  return CreateGenericSerializer(serializerTypeDefinition, type.GetTypeInfo().GetGenericArguments(), serializerRegistry);
102  }
103  }
104 
105  if (typeInfo.IsEnum)
106  {
107  return CreateGenericSerializer(typeof(EnumSerializer<>), new[] { type }, serializerRegistry);
108  }
109 
110  return null;
