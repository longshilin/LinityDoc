47  public override T Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
48  {
49  var bsonReader = context.Reader;
50 
51  var bsonType = bsonReader.GetCurrentBsonType();
52  string message;
53  switch (bsonType)
54  {
55  case BsonType.Document:
56  var dynamicContext = context.With(ConfigureDeserializationContext);
57  bsonReader.ReadStartDocument();
58  var document = CreateDocument();
59  while (bsonReader.ReadBsonType() != BsonType.EndOfDocument)
60  {
61  var name = bsonReader.ReadName();
62  var value = _objectSerializer.Deserialize(dynamicContext);
63  SetValueForMember(document, name, value);
64  }
65  bsonReader.ReadEndDocument();
66  return document;
67 
68  default:
69  message = string.Format("Cannot deserialize a '{0}' from BsonType '{1}'.", BsonUtils.GetFriendlyTypeName(typeof(T)), bsonType);
70  throw new FormatException(message);
71  }
72  }

80  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, T value)
81  {
82  var bsonWriter = context.Writer;
83 
84  var metaObject = value.GetMetaObject(Expression.Constant(value));
85  var memberNames = metaObject.GetDynamicMemberNames();
86  var dynamicContext = context.With(ConfigureSerializationContext);
87 
88  bsonWriter.WriteStartDocument();
89  foreach (var memberName in memberNames)
90  {
91  object memberValue;
92  if (TryGetValueForMember(value, memberName, out memberValue))
93  {
94  bsonWriter.WriteName(memberName);
95  _objectSerializer.Serialize(dynamicContext, memberValue);
96  }
97  }
98  bsonWriter.WriteEndDocument();
99  }
