82  public bool GetDocumentId(
83  object document,

85  out Type idNominalType,
86  out IIdGenerator idGenerator)
87  {
88  var bsonDocument = (BsonDocument)document;
89 
90  BsonValue idBsonValue;
91  if (bsonDocument.TryGetValue("_id", out idBsonValue))
92  {
93  id = idBsonValue;
94  idGenerator = BsonSerializer.LookupIdGenerator(id.GetType());
95 
96  if (idGenerator == null)
97  {
98  var idBinaryData = id as BsonBinaryData;
99  if (idBinaryData != null && (idBinaryData.SubType == BsonBinarySubType.UuidLegacy || idBinaryData.SubType == BsonBinarySubType.UuidStandard))
100  {
101  idGenerator = BsonBinaryDataGuidGenerator.GetInstance(idBinaryData.GuidRepresentation);
102  }
103  }
104  }
105  else
106  {
107  id = null;
108  idGenerator = BsonObjectIdGenerator.Instance;
109  }
110

138  protected override void SerializeValue(BsonSerializationContext context, BsonSerializationArgs args, BsonDocument value)
139  {
140  var bsonWriter = context.Writer;
141  bsonWriter.WriteStartDocument();
142 
143  var alreadySerializedIndex = -1;
144  if (args.SerializeIdFirst)
145  {
146  var idIndex = value.IndexOfName("_id");
147  if (idIndex != -1)
148  {
149  bsonWriter.WriteName("_id");
150  BsonValueSerializer.Instance.Serialize(context, value[idIndex]);
151  alreadySerializedIndex = idIndex;
152  }
153  }
154 
155  var elementCount = value.ElementCount;
156  for (var index = 0; index < elementCount; index++)
157  {
158  if (index == alreadySerializedIndex)
159  {
160  continue;
161  }
162 
163  var element = value.GetElement(index);
164  bsonWriter.WriteName(element.Name);
165  BsonValueSerializer.Instance.Serialize(context, element.Value);
166  }
167 
168  bsonWriter.WriteEndDocument();
169  }
