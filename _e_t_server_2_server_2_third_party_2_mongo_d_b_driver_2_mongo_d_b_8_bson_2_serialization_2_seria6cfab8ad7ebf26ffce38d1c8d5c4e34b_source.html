88  public override TValue Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
89  {
90  var bsonReader = context.Reader;
91 
92  var bsonType = bsonReader.GetCurrentBsonType();
93  switch (bsonType)
94  {
95  case BsonType.Null:
96  bsonReader.ReadNull();
97  return null;
98 
99  case BsonType.Array:
100  bsonReader.ReadStartArray();
101  var accumulator = CreateAccumulator();
102  while (bsonReader.ReadBsonType() != BsonType.EndOfDocument)
103  {
104  var item = _lazyItemSerializer.Value.Deserialize(context);
105  AddItem(accumulator, item);
106  }
107  bsonReader.ReadEndArray();
108  return FinalizeResult(accumulator);
109 
110  case BsonType.Document:
111  var serializer = new DiscriminatedWrapperSerializer<TValue>(_discriminatorConvention, this);
112  if (serializer.IsPositionedAtDiscriminatedWrapper(context))
113  {
114  return (TValue)serializer.Deserialize(context);
115  }
116  else
117  {
118  goto default;
119  }
120 
121  default:
122  throw CreateCannotDeserializeFromBsonTypeException(bsonType);
123  }
124  }
