191  public override DateTime Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
192  {
193  var bsonReader = context.Reader;
194  DateTime value;
195 
196  var bsonType = bsonReader.GetCurrentBsonType();
197  switch (bsonType)
198  {
199  case BsonType.DateTime:
200  // use an intermediate BsonDateTime so MinValue and MaxValue are handled correctly
201  value = new BsonDateTime(bsonReader.ReadDateTime()).ToUniversalTime();
202  break;
203 
204  case BsonType.Document:
205  value = default(DateTime);
206  _helper.DeserializeMembers(context, (elementName, flag) =>
207  {
208  switch (flag)
209  {
210  case Flags.DateTime: bsonReader.SkipValue(); break; // ignore value (use Ticks instead)
211  case Flags.Ticks: value = new DateTime(_int64Serializer.Deserialize(context), DateTimeKind.Utc); break;
212  }
213  });
214  break;
215 
216  case BsonType.Int64:
217  value = DateTime.SpecifyKind(new DateTime(bsonReader.ReadInt64()), DateTimeKind.Utc);
218  break;
219 
220  case BsonType.String:
221  if (_dateOnly)
222  {
223  value = DateTime.SpecifyKind(DateTime.ParseExact(bsonReader.ReadString(), "yyyy-MM-dd", null), DateTimeKind.Utc);
224  }
225  else
226  {
227  value = JsonConvert.ToDateTime(bsonReader.ReadString());
228  }
229  break;
230 
231  default:
232  throw CreateCannotDeserializeFromBsonTypeException(bsonType);
233  }
234 
235  if (_dateOnly)
236  {
237  if (value.TimeOfDay != TimeSpan.Zero)
238  {
239  throw new FormatException("TimeOfDay component for DateOnly DateTime value is not zero.");
240  }
241  value = DateTime.SpecifyKind(value, _kind); // not ToLocalTime or ToUniversalTime!
242  }
243  else
244  {
245  switch (_kind)
246  {
247  case DateTimeKind.Local:
248  case DateTimeKind.Unspecified:
249  value = DateTime.SpecifyKind(BsonUtils.ToLocalTime(value), _kind);
250  break;
251  case DateTimeKind.Utc:
252  value = BsonUtils.ToUniversalTime(value);
253  break;
254  }
255  }
256 
257  return value;
258  }

266  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, DateTime value)
267  {
268  var bsonWriter = context.Writer;
269 
270  DateTime utcDateTime;
271  if (_dateOnly)
272  {
273  if (value.TimeOfDay != TimeSpan.Zero)
274  {
275  throw new BsonSerializationException("TimeOfDay component is not zero.");
276  }
277  utcDateTime = DateTime.SpecifyKind(value, DateTimeKind.Utc); // not ToLocalTime
278  }
279  else
280  {
281  utcDateTime = BsonUtils.ToUniversalTime(value);
282  }
283  var millisecondsSinceEpoch = BsonUtils.ToMillisecondsSinceEpoch(utcDateTime);
284 
285  switch (_representation)
286  {
287  case BsonType.DateTime:
288  bsonWriter.WriteDateTime(millisecondsSinceEpoch);
289  break;
290 
291  case BsonType.Document:
292  bsonWriter.WriteStartDocument();
293  bsonWriter.WriteDateTime("DateTime", millisecondsSinceEpoch);
294  bsonWriter.WriteInt64("Ticks", utcDateTime.Ticks);
295  bsonWriter.WriteEndDocument();
296  break;
297 
298  case BsonType.Int64:
299  bsonWriter.WriteInt64(utcDateTime.Ticks);
300  break;
301 
302  case BsonType.String:
303  if (_dateOnly)
304  {
305  bsonWriter.WriteString(value.ToString("yyyy-MM-dd"));
306  }
307  else
308  {
309  if (value == DateTime.MinValue || value == DateTime.MaxValue)
310  {
311  // serialize MinValue and MaxValue as Unspecified so we do NOT get the time zone offset
312  value = DateTime.SpecifyKind(value, DateTimeKind.Unspecified);
313  }
314  else if (value.Kind == DateTimeKind.Unspecified)
315  {
316  // serialize Unspecified as Local se we get the time zone offset
317  value = DateTime.SpecifyKind(value, DateTimeKind.Local);
318  }
319  bsonWriter.WriteString(JsonConvert.ToString(value));
320  }
321  break;
322 
323  default:
324  var message = string.Format("'{0}' is not a valid DateTime representation.", _representation);
325  throw new BsonSerializationException(message);
326  }
327  }
