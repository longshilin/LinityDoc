99  public override DateTimeOffset Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
100  {
101  var bsonReader = context.Reader;
102  long ticks;
103  TimeSpan offset;
104 
105  BsonType bsonType = bsonReader.GetCurrentBsonType();
106  switch (bsonType)
107  {
108  case BsonType.Array:
109  bsonReader.ReadStartArray();
110  ticks = bsonReader.ReadInt64();
111  offset = TimeSpan.FromMinutes(bsonReader.ReadInt32());
112  bsonReader.ReadEndArray();
113  return new DateTimeOffset(ticks, offset);
114 
115  case BsonType.Document:
116  ticks = 0;
117  offset = TimeSpan.Zero;
118  _helper.DeserializeMembers(context, (elementName, flag) =>
119  {
120  switch (flag)
121  {
122  case Flags.DateTime: bsonReader.SkipValue(); break; // ignore value
123  case Flags.Ticks: ticks = _int64Serializer.Deserialize(context); break;
124  case Flags.Offset: offset = TimeSpan.FromMinutes(_int32Serializer.Deserialize(context)); break;
125  }
126  });
127  return new DateTimeOffset(ticks, offset);
128 
129  case BsonType.String:
130  return JsonConvert.ToDateTimeOffset(bsonReader.ReadString());
131 
132  default:
133  throw CreateCannotDeserializeFromBsonTypeException(bsonType);
134  }
135  }

143  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, DateTimeOffset value)
144  {
145  var bsonWriter = context.Writer;
146 
147  // note: the DateTime portion cannot be serialized as a BsonType.DateTime because it is NOT in UTC
148 
149  switch (_representation)
150  {
151  case BsonType.Array:
152  bsonWriter.WriteStartArray();
153  bsonWriter.WriteInt64(value.Ticks);
154  bsonWriter.WriteInt32((int)value.Offset.TotalMinutes);
155  bsonWriter.WriteEndArray();
156  break;
157 
158  case BsonType.Document:
159  bsonWriter.WriteStartDocument();
160  bsonWriter.WriteDateTime("DateTime", BsonUtils.ToMillisecondsSinceEpoch(value.UtcDateTime));
161  bsonWriter.WriteInt64("Ticks", value.Ticks);
162  bsonWriter.WriteInt32("Offset", (int)value.Offset.TotalMinutes);
163  bsonWriter.WriteEndDocument();
164  break;
165 
166  case BsonType.String:
167  bsonWriter.WriteString(JsonConvert.ToString(value));
168  break;
169 
170  default:
171  var message = string.Format("'{0}' is not a valid DateTimeOffset representation.", _representation);
172  throw new BsonSerializationException(message);
173  }
174  }
