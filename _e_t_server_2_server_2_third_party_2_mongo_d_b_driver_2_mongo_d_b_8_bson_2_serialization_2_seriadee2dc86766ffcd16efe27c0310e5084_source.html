105  public override TimeSpan Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
106  {
107  var bsonReader = context.Reader;
108 
109  BsonType bsonType = bsonReader.GetCurrentBsonType();
110  switch (bsonType)
111  {
112  case BsonType.Double:
113  return FromDouble(bsonReader.ReadDouble(), _units);
114 
115  case BsonType.Int32:
116  return FromInt32(bsonReader.ReadInt32(), _units);
117 
118  case BsonType.Int64:
119  return FromInt64(bsonReader.ReadInt64(), _units);
120 
121  case BsonType.String:
122  return TimeSpan.Parse(bsonReader.ReadString()); // not XmlConvert.ToTimeSpan (we're using .NET's format for TimeSpan)
123 
124  default:
125  throw CreateCannotDeserializeFromBsonTypeException(bsonType);
126  }

237  private long TicksPerUnit(TimeSpanUnits units)
238  {
239  switch (units)
240  {
241  case TimeSpanUnits.Days: return TimeSpan.TicksPerDay;
242  case TimeSpanUnits.Hours: return TimeSpan.TicksPerHour;
243  case TimeSpanUnits.Minutes: return TimeSpan.TicksPerMinute;
244  case TimeSpanUnits.Seconds: return TimeSpan.TicksPerSecond;
245  case TimeSpanUnits.Milliseconds: return TimeSpan.TicksPerMillisecond;
246  case TimeSpanUnits.Microseconds: return TimeSpan.TicksPerMillisecond / 1000;
247  case TimeSpanUnits.Ticks: return 1;
248  default:
249  var message = string.Format("Invalid TimeSpanUnits value: {0}.", units);
250  throw new ArgumentException(message);
