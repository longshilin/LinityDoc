80  public override Guid Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
81  {
82  var bsonReader = context.Reader;
83  string message;
84 
85  var bsonType = bsonReader.GetCurrentBsonType();
86  switch (bsonType)
87  {
88  case BsonType.Binary:
89  var binaryData = bsonReader.ReadBinaryData();
90  var bytes = binaryData.Bytes;
91  var subType = binaryData.SubType;
92  var guidRepresentation = binaryData.GuidRepresentation;
93  if (bytes.Length != 16)
94  {
95  message = string.Format("Expected length to be 16, not {0}.", bytes.Length);
96  throw new FormatException(message);
97  }
98  if (subType != BsonBinarySubType.UuidStandard && subType != BsonBinarySubType.UuidLegacy)
99  {
100  message = string.Format("Expected binary sub type to be UuidStandard or UuidLegacy, not {0}.", subType);
101  throw new FormatException(message);
102  }
103  if (guidRepresentation == GuidRepresentation.Unspecified)
104  {
105  throw new BsonSerializationException("GuidSerializer cannot deserialize a Guid when GuidRepresentation is Unspecified.");
106  }
107  return GuidConverter.FromBytes(bytes, guidRepresentation);
108 
109  case BsonType.String:
110  return new Guid(bsonReader.ReadString());
111 
112  default:
113  throw CreateCannotDeserializeFromBsonTypeException(bsonType);
114  }
115  }

123  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, Guid value)
124  {
125  var bsonWriter = context.Writer;
126 
127  switch (_representation)
128  {
129  case BsonType.Binary:
130  var writerGuidRepresentation = bsonWriter.Settings.GuidRepresentation;
131  if (writerGuidRepresentation == GuidRepresentation.Unspecified)
132  {
133  throw new BsonSerializationException("GuidSerializer cannot serialize a Guid when GuidRepresentation is Unspecified.");
134  }
135  var bytes = GuidConverter.ToBytes(value, writerGuidRepresentation);
136  var subType = (writerGuidRepresentation == GuidRepresentation.Standard) ? BsonBinarySubType.UuidStandard : BsonBinarySubType.UuidLegacy;
137  bsonWriter.WriteBinaryData(new BsonBinaryData(bytes, subType, writerGuidRepresentation));
138  break;
139 
140  case BsonType.String:
141  bsonWriter.WriteString(value.ToString());
142  break;
143 
144  default:
145  var message = string.Format("'{0}' is not a valid Guid representation.", _representation);
146  throw new BsonSerializationException(message);
147  }
148  }
