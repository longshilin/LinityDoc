76  public override object Deserialize(BsonDeserializationContext context, BsonDeserializationArgs args)
77  {
78  var bsonReader = context.Reader;
79 
80  var bsonType = bsonReader.GetCurrentBsonType();
81  switch (bsonType)
82  {
83  case BsonType.Array:
84  if (context.DynamicArraySerializer != null)
85  {
86  return context.DynamicArraySerializer.Deserialize(context);
87  }
88  goto default;
89 
90  case BsonType.Binary:
91  var binaryData = bsonReader.ReadBinaryData();
92  var subType = binaryData.SubType;
93  if (subType == BsonBinarySubType.UuidStandard || subType == BsonBinarySubType.UuidLegacy)
94  {
95  return binaryData.ToGuid();
96  }
97  goto default;
98 
99  case BsonType.Boolean:
100  return bsonReader.ReadBoolean();
101 
102  case BsonType.DateTime:
103  var millisecondsSinceEpoch = bsonReader.ReadDateTime();
104  var bsonDateTime = new BsonDateTime(millisecondsSinceEpoch);
105  return bsonDateTime.ToUniversalTime();
106 
107  case BsonType.Decimal128:
108  return bsonReader.ReadDecimal128();
109 
110  case BsonType.Document:
111  return DeserializeDiscriminatedValue(context, args);
112 
113  case BsonType.Double:
114  return bsonReader.ReadDouble();
115 
116  case BsonType.Int32:
117  return bsonReader.ReadInt32();
118 
119  case BsonType.Int64:
120  return bsonReader.ReadInt64();
121 
122  case BsonType.Null:
123  bsonReader.ReadNull();
124  return null;
125 
126  case BsonType.ObjectId:
127  return bsonReader.ReadObjectId();
128 
129  case BsonType.String:
130  return bsonReader.ReadString();
131 
132  default:
133  var message = string.Format("ObjectSerializer does not support BSON type '{0}'.", bsonType);
134  throw new FormatException(message);
135  }
136  }

144  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, object value)
145  {
146  var bsonWriter = context.Writer;
147 
148  if (value == null)
149  {
150  bsonWriter.WriteNull();
151  }
152  else
153  {
154  var actualType = value.GetType();
155  if (actualType == typeof(object))
156  {
157  bsonWriter.WriteStartDocument();
158  bsonWriter.WriteEndDocument();
159  }
160  else
161  {
162  // certain types can be written directly as BSON value
163  // if we're not at the top level document, or if we're using the JsonWriter
164  if (bsonWriter.State == BsonWriterState.Value || bsonWriter is JsonWriter)
165  {
166  switch (Type.GetTypeCode(actualType))
167  {
168  case TypeCode.Boolean:
169  bsonWriter.WriteBoolean((bool)value);
170  return;
171 
172  case TypeCode.DateTime:
173  // TODO: is this right? will lose precision after round trip
174  var bsonDateTime = new BsonDateTime(BsonUtils.ToUniversalTime((DateTime)value));
175  bsonWriter.WriteDateTime(bsonDateTime.MillisecondsSinceEpoch);
176  return;
177 
178  case TypeCode.Double:
179  bsonWriter.WriteDouble((double)value);
180  return;
181 
182  case TypeCode.Int16:
183  // TODO: is this right? will change type to Int32 after round trip
184  bsonWriter.WriteInt32((short)value);
185  return;
186 
187  case TypeCode.Int32:
188  bsonWriter.WriteInt32((int)value);
189  return;
190 
191  case TypeCode.Int64:
192  bsonWriter.WriteInt64((long)value);
193  return;
194 
195  case TypeCode.Object:
196  if (actualType == typeof(Decimal128))
197  {
198  var decimal128 = (Decimal128)value;
199  bsonWriter.WriteDecimal128(decimal128);
200  return;
201  }
202  if (actualType == typeof(Guid))
203  {
204  var guid = (Guid)value;
205  var guidRepresentation = bsonWriter.Settings.GuidRepresentation;
206  var binaryData = new BsonBinaryData(guid, guidRepresentation);
207  bsonWriter.WriteBinaryData(binaryData);
208  return;
209  }
210  if (actualType == typeof(ObjectId))
211  {
212  bsonWriter.WriteObjectId((ObjectId)value);
213  return;
214  }
215  break;
216 
217  case TypeCode.String:
218  bsonWriter.WriteString((string)value);
219  return;
220  }
221  }
222 
223  SerializeDiscriminatedValue(context, args, value, actualType);
224  }
225  }

229  private object DeserializeDiscriminatedValue(BsonDeserializationContext context, BsonDeserializationArgs args)
230  {
231  var bsonReader = context.Reader;
232 
233  var actualType = _discriminatorConvention.GetActualType(bsonReader, typeof(object));
234  if (actualType == typeof(object))
235  {
236  var type = bsonReader.GetCurrentBsonType();
237  switch(type)
238  {
239  case BsonType.Document:
240  if (context.DynamicDocumentSerializer != null)
241  {
242  return context.DynamicDocumentSerializer.Deserialize(context, args);
243  }
244  break;
245  }
246 
247  bsonReader.ReadStartDocument();
248  bsonReader.ReadEndDocument();
249  return new object();
250  }
251  else
252  {
253  var serializer = BsonSerializer.LookupSerializer(actualType);
254  var polymorphicSerializer = serializer as IBsonPolymorphicSerializer;
255  if (polymorphicSerializer != null && polymorphicSerializer.IsDiscriminatorCompatibleWithObjectSerializer)
256  {
257  return serializer.Deserialize(context, args);
258  }
259  else
260  {
261  object value = null;
262  var wasValuePresent = false;
263 
264  bsonReader.ReadStartDocument();
265  while (bsonReader.ReadBsonType() != 0)
266  {
267  var name = bsonReader.ReadName();
268  if (name == _discriminatorConvention.ElementName)
269  {
270  bsonReader.SkipValue();
271  }
272  else if (name == "_v")
273  {
274  value = serializer.Deserialize(context);
275  wasValuePresent = true;
276  }
277  else
278  {
279  var message = string.Format("Unexpected element name: '{0}'.", name);
280  throw new FormatException(message);
281  }
282  }
283  bsonReader.ReadEndDocument();
284 
285  if (!wasValuePresent)
286  {
287  throw new FormatException("_v element missing.");
288  }
289 
290  return value;
291  }
292  }
293  }

295  private void SerializeDiscriminatedValue(BsonSerializationContext context, BsonSerializationArgs args, object value, Type actualType)
296  {
297  var serializer = BsonSerializer.LookupSerializer(actualType);
298 
299  var polymorphicSerializer = serializer as IBsonPolymorphicSerializer;
300  if (polymorphicSerializer != null && polymorphicSerializer.IsDiscriminatorCompatibleWithObjectSerializer)
301  {
302  serializer.Serialize(context, args, value);
303  }
304  else
305  {
306  if (context.IsDynamicType != null && context.IsDynamicType(value.GetType()))
307  {
308  args.NominalType = actualType;
309  serializer.Serialize(context, args, value);
310  }
311  else
312  {
313  var bsonWriter = context.Writer;
314  var discriminator = _discriminatorConvention.GetDiscriminator(typeof(object), actualType);
315 
316  bsonWriter.WriteStartDocument();
317  bsonWriter.WriteName(_discriminatorConvention.ElementName);
318  BsonValueSerializer.Instance.Serialize(context, discriminator);
319  bsonWriter.WriteName("_v");
320  serializer.Serialize(context, value);
321  bsonWriter.WriteEndDocument();
322  }
323  }
324  }
