40  {
41  if (members == null)
42  {
43  throw new ArgumentNullException("members");
44  }
45  if (members.Length > 64)
46  {
47  throw new ArgumentException("SerializerHelper supports a maximum of 64 members.", "members");
48  }

50  _members = members;

52 
53  foreach (var member in members)

59  }
60

79  public long DeserializeMembers(BsonDeserializationContext context, Action<string, long> memberHandler)
80  {
81  var reader = context.Reader;
82  var foundMemberFlags = 0L;
83 
84  reader.ReadStartDocument();
85  while (reader.ReadBsonType() != 0)
86  {
87  var trieDecoder = new TrieNameDecoder<long>(_trie);
88  var elementName = reader.ReadName(trieDecoder);
89 
90  long memberFlag;
91  if (trieDecoder.Found)
92  {
93  memberFlag = trieDecoder.Value;
94  }
95  else
96  {
97  if (_extraMemberFlag == 0)
98  {
99  throw new BsonSerializationException(string.Format(
100  "Invalid element: '{0}'.", elementName));
101  }
102  else
103  {
104  memberFlag = _extraMemberFlag;
105  }
106  }
107 
108  memberHandler(elementName, memberFlag);
109  foundMemberFlags |= memberFlag;
110  }
111  reader.ReadEndDocument();
112 
113  var missingRequiredMemberFlags = _requiredMemberFlags & ~foundMemberFlags;
114  if (missingRequiredMemberFlags != 0)
115  {
116  var missingRequiredMember = FindFirstMissingRequiredMember(missingRequiredMemberFlags);
117  throw new BsonSerializationException(string.Format(
118  "Missing element: '{0}'.", missingRequiredMember.ElementName));
119  }
120 
121  return foundMemberFlags;
122  }
123

125  private Member FindFirstMissingRequiredMember(long missingRequiredMemberFlags)
126  {
127  foreach (var member in _members)
128  {
129  if ((member.Flag & missingRequiredMemberFlags) != 0)
130  {
131  return member;
132  }
133  }
134  throw new BsonInternalException();
