41  public override IBsonSerializer GetSerializer(Type type, IBsonSerializerRegistry serializerRegistry)
42  {
43  if (type == null)
44  {
45  throw new ArgumentNullException("type");
46  }
47  var typeInfo = type.GetTypeInfo();
48  if (typeInfo.IsGenericType && typeInfo.ContainsGenericParameters)
49  {
50  var message = string.Format("Generic type {0} has unassigned type parameters.", BsonUtils.GetFriendlyTypeName(type));
51  throw new ArgumentException(message, "type");
52  }
53 
54  Type serializerType;
55  if (_serializerTypes.TryGetValue(type, out serializerType))
56  {
57  return CreateSerializer(serializerType, serializerRegistry);
58  }
59 
60  if (typeInfo.IsGenericType && !typeInfo.ContainsGenericParameters)
61  {
62  Type serializerTypeDefinition;
63  if (_serializerTypes.TryGetValue(type.GetGenericTypeDefinition(), out serializerTypeDefinition))
64  {
65  return CreateGenericSerializer(serializerTypeDefinition, type.GetTypeInfo().GetGenericArguments(), serializerRegistry);
66  }

77  public void RegisterMapping(Type type, Type serializerType)
78  {
79  if (type == null)
80  {
81  throw new ArgumentNullException("type");
82  }
83  if (serializerType == null)
84  {
85  throw new ArgumentNullException("serializerType");
86  }
87  var typeInfo = type.GetTypeInfo();
88  var serializerTypeInfo = serializerType.GetTypeInfo();
89  if (typeInfo.ContainsGenericParameters != serializerTypeInfo.ContainsGenericParameters)
90  {
91  throw new ArgumentException("The type and the serializerType must have the same number of type parameters.");
92  }
93  if (typeInfo.ContainsGenericParameters)
94  {
95  if (!typeInfo.IsGenericTypeDefinition || !serializerTypeInfo.IsGenericTypeDefinition)
96  {
97  throw new ArgumentException("A generic type must either have all or none of the type parameters assigned.");
98  }
99  if (type.GetTypeInfo().GetGenericArguments().Length != serializerType.GetTypeInfo().GetGenericArguments().Length)
100  {
101  throw new ArgumentException("The type and the serializerType must have the same number of type parameters.");
102  }
103  }
104 
105  if (!_serializerTypes.TryAdd(type, serializerType))
106  {
107  var message = string.Format("There is already a serializer mapping registered for type {0}.", BsonUtils.GetFriendlyTypeName(type));
108  throw new BsonSerializationException(message);
109  }
110  }
