83  // added tokens for the current line
84  void ScanOne(Token tok)
85  {
86  while (true)
87  {
88  while (pos < length && Char.IsWhiteSpace(text[pos]))
89  {
90  UpdateNewlineStats(text[pos]);
91  pos++;
92  }
93 
94  if (pos >= length)
95  {
96  tok.type = TokenType.EOF;
97  break;
98  }
99 
100  char ch = text[pos];
101  char ch2 = text[pos+1];
102 
103  if (ch == '\"')
104  ScanQuotedString(tok);
105  else if (ch == '/' && ch2 == '*')
106  ScanMultilineComment(tok);
107  else if (ch == '/' && ch2 == '/')
108  ScanComment(tok);
109  else if (IsOperator(ch))
110  ScanOperator(tok);
111  else
112  ScanString(tok); // be more robust and accept whatever is left
113  return;
114  }
115  }
