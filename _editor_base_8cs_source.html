34  static string FindURL (System.Type type, string path) {
35  // Find the correct type if the path was not an immediate member of #type
36  while (true) {
37  var index = path.IndexOf('.');
38  if (index == -1) break;
39  var fieldName = path.Substring(0, index);
40  var remaining = path.Substring(index + 1);
41  var field = type.GetField(fieldName, System.Reflection.BindingFlags.Instance | System.Reflection.BindingFlags.Public | System.Reflection.BindingFlags.NonPublic);
42  if (field != null) {
43  type = field.FieldType;
44  path = remaining;
45  } else {
46  // Could not find the correct field
47  return null;
48  }
49  }
50 
51  // Find a documentation entry for the field, fall back to parent classes if necessary
52  while (type != null) {
53  var url = FindURL(type.FullName + "." + path);
54  if (url != null) return url;
55  type = type.BaseType;
56  }
57  return null;
58  }
