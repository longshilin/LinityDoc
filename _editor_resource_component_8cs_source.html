556  private void Update()
557  {
558  if (m_LoadAssetInfos.Count > 0)
559  {
560  int count = 0;
561  LinkedListNode<LoadAssetInfo> current = m_LoadAssetInfos.First;
562  while (current != null && count < m_LoadAssetCountPerFrame)
563  {
564  LoadAssetInfo loadAssetInfo = current.Value;
565  float elapseSeconds = (float)(DateTime.UtcNow - loadAssetInfo.StartTime).TotalSeconds;
566  if (elapseSeconds >= loadAssetInfo.DelaySeconds)
567  {
568  UnityEngine.Object asset = GetCachedAsset(loadAssetInfo.AssetName);
569  if (asset == null)
570  {
571 #if UNITY_EDITOR
572  if (loadAssetInfo.AssetType != null)
573  {
574  asset = UnityEditor.AssetDatabase.LoadAssetAtPath(loadAssetInfo.AssetName, loadAssetInfo.AssetType);
575  }
576  else
577  {
578  asset = UnityEditor.AssetDatabase.LoadMainAssetAtPath(loadAssetInfo.AssetName);
579  }
580 
581  if (m_EnableCachedAssets && asset != null)
582  {
583  m_CachedAssets.Add(loadAssetInfo.AssetName, asset);
584  }
585 #endif
586  }
587 
588  if (asset != null)
589  {
590  if (loadAssetInfo.LoadAssetCallbacks.LoadAssetSuccessCallback != null)
591  {
592  loadAssetInfo.LoadAssetCallbacks.LoadAssetSuccessCallback(loadAssetInfo.AssetName, asset, elapseSeconds, loadAssetInfo.UserData);
593  }
594  }
595  else
596  {
597  if (loadAssetInfo.LoadAssetCallbacks.LoadAssetFailureCallback != null)
598  {
599  loadAssetInfo.LoadAssetCallbacks.LoadAssetFailureCallback(loadAssetInfo.AssetName, LoadResourceStatus.AssetError, "Can not load this asset from asset database.", loadAssetInfo.UserData);
600  }
601  }
602 
603  LinkedListNode<LoadAssetInfo> next = current.Next;
604  m_LoadAssetInfos.Remove(loadAssetInfo);
605  current = next;
606  count++;
607  }
608  else
609  {
610  if (loadAssetInfo.LoadAssetCallbacks.LoadAssetUpdateCallback != null)
611  {
612  loadAssetInfo.LoadAssetCallbacks.LoadAssetUpdateCallback(loadAssetInfo.AssetName, elapseSeconds / loadAssetInfo.DelaySeconds, loadAssetInfo.UserData);
613  }
614 
615  current = current.Next;
616  }
617  }
618  }
619 
620  if (m_LoadSceneInfos.Count > 0)
621  {
622  LinkedListNode<LoadSceneInfo> current = m_LoadSceneInfos.First;
623  while (current != null)
624  {
625  LoadSceneInfo loadSceneInfo = current.Value;
626  if (loadSceneInfo.AsyncOperation.isDone)
627  {
628  if (loadSceneInfo.AsyncOperation.allowSceneActivation)
629  {
630  if (loadSceneInfo.LoadSceneCallbacks.LoadSceneSuccessCallback != null)
631  {
632  loadSceneInfo.LoadSceneCallbacks.LoadSceneSuccessCallback(loadSceneInfo.SceneAssetName, (float)(DateTime.UtcNow - loadSceneInfo.StartTime).TotalSeconds, loadSceneInfo.UserData);
633  }
634  }
635  else
636  {
637  if (loadSceneInfo.LoadSceneCallbacks.LoadSceneFailureCallback != null)
638  {
639  loadSceneInfo.LoadSceneCallbacks.LoadSceneFailureCallback(loadSceneInfo.SceneAssetName, LoadResourceStatus.AssetError, "Can not load this scene from asset database.", loadSceneInfo.UserData);
640  }
641  }
642 
643  LinkedListNode<LoadSceneInfo> next = current.Next;
644  m_LoadSceneInfos.Remove(loadSceneInfo);
645  current = next;
646  }
647  else
648  {
649  if (loadSceneInfo.LoadSceneCallbacks.LoadSceneUpdateCallback != null)
650  {
651  loadSceneInfo.LoadSceneCallbacks.LoadSceneUpdateCallback(loadSceneInfo.SceneAssetName, loadSceneInfo.AsyncOperation.progress, loadSceneInfo.UserData);
652  }
653 
654  current = current.Next;
655  }
656  }
657  }
658 
659  if (m_UnloadSceneInfos.Count > 0)
660  {
661  LinkedListNode<UnloadSceneInfo> current = m_UnloadSceneInfos.First;
662  while (current != null)
663  {
664  UnloadSceneInfo unloadSceneInfo = current.Value;
665  if (unloadSceneInfo.AsyncOperation.isDone)
666  {
667  if (unloadSceneInfo.AsyncOperation.allowSceneActivation)
668  {
669  if (unloadSceneInfo.UnloadSceneCallbacks.UnloadSceneSuccessCallback != null)
670  {
671  unloadSceneInfo.UnloadSceneCallbacks.UnloadSceneSuccessCallback(unloadSceneInfo.SceneAssetName, unloadSceneInfo.UserData);
672  }
673  }
674  else
675  {
676  if (unloadSceneInfo.UnloadSceneCallbacks.UnloadSceneFailureCallback != null)
677  {
678  unloadSceneInfo.UnloadSceneCallbacks.UnloadSceneFailureCallback(unloadSceneInfo.SceneAssetName, unloadSceneInfo.UserData);
679  }
680  }
681 
682  LinkedListNode<UnloadSceneInfo> next = current.Next;
683  m_UnloadSceneInfos.Remove(unloadSceneInfo);
684  current = next;
685  }
686  else
687  {
688  current = current.Next;
689  }
690  }
