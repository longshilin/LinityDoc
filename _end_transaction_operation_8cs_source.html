197  private bool ShouldReplaceTransientTransactionErrorWithUnknownTransactionCommitResult(MongoException exception)
198  {
199  if (exception is MongoConnectionException)
200  {
201  return true;
202  }
203 
204  if (exception is MongoNotPrimaryException || exception is MongoNodeIsRecoveringException)
205  {
206  return true;
207  }
208 
209  var writeConcernException = exception as MongoWriteConcernException;
210  if (writeConcernException != null)
211  {
212  var writeConcernError = writeConcernException.WriteConcernResult.Response?.GetValue("writeConcernError", null)?.AsBsonDocument;
213  if (writeConcernError != null)
214  {
215  var code = (ServerErrorCode)writeConcernError.GetValue("code", -1).ToInt32();
216  switch (code)
217  {
218  case ServerErrorCode.UnsatisfiableWriteConcern:
219  case ServerErrorCode.UnknownReplWriteConcern:
220  return false;
221 
222  default:
223  return true;
224  }
225  }
226  }
227 
228  return false;
229  }
