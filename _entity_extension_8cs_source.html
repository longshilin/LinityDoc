96  private static void ShowEntity(this EntityComponent entityComponent, Type logicType, string entityGroup, int priority, EntityData data)
97  {
98  if (data == null)
99  {
100  Log.Warning("Data is invalid.");
101  return;
102  }
103 
104  IDataTable<DREntity> dtEntity = GameEntry.DataTable.GetDataTable<DREntity>();
105  DREntity drEntity = dtEntity.GetDataRow(data.TypeId);
106  if (drEntity == null)
107  {
108  Log.Warning("Can not load entity id '{0}' from data table.", data.TypeId.ToString());
109  return;
110  }
111 
112  entityComponent.ShowEntity(data.Id, logicType, AssetUtility.GetEntityAsset(drEntity.AssetName), entityGroup, priority, data);
113  }
