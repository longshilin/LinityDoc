162  internal override void Update(float elapseSeconds, float realElapseSeconds)
163  {
164  while (m_RecycleQueue.Count > 0)
165  {
166  EntityInfo entityInfo = m_RecycleQueue.Dequeue();
167  IEntity entity = entityInfo.Entity;
168  EntityGroup entityGroup = (EntityGroup)entity.EntityGroup;
169  if (entityGroup == null)
170  {
171  throw new GameFrameworkException("Entity group is invalid.");
172  }
173 
174  entityInfo.Status = EntityStatus.WillRecycle;
175  entity.OnRecycle();
176  entityInfo.Status = EntityStatus.Recycled;
177  entityGroup.UnspawnEntity(entity);
178  ReferencePool.Release(entityInfo);
179  }
180 
181  foreach (KeyValuePair<string, EntityGroup> entityGroup in m_EntityGroups)
182  {
183  entityGroup.Value.Update(elapseSeconds, realElapseSeconds);
184  }
185  }

608  public void ShowEntity(int entityId, string entityAssetName, string entityGroupName, int priority, object userData)
609  {
610  if (m_ResourceManager == null)
611  {
612  throw new GameFrameworkException("You must set resource manager first.");
613  }
614 
615  if (m_EntityHelper == null)
616  {
617  throw new GameFrameworkException("You must set entity helper first.");
618  }
619 
620  if (string.IsNullOrEmpty(entityAssetName))
621  {
622  throw new GameFrameworkException("Entity asset name is invalid.");
623  }
624 
625  if (string.IsNullOrEmpty(entityGroupName))
626  {
627  throw new GameFrameworkException("Entity group name is invalid.");
628  }
629 
630  if (HasEntity(entityId))
631  {
632  throw new GameFrameworkException(Utility.Text.Format("Entity id '{0}' is already exist.", entityId.ToString()));
633  }
634 
635  if (IsLoadingEntity(entityId))
636  {
637  throw new GameFrameworkException(Utility.Text.Format("Entity '{0}' is already being loaded.", entityId.ToString()));
638  }
639 
640  EntityGroup entityGroup = (EntityGroup)GetEntityGroup(entityGroupName);
641  if (entityGroup == null)
642  {
643  throw new GameFrameworkException(Utility.Text.Format("Entity group '{0}' is not exist.", entityGroupName));
644  }
645 
646  EntityInstanceObject entityInstanceObject = entityGroup.SpawnEntityInstanceObject(entityAssetName);
647  if (entityInstanceObject == null)
648  {
649  int serialId = ++m_Serial;
650  m_EntitiesBeingLoaded.Add(entityId, serialId);
651  m_ResourceManager.LoadAsset(entityAssetName, priority, m_LoadAssetCallbacks, ShowEntityInfo.Create(serialId, entityId, entityGroup, userData));
652  return;
653  }
654 
655  InternalShowEntity(entityId, entityAssetName, entityGroup, entityInstanceObject.Target, false, 0f, userData);
656  }

907  public void AttachEntity(int childEntityId, int parentEntityId, object userData)
908  {
909  if (childEntityId == parentEntityId)
910  {
911  throw new GameFrameworkException(Utility.Text.Format("Can not attach entity when child entity id equals to parent entity id '{0}'.", parentEntityId.ToString()));
912  }

916  {
917  throw new GameFrameworkException(Utility.Text.Format("Can not find child entity '{0}'.", childEntityId.ToString()));
918  }
919 
920  if (childEntityInfo.Status >= EntityStatus.WillHide)
921  {
922  throw new GameFrameworkException(Utility.Text.Format("Can not attach entity when child entity status is '{0}'.", childEntityInfo.Status.ToString()));
923  }
924 
925  EntityInfo parentEntityInfo = GetEntityInfo(parentEntityId);
926  if (parentEntityInfo == null)
927  {
928  throw new GameFrameworkException(Utility.Text.Format("Can not find parent entity '{0}'.", parentEntityId.ToString()));
929  }
930 
931  if (parentEntityInfo.Status >= EntityStatus.WillHide)
932  {
933  throw new GameFrameworkException(Utility.Text.Format("Can not attach entity when parent entity status is '{0}'.", parentEntityInfo.Status.ToString()));
934  }
935 
936  IEntity childEntity = childEntityInfo.Entity;
937  IEntity parentEntity = parentEntityInfo.Entity;
938  DetachEntity(childEntity.Id, userData);
939  childEntityInfo.ParentEntity = parentEntity;
940  parentEntityInfo.AddChildEntity(childEntity);
941  parentEntity.OnAttached(childEntity, userData);
942  childEntity.OnAttachTo(parentEntity, userData);
943  }
