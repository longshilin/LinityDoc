36  public static Exception Map(ConnectionId connectionId, BsonDocument response)
37  {
38  BsonValue code;
39  if (response.TryGetValue("code", out code) && code.IsNumeric)
40  {
41  switch (code.ToInt32())
42  {
43  case 50:
44  case 13475:
45  case 16986:
46  case 16712:
47  return new MongoExecutionTimeoutException(connectionId, message: "Operation exceeded time limit.");
48  }
49  }
50 
51  // the server sometimes sends a response that is missing the "code" field but does have an "errmsg" field
52  BsonValue errmsg;
53  if (response.TryGetValue("errmsg", out errmsg) && errmsg.IsString)
54  {
55  if (errmsg.AsString.Contains("exceeded time limit") ||
56  errmsg.AsString.Contains("execution terminated"))
57  {
58  return new MongoExecutionTimeoutException(connectionId, message: "Operation exceeded time limit.");
59  }
60  }
61 
62  return null;
63  }

73  public static Exception Map(ConnectionId connectionId, WriteConcernResult writeConcernResult)
74  {
75  var code = GetCode(writeConcernResult.Response);
76  if (code.HasValue)
77  {
78  switch (code.Value)
79  {
80  case 11000:
81  case 11001:
82  case 12582:
83  var errorMessage = string.Format(
84  "WriteConcern detected an error '{0}'. (Response was {1}).",
85  writeConcernResult.LastErrorMessage, writeConcernResult.Response.ToJson());
86  return new MongoDuplicateKeyException(connectionId, errorMessage, writeConcernResult);
87  }
88  }
89 
90  bool ok = writeConcernResult.Response.GetValue("ok", false).ToBoolean();
91 
92  if (!ok)
93  {
94  var errorMessage = string.Format(
95  "WriteConcern detected an error '{0}'. (Response was {1}).",
96  writeConcernResult.LastErrorMessage, writeConcernResult.Response.ToJson());
97  return new MongoWriteConcernException(connectionId, errorMessage, writeConcernResult);
98  }

104  writeConcernResult.LastErrorMessage,
105  writeConcernResult.Response.ToJson());

120  public static Exception MapNotPrimaryOrNodeIsRecovering(ConnectionId connectionId, BsonDocument command, BsonDocument response, string errorMessageFieldName)
121  {
122  BsonValue codeBsonValue;
123  if (response.TryGetValue("code", out codeBsonValue) && codeBsonValue.IsNumeric)
124  {
125  var code = (ServerErrorCode)codeBsonValue.ToInt32();
126  switch (code)
127  {
128  case ServerErrorCode.NotMaster:
129  case ServerErrorCode.NotMasterNoSlaveOk:
130  return new MongoNotPrimaryException(connectionId, command, response);
131 
132  case ServerErrorCode.InterruptedAtShutdown:
133  case ServerErrorCode.InterruptedDueToReplStateChange:
134  case ServerErrorCode.NotMasterOrSecondary:
135  case ServerErrorCode.PrimarySteppedDown:
136  case ServerErrorCode.ShutdownInProgress:
137  return new MongoNodeIsRecoveringException(connectionId, command, response);
138  }
139  }
140 
141  BsonValue errorMessageBsonValue;
142  if (response.TryGetValue(errorMessageFieldName, out errorMessageBsonValue) && errorMessageBsonValue.IsString)
143  {
144  var errorMessage = errorMessageBsonValue.ToString();
145  if (errorMessage.IndexOf("node is recovering", StringComparison.OrdinalIgnoreCase) != -1 ||
146  errorMessage.IndexOf("not master or secondary", StringComparison.OrdinalIgnoreCase) != -1)
147  {
148  return new MongoNodeIsRecoveringException(connectionId, command, response);
149  }
150  else if (errorMessage.IndexOf("not master", StringComparison.OrdinalIgnoreCase) != -1)
151  {
152  return new MongoNotPrimaryException(connectionId, command, response);
153  }
154  }
155 
156  return null;
157  }

159  private static int? GetCode(BsonDocument response)
160  {
161  BsonValue code;
162  if (!response.TryGetValue("code", out code))
163  {
164  BsonValue err;
165  BsonValue errObjects;
166  if (response.TryGetValue("err", out err) && response.TryGetValue("errObjects", out errObjects) && errObjects.IsBsonArray)
167  {
168  foreach (var errObject in errObjects.AsBsonArray.OfType<BsonDocument>())
169  {
170  BsonValue currentErr = errObject.GetValue("err", null);
171  if (err.Equals(currentErr))
172  {
173  code = errObject.GetValue("code", null);
174  break;
175  }
176  }
177  }
178  }
179 
180  return (code != null) ? code.ToInt32() : (int?)null;
181  }
182  }
