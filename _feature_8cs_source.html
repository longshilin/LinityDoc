354  public bool IsSupported(SemanticVersion serverVersion)
355  {
356  return _supportRemovedVersion != null
357  ? serverVersion >= _firstSupportedVersion && serverVersion < _supportRemovedVersion
358  : serverVersion >= _firstSupportedVersion;
359  }

366  public SemanticVersion SupportedOrNotSupportedVersion(bool isSupported)
367  {
368  return isSupported ? _firstSupportedVersion : VersionBefore(_firstSupportedVersion);
369  }

375  public void ThrowIfNotSupported(SemanticVersion serverVersion)
376  {
377  if (!IsSupported(serverVersion))
378  {
379  throw new NotSupportedException($"Server version {serverVersion} does not support the {_name} feature.");
380  }
381  }

383  private SemanticVersion VersionBefore(SemanticVersion version)
384  {
385  if (version.Patch > 0)
386  {
387  return new SemanticVersion(version.Major, version.Minor, version.Patch - 1);
388  }
389  else if (version.Minor > 0)
390  {
391  return new SemanticVersion(version.Major, version.Minor - 1, 99);
392  }
393  else if (version.Major > 0)
394  {
395  return new SemanticVersion(version.Major - 1, 99, 99);
396  }
397  else
398  {
399  throw new ArgumentException("There is no version before 0.0.0.", nameof(version));
400  }
401  }
