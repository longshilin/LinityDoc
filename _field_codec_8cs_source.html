241  public sealed class FieldCodec<T>
242  {
243  private static readonly T DefaultDefault;
244  // Only non-nullable value types support packing. This is the simplest way of detecting that.
245  private static readonly bool TypeSupportsPacking = default(T) != null;
246 
247  static FieldCodec()
248  {
249  if (typeof(T) == typeof(string))
250  {
251  DefaultDefault = (T)(object)"";
252  }
253  else if (typeof(T) == typeof(ByteString))
254  {
255  DefaultDefault = (T)(object)ByteString.Empty;
256  }
257  // Otherwise it's the default value of the CLR type
258  }
259 
260  internal static bool IsPackedRepeatedField(uint tag)
261  {
262  return TypeSupportsPacking && WireFormat.GetTagWireType(tag) == WireFormat.WireType.LengthDelimited;
263  }
264 
265  internal readonly bool PackedRepeatedField;
266 
270  internal readonly Action<CodedOutputStream, T> ValueWriter;
271 
275  internal readonly Func<T, int> ValueSizeCalculator;
276 
281  internal readonly Func<CodedInputStream, T> ValueReader;
282 
286  internal readonly int FixedSize;
287 
294  internal readonly uint Tag;
295 
304  internal readonly T DefaultValue;
305 
306  private readonly int tagSize;
307 
308  internal FieldCodec(
309  Func<CodedInputStream, T> reader,
310  Action<CodedOutputStream, T> writer,
311  int fixedSize,
312  uint tag) : this(reader, writer, _ => fixedSize, tag)
313  {
314  FixedSize = fixedSize;
315  }
316 
317  internal FieldCodec(
318  Func<CodedInputStream, T> reader,
319  Action<CodedOutputStream, T> writer,
320  Func<T, int> sizeCalculator,
321  uint tag) : this(reader, writer, sizeCalculator, tag, DefaultDefault)
322  {
323  }
324 
325  internal FieldCodec(
326  Func<CodedInputStream, T> reader,
327  Action<CodedOutputStream, T> writer,
328  Func<T, int> sizeCalculator,
329  uint tag,
330  T defaultValue)
331  {
332  ValueReader = reader;
333  ValueWriter = writer;
334  ValueSizeCalculator = sizeCalculator;
335  FixedSize = 0;
336  Tag = tag;
337  DefaultValue = defaultValue;
338  tagSize = CodedOutputStream.ComputeRawVarint32Size(tag);
339  // Detect packed-ness once, so we can check for it within RepeatedField<T>.
340  PackedRepeatedField = IsPackedRepeatedField(tag);
341  }
342 
346  public void WriteTagAndValue(CodedOutputStream output, T value)
347  {
348  if (!IsDefault(value))
349  {
350  output.WriteTag(Tag);
351  ValueWriter(output, value);
352  }
353  }
354 
360  public T Read(CodedInputStream input)
361  {
362  return ValueReader(input);
363  }
364 
369  public int CalculateSizeWithTag(T value)
370  {
371  return IsDefault(value) ? 0 : ValueSizeCalculator(value) + tagSize;
372  }
373 
374  private bool IsDefault(T value)
375  {
376  return EqualityComparer<T>.Default.Equals(value, DefaultValue);
377  }
378  }
