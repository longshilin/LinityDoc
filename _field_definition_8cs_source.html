248  public override RenderedFieldDefinition Render(IBsonSerializer<TDocument> documentSerializer, IBsonSerializerRegistry serializerRegistry)
249  {
250  var bindingContext = new PipelineBindingContext(serializerRegistry);
251  var lambda = ExpressionHelper.GetLambda(PartialEvaluator.Evaluate(_expression));
252  var parameterExpression = new DocumentExpression(documentSerializer);
253  bindingContext.AddExpressionMapping(lambda.Parameters[0], parameterExpression);
254  var bound = bindingContext.Bind(lambda.Body);
255  bound = FieldExpressionFlattener.FlattenFields(bound);
256  IFieldExpression field;
257  if (!ExpressionHelper.TryGetExpression(bound, out field))
258  {
259  var message = string.Format("Unable to determine the serialization information for {0}.", _expression);
260  throw new InvalidOperationException(message);
261  }
262 
263  return new RenderedFieldDefinition(field.FieldName, field.Serializer);
264  }

369  string resolvedName;
370  IBsonSerializer underlyingSerializer;
371  StringFieldDefinitionHelper.Resolve<TDocument>(_fieldName, documentSerializer, out resolvedName, out underlyingSerializer);
372

374 
375  IBsonSerializer<TField> valueSerializer;
376  if (_fieldSerializer != null)
377  {
378  valueSerializer = _fieldSerializer;

387  }
388 
389  return new RenderedFieldDefinition<TField>(resolvedName, fieldSerializer, valueSerializer, underlyingSerializer);

393  internal static class StringFieldDefinitionHelper
394  {
395  public static void Resolve<TDocument>(string fieldName, IBsonSerializer<TDocument> serializer, out string resolvedFieldName, out IBsonSerializer resolvedFieldSerializer)
396  {
397  resolvedFieldName = fieldName;
398  resolvedFieldSerializer = null;
399 
400  var documentSerializer = serializer as IBsonDocumentSerializer;
401  if (documentSerializer == null)
402  {
403  return;
404  }
405 
406  // shortcut BsonDocumentSerializer since it is so common
407  if (serializer.GetType() == typeof(BsonDocumentSerializer))
408  {
409  return;
410  }
411 
412  BsonSerializationInfo serializationInfo;
413 
414  // first, lets try the quick and easy one, which will be a majority of cases
415  if (documentSerializer.TryGetMemberSerializationInfo(fieldName, out serializationInfo))
416  {
417  resolvedFieldName = serializationInfo.ElementName;
418  resolvedFieldSerializer = serializationInfo.Serializer;
419  return;
420  }
421 
422  // now lets go and do the more difficult variant
423  var nameParts = fieldName.Split('.');
424  if (nameParts.Length <= 1)
425  {
426  // if we only have 1, then it's no different than what we did above
427  // when we found nothing.
428  return;
429  }
430 
431  IBsonArraySerializer arraySerializer;
432  resolvedFieldSerializer = documentSerializer;
433  for (int i = 0; i < nameParts.Length; i++)
434  {
435  if (nameParts[i] == "$" || nameParts[i].All(char.IsDigit))
436  {
437  arraySerializer = resolvedFieldSerializer as IBsonArraySerializer;
438  if (arraySerializer != null && arraySerializer.TryGetItemSerializationInfo(out serializationInfo))
439  {
440  resolvedFieldSerializer = serializationInfo.Serializer;
441  continue;
442  }
443 
444  resolvedFieldSerializer = null;
445  break;
446  }
447 
448  documentSerializer = resolvedFieldSerializer as IBsonDocumentSerializer;
449  if (documentSerializer == null || !documentSerializer.TryGetMemberSerializationInfo(nameParts[i], out serializationInfo))
450  {
451  // need to check if this is an any element array match
452  arraySerializer = resolvedFieldSerializer as IBsonArraySerializer;
453  if (arraySerializer != null && arraySerializer.TryGetItemSerializationInfo(out serializationInfo))
454  {
455  documentSerializer = serializationInfo.Serializer as IBsonDocumentSerializer;
456  if (documentSerializer == null || !documentSerializer.TryGetMemberSerializationInfo(nameParts[i], out serializationInfo))
457  {
458  resolvedFieldSerializer = null;
459  break;
460  }
461  }
462  else
463  {
464  resolvedFieldSerializer = null;
465  break;
466  }
467  }
468 
469  nameParts[i] = serializationInfo.ElementName;
470  resolvedFieldSerializer = serializationInfo.Serializer;
471  }
472 
473  resolvedFieldName = string.Join(".", nameParts);
474  }
475  }
