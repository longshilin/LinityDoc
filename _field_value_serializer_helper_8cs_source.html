34  private static IBsonSerializer GetSerializerForValueType(IBsonSerializer fieldSerializer, IBsonSerializerRegistry serializerRegistry, Type valueType, int recursionLevel)
35  {
36  if (recursionLevel > 1)
37  {
38  throw new ArgumentException("Unexpectedly high recursion level.", nameof(recursionLevel));
39  }
40 
41  var fieldType = fieldSerializer.ValueType;
42 
43  // these will normally be equal unless we've removed some Convert(s) that the compiler put in
44  if (fieldType == valueType)
45  {
46  return fieldSerializer;
47  }
48 
49  // serialize numeric values without converting them
50  if (fieldType.IsNumeric() && valueType.IsNumeric())
51  {
52  var valueSerializer = BsonSerializer.SerializerRegistry.GetSerializer(valueType);
53  if (HasStringRepresentation(fieldSerializer))
54  {
55  valueSerializer = WithStringRepresentation(valueSerializer);
56  }
57  return valueSerializer;
58  }
59 
60  var fieldTypeInfo = fieldType.GetTypeInfo();
61  var fieldSerializerInterfaceType = typeof(IBsonSerializer<>).MakeGenericType(fieldType);
62  var valueTypeInfo = valueType.GetTypeInfo();
63 
64  // synthesize a NullableSerializer using the field serializer
65  if (valueType.IsNullable() && valueType.GetNullableUnderlyingType() == fieldType)
66  {
67  var nullableSerializerType = typeof(NullableSerializer<>).MakeGenericType(fieldType);
68  var nullableSerializerConstructor = nullableSerializerType.GetTypeInfo().GetConstructor(new[] { fieldSerializerInterfaceType });
69  return (IBsonSerializer)nullableSerializerConstructor.Invoke(new object[] { fieldSerializer });
70  }
71 
72  // synthesize an EnumConvertingSerializer using the field serializer
73  if (fieldTypeInfo.IsEnum)
74  {
75  if (valueType.IsConvertibleToEnum())
76  {
77  var enumConvertingSerializerType = typeof(EnumConvertingSerializer<,>).MakeGenericType(valueType, fieldType);
78  var enumConvertingSerializerConstructor = enumConvertingSerializerType.GetTypeInfo().GetConstructor(new[] { fieldSerializerInterfaceType });
79  return (IBsonSerializer)enumConvertingSerializerConstructor.Invoke(new object[] { fieldSerializer });
80  }
81 
82  if (valueType.IsNullable() && valueType.GetNullableUnderlyingType().IsConvertibleToEnum())
83  {
84  var underlyingValueType = valueType.GetNullableUnderlyingType();
85  var underlyingValueSerializerInterfaceType = typeof(IBsonSerializer<>).MakeGenericType(underlyingValueType);
86  var enumConvertingSerializerType = typeof(EnumConvertingSerializer<,>).MakeGenericType(underlyingValueType, fieldType);
87  var enumConvertingSerializerConstructor = enumConvertingSerializerType.GetTypeInfo().GetConstructor(new[] { fieldSerializerInterfaceType });
88  var enumConvertingSerializer = enumConvertingSerializerConstructor.Invoke(new object[] { fieldSerializer });
89  var nullableSerializerType = typeof(NullableSerializer<>).MakeGenericType(underlyingValueType);
90  var nullableSerializerConstructor = nullableSerializerType.GetTypeInfo().GetConstructor(new[] { underlyingValueSerializerInterfaceType });
91  return (IBsonSerializer)nullableSerializerConstructor.Invoke(new object[] { enumConvertingSerializer });
92  }
93  }
94 
95  // synthesize a NullableEnumConvertingSerializer using the field serializer
96  if (fieldType.IsNullableEnum() && valueType.IsNullable())
97  {
98  var nonNullableFieldType = fieldType.GetNullableUnderlyingType();
99  var nonNullableValueType = valueType.GetNullableUnderlyingType();
100  var nonNullableFieldSerializer = ((IChildSerializerConfigurable)fieldSerializer).ChildSerializer;
101  var nonNullableFieldSerializerInterfaceType = typeof(IBsonSerializer<>).MakeGenericType(nonNullableFieldType);
102  var nullableEnumConvertingSerializerType = typeof(NullableEnumConvertingSerializer<,>).MakeGenericType(nonNullableValueType, nonNullableFieldType);
103  var nullableEnumConvertingSerializerConstructor = nullableEnumConvertingSerializerType.GetTypeInfo().GetConstructor(new[] { nonNullableFieldSerializerInterfaceType });
104  return (IBsonSerializer)nullableEnumConvertingSerializerConstructor.Invoke(new object[] { nonNullableFieldSerializer });
105  }
106 
107  // synthesize an IEnumerableSerializer serializer using the item serializer from the field serializer
108  Type fieldIEnumerableInterfaceType;
109  Type valueIEnumerableInterfaceType;
110  Type itemType;
111  if (
112  (fieldIEnumerableInterfaceType = fieldType.FindIEnumerable()) != null &&
113  (valueIEnumerableInterfaceType = valueType.FindIEnumerable()) != null &&
114  (itemType = fieldIEnumerableInterfaceType.GetSequenceElementType()) == valueIEnumerableInterfaceType.GetSequenceElementType() &&
115  fieldSerializer is IChildSerializerConfigurable)
116  {
117  var itemSerializer = ((IChildSerializerConfigurable)fieldSerializer).ChildSerializer;
118  var itemSerializerInterfaceType = typeof(IBsonSerializer<>).MakeGenericType(itemType);
119  var ienumerableSerializerType = typeof(IEnumerableSerializer<>).MakeGenericType(itemType);
120  var ienumerableSerializerConstructor = ienumerableSerializerType.GetTypeInfo().GetConstructor(new[] { itemSerializerInterfaceType });
121  return (IBsonSerializer)ienumerableSerializerConstructor.Invoke(new object[] { itemSerializer });
122  }
123 
124  // if the fieldSerializer is an array serializer try to adapt its itemSerializer for valueType
125  IBsonArraySerializer arraySerializer;
126  if ((arraySerializer = fieldSerializer as IBsonArraySerializer) != null)
127  {
128  BsonSerializationInfo itemSerializationInfo;
129  if (arraySerializer.TryGetItemSerializationInfo(out itemSerializationInfo))
130  {
131  if (recursionLevel == 0)
132  {
133  var itemSerializer = itemSerializationInfo.Serializer;
134  return GetSerializerForValueType(itemSerializer, serializerRegistry, valueType, recursionLevel + 1);
135  }
136  }
137  }
138 
139  // if we can't return a value serializer based on the field serializer return a converting serializer
140  var convertIfPossibleSerializerType = typeof(ConvertIfPossibleSerializer<,>).MakeGenericType(valueType, fieldType);
141  var convertIfPossibleSerializerConstructor = convertIfPossibleSerializerType.GetTypeInfo().GetConstructor(new[] { fieldSerializerInterfaceType, typeof(IBsonSerializerRegistry) });
142  return (IBsonSerializer)convertIfPossibleSerializerConstructor.Invoke(new object[] { fieldSerializer, serializerRegistry });
143  }

145  public static IBsonSerializer GetSerializerForValueType(IBsonSerializer fieldSerializer, IBsonSerializerRegistry serializerRegistry, Type valueType, object value)
146  {
147  if (!valueType.GetTypeInfo().IsValueType && value == null)
148  {
149  return fieldSerializer;
150  }
151  else
152  {
153  return GetSerializerForValueType(fieldSerializer, serializerRegistry, valueType);
154  }
155  }

158  private static bool HasStringRepresentation(IBsonSerializer serializer)
159  {
160  var configurableSerializer = serializer as IRepresentationConfigurable;
161  if (configurableSerializer != null)
162  {
163  return configurableSerializer.Representation == BsonType.String;
164  }
165  else
166  {
167  return false;
168  }
169  }

171  private static IBsonSerializer WithStringRepresentation(IBsonSerializer serializer)
172  {
173  var configurableSerializer = serializer as IRepresentationConfigurable;
174  if (configurableSerializer != null)
175  {
176  return configurableSerializer.WithRepresentation(BsonType.String);
177  }
178  else
179  {
180  return serializer;
181  }

196  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, TFrom value)
197  {
198  TTo convertedValue;
199  if (TryConvertValue(value, out convertedValue))
200  {
201  args.NominalType = typeof(TTo);
202  _serializer.Serialize(context, args, convertedValue);
203  }
204  else
205  {
206  var serializer = _serializerRegistry.GetSerializer<TFrom>();
207  serializer.Serialize(context, args, value);
208  }
209  }

211  private bool TryConvertValue(TFrom value, out TTo convertedValue)
212  {
213  if (object.ReferenceEquals(value, null))
214  {
215  convertedValue = (TTo)(object)null;
216  return true;
217  }
218 
219  Type fromType = value.GetType();
220  Type toType = typeof(TTo);
221 
222  if (toType.GetTypeInfo().IsAssignableFrom(fromType))
223  {
224  convertedValue = (TTo)(object)value;
225  return true;
226  }
227 
228  var toConverter = TypeDescriptor.GetConverter(toType);
229  if (toConverter.CanConvertFrom(fromType))
230  {
231  convertedValue = (TTo)toConverter.ConvertFrom(value);
232  return true;
233  }
234 
235  var fromConverter = TypeDescriptor.GetConverter(fromType);
236  if (fromConverter.CanConvertTo(toType))
237  {
238  convertedValue = (TTo)fromConverter.ConvertTo(value, toType);
239  return true;
240  }
241 
242  try
243  {
244  convertedValue = (TTo)Convert.ChangeType(value, toType);
245  return true;
246  }
247  catch { }
248 
249  convertedValue = default(TTo);
250  return false;
251  }
252  }

264  {
265  TTo convertedValue;
266  if (typeof(TFrom) == typeof(string))
267  {
268  convertedValue = (TTo)Enum.Parse(typeof(TTo), (string)(object)value);
269  }
270  else

272  convertedValue = (TTo)Enum.ToObject(typeof(TTo), (object)value);
273  }

287  public override void Serialize(BsonSerializationContext context, BsonSerializationArgs args, IEnumerable<TItem> value)
288  {
289  var bsonWriter = context.Writer;
290  if (value == null)
291  {
292  bsonWriter.WriteNull();
293  }
294  else
295  {
296  bsonWriter.WriteStartArray();
297  foreach (var item in value)
298  {
299  _itemSerializer.Serialize(context, item);
300  }
301  bsonWriter.WriteEndArray();
302  }
303  }
