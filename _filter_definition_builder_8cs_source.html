1524  private static void AddClause(BsonDocument document, BsonElement clause)
1525  {
1526  if (clause.Name == "$and")

1528  // flatten out nested $and
1529  foreach (var item in (BsonArray)clause.Value)
1530  {
1531  foreach (var element in (BsonDocument)item)
1532  {
1533  AddClause(document, element);
1534  }
1535  }
1536  }
1537  else if (document.ElementCount == 1 && document.GetElement(0).Name == "$and")
1538  {

1556  }
1557  else
