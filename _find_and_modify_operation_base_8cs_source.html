152  {

204  {

208  GuidRepresentation = _messageEncoderSettings.GetOrDefault<GuidRepresentation>(MessageEncoderSettingsName.GuidRepresentation, GuidRepresentation.CSharpLegacy)

210 
211  using (var stream = new ByteBufferStream(rawBsonDocument.Slice, ownsBuffer: false))
212  using (var reader = new BsonBinaryReader(stream, binaryReaderSettings))
