419  internal BsonDocument CreateCommand(ConnectionDescription connectionDescription, ICoreSession session)
420  {
421  Feature.ReadConcern.ThrowIfNotSupported(connectionDescription.ServerVersion, _readConcern);
422  Feature.Collation.ThrowIfNotSupported(connectionDescription.ServerVersion, _collation);
423 
424  var firstBatchSize = _firstBatchSize ?? (_batchSize > 0 ? _batchSize : null);
425  var isShardRouter = connectionDescription.IsMasterResult.ServerType == ServerType.ShardRouter;
426 
427  var readConcern = ReadConcernHelper.GetReadConcernForCommand(session, connectionDescription, _readConcern);
428  return new BsonDocument
429  {
430  { "find", _collectionNamespace.CollectionName },
431  { "filter", _filter, _filter != null },
432  { "sort", _sort, _sort != null },
433  { "projection", _projection, _projection != null },
434  { "hint", _hint, _hint != null },
435  { "skip", () => _skip.Value, _skip.HasValue },
436  { "limit", () => Math.Abs(_limit.Value), _limit.HasValue && _limit != 0 },
437  { "batchSize", () => firstBatchSize.Value, firstBatchSize.HasValue },
438  { "singleBatch", () => _limit < 0 || _singleBatch.Value, _limit < 0 || _singleBatch.HasValue },
439  { "comment", _comment, _comment != null },
440  { "maxScan", () => _maxScan.Value, _maxScan.HasValue },
441  { "maxTimeMS", () => MaxTimeHelper.ToMaxTimeMS(_maxTime.Value), _maxTime.HasValue },
442  { "max", _max, _max != null },
443  { "min", _min, _min != null },
444  { "returnKey", () => _returnKey.Value, _returnKey.HasValue },
445  { "showRecordId", () => _showRecordId.Value, _showRecordId.HasValue },
446  { "snapshot", () => _snapshot.Value, _snapshot.HasValue },
447  { "tailable", true, _cursorType == CursorType.Tailable || _cursorType == CursorType.TailableAwait },
448  { "oplogReplay", () => _oplogReplay.Value, _oplogReplay.HasValue },
449  { "noCursorTimeout", () => _noCursorTimeout.Value, _noCursorTimeout.HasValue },
450  { "awaitData", true, _cursorType == CursorType.TailableAwait },
451  { "allowPartialResults", () => _allowPartialResults.Value, _allowPartialResults.HasValue && isShardRouter },
452  { "collation", () => _collation.ToBsonDocument(), _collation != null },
453  { "readConcern", readConcern, readConcern != null }
454  };
455  }

457  private AsyncCursor<TDocument> CreateCursor(IChannelSourceHandle channelSource, BsonDocument commandResult)
458  {
459  var getMoreChannelSource = new ServerChannelSource(channelSource.Server, channelSource.Session.Fork());
460  var firstBatch = CreateCursorBatch(commandResult);
461 
462  return new AsyncCursor<TDocument>(
463  getMoreChannelSource,
464  _collectionNamespace,
465  _filter ?? new BsonDocument(),
466  firstBatch.Documents,
467  firstBatch.CursorId,
468  _batchSize,
469  _limit < 0 ? Math.Abs(_limit.Value) : _limit,
470  _resultSerializer,
471  _messageEncoderSettings,
472  _cursorType == CursorType.TailableAwait ? _maxAwaitTime : null);
473  }

475  private CursorBatch<TDocument> CreateCursorBatch(BsonDocument commandResult)
476  {
477  var cursorDocument = commandResult["cursor"].AsBsonDocument;
478  var cursorId = cursorDocument["id"].ToInt64();
479  var batch = (RawBsonArray)cursorDocument["firstBatch"];
480 
481  using (batch)
482  {
483  var documents = CursorBatchDeserializationHelper.DeserializeBatch(batch, _resultSerializer, _messageEncoderSettings);
484  return new CursorBatch<TDocument>(cursorId, documents);
485  }

490  {
491  Ensure.IsNotNull(binding, nameof(binding));
492 
493  using (EventContext.BeginOperation())
494  using (var channelSource = binding.GetReadChannelSource(cancellationToken))
495  using (var channel = channelSource.GetChannel(cancellationToken))
496  using (var channelBinding = new ChannelReadBinding(channelSource.Server, channel, binding.ReadPreference, binding.Session.Fork()))
497  {
498  var readPreference = binding.ReadPreference;
499 
500  using (EventContext.BeginFind(_batchSize, _limit))
501  {
502  var operation = CreateOperation(channel, channelBinding);
503  var commandResult = operation.Execute(channelBinding, cancellationToken);
504  return CreateCursor(channelSource, commandResult);
