404  internal BsonDocument CreateWrappedQuery(ServerType serverType, ReadPreference readPreference)
405  {
406  var readPreferenceDocument = QueryHelper.CreateReadPreferenceDocument(serverType, readPreference);
407 
408  var wrappedQuery = new BsonDocument
409  {
410  { "$query", _filter ?? new BsonDocument() },
411  { "$readPreference", readPreferenceDocument, readPreferenceDocument != null },
412  { "$orderby", _sort, _sort != null },
413  { "$comment", _comment, _comment != null },
414  { "$maxTimeMS", () => MaxTimeHelper.ToMaxTimeMS(_maxTime.Value), _maxTime.HasValue },
415  { "$hint", _hint, _hint != null },
416  { "$max", _max, _max != null },
417  { "$maxScan", () => _maxScan.Value, _maxScan.HasValue },
418  { "$min", _min, _min != null },
419  { "$showDiskLoc", () => _showRecordId.Value, _showRecordId.HasValue },
420  { "$snapshot", () => _snapshot.Value, _snapshot.HasValue }
421  };
422 
423  if (_modifiers != null)
424  {
425  wrappedQuery.Merge(_modifiers, overwriteExistingElements: false);
426  }
427 
428  if (wrappedQuery.ElementCount == 1 && wrappedQuery.GetElement(0).Name == "$query")
429  {
430  var unwrappedQuery = wrappedQuery[0].AsBsonDocument;
431  if (!unwrappedQuery.Contains("query"))
432  {
433  return unwrappedQuery;
434  }
435  }
436 
437  return wrappedQuery;
438  }

441  public IAsyncCursor<TDocument> Execute(IReadBinding binding, CancellationToken cancellationToken)
442  {
443  Ensure.IsNotNull(binding, nameof(binding));
444 
445  using (EventContext.BeginOperation())
446  using (var channelSource = binding.GetReadChannelSource(cancellationToken))
447  using (var channel = channelSource.GetChannel(cancellationToken))
448  {
449  var readPreference = binding.ReadPreference;
450  var serverDescription = channelSource.ServerDescription;
451  var wrappedQuery = CreateWrappedQuery(serverDescription.Type, readPreference);
452  var slaveOk = readPreference != null && readPreference.ReadPreferenceMode != ReadPreferenceMode.Primary;
453 
454  using (EventContext.BeginFind(_batchSize, _limit))
455  {
456  var batch = ExecuteProtocol(channel, wrappedQuery, slaveOk, cancellationToken);
457  return CreateCursor(channelSource, wrappedQuery, batch);
458  }
459  }
460  }
