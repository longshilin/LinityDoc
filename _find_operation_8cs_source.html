447  internal FindCommandOperation<TDocument> CreateFindCommandOperation()
448  {
449  var comment = _comment;
450  var hint = _hint;
451  var max = _max;
452  var maxScan = _maxScan;
453  var maxTime = _maxTime;
454  var min = _min;
455  var returnKey = _returnKey;
456  var showRecordId = _showRecordId;
457  var snapshot = _snapshot;
458  var sort = _sort;
459 
460  if (_modifiers != null)
461  {
462  foreach (var element in _modifiers)
463  {
464  var value = element.Value;
465  switch (element.Name)
466  {
467  case "$comment": comment = _comment ?? value.AsString; break;
468  case "$hint": hint = _hint ?? value; break;
469  case "$max": max = _max ?? value.AsBsonDocument; break;
470  case "$maxScan": maxScan = _maxScan ?? value.ToInt32(); break;
471  case "$maxTimeMS": maxTime = _maxTime ?? TimeSpan.FromMilliseconds(value.ToDouble()); break;
472  case "$min": min = _min ?? value.AsBsonDocument; break;
473  case "$orderby": sort = _sort ?? value.AsBsonDocument; break;
474  case "$returnKey": returnKey = _returnKey ?? value.ToBoolean(); break;
475  case "$showDiskLoc": showRecordId = _showRecordId ?? value.ToBoolean(); break;
476  case "$snapshot": snapshot = _snapshot ?? value.ToBoolean(); break;
477  default: throw new ArgumentException($"Modifier not supported by the Find command: '{element.Name}'.");
478  }
479  }
480  }

565  {
