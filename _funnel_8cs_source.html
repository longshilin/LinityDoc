31  public static List<PathPart> SplitIntoParts (Path path) {
32  var nodes = path.path;
33 
34  var result = ListPool<PathPart>.Claim();
35 
36  if (nodes == null || nodes.Count == 0) {
37  return result;
38  }
39 
40  // Loop through the path and split it into
41  // parts joined by links
42  for (int i = 0; i < nodes.Count; i++) {
43  if (nodes[i] is TriangleMeshNode) {
44  var part = new PathPart();
45  part.startIndex = i;
46  uint currentGraphIndex = nodes[i].GraphIndex;
47 
48  // Loop up until we find a node in another graph
49  // Ignore NodeLink3 nodes
50  for (; i < nodes.Count; i++) {
51  if (nodes[i].GraphIndex != currentGraphIndex) {
52  break;
53  }
54  }
55 
56  i--;
57  part.endIndex = i;
58 
59  // If this is the first part in the path, use the exact start point
60  // otherwise use the position of the node right before the start of this
61  // part which is likely the end of the link to this part
62  if (part.startIndex == 0) {
63  part.startPoint = path.vectorPath[0];
64  } else {
65  part.startPoint = (Vector3)nodes[part.startIndex-1].position;
66  }
67 
68  if (part.endIndex == nodes.Count-1) {
69  part.endPoint = path.vectorPath[path.vectorPath.Count-1];
70  } else {
71  part.endPoint = (Vector3)nodes[part.endIndex+1].position;
72  }
73 
74  result.Add(part);
75  } else {
76  throw new System.Exception("Unsupported node type or null node");
77  }
78  }

83  public static FunnelPortals ConstructFunnelPortals (List<GraphNode> nodes, PathPart part) {
84  if (nodes == null || nodes.Count == 0) {
85  return new FunnelPortals { left = ListPool<Vector3>.Claim(0), right = ListPool<Vector3>.Claim(0) };
86  }
87 
88  if (part.endIndex < part.startIndex || part.startIndex < 0 || part.endIndex > nodes.Count) throw new System.ArgumentOutOfRangeException();
89 
90  // Claim temporary lists and try to find lists with a high capacity
91  var left = ListPool<Vector3>.Claim(nodes.Count+1);
92  var right = ListPool<Vector3>.Claim(nodes.Count+1);
93 
94  // Add start point
95  left.Add(part.startPoint);
96  right.Add(part.startPoint);
97 
98  // Loop through all nodes in the path (except the last one)
99  for (int i = part.startIndex; i < part.endIndex; i++) {
100  // Get the portal between path[i] and path[i+1] and add it to the left and right lists
101  bool portalWasAdded = nodes[i].GetPortal(nodes[i+1], left, right, false);
102 
103  if (!portalWasAdded) {
104  // Fallback, just use the positions of the nodes
105  left.Add((Vector3)nodes[i].position);
106  right.Add((Vector3)nodes[i].position);
107 
108  left.Add((Vector3)nodes[i+1].position);
109  right.Add((Vector3)nodes[i+1].position);
110  }
111  }
112 
113  // Add end point
114  left.Add(part.endPoint);
115  right.Add(part.endPoint);
116 
117  return new FunnelPortals { left = left, right = right };

120  public static void ShrinkPortals (FunnelPortals portals, float shrink) {
121  if (shrink <= 0.00001f) return;
122 
123  for (int i = 0; i < portals.left.Count; i++) {
124  var left = portals.left[i];
125  var right = portals.right[i];
126 
127  var length = (left - right).magnitude;
128  if (length > 0) {
129  float s = Mathf.Min(shrink / length, 0.4f);
130  portals.left[i] = Vector3.Lerp(left, right, s);
131  portals.right[i] = Vector3.Lerp(left, right, 1 - s);
132  }

136  static bool UnwrapHelper (Vector3 portalStart, Vector3 portalEnd, Vector3 prevPoint, Vector3 nextPoint, ref Quaternion mRot, ref Vector3 mOffset) {
137  // Skip the point if it was on the rotation axis
138  if (VectorMath.IsColinear(portalStart, portalEnd, nextPoint)) {
139  return false;
140  }
141 
142  var axis = portalEnd - portalStart;
143  var sqrMagn = axis.sqrMagnitude;
144  prevPoint -= Vector3.Dot(prevPoint - portalStart, axis)/sqrMagn * axis;
145  nextPoint -= Vector3.Dot(nextPoint - portalStart, axis)/sqrMagn * axis;
146  var rot = Quaternion.FromToRotation(nextPoint - portalStart, portalStart - prevPoint);
147 
148  // The code below is equivalent to these matrix operations (but a lot faster)
149  // This represents a rotation around a line in 3D space
150  //mat = mat * Matrix4x4.TRS(portalStart, rot, Vector3.one) * Matrix4x4.TRS(-portalStart, Quaternion.identity, Vector3.one);
151  mOffset += mRot * (portalStart - rot * portalStart);
152  mRot *= rot;
153 
154  return true;
155  }

167  public static void Unwrap (FunnelPortals funnel, Vector2[] left, Vector2[] right) {
168  int startingIndex = 1;
169  var normal = Vector3.Cross(funnel.right[1] - funnel.left[0], funnel.left[1] - funnel.left[0]);
170 
171  // This handles the case when the starting point is colinear with the first portal.
172  // Note that left.Length is only guaranteed to be at least as large as funnel.left.Count, it may be larger.
173  while (normal.sqrMagnitude <= 0.00000001f && startingIndex + 1 < funnel.left.Count) {
174  startingIndex++;
175  normal = Vector3.Cross(funnel.right[startingIndex] - funnel.left[0], funnel.left[startingIndex] - funnel.left[0]);
176  }
177 
178  left[0] = right[0] = Vector2.zero;
179 
180  var portalLeft = funnel.left[1];
181  var portalRight = funnel.right[1];
182  var prevPoint = funnel.left[0];
183 
184  // The code below is equivalent to this matrix (but a lot faster)
185  // This represents a rotation around a line in 3D space
186  // Matrix4x4 m = Matrix4x4.TRS(Vector3.zero, Quaternion.FromToRotation(normal, Vector3.forward), Vector3.one) * Matrix4x4.TRS(-funnel.right[0], Quaternion.identity, Vector3.one);
187  Quaternion mRot = Quaternion.FromToRotation(normal, Vector3.forward);
188  Vector3 mOffset = mRot * (-funnel.right[0]);
189 
190  for (int i = 1; i < funnel.left.Count; i++) {
191  if (UnwrapHelper(portalLeft, portalRight, prevPoint, funnel.left[i], ref mRot, ref mOffset)) {
192  prevPoint = portalLeft;
193  portalLeft = funnel.left[i];
194  }
195 
196  left[i] = (mRot * funnel.left[i] + mOffset);
197 
198  if (UnwrapHelper(portalLeft, portalRight, prevPoint, funnel.right[i], ref mRot, ref mOffset)) {
199  prevPoint = portalRight;
200  portalRight = funnel.right[i];
201  }
202 
203  right[i] = (mRot * funnel.right[i] + mOffset);
204  }
205  }

210  static int FixFunnel (Vector2[] left, Vector2[] right, int numPortals) {
211  if (numPortals > left.Length || numPortals > right.Length) throw new System.ArgumentException("Arrays do not have as many elements as specified");
212 
213  if (numPortals < 3) {
214  return -1;
215  }
216 
217  // Remove duplicate vertices
218  int startIndex = 0;
219  while (left[startIndex + 1] == left[startIndex + 2] && right[startIndex + 1] == right[startIndex + 2]) {
220  // Equivalent to RemoveAt(1) if they would have been lists
221  left[startIndex + 1] = left[startIndex + 0];
222  right[startIndex + 1] = right[startIndex + 0];
223  startIndex++;
224 
225  if (numPortals - startIndex < 3) {
226  return -1;
227  }
228  }
229 
230  return startIndex;

268  public static List<Vector3> Calculate (FunnelPortals funnel, bool unwrap, bool splitAtEveryPortal) {
269  if (funnel.left.Count != funnel.right.Count) throw new System.ArgumentException("funnel.left.Count != funnel.right.Count");
270 
271  // Get arrays at least as large as the number of portals
272  var leftArr = ArrayPool<Vector2>.Claim(funnel.left.Count);
273  var rightArr = ArrayPool<Vector2>.Claim(funnel.left.Count);
274 
275  if (unwrap) {
276  Unwrap(funnel, leftArr, rightArr);
277  } else {
278  // Copy to arrays
279  for (int i = 0; i < funnel.left.Count; i++) {
280  leftArr[i] = ToXZ(funnel.left[i]);
281  rightArr[i] = ToXZ(funnel.right[i]);
282  }
283  }
284 
285  int startIndex = FixFunnel(leftArr, rightArr, funnel.left.Count);
286  var intermediateResult = ListPool<int>.Claim();
287  if (startIndex == -1) {
288  // If funnel algorithm failed, fall back to a simple line
289  intermediateResult.Add(0);
290  intermediateResult.Add(funnel.left.Count - 1);
291  } else {
292  bool lastCorner;
293  Calculate(leftArr, rightArr, funnel.left.Count, startIndex, intermediateResult, int.MaxValue, out lastCorner);
294  }
295 
296  // Get list for the final result
297  var result = ListPool<Vector3>.Claim(intermediateResult.Count);
298 
299  Vector2 prev2D = leftArr[0];
300  var prevIdx = 0;
301  for (int i = 0; i < intermediateResult.Count; i++) {
302  var idx = intermediateResult[i];
303 
304  if (splitAtEveryPortal) {
305  // Check intersections with every portal segment
306  var next2D = idx >= 0 ? leftArr[idx] : rightArr[-idx];
307  for (int j = prevIdx + 1; j < System.Math.Abs(idx); j++) {
308  var factor = VectorMath.LineIntersectionFactorXZ(FromXZ(leftArr[j]), FromXZ(rightArr[j]), FromXZ(prev2D), FromXZ(next2D));
309  result.Add(Vector3.Lerp(funnel.left[j], funnel.right[j], factor));
310  }
311 
312  prevIdx = Mathf.Abs(idx);
313  prev2D = next2D;
314  }
315 
316  if (idx >= 0) {
317  result.Add(funnel.left[idx]);
318  } else {
319  result.Add(funnel.right[-idx]);
320  }
321  }
322 
323  // Release lists back to the pool
324  ListPool<int>.Release(ref intermediateResult);
325  ArrayPool<Vector2>.Release(ref leftArr);
326  ArrayPool<Vector2>.Release(ref rightArr);
327  return result;
328  }

338  static void Calculate (Vector2[] left, Vector2[] right, int numPortals, int startIndex, List<int> funnelPath, int maxCorners, out bool lastCorner) {
339  if (left.Length != right.Length) throw new System.ArgumentException();
340 
341  lastCorner = false;
342 
343  int apexIndex = startIndex + 0;
344  int rightIndex = startIndex + 1;
345  int leftIndex = startIndex + 1;
346 
347  Vector2 portalApex = left[apexIndex];
348  Vector2 portalLeft = left[leftIndex];
349  Vector2 portalRight = right[rightIndex];
350 
351  funnelPath.Add(apexIndex);
352 
353  for (int i = startIndex + 2; i < numPortals; i++) {
354  if (funnelPath.Count >= maxCorners) {
355  return;
356  }
357 
358  if (funnelPath.Count > 2000) {
359 #if !SERVER
360  UnityEngine.Debug.LogWarning("Avoiding infinite loop. Remove this check if you have this long paths.");
361 #endif
362  break;
363  }
364 
365  Vector2 pLeft = left[i];
366  Vector2 pRight = right[i];
367 
368  if (LeftOrColinear(portalRight - portalApex, pRight - portalApex)) {
369  if (portalApex == portalRight || RightOrColinear(portalLeft - portalApex, pRight - portalApex)) {
370  portalRight = pRight;
371  rightIndex = i;
372  } else {
373  portalApex = portalRight = portalLeft;
374  i = apexIndex = rightIndex = leftIndex;
375 
376  funnelPath.Add(apexIndex);
377  continue;
378  }
379  }
380 
381  if (RightOrColinear(portalLeft - portalApex, pLeft - portalApex)) {
382  if (portalApex == portalLeft || LeftOrColinear(portalRight - portalApex, pLeft - portalApex)) {
383  portalLeft = pLeft;
384  leftIndex = i;
385  } else {
386  portalApex = portalLeft = portalRight;
387  i = apexIndex = leftIndex = rightIndex;
388 
389  // Negative value because we are referring
390  // to the right side
391  funnelPath.Add(-apexIndex);
392 
393  continue;
394  }
395  }
396  }
397 
398  lastCorner = true;
399  funnelPath.Add(numPortals-1);
400  }
401  }
