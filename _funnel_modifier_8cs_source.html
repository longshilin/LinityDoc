56  public override void Apply (Path p) {
57  if (p.path == null || p.path.Count == 0 || p.vectorPath == null || p.vectorPath.Count == 0) {
58  return;
59  }
60 
61  List<Vector3> funnelPath = ListPool<Vector3>.Claim();
62 
63  // Split the path into different parts (separated by custom links)
64  // and run the funnel algorithm on each of them in turn
65  var parts = Funnel.SplitIntoParts(p);
66 
67  if (parts.Count == 0) {
68  // As a really special case, it might happen that the path contained only a single node
69  // and that node was part of a custom link (e.g added by the NodeLink2 component).
70  // In that case the SplitIntoParts method will not know what to do with it because it is
71  // neither a link (as only 1 of the 2 nodes of the link was part of the path) nor a normal
72  // path part. So it will skip it. This will cause it to return an empty list.
73  // In that case we want to simply keep the original path, which is just a single point.
74  return;
75  }
76 
77  for (int i = 0; i < parts.Count; i++) {
78  var part = parts[i];
79  if (!part.isLink) {
80  var portals = Funnel.ConstructFunnelPortals(p.path, part);
81  var result = Funnel.Calculate(portals, unwrap, splitAtEveryPortal);
82  funnelPath.AddRange(result);
83  ListPool<Vector3>.Release(ref portals.left);
84  ListPool<Vector3>.Release(ref portals.right);
85  ListPool<Vector3>.Release(ref result);
86  } else {
87  // non-link parts will add the start/end points for the adjacent parts.
88  // So if there is no non-link part before this one, then we need to add the start point of the link
89  // and if there is no non-link part after this one, then we need to add the end point.
90  if (i == 0 || parts[i-1].isLink) {
91  funnelPath.Add(part.startPoint);
92  }
93  if (i == parts.Count - 1 || parts[i+1].isLink) {
94  funnelPath.Add(part.endPoint);
95  }
96  }
97  }
98 
99  UnityEngine.Assertions.Assert.IsTrue(funnelPath.Count >= 1);
100  ListPool<Funnel.PathPart>.Release(ref parts);
101  // Pool the previous vectorPath
102  ListPool<Vector3>.Release(ref p.vectorPath);
103  p.vectorPath = funnelPath;
104  }
105  }
