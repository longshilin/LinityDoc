60  DRScene drScene = dtScene.GetDataRow(sceneId);

64  return;
65  }
66 
67  GameEntry.Scene.LoadScene(AssetUtility.GetSceneAsset(drScene.AssetName), Constant.AssetPriority.SceneAsset, this);

69  }
