158  private void OnButtonClick()
159  {
160  if (!GetResourceGroupIsReady())
161  {
162  IResourceGroup resourceGroup = GetResourceGroup();
163  IResourceGroup updatingResouceGroup = GameEntry.Resource.UpdatingResourceGroup;
164  if (updatingResouceGroup != null && resourceGroup.Name != updatingResouceGroup.Name)
165  {
166  Log.Error(string.Format("There is already a resource group '{0}' being updated.", updatingResouceGroup.Name));
167  return;
168  }
169 
170  if (!updateResourceGroup)
171  {
172  updateResourceGroup = true;
173  GameEntry.Resource.UpdateResources(resourceGroupName, OnUpdateResourcesComplete);
174  }
175 
176  return;
177  }
178  }

180  private IResourceGroup GetResourceGroup()

234  private void OnUpdateResourcesComplete(IResourceGroup resourceGroup, bool result)
235  {
236  //这里可能下载完的时候这个UI已经被销毁了，做个gameObject判空
237  if (gameObject == null)
238  return;
239 
240  if (resourceGroup.Name != resourceGroupName)
241  {
242  if (!GetResourceGroupIsReady())
243  progressText.text = GameEntry.Localization.GetString("Download");
244 
245  return;
246  }
247 
248  if (result)
249  {
250  SetDownFinishState();
251  StartCoroutine(DelayCloseButton());
252  Log.Info("Update resources complete with no errors.");
253  }
254  else
255  {
256  Log.Error("Update resources complete with errors.");
257  }
258  }
259
