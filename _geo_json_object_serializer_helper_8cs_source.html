59  private static IEnumerable<SerializerHelper.Member> CreateBaseMembers()
60  {
61  return new[]
62  {
63  new SerializerHelper.Member("type", Flags.Type),
64  new SerializerHelper.Member("bbox", Flags.BoundingBox, isOptional: true),
65  new SerializerHelper.Member("crs", Flags.CoordinateReferenceSystem, isOptional: true),
66  new SerializerHelper.Member("*", Flags.ExtraMember, isOptional: true)

70  private static SerializerHelper.Member[] CreateCombinedMembers(IEnumerable<SerializerHelper.Member> derivedMembers)
71  {
72  var combinedMembers = CreateBaseMembers().Concat(derivedMembers).ToArray();
73  ThrowIfDuplicateMemberFlags(combinedMembers);
74  return combinedMembers;
75  }

77  private static void ThrowIfDuplicateMemberFlags(SerializerHelper.Member[] members)
78  {
79  var distinctFlags = new HashSet<long>(members.Select(m => m.Flag));
80  if (distinctFlags.Count < members.Length)
81  {
82  throw new BsonInternalException("Duplicate GeoJsonObject member flags.");
83  }
84  }

86  // public methods
94  public void DeserializeBaseMember(BsonDeserializationContext context, string elementName, long flag, GeoJsonObjectArgs<TCoordinates> args)
95  {
96  switch (flag)
97  {
98  case Flags.Type: EnsureTypeIsValid(context); break;
99  case Flags.CoordinateReferenceSystem: args.CoordinateReferenceSystem = DeserializeCoordinateReferenceSystem(context); break;
100  case Flags.BoundingBox: args.BoundingBox = DeserializeBoundingBox(context); break;
101  case Flags.ExtraMember: DeserializeExtraMember(context, elementName, args); break;
102  default: throw new BsonInternalException();
103  }
104  }

113  public void SerializeMembers<TValue>(BsonSerializationContext context, TValue value, Action<BsonSerializationContext, TValue> serializeDerivedMembers)
114  where TValue : GeoJsonObject<TCoordinates>
115  {
116  context.Writer.WriteStartDocument();
117  SerializeType(context, value.Type);
118  SerializeCoordinateReferenceSystem(context, value.CoordinateReferenceSystem);
119  SerializeBoundingBox(context, value.BoundingBox);
120  serializeDerivedMembers(context, value);
121  SerializeExtraMembers(context, value.ExtraMembers);
122  context.Writer.WriteEndDocument();
123  }

137  {

146  private void EnsureTypeIsValid(BsonDeserializationContext context)
147  {
148  var type = context.Reader.ReadString();
149  if (type != _type)
150  {
151  throw new FormatException(string.Format(
152  "Invalid GeoJson type: '{0}'. Expected: '{1}'.", type, _type));
153  }
154  }

157  {

160  context.Writer.WriteName("bbox");
161  _boundingBoxSerializer.Serialize(context, boundingBox);

165  private void SerializeCoordinateReferenceSystem(BsonSerializationContext context, GeoJsonCoordinateReferenceSystem coordinateReferenceSystem)
166  {
167  if (coordinateReferenceSystem != null)
168  {
169  context.Writer.WriteName("crs");
170  _coordinateReferenceSystemSerializer.Serialize(context, coordinateReferenceSystem);
171  }
172  }
173 
174  private void SerializeExtraMembers(BsonSerializationContext context, BsonDocument value)
175  {
176  if (value != null)
177  {
178  foreach (var element in value)
179  {
180  context.Writer.WriteName(element.Name);
181  BsonValueSerializer.Instance.Serialize(context, element.Value);
182  }
183  }
184  }
185 
186  private void SerializeType(BsonSerializationContext context, GeoJsonObjectType type)
187  {
188  context.Writer.WriteString("type", type.ToString());
189  }
