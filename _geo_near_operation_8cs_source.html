213  internal BsonDocument CreateCommand(ConnectionDescription connectionDescription, ICoreSession session)
214  {
215  Feature.ReadConcern.ThrowIfNotSupported(connectionDescription.ServerVersion, _readConcern);
216  Feature.Collation.ThrowIfNotSupported(connectionDescription.ServerVersion, _collation);
217 
218  var readConcern = ReadConcernHelper.GetReadConcernForCommand(session, connectionDescription, _readConcern);
219  return new BsonDocument
220  {
221  { "geoNear", _collectionNamespace.CollectionName },
222  { "near", _near },
223  { "limit", () => _limit.Value, _limit.HasValue },
224  { "maxDistance", () => _maxDistance.Value, _maxDistance.HasValue },
225  { "query", _filter, _filter != null },
226  { "spherical", () => _spherical.Value, _spherical.HasValue },
227  { "distanceMultiplier", () => _distanceMultiplier.Value, _distanceMultiplier.HasValue },
228  { "includeLocs", () => _includeLocs.Value, _includeLocs.HasValue },
229  { "uniqueDocs", () => _uniqueDocs.Value, _uniqueDocs.HasValue },
230  { "maxTimeMS", () => MaxTimeHelper.ToMaxTimeMS(_maxTime.Value), _maxTime.HasValue },
231  { "collation", () => _collation.ToBsonDocument(), _collation != null },
232  { "readConcern", readConcern, readConcern != null }
233  };
234  }
