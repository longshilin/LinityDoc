74  public Color NodeColor (GraphNode node) {
75  if (showSearchTree && !InSearchTree(node, debugData, debugPathID)) return Color.clear;
76 
77  Color color;
78 
79  if (node.Walkable) {
80  switch (debugMode) {
81  case GraphDebugMode.Areas:
82  color = AstarColor.GetAreaColor(node.Area);
83  break;
84  case GraphDebugMode.Penalty:
85  color = Color.Lerp(AstarColor.ConnectionLowLerp, AstarColor.ConnectionHighLerp, ((float)node.Penalty-debugFloor) / (debugRoof-debugFloor));
86  break;
87  case GraphDebugMode.Tags:
88  color = AstarColor.GetAreaColor(node.Tag);
89  break;
90  case GraphDebugMode.Connections:
91  color = AstarColor.NodeConnection;
92  break;
93  default:
94  if (debugData == null) {
95  color = AstarColor.NodeConnection;
96  break;
97  }
98 
99  PathNode pathNode = debugData.GetPathNode(node);
100  float value;
101  if (debugMode == GraphDebugMode.G) {
102  value = pathNode.G;
103  } else if (debugMode == GraphDebugMode.H) {
104  value = pathNode.H;
105  } else {
106  // mode == F
107  value = pathNode.F;
108  }
109 
110  color = Color.Lerp(AstarColor.ConnectionLowLerp, AstarColor.ConnectionHighLerp, (value-debugFloor) / (debugRoof-debugFloor));
111  break;
112  }
113  } else {
114  color = AstarColor.UnwalkableNode;
115  }
116 
117  return color;
118  }
