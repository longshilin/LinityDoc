500  public void AddConnection (GraphNode node, uint cost, int shapeEdge) {

502 
503  // Check if we already have a connection to the node
504  if (connections != null) {
505  for (int i = 0; i < connections.Length; i++) {
506  if (connections[i].node == node) {
507  // Just update the cost for the existing connection
508  connections[i].cost = cost;
509  // Update edge only if it was a definite edge, otherwise reuse the existing one
510  // This makes it possible to use the AddConnection(node,cost) overload to only update the cost
511  // without changing the edge which is required for backwards compatibility.
512  connections[i].shapeEdge = shapeEdge >= 0 ? (byte)shapeEdge : connections[i].shapeEdge;
