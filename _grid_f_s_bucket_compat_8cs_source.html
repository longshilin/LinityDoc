248  private GridFSFindOptions<ObjectId> WrapFindOptions(GridFSFindOptions options)
249  {
250  if (options != null)
251  {
252  var renderedSort = options.Sort == null ? null : options.Sort.Render(GridFSFileInfoSerializer.Instance, BsonSerializer.SerializerRegistry);
253  var wrappedSort = renderedSort == null ? null : new BsonDocumentSortDefinition<GridFSFileInfo<ObjectId>>(renderedSort);
254  return new GridFSFindOptions<ObjectId>
255  {
256  BatchSize = options.BatchSize,
257  Limit = options.Limit,
258  MaxTime = options.MaxTime,
259  NoCursorTimeout = options.NoCursorTimeout,
260  Skip = options.Skip,
261  Sort = wrappedSort
262  };
263  }
264  else
265  {
266  return null;
