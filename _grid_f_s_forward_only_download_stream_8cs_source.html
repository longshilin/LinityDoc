255  private void ProcessNextBatch(List<BsonDocument> batch)
256  {
257  if (batch == null)
258  {
259 #pragma warning disable 618
260  throw new GridFSChunkException(_idAsBsonValue, _nextChunkNumber, "missing");
261 #pragma warning restore
262  }
263 
264  var previousBatch = _batch;
265  _batch = batch;
266 
267  if (previousBatch != null)
268  {
269  _batchPosition += previousBatch.Count * FileInfo.ChunkSizeBytes; ;
270  }
271 
272  var lastChunkInBatch = _batch.Last();
273  if (lastChunkInBatch["n"].ToInt32() == _lastChunkNumber + 1 && lastChunkInBatch["data"].AsBsonBinaryData.Bytes.Length == 0)
274  {
275  _batch.RemoveAt(_batch.Count - 1);
276  }
277 
278  foreach (var chunk in _batch)
279  {
280  var n = chunk["n"].ToInt32();
281  var bytes = chunk["data"].AsBsonBinaryData.Bytes;
282 
283  if (n != _nextChunkNumber)
284  {
285 #pragma warning disable 618
286  throw new GridFSChunkException(_idAsBsonValue, _nextChunkNumber, "missing");
287 #pragma warning restore
288  }
289  _nextChunkNumber++;
290 
291  var expectedChunkSize = n == _lastChunkNumber ? _lastChunkSize : FileInfo.ChunkSizeBytes;
292  if (bytes.Length != expectedChunkSize)
293  {
294 #pragma warning disable 618
295  throw new GridFSChunkException(_idAsBsonValue, _nextChunkNumber, "the wrong size");
296 #pragma warning restore
297  }
298 
299  if (_checkMD5)
300  {
301  _md5.AppendData(bytes, 0, bytes.Length);
302  }
303  }
304  }
