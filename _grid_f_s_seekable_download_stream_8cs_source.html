121  public override long Seek(long offset, SeekOrigin origin)
122  {
123  long newPosition;
124  switch (origin)
125  {
126  case SeekOrigin.Begin: newPosition = offset; break;
127  case SeekOrigin.Current: newPosition = _position + offset; break;
128  case SeekOrigin.End: newPosition = Length + offset; break;
129  default: throw new ArgumentException("Invalid origin.", "origin");
130  }
131  if (newPosition < 0)
132  {
133  throw new IOException("Position must be greater than or equal to zero.");
134  }
135  if (newPosition > Length)
136  {
137  throw new IOException("Position must be less than or equal to the length of the stream.");
138  }
139  Position = newPosition;
140  return newPosition;
141  }

189  private byte[] GetChunkHelper(long n, List<BsonDocument> documents)
190  {
191  if (documents.Count == 0)
192  {
193 #pragma warning disable 618
194  throw new GridFSChunkException(_idAsBsonValue, n, "missing");
195 #pragma warning restore
196  }
197 
198  var document = documents[0];
199  var data = document["data"].AsBsonBinaryData.Bytes;
200 
201  var chunkSizeBytes = FileInfo.ChunkSizeBytes;
202  var lastChunk = FileInfo.Length / FileInfo.ChunkSizeBytes;
203  var expectedChunkSize = n == lastChunk ? FileInfo.Length % chunkSizeBytes : chunkSizeBytes;
204  if (data.Length != expectedChunkSize)
205  {
206 #pragma warning disable 618
207  throw new GridFSChunkException(_idAsBsonValue, n, "the wrong size");
208 #pragma warning restore
209  }
210 
211  return data;
212  }

214  private ArraySegment<byte> GetSegment(CancellationToken cancellationToken)
215  {
216  var n = _position / FileInfo.ChunkSizeBytes;
217  if (_n != n)
218  {
219  GetChunk(n, cancellationToken);
220  }
221 
222  var segmentOffset = (int)(_position % FileInfo.ChunkSizeBytes);
223  var segmentCount = _chunk.Length - segmentOffset;
224 
225  return new ArraySegment<byte>(_chunk, segmentOffset, segmentCount);
226  }

228  private async Task<ArraySegment<byte>> GetSegmentAsync(CancellationToken cancellationToken)
229  {
230  var n = _position / FileInfo.ChunkSizeBytes;

233  await GetChunkAsync(n, cancellationToken).ConfigureAwait(false);
234  }
235 
236  var segmentOffset = (int)(_position % FileInfo.ChunkSizeBytes);
237  var segmentCount = _chunk.Length - segmentOffset;
238 
239  return new ArraySegment<byte>(_chunk, segmentOffset, segmentCount);
