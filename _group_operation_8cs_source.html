264  private ReadCommandOperation<TResult[]> CreateOperation(SemanticVersion serverVersion)
265  {
266  var command = CreateCommand(serverVersion);
267  var resultSerializer = _resultSerializer ?? BsonSerializer.LookupSerializer<TResult>();
268  var resultArraySerializer = new ArraySerializer<TResult>(resultSerializer);
269  var commandResultSerializer = new ElementDeserializer<TResult[]>("retval", resultArraySerializer);
270  return new ReadCommandOperation<TResult[]>(_collectionNamespace.DatabaseNamespace, command, commandResultSerializer, _messageEncoderSettings);
