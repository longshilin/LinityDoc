144  private static GssapiMechanism CreateMechanism(string username, SecureString password, IEnumerable<KeyValuePair<string, string>> properties)
145  {
146  var serviceName = DefaultServiceName;
147  var canonicalizeHostName = false;
148  string realm = null;
149  if (properties != null)
150  {
151  foreach (var pair in properties)
152  {
153  switch (pair.Key.ToUpperInvariant())
154  {
155  case __serviceNamePropertyName:
156  serviceName = (string)pair.Value;
157  break;
158  case __serviceRealmPropertyName:
159  case __realmPropertyName:
160  realm = (string)pair.Value;
161  break;
162  case __canonicalizeHostNamePropertyName:
163  canonicalizeHostName = bool.Parse(pair.Value);
164  break;
165  default:
166  var message = string.Format("Unknown GSSAPI property '{0}'.", pair.Key);
167  throw new ArgumentException(message, "properties");
168  }
169  }
170  }
171 
172  return new GssapiMechanism(serviceName, canonicalizeHostName, realm, username, password);
173  }

199  public ISaslStep Initialize(IConnection connection, SaslConversation conversation, ConnectionDescription description)
200  {
201  Ensure.IsNotNull(connection, nameof(connection));
202  Ensure.IsNotNull(description, nameof(description));
203 
204  string hostName;
205  var dnsEndPoint = connection.EndPoint as DnsEndPoint;
206  if (dnsEndPoint != null)
207  {
208  hostName = dnsEndPoint.Host;
209  }
210  else if (connection.EndPoint is IPEndPoint)
211  {
212  hostName = ((IPEndPoint)connection.EndPoint).Address.ToString();
213  }
214  else
215  {
216  throw new MongoAuthenticationException(connection.ConnectionId, "Only DnsEndPoint and IPEndPoint are supported for GSSAPI authentication.");
217  }
218 
219  if (_canonicalizeHostName)
220  {
221 #if NETSTANDARD1_5 || NETSTANDARD1_6
222  var entry = Dns.GetHostEntryAsync(hostName).GetAwaiter().GetResult();
223 #else
224  var entry = Dns.GetHostEntry(hostName);
225 #endif
226  if (entry != null)
227  {
228  hostName = entry.HostName;
229  }
230  }
231 
232  return new FirstStep(_serviceName, hostName, _realm, _username, _password, conversation);
233  }

236  private class FirstStep : ISaslStep
237  {
238  private readonly string _authorizationId;

244  public FirstStep(string serviceName, string hostName, string realm, string username, SecureString password, SaslConversation conversation)
245  {
246  _authorizationId = username;
247  _password = password;
248  _servicePrincipalName = string.Format("{0}/{1}", serviceName, hostName);
249  if (!string.IsNullOrEmpty(realm))
250  {
251  _servicePrincipalName += "@" + realm;
252  }
253 
254  SecurityCredential securityCredential;
255  try
256  {
257  securityCredential = SecurityCredential.Acquire(SspiPackage.Kerberos, _authorizationId, _password);
258  conversation.RegisterItemForDisposal(securityCredential);
259  }
260  catch (Win32Exception ex)
261  {
262  throw new MongoAuthenticationException(conversation.ConnectionId, "Unable to acquire security credential.", ex);
263  }
264 
265  try
266  {
267  _context = Sspi.SecurityContext.Initialize(securityCredential, _servicePrincipalName, null, out _bytesToSendToServer);
268  }
269  catch (Win32Exception ex)
270  {
271  if (_password != null)
272  {
273  throw new MongoAuthenticationException(conversation.ConnectionId, "Unable to initialize security context. Ensure the username and password are correct.", ex);
274  }
275  else
276  {
277  throw new MongoAuthenticationException(conversation.ConnectionId, "Unable to initialize security context.", ex);
278  }
279  }
280  }

382  public ISaslStep Transition(SaslConversation conversation, byte[] bytesReceivedFromServer)
383  {
384  byte[] decryptedBytes;
385  try
386  {
387  _context.DecryptMessage(0, bytesReceivedFromServer, out decryptedBytes);
388  }
389  catch (Win32Exception ex)
390  {
391  throw new MongoAuthenticationException(conversation.ConnectionId, "Unabled to decrypt message.", ex);
392  }
393 
394  int length = 4;
395  if (_authorizationId != null)
396  {
397  length += _authorizationId.Length;
398  }
399 
400  bytesReceivedFromServer = new byte[length];
401  bytesReceivedFromServer[0] = 0x1; // NO_PROTECTION
402  bytesReceivedFromServer[1] = 0x0; // NO_PROTECTION
403  bytesReceivedFromServer[2] = 0x0; // NO_PROTECTION
404  bytesReceivedFromServer[3] = 0x0; // NO_PROTECTION
405 
406  if (_authorizationId != null)
407  {
408  var authorizationIdBytes = Encoding.UTF8.GetBytes(_authorizationId);
409  authorizationIdBytes.CopyTo(bytesReceivedFromServer, 4);
410  }
411 
412  byte[] bytesToSendToServer;
413  try
414  {
415  _context.EncryptMessage(bytesReceivedFromServer, out bytesToSendToServer);
416  }
417  catch (Win32Exception ex)
418  {
419  throw new MongoAuthenticationException(conversation.ConnectionId, "Unabled to encrypt message.", ex);
420  }
421 
422  return new CompletedStep(bytesToSendToServer);
423  }
