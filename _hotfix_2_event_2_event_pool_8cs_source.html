223  private void HandleEvent(object sender, T e)
224  {
225  int eventId = e.Id;
226  bool noHandlerException = false;
227  LinkedList<EventHandler<T>> handlers = null;
228  if (m_EventHandlers.TryGetValue(eventId, out handlers) && handlers.Count > 0)
229  {
230  LinkedListNode<EventHandler<T>> current = handlers.First;
231  while (current != null)
232  {
233  LinkedListNode<EventHandler<T>> next = current.Next;
234  current.Value(sender, e);
235  current = next;
236  }
237  }
238  else if (m_DefaultHandler != null)
239  {
240  m_DefaultHandler(sender, e);
241  }
242  else if ((m_EventPoolMode & EventPoolMode.AllowNoHandler) == 0)
243  {
244  noHandlerException = true;
245  }
246 
247  ReferencePool.Release((IReference)e);
248 
249  if (noHandlerException)
250  {
251  throw new GameFrameworkException(GameFramework.Utility.Text.Format("Event '{0}' not allow no handler.", eventId.ToString()));
252  }
253  }
254  }
