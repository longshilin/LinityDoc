183  private static void InternalCheckReferenceType(Type referenceType)
184  {
185  if (!m_EnableStrictCheck)
186  {
187  return;
188  }
189 
190  if (referenceType == null)
191  {
192  throw new GameFrameworkException("Reference type is invalid.");
193  }
194 
195  if (!referenceType.IsClass || referenceType.IsAbstract)
196  {
197  throw new GameFrameworkException("Reference type is not a non-abstract class type.");
198  }
199 
200  if (!typeof(IReference).IsAssignableFrom(referenceType))
201  {
202  throw new GameFrameworkException(GameFramework.Utility.Text.Format("Reference type '{0}' is invalid.", referenceType.FullName));
203  }
204  }

206  private static ReferenceCollection GetReferenceCollection(Type referenceType)

212
