198  private async ETTask InvokeHandler(HttpListenerContext context)
199  {
200  context.Response.StatusCode = 404;
201 
202  MethodInfo methodInfo = null;
203  IHttpHandler httpHandler = null;
204  string postbody = "";
205  switch (context.Request.HttpMethod)
206  {
207  case "GET":
208  this.getHandlers.TryGetValue(context.Request.Url.AbsolutePath, out methodInfo);
209  if (methodInfo != null)
210  {
211  this.handlersMapping.TryGetValue(methodInfo, out httpHandler);
212  }
213  break;
214  case "POST":
215  this.postHandlers.TryGetValue(context.Request.Url.AbsolutePath, out methodInfo);
216  if (methodInfo != null)
217  {
218  this.handlersMapping.TryGetValue(methodInfo, out httpHandler);
219 
220  using (StreamReader sr = new StreamReader(context.Request.InputStream))
221  {
222  postbody = sr.ReadToEnd();
223  }
224  }
225  break;
226  default:
227  context.Response.StatusCode = 405;
228  break;
229  }
230 
231  if (httpHandler != null)
232  {
233  object[] args = InjectParameters(context, methodInfo, postbody);
234 
235  // 自动把返回值，以json方式响应。
236  object resp = methodInfo.Invoke(httpHandler, args);
237  object result = resp;
238  if (resp is ETTask<HttpResult> t)
239  {
240  await t;
241  result = t.GetType().GetProperty("Result").GetValue(t, null);
242  }
243 
244  if (result != null)
245  {
246  using (StreamWriter sw = new StreamWriter(context.Response.OutputStream,Encoding.UTF8))
247  {
248  if (result.GetType() == typeof(string))
249  {
250  sw.Write(result.ToString());
251  }
252  else
253  {
254  sw.Write(JsonHelper.ToJson(result));
255  }
256  }
257  }
258  }

268  private static object[] InjectParameters(HttpListenerContext context, MethodInfo methodInfo, string postbody)
269  {
270  context.Response.StatusCode = 200;
271  ParameterInfo[] parameterInfos = methodInfo.GetParameters();
272  object[] args = new object[parameterInfos.Length];
273  for (int i = 0; i < parameterInfos.Length; i++)
274  {
275  ParameterInfo item = parameterInfos[i];
276 
277  if (item.ParameterType == typeof(HttpListenerRequest))
278  {
279  args[i] = context.Request;
280  continue;
281  }
282 
283  if (item.ParameterType == typeof(HttpListenerResponse))
284  {
285  args[i] = context.Response;
286  continue;
287  }
288 
289  try
290  {
291  switch (context.Request.HttpMethod)
292  {
293  case "POST":
294  if (item.Name == "postBody") // 约定参数名称为postBody,只传string类型。本来是byte[]，有需求可以改。
295  {
296  args[i] = postbody;
297  }
298  else if (item.ParameterType.IsClass && item.ParameterType != typeof(string) && !string.IsNullOrEmpty(postbody))
299  {
300  object entity = JsonHelper.FromJson(item.ParameterType, postbody);
301  args[i] = entity;
302  }
303 
304  break;
305  case "GET":
306  string query = context.Request.QueryString[item.Name];
307  if (query != null)
308  {
309  object value = Convert.ChangeType(query, item.ParameterType);
310  args[i] = value;
311  }
312 
313  break;
314  default:
315  args[i] = null;
316  break;
317  }
318  }
319  catch (Exception e)
320  {
321  Log.Error(e);
322  args[i] = null;
323  }
324  }
325 
326  return args;
