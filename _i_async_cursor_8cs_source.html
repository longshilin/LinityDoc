177  public static async Task ForEachAsync<TDocument>(this IAsyncCursor<TDocument> source, Func<TDocument, int, Task> processor, CancellationToken cancellationToken = default(CancellationToken))
178  {
179  Ensure.IsNotNull(source, nameof(source));
180  Ensure.IsNotNull(processor, nameof(processor));
181 
182  // yes, we are taking ownership... assumption being that they've
183  // exhausted the thing and don't need it anymore.
184  using (source)
185  {
186  var index = 0;
187  while (await source.MoveNextAsync(cancellationToken).ConfigureAwait(false))
188  {
189  foreach (var document in source.Current)
190  {
191  await processor(document, index++).ConfigureAwait(false);
192  cancellationToken.ThrowIfCancellationRequested();
193  }
194  }
195  }
196  }
