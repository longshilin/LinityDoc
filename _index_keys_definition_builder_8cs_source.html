419  internal sealed class CombinedIndexKeysDefinition<TDocument> : IndexKeysDefinition<TDocument>
420  {
421  private readonly List<IndexKeysDefinition<TDocument>> _keys;
422 
423  public CombinedIndexKeysDefinition(IEnumerable<IndexKeysDefinition<TDocument>> keys)
424  {
425  _keys = Ensure.IsNotNull(keys, nameof(keys)).ToList();
426  }
427 
428  public override BsonDocument Render(IBsonSerializer<TDocument> documentSerializer, IBsonSerializerRegistry serializerRegistry)
429  {
430  var document = new BsonDocument();
431 
432  foreach (var key in _keys)
433  {
434  var renderedKey = key.Render(documentSerializer, serializerRegistry);
435 
436  foreach (var element in renderedKey.Elements)
437  {
438  if (document.Contains(element.Name))
439  {
440  var message = string.Format(
441  "The index keys definition contains multiple values for the field '{0}'.",
442  element.Name);
443  throw new MongoException(message);
444  }
445  document.Add(element);
446  }
447  }
448 
449  return document;
450  }
451  }

453  internal sealed class DirectionalIndexKeyDefinition<TDocument> : IndexKeysDefinition<TDocument>
454  {
455  private readonly FieldDefinition<TDocument> _field;
456  private readonly SortDirection _direction;
457 
458  public DirectionalIndexKeyDefinition(FieldDefinition<TDocument> field, SortDirection direction)
459  {
460  _field = Ensure.IsNotNull(field, nameof(field));
461  _direction = direction;
462  }
463 
464  public override BsonDocument Render(IBsonSerializer<TDocument> documentSerializer, IBsonSerializerRegistry serializerRegistry)
465  {
466  var renderedField = _field.Render(documentSerializer, serializerRegistry);
467 
468  BsonValue value;
469  switch (_direction)
470  {
471  case SortDirection.Ascending:
472  value = 1;
473  break;
474  case SortDirection.Descending:
475  value = -1;
476  break;
477  default:
478  throw new InvalidOperationException("Unknown value for " + typeof(SortDirection) + ".");
479  }
480 
481  return new BsonDocument(renderedField.FieldName, value);
482  }
483  }

486  {
487  private readonly FieldDefinition<TDocument> _field;
488  private readonly FieldDefinition<TDocument> _additionalFieldName;
489 
490  public GeoHaystackIndexKeyDefinition(FieldDefinition<TDocument> field, FieldDefinition<TDocument> additionalFieldName = null)
491  {
492  _field = Ensure.IsNotNull(field, nameof(field));
493  _additionalFieldName = additionalFieldName;
494  }
495 
496  public override BsonDocument Render(IBsonSerializer<TDocument> documentSerializer, IBsonSerializerRegistry serializerRegistry)
497  {
498  var renderedField = _field.Render(documentSerializer, serializerRegistry);
499 
500  var document = new BsonDocument(renderedField.FieldName, "geoHaystack");
501  if (_additionalFieldName != null)
502  {
503  var additionalRenderedField = _additionalFieldName.Render(documentSerializer, serializerRegistry);
504  document.Add(additionalRenderedField.FieldName, 1);
505  }
506 
507  return document;
508  }
