55  {

87  public void WriteMessage(InsertMessage<TDocument> message)
88  {

91  var writer = CreateBinaryWriter();

94

99  stream.WriteInt32((int)BuildInsertFlags(message));
100  stream.WriteCString(message.CollectionNamespace.FullName);
101  WriteDocuments(writer, messageStartPosition, message);

116  private List<TDocument> ReadDocuments(BsonBinaryReader reader, long messageStartPosition, int messageSize)
117  {
118  var stream = reader.BsonStream;
119  var context = BsonDeserializationContext.CreateRoot(reader);
120  var documents = new List<TDocument>();
121 
122  while (stream.Position < messageStartPosition + messageSize)
123  {

125  documents.Add(document);
126  }

131  private void WriteDocuments(BsonBinaryWriter writer, long messageStartPosition, InsertMessage<TDocument> message)
132  {
133  var stream = writer.BsonStream;
134  var context = BsonSerializationContext.CreateRoot(writer);
135 
136  var collectionNamespace = message.CollectionNamespace;
137  var isSystemIndexesCollection = collectionNamespace.Equals(collectionNamespace.DatabaseNamespace.SystemIndexesCollection);
138  var elementNameValidator = isSystemIndexesCollection ? (IElementNameValidator)NoOpElementNameValidator.Instance : CollectionElementNameValidator.Instance;
139 
140  writer.PushElementNameValidator(elementNameValidator);
141  try
142  {
143  var documentSource = message.DocumentSource;
144  var batchCount = Math.Min(documentSource.Count, message.MaxBatchCount);
145  if (batchCount < documentSource.Count && !documentSource.CanBeSplit)
146  {
147  throw new BsonSerializationException("Batch is too large.");
148  }
149 
150  for (var i = 0; i < batchCount; i++)
151  {
152  var document = documentSource.Items[documentSource.Offset + i];
153  var documentStartPosition = stream.Position;
154 
155  _serializer.Serialize(context, document);
156 
157  var messageSize = stream.Position - messageStartPosition;
158  if (messageSize > message.MaxMessageSize)
159  {
160  if (i > 0 && documentSource.CanBeSplit)
161  {
162  stream.Position = documentStartPosition;
163  stream.SetLength(documentStartPosition);
164  documentSource.SetProcessedCount(i);
165  return;
166  }
167  else
168  {
169  throw new BsonSerializationException("Batch is too large.");
170  }
171  }
172  }
173  documentSource.SetProcessedCount(batchCount);
174  }
175  finally
176  {
177  writer.PopElementNameValidator();
178  }
179  }
