55  public InsertMessage<TDocument> ReadMessage()
56  {
57  var jsonReader = CreateJsonReader();
58  var messageContext = BsonDeserializationContext.CreateRoot(jsonReader);
59  var messageDocument = BsonDocumentSerializer.Instance.Deserialize(messageContext);
60 
61  var opcode = messageDocument["opcode"].AsString;
62  if (opcode != "insert")
63  {
64  throw new FormatException("Opcode is not insert.");
65  }
66 
67  var requestId = messageDocument["requestId"].ToInt32();
68  var databaseName = messageDocument["database"].AsString;
69  var collectionName = messageDocument["collection"].AsString;
70  var maxBatchCount = messageDocument["maxBatchCount"].ToInt32();
71  var maxMessageSize = messageDocument["maxMessageSize"].ToInt32();
72  var continueOnError = messageDocument["continueOnError"].ToBoolean();
73  var documents = messageDocument["documents"];
74 
75  if (documents.IsBsonNull)
76  {
77  throw new FormatException("InsertMessageJsonEncoder requires documents to not be null.");
78  }
79 
80  var batch = new List<TDocument>();
81  foreach (BsonDocument serializedDocument in documents.AsBsonArray)
82  {
83  using (var documentReader = new BsonDocumentReader(serializedDocument))
84  {
85  var documentContext = BsonDeserializationContext.CreateRoot(documentReader);
86  var document = _serializer.Deserialize(documentContext);
87  batch.Add(document);
88  }
89  }
90  var documentSource = new BatchableSource<TDocument>(batch);
91 
92  return new InsertMessage<TDocument>(
93  requestId,
94  new CollectionNamespace(databaseName, collectionName),
95  _serializer,
96  documentSource,
97  maxBatchCount,
98  maxMessageSize,
99  continueOnError);
100  }

106  public void WriteMessage(InsertMessage<TDocument> message)
107  {
108  Ensure.IsNotNull(message, nameof(message));
109 
110  var messageDocument = new BsonDocument
111  {
112  { "opcode", "insert" },
113  { "requestId", message.RequestId },
114  { "database", message.CollectionNamespace.DatabaseNamespace.DatabaseName },
115  { "collection", message.CollectionNamespace.CollectionName },
116  { "maxBatchCount", message.MaxBatchCount },
117  { "maxMessageSize", message.MaxMessageSize },
118  { "continueOnError", message.ContinueOnError },
119  { "documents", WrapDocuments(message) }
120  };
121 
122  var jsonWriter = CreateJsonWriter();
123  var messageContext = BsonSerializationContext.CreateRoot(jsonWriter);
124  BsonDocumentSerializer.Instance.Serialize(messageContext, messageDocument);
125  }

139  private BsonArray WrapDocuments(InsertMessage<TDocument> message)
140  {
141  var documentSource = message.DocumentSource;
142  var batchCount = Math.Min(documentSource.Count, message.MaxBatchCount);
143  if (batchCount < documentSource.Count && !documentSource.CanBeSplit)
144  {
145  throw new BsonSerializationException("Batch is too large.");
146  }
147 
148  var wrappedDocuments = new BsonArray(batchCount);
149  for (var i = 0; i < batchCount; i++)
150  {
151  var document = documentSource.Items[documentSource.Offset + i];
152  var wrappedDocument = new BsonDocumentWrapper(document, _serializer);
153  wrappedDocuments.Add(wrappedDocument);
154  }
155  documentSource.SetProcessedCount(batchCount);
156 
157  return wrappedDocuments;
158  }
