385  // public methods
386  public IEnumerable<Batch> GetBatches()
387  {
388  while (_documentSource.Count > 0)
389  {
390  var writeConcern = _writeConcern;
391  Func<bool> shouldSendGetLastError = null;
392  if (!writeConcern.IsAcknowledged && !_continueOnError)
393  {
394  writeConcern = WriteConcern.W1;
395  shouldSendGetLastError = () => !_documentSource.AllItemsWereProcessed;
396  }
397 
398  var batch = new Batch
399  {
400  Documents = _documentSource,
401  WriteConcern = writeConcern,
402  ShouldSendGetLastError = shouldSendGetLastError
403  };
404 
405  yield return batch;
406 
407  _results.Add(batch.Result);
408 
409  _documentSource.AdvancePastProcessedItems();
410  }
411  }

413  public WriteConcernResult HandleException(MongoWriteConcernException exception)
414  {
415  var result = exception.WriteConcernResult;
416 
417  if (_continueOnError)
418  {
419  _finalException = exception;
420  }
421  else if (_writeConcern.IsAcknowledged)
422  {

424  exception.Data["results"] = _results;
425  throw exception;
