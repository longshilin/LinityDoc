169  private static void AddArrayMetadata (Type type)
170  {
171  if (array_metadata.ContainsKey (type))
172  return;
173 
174  ArrayMetadata data = new ArrayMetadata ();
175 
176  data.IsArray = type.IsArray;
177 
178  if (type.GetInterface ("System.Collections.IList") != null)
179  data.IsList = true;
180 
181  foreach (PropertyInfo p_info in type.GetProperties ()) {
182  if (p_info.Name != "Item")
183  continue;
184 
185  ParameterInfo[] parameters = p_info.GetIndexParameters ();
186 
187  if (parameters.Length != 1)
188  continue;
189 
190  if (parameters[0].ParameterType == typeof (int))
191  data.ElementType = p_info.PropertyType;
192  }
193 
194  lock (array_metadata_lock) {
195  try {
196  array_metadata.Add (type, data);
197  } catch (ArgumentException) {
198  return;
199  }
200  }

288  {
289  lock (conv_ops_lock) {
290  if (! conv_ops.ContainsKey (t1))
291  conv_ops.Add (t1, new Dictionary<Type, MethodInfo> ());

295  return conv_ops[t1][t2];

311  private static object ReadValue (Type inst_type, JsonReader reader)
312  {
313  reader.Read ();
314 
315  if (reader.Token == JsonToken.ArrayEnd)
316  return null;
317 
318  Type underlying_type = Nullable.GetUnderlyingType(inst_type);
319  Type value_type = underlying_type ?? inst_type;
320 
321  if (reader.Token == JsonToken.Null) {
322  if (inst_type.IsClass || underlying_type != null) {
323  return null;
324  }
325 
326  throw new JsonException (String.Format (
327  "Can't assign null to an instance of type {0}",
328  inst_type));
329  }
330 
331  if (reader.Token == JsonToken.Double ||
332  reader.Token == JsonToken.Int ||
333  reader.Token == JsonToken.Long ||
334  reader.Token == JsonToken.String ||
335  reader.Token == JsonToken.Boolean) {
336 
337  Type json_type = reader.Value.GetType ();
338 
339  if (value_type.IsAssignableFrom (json_type))
340  return reader.Value;
341 
342  // If there's a custom importer that fits, use it
343  if (custom_importers_table.ContainsKey (json_type) &&
344  custom_importers_table[json_type].ContainsKey (
345  value_type)) {
346 
347  ImporterFunc importer =
348  custom_importers_table[json_type][value_type];
349 
350  return importer (reader.Value);
351  }
352 
353  // Maybe there's a base importer that works
354  if (base_importers_table.ContainsKey (json_type) &&
355  base_importers_table[json_type].ContainsKey (
356  value_type)) {
357 
358  ImporterFunc importer =
359  base_importers_table[json_type][value_type];
360 
361  return importer (reader.Value);
362  }
363 
364  // Maybe it's an enum
365  if (value_type.IsEnum)
366  return Enum.ToObject (value_type, reader.Value);
367 
368  // Try using an implicit conversion operator
369  MethodInfo conv_op = GetConvOp (value_type, json_type);
370 
371  if (conv_op != null)
372  return conv_op.Invoke (null,
373  new object[] { reader.Value });
374 
375  // No luck
376  throw new JsonException (String.Format (
377  "Can't assign value '{0}' (type {1}) to type {2}",
378  reader.Value, json_type, inst_type));
379  }
380 
381  object instance = null;
382 
383  if (reader.Token == JsonToken.ArrayStart) {
384 
385  AddArrayMetadata (inst_type);
386  ArrayMetadata t_data = array_metadata[inst_type];
387 
388  if (! t_data.IsArray && ! t_data.IsList)
389  throw new JsonException (String.Format (
390  "Type {0} can't act as an array",
391  inst_type));
392 
393  IList list;
394  Type elem_type;
395 
396  if (! t_data.IsArray) {
397  list = (IList) Activator.CreateInstance (inst_type);
398  elem_type = t_data.ElementType;
399  } else {
400  list = new ArrayList ();
401  elem_type = inst_type.GetElementType ();
402  }
403 
404  while (true) {
405  object item = ReadValue (elem_type, reader);
406  if (item == null && reader.Token == JsonToken.ArrayEnd)
407  break;
408 
409  list.Add (item);
410  }
411 
412  if (t_data.IsArray) {
413  int n = list.Count;
414  instance = Array.CreateInstance (elem_type, n);
415 
416  for (int i = 0; i < n; i++)
417  ((Array) instance).SetValue (list[i], i);
418  } else
419  instance = list;
420 
421  } else if (reader.Token == JsonToken.ObjectStart) {
422  AddObjectMetadata (value_type);
423  ObjectMetadata t_data = object_metadata[value_type];
424 
425  instance = Activator.CreateInstance (value_type);
426 
427  while (true) {
428  reader.Read ();
429 
430  if (reader.Token == JsonToken.ObjectEnd)
431  break;
432 
433  string property = (string) reader.Value;
434 
435  if (t_data.Properties.ContainsKey (property)) {
436  PropertyMetadata prop_data =
437  t_data.Properties[property];
438 
439  if (prop_data.IsField) {
440  ((FieldInfo) prop_data.Info).SetValue (
441  instance, ReadValue (prop_data.Type, reader));
442  } else {
443  PropertyInfo p_info =
444  (PropertyInfo) prop_data.Info;
445 
446  if (p_info.CanWrite)
447  p_info.SetValue (
448  instance,
449  ReadValue (prop_data.Type, reader),
450  null);
451  else
452  ReadValue (prop_data.Type, reader);
453  }
454 
455  } else {
456  if (! t_data.IsDictionary) {
457 
458  if (! reader.SkipNonMembers) {
459  throw new JsonException (String.Format (
460  "The type {0} doesn't have the " +
461  "property '{1}'",
462  inst_type, property));
463  } else {
464  ReadSkip (reader);
465  continue;
466  }
467  }
468 
469  ((IDictionary) instance).Add (
470  property, ReadValue (
471  t_data.ElementType, reader));
472  }
473 
474  }
475 
476  }
477 
478  return instance;
479  }

481  private static IJsonWrapper ReadValue (WrapperFactory factory,
482  JsonReader reader)
483  {
484  reader.Read ();
485 
486  if (reader.Token == JsonToken.ArrayEnd ||
487  reader.Token == JsonToken.Null)
488  return null;
489 
490  IJsonWrapper instance = factory ();
491 
492  if (reader.Token == JsonToken.String) {
493  instance.SetString ((string) reader.Value);
494  return instance;
495  }
496 
497  if (reader.Token == JsonToken.Double) {
498  instance.SetDouble ((double) reader.Value);
499  return instance;
500  }
501 
502  if (reader.Token == JsonToken.Int) {
503  instance.SetInt ((int) reader.Value);
504  return instance;
505  }
506 
507  if (reader.Token == JsonToken.Long) {
508  instance.SetLong ((long) reader.Value);
509  return instance;
510  }
511 
512  if (reader.Token == JsonToken.Boolean) {
513  instance.SetBoolean ((bool) reader.Value);
514  return instance;
515  }
516 
517  if (reader.Token == JsonToken.ArrayStart) {
518  instance.SetJsonType (JsonType.Array);
519 
520  while (true) {
521  IJsonWrapper item = ReadValue (factory, reader);
522  if (item == null && reader.Token == JsonToken.ArrayEnd)
523  break;
524 
525  ((IList) instance).Add (item);
526  }
527  }
528  else if (reader.Token == JsonToken.ObjectStart) {
529  instance.SetJsonType (JsonType.Object);
530 
531  while (true) {
532  reader.Read ();
533 
534  if (reader.Token == JsonToken.ObjectEnd)
535  break;
536 
537  string property = (string) reader.Value;
538 
539  ((IDictionary) instance)[property] = ReadValue (
540  factory, reader);
541  }
542 
543  }
544 
545  return instance;
546  }

691  private static void WriteValue (object obj, JsonWriter writer,
692  bool writer_is_private,
693  int depth)
694  {
695  if (depth > max_nesting_depth)
696  throw new JsonException (
697  String.Format ("Max allowed object depth reached while " +
698  "trying to export from type {0}",
699  obj.GetType ()));
700 
701  if (obj == null) {
702  writer.Write (null);
703  return;
704  }
705 
706  if (obj is IJsonWrapper) {
707  if (writer_is_private)
708  writer.TextWriter.Write (((IJsonWrapper) obj).ToJson ());
709  else
710  ((IJsonWrapper) obj).ToJson (writer);
711 
712  return;
713  }
714 
715  if (obj is String) {
716  writer.Write ((string) obj);
717  return;
718  }
719 
720  if (obj is Double) {
721  writer.Write ((double) obj);
722  return;
723  }
724 
725  if (obj is Int32) {
726  writer.Write ((int) obj);
727  return;
728  }
729 
730  if (obj is Boolean) {
731  writer.Write ((bool) obj);
732  return;
733  }
734 
735  if (obj is Int64) {
736  writer.Write ((long) obj);
737  return;
738  }
739 
740  if (obj is Array) {
741  writer.WriteArrayStart ();
742 
743  foreach (object elem in (Array) obj)
744  WriteValue (elem, writer, writer_is_private, depth + 1);
745 
746  writer.WriteArrayEnd ();
747 
748  return;
749  }
750 
751  if (obj is IList) {
752  writer.WriteArrayStart ();
753  foreach (object elem in (IList) obj)
754  WriteValue (elem, writer, writer_is_private, depth + 1);
755  writer.WriteArrayEnd ();
756 
757  return;
758  }
759 
760  if (obj is IDictionary) {
761  writer.WriteObjectStart ();
762  foreach (DictionaryEntry entry in (IDictionary) obj) {
763  writer.WritePropertyName ((string) entry.Key);
764  WriteValue (entry.Value, writer, writer_is_private,
765  depth + 1);
766  }
767  writer.WriteObjectEnd ();
768 
769  return;
770  }
771 
772  Type obj_type = obj.GetType ();
773 
774  // See if there's a custom exporter for the object
775  if (custom_exporters_table.ContainsKey (obj_type)) {
776  ExporterFunc exporter = custom_exporters_table[obj_type];
777  exporter (obj, writer);
778 
779  return;
780  }
781 
782  // If not, maybe there's a base exporter
783  if (base_exporters_table.ContainsKey (obj_type)) {
784  ExporterFunc exporter = base_exporters_table[obj_type];
785  exporter (obj, writer);
786 
787  return;
788  }
789 
790  // Last option, let's see if it's an enum
791  if (obj is Enum) {
792  Type e_type = Enum.GetUnderlyingType (obj_type);
793 
794  if (e_type == typeof (long)
795  || e_type == typeof (uint)
796  || e_type == typeof (ulong))
797  writer.Write ((ulong) obj);
798  else
799  writer.Write ((int) obj);
800 
801  return;
802  }
803 
804  // Okay, so it looks like the input should be exported as an
805  // object
806  AddTypeProperties (obj_type);
807  IList<PropertyMetadata> props = type_properties[obj_type];
808 
809  writer.WriteObjectStart ();
810  foreach (PropertyMetadata p_data in props) {
811  if (p_data.IsField) {
812  writer.WritePropertyName (p_data.Info.Name);
813  WriteValue (((FieldInfo) p_data.Info).GetValue (obj),
814  writer, writer_is_private, depth + 1);
815  }
816  else {
817  PropertyInfo p_info = (PropertyInfo) p_data.Info;
818 
819  if (p_info.CanRead) {
820  writer.WritePropertyName (p_data.Info.Name);
821  WriteValue (p_info.GetValue (obj, null),
822  writer, writer_is_private, depth + 1);
823  }
824  }
825  }
826  writer.WriteObjectEnd ();
827  }
