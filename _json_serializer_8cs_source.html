482 
483  // Graph was null when saving, ignore
484  if (System.Type.Equals(graphType, null)) return null;

704  private GraphMeta DeserializeBinaryMeta (ZipEntry entry) {
705  var meta = new GraphMeta();
706 
707  var reader = GetBinaryReader(entry);
708 
709  if (reader.ReadString() != "A*") throw new System.Exception("Invalid magic number in saved data");
710  int major = reader.ReadInt32();
711  int minor = reader.ReadInt32();
712  int build = reader.ReadInt32();
713  int revision = reader.ReadInt32();
714 
715  // Required because when saving a version with a field not set, it will save it as -1
716  // and then the Version constructor will throw an exception (which we do not want)
717  if (major < 0) meta.version = new Version(0, 0);
718  else if (minor < 0) meta.version = new Version(major, 0);
719  else if (build < 0) meta.version = new Version(major, minor);
720  else if (revision < 0) meta.version = new Version(major, minor, build);
721  else meta.version = new Version(major, minor, build, revision);
722 
723  meta.graphs = reader.ReadInt32();
724 
725  meta.guids = new List<string>();
726  int count = reader.ReadInt32();
727  for (int i = 0; i < count; i++) meta.guids.Add(reader.ReadString());
728 
729  meta.typeNames = new List<string>();
730  count = reader.ReadInt32();
731  for (int i = 0; i < count; i++) meta.typeNames.Add(reader.ReadString());
732 
733  return meta;
734  }
