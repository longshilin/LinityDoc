61  if (objects.Length == 0)

67  this.types.Add(baseAttribute.AttributeType, type); // types中存放了所有<以BaseAttribute为基类的自定义属性类型，添加该自定义属性的具体方法类型>
68  }

83 
84  if (attrs.Length == 0)

87  }

92  {

98  break;

101  break;
102  case IStartSystem startSystem:
103  this.startSystems.Add(startSystem.Type(), startSystem);
104  break;
105  case IDestroySystem destroySystem:
106  this.destroySystems.Add(destroySystem.Type(), destroySystem);
107  break;
108  case ILoadSystem loadSystem:
109  this.loadSystems.Add(loadSystem.Type(), loadSystem);
110  break;
111  case IChangeSystem changeSystem:
112  this.changeSystems.Add(changeSystem.Type(), changeSystem);
113  break;
