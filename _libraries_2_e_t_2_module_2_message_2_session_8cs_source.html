128  private void Run(MemoryStream memoryStream)
129  {
130  memoryStream.Seek(Packet.MessageIndex, SeekOrigin.Begin);
131  ushort opcode = BitConverter.ToUInt16(memoryStream.GetBuffer(), Packet.OpcodeIndex);
132 
133 #if !SERVER
134  if (OpcodeHelper.IsClientHotfixMessage(opcode))
135  {
136  this.GetComponent<SessionCallbackComponent>().MessageCallback.Invoke(this, opcode, memoryStream);
137  return;
138  }
139 #endif
140 
141  object message;
142  try
143  {
144  OpcodeTypeComponent opcodeTypeComponent = this.Network.Entity.GetComponent<OpcodeTypeComponent>();
145  object instance = opcodeTypeComponent.GetInstance(opcode);
146  message = this.Network.MessagePacker.DeserializeFrom(instance, memoryStream);
147 
148  if (OpcodeHelper.IsNeedDebugLogMessage(opcode))
149  {
150  Log.Msg(message);
151  }
152  }
153  catch (Exception e)
154  {
155  // 出现任何消息解析异常都要断开Session，防止客户端伪造消息
156  Log.Error($"opcode: {opcode} {this.Network.Count} {e}, ip: {this.RemoteAddress}");
157  this.Error = ErrorCode.ERR_PacketParserError;
158  this.Network.Remove(this.Id);
159  return;
160  }
161 
162  if (!(message is IResponse response))
163  {
164  this.Network.MessageDispatcher.Dispatch(this, opcode, message);
165  return;
166  }
167 
168  Action<IResponse> action;
169  if (!this.requestCallback.TryGetValue(response.RpcId, out action))
170  {
171  throw new Exception($"not found rpc, response message: {StringHelper.MessageToStr(response)}");
172  }
173  this.requestCallback.Remove(response.RpcId);
174 
175  action(response);
176  }
