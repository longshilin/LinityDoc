252  private void HandleEvent(object sender, T e)
253  {
254  bool noHandlerException = false;
255  GameFrameworkLinkedListRange<EventHandler<T>> range = default(GameFrameworkLinkedListRange<EventHandler<T>>);
256  if (m_EventHandlers.TryGetValue(e.Id, out range))
257  {
258  LinkedListNode<EventHandler<T>> current = range.First;
259  while (current != null && current != range.Terminal)
260  {
261  m_CachedNodes[e] = current.Next != range.Terminal ? current.Next : null;
262  current.Value(sender, e);
263  current = m_CachedNodes[e];
264  }
265 
266  m_CachedNodes.Remove(e);
267  }
268  else if (m_DefaultHandler != null)
269  {
270  m_DefaultHandler(sender, e);
271  }
272  else if ((m_EventPoolMode & EventPoolMode.AllowNoHandler) == 0)
273  {
274  noHandlerException = true;
275  }
276 
277  ReferencePool.Release(e);
278 
279  if (noHandlerException)
280  {
281  throw new GameFrameworkException(Utility.Text.Format("Event '{0}' not allow no handler.", e.Id.ToString()));
282  }
283  }
