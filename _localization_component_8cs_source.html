118  private void Start()
119  {
120  BaseComponent baseComponent = GameEntry.GetComponent<BaseComponent>();
121  if (baseComponent == null)
122  {
123  Log.Fatal("Base component is invalid.");
124  return;

128  if (m_EventComponent == null)

131  return;
132  }
133 
134  if (baseComponent.EditorResourceMode)
135  {
136  m_LocalizationManager.SetResourceManager(baseComponent.EditorResourceHelper);
137  }
138  else
139  {
140  m_LocalizationManager.SetResourceManager(GameFrameworkEntry.GetModule<IResourceManager>());
141  }
142 
143  LocalizationHelperBase localizationHelper = Helper.CreateHelper(m_LocalizationHelperTypeName, m_CustomLocalizationHelper);
144  if (localizationHelper == null)
145  {
146  Log.Error("Can not create localization helper.");
147  return;
148  }
149 
150  localizationHelper.name = "Localization Helper";
151  Transform transform = localizationHelper.transform;
152  transform.SetParent(this.transform);
153  transform.localScale = Vector3.one;
154 
155  m_LocalizationManager.SetDataProviderHelper(localizationHelper);
156  m_LocalizationManager.SetLocalizationHelper(localizationHelper);
157  m_LocalizationManager.Language = baseComponent.EditorResourceMode && baseComponent.EditorLanguage != Language.Unspecified ? baseComponent.EditorLanguage : m_LocalizationManager.SystemLanguage;
158  if (m_CachedBytesSize > 0)
159  {
160  EnsureCachedBytesSize(m_CachedBytesSize);
161  }
162  }
