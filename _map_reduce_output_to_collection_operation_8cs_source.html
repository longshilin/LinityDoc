140  protected internal override BsonDocument CreateCommand(ICoreSessionHandle session, ConnectionDescription connectionDescription)
141  {

143 
144  var serverVersion = connectionDescription.ServerVersion;
145  if (_bypassDocumentValidation.HasValue && Feature.BypassDocumentValidation.IsSupported(serverVersion))

147  command.Add("bypassDocumentValidation", _bypassDocumentValidation.Value);
