63  Vector3 CalculateBlendPoint () {
64  // Fall back to rotating around the transform position if no feet could be found
65  if (footTransforms[0] == null || footTransforms[1] == null) return tr.position;
66 
67  var leftFootPos = footTransforms[0].position;
68  var rightFootPos = footTransforms[1].position;
69 
70  // This is the same calculation that Unity uses for
71  // Animator.pivotWeight and Animator.pivotPosition
72  // but those properties do not work for all animations apparently.
73  var footVelocity1 = (leftFootPos - prevFootPos[0]) / Time.deltaTime;
74  var footVelocity2 = (rightFootPos - prevFootPos[1]) / Time.deltaTime;
75  float denominator = footVelocity1.magnitude + footVelocity2.magnitude;
76  var pivotWeight = denominator > 0 ? footVelocity1.magnitude / denominator : 0.5f;
77  prevFootPos[0] = leftFootPos;
78  prevFootPos[1] = rightFootPos;
79  var pivotPosition = Vector3.Lerp(leftFootPos, rightFootPos, pivotWeight);
80  return pivotPosition;
81  }
