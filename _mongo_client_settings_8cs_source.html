607  public static MongoClientSettings FromUrl(MongoUrl url)
608  {
609  if (url.Scheme == ConnectionStringScheme.MongoDBPlusSrv)
610  {
611  url = url.Resolve();
612  }
613 
614  var credential = url.GetCredential();
615 
616  var clientSettings = new MongoClientSettings();
617  clientSettings.ApplicationName = url.ApplicationName;
618  clientSettings.ConnectionMode = url.ConnectionMode;
619  clientSettings.ConnectTimeout = url.ConnectTimeout;
620  if (credential != null)
621  {
622  foreach (var property in url.AuthenticationMechanismProperties)
623  {
624  if (property.Key.Equals("CANONICALIZE_HOST_NAME", StringComparison.OrdinalIgnoreCase))
625  {
626  credential = credential.WithMechanismProperty(property.Key, bool.Parse(property.Value));
627  }
628  else
629  {
630  credential = credential.WithMechanismProperty(property.Key, property.Value);
631  }
632  }
633  clientSettings.Credential = credential;
634  }
635  clientSettings.GuidRepresentation = url.GuidRepresentation;
636  clientSettings.HeartbeatInterval = url.HeartbeatInterval;
637  clientSettings.HeartbeatTimeout = url.HeartbeatTimeout;
638  clientSettings.IPv6 = url.IPv6;
639  clientSettings.MaxConnectionIdleTime = url.MaxConnectionIdleTime;
640  clientSettings.MaxConnectionLifeTime = url.MaxConnectionLifeTime;
641  clientSettings.MaxConnectionPoolSize = url.MaxConnectionPoolSize;
642  clientSettings.MinConnectionPoolSize = url.MinConnectionPoolSize;
643  clientSettings.ReadConcern = new ReadConcern(url.ReadConcernLevel);
644  clientSettings.ReadEncoding = null; // ReadEncoding must be provided in code
645  clientSettings.ReadPreference = (url.ReadPreference == null) ? ReadPreference.Primary : url.ReadPreference;
646  clientSettings.ReplicaSetName = url.ReplicaSetName;
647  clientSettings.RetryWrites = url.RetryWrites.GetValueOrDefault(false);
648  clientSettings.LocalThreshold = url.LocalThreshold;
649  clientSettings.Servers = new List<MongoServerAddress>(url.Servers);
650  clientSettings.ServerSelectionTimeout = url.ServerSelectionTimeout;
651  clientSettings.SocketTimeout = url.SocketTimeout;
652  clientSettings.SslSettings = null; // SSL settings must be provided in code
653  clientSettings.UseSsl = url.UseSsl;
654  clientSettings.VerifySslCertificate = url.VerifySslCertificate;
655  clientSettings.WaitQueueSize = url.ComputedWaitQueueSize;
656  clientSettings.WaitQueueTimeout = url.WaitQueueTimeout;
657  clientSettings.WriteConcern = url.GetWriteConcern(true); // WriteConcern is enabled by default for MongoClient
658  clientSettings.WriteEncoding = null; // WriteEncoding must be provided in code
659  return clientSettings;
660  }
