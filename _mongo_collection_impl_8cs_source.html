99  public override IAsyncCursor<TResult> Aggregate<TResult>(IClientSessionHandle session, PipelineDefinition<TDocument, TResult> pipeline, AggregateOptions options, CancellationToken cancellationToken = default(CancellationToken))
100  {
101  Ensure.IsNotNull(session, nameof(session));
102  var renderedPipeline = Ensure.IsNotNull(pipeline, nameof(pipeline)).Render(_documentSerializer, _settings.SerializerRegistry);
103  options = options ?? new AggregateOptions();
104 
105  var last = renderedPipeline.Documents.LastOrDefault();
106  if (last != null && last.GetElement(0).Name == "$out")
107  {
108  var aggregateOperation = CreateAggregateToCollectionOperation(renderedPipeline, options);
109  ExecuteWriteOperation(session, aggregateOperation, cancellationToken);
110 
111  // we want to delay execution of the find because the user may
112  // not want to iterate the results at all...
113  var findOperation = CreateAggregateToCollectionFindOperation(last, renderedPipeline.OutputSerializer, options);
114  var forkedSession = session.Fork();
115  var deferredCursor = new DeferredAsyncCursor<TResult>(
116  () => forkedSession.Dispose(),
117  ct => ExecuteReadOperation(forkedSession, findOperation, ReadPreference.Primary, ct),
118  ct => ExecuteReadOperationAsync(forkedSession, findOperation, ReadPreference.Primary, ct));
119  return deferredCursor;
120  }
121  else
122  {
123  var aggregateOperation = CreateAggregateOperation(renderedPipeline, options);
124  return ExecuteReadOperation(session, aggregateOperation, cancellationToken);
125  }
126  }
