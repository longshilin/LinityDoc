397  internal IAuthenticator ToAuthenticator()
398  {
399  var passwordEvidence = _evidence as PasswordEvidence;
400  if (passwordEvidence != null)
401  {
402  var insecurePassword = MongoUtils.ToInsecureString(passwordEvidence.SecurePassword);
403  var credential = new UsernamePasswordCredential(
404  _identity.Source,
405  _identity.Username,
406  insecurePassword);
407  if (_mechanism == null)
408  {
409  return new DefaultAuthenticator(credential);
410  }
411 #pragma warning disable 618
412  else if (_mechanism == MongoDBCRAuthenticator.MechanismName)
413  {
414  return new MongoDBCRAuthenticator(credential);
415 #pragma warning restore 618
416  }
417  else if (_mechanism == ScramSha1Authenticator.MechanismName)
418  {
419  return new ScramSha1Authenticator(credential);
420  }
421  else if (_mechanism == ScramSha256Authenticator.MechanismName)
422  {
423  return new ScramSha256Authenticator(credential);
424  }
425  else if (_mechanism == PlainAuthenticator.MechanismName)
426  {
427  return new PlainAuthenticator(credential);
428  }
429  else if (_mechanism == GssapiAuthenticator.MechanismName)
430  {
431  return new GssapiAuthenticator(
432  credential,
433  _mechanismProperties.Select(x => new KeyValuePair<string, string>(x.Key, x.Value.ToString())));
434  }
435  }
436  else if (_identity.Source == "$external" && _evidence is ExternalEvidence)
437  {
438  if (_mechanism == MongoDBX509Authenticator.MechanismName)
439  {
440  return new MongoDBX509Authenticator(_identity.Username);
441  }
442  else if (_mechanism == GssapiAuthenticator.MechanismName)
443  {
444  return new GssapiAuthenticator(
445  _identity.Username,
446  _mechanismProperties.Select(x => new KeyValuePair<string, string>(x.Key, x.Value.ToString())));
447  }
448  }
449 
450  throw new NotSupportedException("Unable to create an authenticator.");
451  }

473  // private static methods
474  private static MongoCredential FromComponents(string mechanism, string source, string username, MongoIdentityEvidence evidence)
475  {
476  var defaultedMechanism = (mechanism ?? "DEFAULT").Trim().ToUpperInvariant();
477  switch (defaultedMechanism)
478  {
479  case "DEFAULT":
480  case "MONGODB-CR":
481  case "SCRAM-SHA-1":
482  case "SCRAM-SHA-256":
483  // it is allowed for a password to be an empty string, but not a username
484  source = source ?? "admin";
485  if (evidence == null || !(evidence is PasswordEvidence))
486  {
487  var message = string.Format("A {0} credential must have a password.", defaultedMechanism);
488  throw new ArgumentException(message);
489  }
490 
491  return new MongoCredential(
492  mechanism,
493  new MongoInternalIdentity(source, username),
494  evidence);
495  case "MONGODB-X509":
496  // always $external for X509.
497  source = "$external";
498  if (evidence == null || !(evidence is ExternalEvidence))
499  {
500  throw new ArgumentException("A MONGODB-X509 does not support a password.");
501  }
502 
503  return new MongoCredential(
504  mechanism,
505  new MongoX509Identity(username),
506  evidence);
507  case "GSSAPI":
508  // always $external for GSSAPI.
509  source = "$external";
510 
511  return new MongoCredential(
512  "GSSAPI",
513  new MongoExternalIdentity(source, username),
514  evidence);
515  case "PLAIN":
516  source = source ?? "admin";
517  if (evidence == null || !(evidence is PasswordEvidence))
518  {
519  throw new ArgumentException("A PLAIN credential must have a password.");
520  }
521 
522  MongoIdentity identity;
523  if (source == "$external")
524  {
525  identity = new MongoExternalIdentity(source, username);
526  }
527  else
528  {
529  identity = new MongoInternalIdentity(source, username);
530  }
531 
532  return new MongoCredential(
533  mechanism,
534  identity,
535  evidence);
536  default:
537  throw new NotSupportedException(string.Format("Unsupported MongoAuthenticationMechanism {0}.", mechanism));
538  }
539  }
