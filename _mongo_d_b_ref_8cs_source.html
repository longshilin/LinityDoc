220  // public methods
229  public bool TryGetMemberSerializationInfo(string memberName, out BsonSerializationInfo serializationInfo)
230  {
231  string elementName;
232  IBsonSerializer serializer;
233 
234  switch (memberName)
235  {
236  case "DatabaseName":
237  elementName = "$db";
238  serializer = new StringSerializer();
239  break;
240  case "CollectionName":
241  elementName = "$ref";
242  serializer = new StringSerializer();
243  break;
244  case "Id":
245  elementName = "$id";
246  serializer = BsonValueSerializer.Instance;
247  break;
248  default:
249  serializationInfo = null;
250  return false;
251  }
252 
253  serializationInfo = new BsonSerializationInfo(elementName, serializer, serializer.ValueType);
254  return true;
255  }

257  // protected methods
264  protected override MongoDBRef DeserializeValue(BsonDeserializationContext context, BsonDeserializationArgs args)
265  {
266  var bsonReader = context.Reader;
267 
268  string databaseName = null;
269  string collectionName = null;
270  BsonValue id = null;
271 
272  _helper.DeserializeMembers(context, (elementName, flag) =>
273  {
274  switch (flag)
275  {
276  case Flags.CollectionName: collectionName = bsonReader.ReadString(); break;
277  case Flags.Id: id = BsonValueSerializer.Instance.Deserialize(context); break;
278  case Flags.DatabaseName: databaseName = bsonReader.ReadString(); break;
279  }
280  });
281 
282  return new MongoDBRef(databaseName, collectionName, id);
283  }
284 
291  protected override void SerializeValue(BsonSerializationContext context, BsonSerializationArgs args, MongoDBRef value)
292  {
293  var bsonWriter = context.Writer;
294 
295  bsonWriter.WriteStartDocument();
296  bsonWriter.WriteString("$ref", value.CollectionName);
297  bsonWriter.WriteName("$id");
298  BsonValueSerializer.Instance.Serialize(context, value.Id);
299  if (value.DatabaseName != null)
300  {
301  bsonWriter.WriteString("$db", value.DatabaseName);
302  }
303  bsonWriter.WriteEndDocument();
304  }
305  }
