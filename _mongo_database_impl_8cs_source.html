415  private CreateCollectionOperation CreateCreateCollectionOperation<TDocument>(string name, CreateCollectionOptions<TDocument> options)
416  {
417  var messageEncoderSettings = GetMessageEncoderSettings();
418  BsonDocument validator = null;
419  if (options.Validator != null)
420  {
421  var serializerRegistry = options.SerializerRegistry ?? BsonSerializer.SerializerRegistry;
422  var documentSerializer = options.DocumentSerializer ?? serializerRegistry.GetSerializer<TDocument>();
423  validator = options.Validator.Render(documentSerializer, serializerRegistry);
424  }
425 
426 #pragma warning disable 618
427  return new CreateCollectionOperation(new CollectionNamespace(_databaseNamespace, name), messageEncoderSettings)
428  {
429  AutoIndexId = options.AutoIndexId,
430  Capped = options.Capped,
431  Collation = options.Collation,
432  IndexOptionDefaults = options.IndexOptionDefaults?.ToBsonDocument(),
433  MaxDocuments = options.MaxDocuments,
434  MaxSize = options.MaxSize,
435  NoPadding = options.NoPadding,
436  StorageEngine = options.StorageEngine,
437  UsePowerOf2Sizes = options.UsePowerOf2Sizes,
438  ValidationAction = options.ValidationAction,
439  ValidationLevel = options.ValidationLevel,
440  Validator = validator,
441  WriteConcern = _settings.WriteConcern
442  };
443 #pragma warning restore
444  }

488  {

492  }
