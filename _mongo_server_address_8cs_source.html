84  public static bool TryParse(string value, out MongoServerAddress address)
85  {
86  // don't throw ArgumentNullException if value is null
87  if (value != null)
88  {
89  Match match = Regex.Match(value, @"^(?<host>(\[[^]]+\]|[^:\[\]]+))(:(?<port>\d+))?$");
90  if (match.Success)
91  {
92  string host = match.Groups["host"].Value;
93  string portString = match.Groups["port"].Value;
94  int port = (portString == "") ? 27017 : JsonConvert.ToInt32(portString);
95  address = new MongoServerAddress(host, port);
96  return true;
97  }
98  }
99 
100  address = null;
101  return false;
