493  public static MongoUrl Create(string url)
494  {
495  // cache previously seen urls to avoid repeated parsing
496  lock (__staticLock)
497  {
498  MongoUrl mongoUrl;
499  if (!__cache.TryGetValue(url, out mongoUrl))
500  {
501  mongoUrl = new MongoUrl(url);
502  var canonicalUrl = mongoUrl.ToString();
503  if (canonicalUrl != url)
504  {
505  if (__cache.ContainsKey(canonicalUrl))
506  {
507  mongoUrl = __cache[canonicalUrl]; // use existing MongoUrl
508  }
509  else
510  {
511  __cache[canonicalUrl] = mongoUrl; // cache under canonicalUrl also
512  }
513  }
514  __cache[url] = mongoUrl;
515  }
516  return mongoUrl;
517  }
518  }
