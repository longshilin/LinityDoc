619  public void Parse(string url)
620  {
621  var connectionString = new ConnectionString(url);
622  _applicationName = connectionString.ApplicationName;
623  _authenticationMechanism = connectionString.AuthMechanism;
624  _authenticationMechanismProperties = connectionString.AuthMechanismProperties.ToDictionary(x => x.Key, x => x.Value, StringComparer.OrdinalIgnoreCase);
625  _authenticationSource = connectionString.AuthSource;
626  switch (connectionString.Connect)
627  {
628  case ClusterConnectionMode.Direct:
629  _connectionMode = Driver.ConnectionMode.Direct;
630  break;
631  case ClusterConnectionMode.ReplicaSet:
632  _connectionMode = Driver.ConnectionMode.ReplicaSet;
633  break;
634  case ClusterConnectionMode.Sharded:
635  _connectionMode = Driver.ConnectionMode.ShardRouter;
636  break;
637  case ClusterConnectionMode.Standalone:
638  _connectionMode = Driver.ConnectionMode.Standalone;
639  break;
640  default:
641  _connectionMode = Driver.ConnectionMode.Automatic;
642  break;
643  }
644  _connectTimeout = connectionString.ConnectTimeout.GetValueOrDefault(MongoDefaults.ConnectTimeout);
645  _databaseName = connectionString.DatabaseName;
646  _fsync = connectionString.FSync;
647  _guidRepresentation = connectionString.UuidRepresentation.GetValueOrDefault(MongoDefaults.GuidRepresentation);
648  _heartbeatInterval = connectionString.HeartbeatInterval ?? ServerSettings.DefaultHeartbeatInterval;
649  _heartbeatTimeout = connectionString.HeartbeatTimeout ?? ServerSettings.DefaultHeartbeatTimeout;
650  _ipv6 = connectionString.Ipv6.GetValueOrDefault(false);
651  _journal = connectionString.Journal;
652  _maxConnectionIdleTime = connectionString.MaxIdleTime.GetValueOrDefault(MongoDefaults.MaxConnectionIdleTime);
653  _maxConnectionLifeTime = connectionString.MaxLifeTime.GetValueOrDefault(MongoDefaults.MaxConnectionLifeTime);
654  _maxConnectionPoolSize = connectionString.MaxPoolSize.GetValueOrDefault(MongoDefaults.MaxConnectionPoolSize);
655  _minConnectionPoolSize = connectionString.MinPoolSize.GetValueOrDefault(MongoDefaults.MinConnectionPoolSize);
656  _password = connectionString.Password;
657  _readConcernLevel = connectionString.ReadConcernLevel;
658  if (connectionString.ReadPreference.HasValue || connectionString.ReadPreferenceTags != null || connectionString.MaxStaleness.HasValue)
659  {
660  if (!connectionString.ReadPreference.HasValue)
661  {
662  throw new MongoConfigurationException("readPreference mode is required when using tag sets or max staleness.");
663  }
664  _readPreference = new ReadPreference(connectionString.ReadPreference.Value, connectionString.ReadPreferenceTags, connectionString.MaxStaleness);
665  }
666  _replicaSetName = connectionString.ReplicaSet;
667  _retryWrites = connectionString.RetryWrites;
668  _localThreshold = connectionString.LocalThreshold.GetValueOrDefault(MongoDefaults.LocalThreshold);
669  _scheme = connectionString.Scheme;
670  _servers = connectionString.Hosts.Select(endPoint =>
671  {
672  DnsEndPoint dnsEndPoint;
673  IPEndPoint ipEndPoint;
674  if ((dnsEndPoint = endPoint as DnsEndPoint) != null)
675  {
676  return new MongoServerAddress(dnsEndPoint.Host, dnsEndPoint.Port);
677  }
678  else if ((ipEndPoint = endPoint as IPEndPoint) != null)
679  {
680  var address = ipEndPoint.Address.ToString();
681  if (ipEndPoint.AddressFamily == System.Net.Sockets.AddressFamily.InterNetworkV6)
682  {
683  address = "[" + address + "]";
684  }
685  return new MongoServerAddress(address, ipEndPoint.Port);
686  }
687  else
688  {
689  throw new NotSupportedException("Only DnsEndPoint and IPEndPoints are supported in the connection string.");
690  }
691  });
692  _serverSelectionTimeout = connectionString.ServerSelectionTimeout.GetValueOrDefault(MongoDefaults.ServerSelectionTimeout);
693  _socketTimeout = connectionString.SocketTimeout.GetValueOrDefault(MongoDefaults.SocketTimeout);
694  _username = connectionString.Username;
695  _useSsl = connectionString.Ssl.GetValueOrDefault(false);
696  _verifySslCertificate = connectionString.SslVerifyCertificate.GetValueOrDefault(true);
697  _w = connectionString.W;
698  if (connectionString.WaitQueueSize != null)
699  {
700  _waitQueueSize = connectionString.WaitQueueSize.Value;
701  _waitQueueMultiple = 0.0;
702  }
703  else if (connectionString.WaitQueueMultiple != null)
704  {
705  _waitQueueMultiple = connectionString.WaitQueueMultiple.Value;
706  _waitQueueSize = 0;
707  }
708  _waitQueueTimeout = connectionString.WaitQueueTimeout.GetValueOrDefault(MongoDefaults.WaitQueueTimeout);
709  _wTimeout = connectionString.WTimeout;
710  }

725  public override string ToString()
726  {
727  StringBuilder url = new StringBuilder();
728  if (_scheme == ConnectionStringScheme.MongoDB)
729  {
730  url.Append("mongodb://");
731  }
732  else
733  {
734  url.Append("mongodb+srv://");
735  }
736  if (!string.IsNullOrEmpty(_username))
737  {
738  url.Append(Uri.EscapeDataString(_username));
739  if (_password != null)
740  {
741  url.AppendFormat(":{0}", Uri.EscapeDataString(_password));
742  }
743  url.Append("@");
744  }
745  else if (_password != null)
746  {
747  // this would be weird and we really shouldn't be here...
748  url.AppendFormat(":{0}@", _password);
749  }
750  if (_servers != null)
751  {
752  bool firstServer = true;
753  foreach (MongoServerAddress server in _servers)
754  {
755  if (!firstServer) { url.Append(","); }
756  if (server.Port == 27017)
757  {
758  url.Append(server.Host);
759  }
760  else
761  {
762  url.AppendFormat("{0}:{1}", server.Host, server.Port);
763  }
764  firstServer = false;
765  }
766  }
767  if (_databaseName != null)
768  {
769  url.Append("/");
770  url.Append(_databaseName);
771  }
772  var query = new StringBuilder();
773  if (_authenticationMechanism != null)
774  {
775  query.AppendFormat("authMechanism={0};", _authenticationMechanism);
776  }
777  if (_authenticationMechanismProperties.Any())
778  {
779  query.AppendFormat(
780  "authMechanismProperties={0};",
781  string.Join(",", _authenticationMechanismProperties
782  .Select(x => string.Format("{0}:{1}", x.Key, x.Value)).ToArray()));
783  }
784  if (_authenticationSource != null)
785  {
786  query.AppendFormat("authSource={0};", _authenticationSource);
787  }
788  if (_applicationName != null)
789  {
790  query.AppendFormat("appname={0};", _applicationName);
791  }
792  if (_ipv6)
793  {
794  query.AppendFormat("ipv6=true;");
795  }
796  if (_useSsl)
797  {
798  query.AppendFormat("ssl=true;");
799  }
800  if (!_verifySslCertificate)
801  {
802  query.AppendFormat("sslVerifyCertificate=false;");
803  }
804  if (_connectionMode != ConnectionMode.Automatic)
805  {
806  query.AppendFormat("connect={0};", MongoUtils.ToCamelCase(_connectionMode.ToString()));
807  }
808  if (!string.IsNullOrEmpty(_replicaSetName))
809  {
810  query.AppendFormat("replicaSet={0};", _replicaSetName);
811  }
812  if (_readConcernLevel != null)
813  {
814  query.AppendFormat("readConcernLevel={0};", MongoUtils.ToCamelCase(_readConcernLevel.Value.ToString()));
815  }
816  if (_readPreference != null)
817  {
818  query.AppendFormat("readPreference={0};", MongoUtils.ToCamelCase(_readPreference.ReadPreferenceMode.ToString()));
819  if (_readPreference.TagSets != null)
820  {
821  foreach (var tagSet in _readPreference.TagSets)
822  {
823  query.AppendFormat("readPreferenceTags={0};", string.Join(",", tagSet.Tags.Select(t => string.Format("{0}:{1}", t.Name, t.Value)).ToArray()));
824  }
825  }
826  if (_readPreference.MaxStaleness.HasValue)
827  {
828  query.AppendFormat("maxStaleness={0};", FormatTimeSpan(_readPreference.MaxStaleness.Value));
829  }
830  }
831  if (_fsync != null)
832  {
833  query.AppendFormat("fsync={0};", JsonConvert.ToString(_fsync.Value));
834  }
835  if (_journal != null)
836  {
837  query.AppendFormat("journal={0};", JsonConvert.ToString(_journal.Value));
838  }
839  if (_w != null)
840  {
841  query.AppendFormat("w={0};", _w);
842  }
843  if (_wTimeout != null)
844  {
845  query.AppendFormat("wtimeout={0};", FormatTimeSpan(_wTimeout.Value));
846  }
847  if (_connectTimeout != MongoDefaults.ConnectTimeout)
848  {
849  query.AppendFormat("connectTimeout={0};", FormatTimeSpan(_connectTimeout));
850  }
851  if (_heartbeatInterval != ServerSettings.DefaultHeartbeatInterval)
852  {
853  query.AppendFormat("heartbeatInterval={0};", FormatTimeSpan(_heartbeatInterval));
854  }
855  if (_heartbeatTimeout != ServerSettings.DefaultHeartbeatTimeout)
856  {
857  query.AppendFormat("heartbeatTimeout={0};", FormatTimeSpan(_heartbeatTimeout));
858  }
859  if (_maxConnectionIdleTime != MongoDefaults.MaxConnectionIdleTime)
860  {
861  query.AppendFormat("maxIdleTime={0};", FormatTimeSpan(_maxConnectionIdleTime));
862  }
863  if (_maxConnectionLifeTime != MongoDefaults.MaxConnectionLifeTime)
864  {
865  query.AppendFormat("maxLifeTime={0};", FormatTimeSpan(_maxConnectionLifeTime));
866  }
867  if (_maxConnectionPoolSize != MongoDefaults.MaxConnectionPoolSize)
868  {
869  query.AppendFormat("maxPoolSize={0};", _maxConnectionPoolSize);
870  }
871  if (_minConnectionPoolSize != MongoDefaults.MinConnectionPoolSize)
872  {
873  query.AppendFormat("minPoolSize={0};", _minConnectionPoolSize);
874  }
875  if (_localThreshold != MongoDefaults.LocalThreshold)
876  {
877  query.AppendFormat("localThreshold={0};", FormatTimeSpan(_localThreshold));
878  }
879  if (_serverSelectionTimeout != MongoDefaults.ServerSelectionTimeout)
880  {
881  query.AppendFormat("serverSelectionTimeout={0};", FormatTimeSpan(_serverSelectionTimeout));
882  }
883  if (_socketTimeout != MongoDefaults.SocketTimeout)
884  {
885  query.AppendFormat("socketTimeout={0};", FormatTimeSpan(_socketTimeout));
886  }
887  if (_waitQueueMultiple != 0.0 && _waitQueueMultiple != MongoDefaults.WaitQueueMultiple)
888  {
889  query.AppendFormat("waitQueueMultiple={0};", _waitQueueMultiple);
890  }
891  if (_waitQueueSize != 0 && _waitQueueSize != MongoDefaults.WaitQueueSize)
892  {
893  query.AppendFormat("waitQueueSize={0};", _waitQueueSize);
894  }
895  if (_waitQueueTimeout != MongoDefaults.WaitQueueTimeout)
896  {
897  query.AppendFormat("waitQueueTimeout={0};", FormatTimeSpan(WaitQueueTimeout));
898  }
899  if (_guidRepresentation != MongoDefaults.GuidRepresentation)
900  {
901  query.AppendFormat("uuidRepresentation={0};", (_guidRepresentation == GuidRepresentation.CSharpLegacy) ? "csharpLegacy" : MongoUtils.ToCamelCase(_guidRepresentation.ToString()));
902  }
903  if (_retryWrites.GetValueOrDefault(false))
904  {
