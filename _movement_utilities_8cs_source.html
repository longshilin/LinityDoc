21  public static Vector2 ClampVelocity (Vector2 velocity, float maxSpeed, float slowdownFactor, bool slowWhenNotFacingTarget, Vector2 forward) {
22  // Max speed to use for this frame
23  var currentMaxSpeed = maxSpeed * slowdownFactor;
24 
25  // Check if the agent should slow down in case it is not facing the direction it wants to move in
26  if (slowWhenNotFacingTarget && (forward.x != 0 || forward.y != 0)) {
27  float currentSpeed;
28  var normalizedVelocity = VectorMath.Normalize(velocity.ToPFV2(), out currentSpeed);
29  float dot = Vector2.Dot(normalizedVelocity.ToUnityV2(), forward);
30 
31  // Lower the speed when the character's forward direction is not pointing towards the desired velocity
32  // 1 when velocity is in the same direction as forward
33  // 0.2 when they point in the opposite directions
34  float directionSpeedFactor = Mathf.Clamp(dot+0.707f, 0.2f, 1.0f);
35  currentMaxSpeed *= directionSpeedFactor;
36  currentSpeed = Mathf.Min(currentSpeed, currentMaxSpeed);
37 
38  // Angle between the forwards direction of the character and our desired velocity
39  float angle = Mathf.Acos(Mathf.Clamp(dot, -1, 1));
40 
41  // Clamp the angle to 20 degrees
42  // We cannot keep the velocity exactly in the forwards direction of the character
43  // because we use the rotation to determine in which direction to rotate and if
44  // the velocity would always be in the forwards direction of the character then
45  // the character would never rotate.
46  // Allow larger angles when near the end of the path to prevent oscillations.
47  angle = Mathf.Min(angle, (20f + 180f*(1 - slowdownFactor*slowdownFactor))*Mathf.Deg2Rad);
48 
49  float sin = Mathf.Sin(angle);
50  float cos = Mathf.Cos(angle);
51 
52  // Determine if we should rotate clockwise or counter-clockwise to move towards the current velocity
53  sin *= Mathf.Sign(normalizedVelocity.x*forward.y - normalizedVelocity.y*forward.x);
54  // Rotate the #forward vector by #angle radians
55  // The rotation is done using an inlined rotation matrix.
56  // See https://en.wikipedia.org/wiki/Rotation_matrix
57  return new Vector2(forward.x*cos + forward.y*sin, forward.y*cos - forward.x*sin) * currentSpeed;
58  } else {
59  return Vector2.ClampMagnitude(velocity, currentMaxSpeed);
60  }
61  }

64  public static Vector2 CalculateAccelerationToReachPoint (Vector2 deltaPosition, Vector2 targetVelocity, Vector2 currentVelocity, float forwardsAcceleration, float rotationSpeed, float maxSpeed, Vector2 forwardsVector) {
65  // Guard against div by zero
66  if (forwardsAcceleration <= 0) return Vector2.zero;
67 
68  float currentSpeed = currentVelocity.magnitude;
69 
70  // Convert rotation speed to an acceleration
71  // See https://en.wikipedia.org/wiki/Centripetal_force
72  var sidewaysAcceleration = currentSpeed * rotationSpeed * Mathf.Deg2Rad;
73 
74  // To avoid weird behaviour when the rotation speed is very low we allow the agent to accelerate sideways without rotating much
75  // if the rotation speed is very small. Also guards against division by zero.
76  sidewaysAcceleration = Mathf.Max(sidewaysAcceleration, forwardsAcceleration);
77  sidewaysAcceleration = forwardsAcceleration;
78 
79  // Transform coordinates to local space where +X is the forwards direction
80  // This is essentially equivalent to Transform.InverseTransformDirection.
81  deltaPosition = VectorMath.ComplexMultiplyConjugate(deltaPosition.ToPFV2(), forwardsVector.ToPFV2()).ToUnityV2();
82  targetVelocity = VectorMath.ComplexMultiplyConjugate(targetVelocity.ToPFV2(), forwardsVector.ToPFV2()).ToUnityV2();
83  currentVelocity = VectorMath.ComplexMultiplyConjugate(currentVelocity.ToPFV2(), forwardsVector.ToPFV2()).ToUnityV2();
84  float ellipseSqrFactorX = 1 / (forwardsAcceleration*forwardsAcceleration);
85  float ellipseSqrFactorY = 1 / (sidewaysAcceleration*sidewaysAcceleration);
86 
87  // If the target velocity is zero we can use a more fancy approach
88  // and calculate a nicer path.
89  // In particular, this is the case at the end of the path.
90  if (targetVelocity == Vector2.zero) {
91  // Run a binary search over the time to get to the target point.
92  float mn = 0.01f;
93  float mx = 10;
94  while (mx - mn > 0.01f) {
95  var time = (mx + mn) * 0.5f;
96 
97  // Given that we want to move deltaPosition units from out current position, that our current velocity is given
98  // and that when we reach the target we want our velocity to be zero. Also assume that our acceleration will
99  // vary linearly during the slowdown. Then we can calculate what our acceleration should be during this frame.
100 
101  //{ t = time
102  //{ deltaPosition = vt + at^2/2 + qt^3/6
103  //{ 0 = v + at + qt^2/2
104  //{ solve for a
105  // a = acceleration vector
106  // q = derivative of the acceleration vector
107  var a = (6*deltaPosition - 4*time*currentVelocity)/(time*time);
108  var q = 6*(time*currentVelocity - 2*deltaPosition)/(time*time*time);
109 
110  // Make sure the acceleration is not greater than our maximum allowed acceleration.
111  // If it is we increase the time we want to use to get to the target
112  // and if it is not, we decrease the time to get there faster.
113  // Since the acceleration is described by acceleration = a + q*t
114  // we only need to check at t=0 and t=time.
115  // Note that the acceleration limit is described by an ellipse, not a circle.
116  var nextA = a + q*time;
117  if (a.x*a.x*ellipseSqrFactorX + a.y*a.y*ellipseSqrFactorY > 1.0f || nextA.x*nextA.x*ellipseSqrFactorX + nextA.y*nextA.y*ellipseSqrFactorY > 1.0f) {
118  mn = time;
119  } else {
120  mx = time;
121  }
122  }
123 
124  var finalAcceleration = (6*deltaPosition - 4*mx*currentVelocity)/(mx*mx);
125 
126  // Boosting
127  {
128  // The trajectory calculated above has a tendency to use very wide arcs
129  // and that does unfortunately not look particularly good in some cases.
130  // Here we amplify the component of the acceleration that is perpendicular
131  // to our current velocity. This will make the agent turn towards the
132  // target quicker.
133  // How much amplification to use. Value is unitless.
134  const float Boost = 1;
135  finalAcceleration.y *= 1 + Boost;
136 
137  // Clamp the velocity to the maximum acceleration.
138  // Note that the maximum acceleration constraint is shaped like an ellipse, not like a circle.
139  float ellipseMagnitude = finalAcceleration.x*finalAcceleration.x*ellipseSqrFactorX + finalAcceleration.y*finalAcceleration.y*ellipseSqrFactorY;
140  if (ellipseMagnitude > 1.0f) finalAcceleration /= Mathf.Sqrt(ellipseMagnitude);
141  }
142 
143  return VectorMath.ComplexMultiply(finalAcceleration.ToPFV2(), forwardsVector.ToPFV2()).ToUnityV2();
144  } else {
145  // Here we try to move towards the next waypoint which has been modified slightly using our
146  // desired velocity at that point so that the agent will more smoothly round the corner.
147 
148  // How much to strive for making sure we reach the target point with the target velocity. Unitless.
149  const float TargetVelocityWeight = 0.5f;
150 
151  // Limit to how much to care about the target velocity. Value is in seconds.
152  // This prevents the character from moving away from the path too much when the target point is far away
153  const float TargetVelocityWeightLimit = 1.5f;
154  float targetSpeed;
155  var normalizedTargetVelocity = VectorMath.Normalize(targetVelocity.ToPFV2(), out targetSpeed);
156 
157  var distance = deltaPosition.magnitude;
158  var targetPoint = deltaPosition.ToPFV2() - normalizedTargetVelocity * System.Math.Min(TargetVelocityWeight * distance * targetSpeed / (currentSpeed + targetSpeed), maxSpeed*TargetVelocityWeightLimit);
159 
160  // How quickly the agent will try to reach the velocity that we want it to have.
161  // We need this to prevent oscillations and jitter which is what happens if
162  // we let the constant go towards zero. Value is in seconds.
163  const float TimeToReachDesiredVelocity = 0.1f;
164  // TODO: Clamp to ellipse using more accurate acceleration (use rotation speed as well)
165  var finalAcceleration = (targetPoint.normalized*maxSpeed - currentVelocity.ToPFV2()) * (1f/TimeToReachDesiredVelocity);
166 
167  // Clamp the velocity to the maximum acceleration.
168  // Note that the maximum acceleration constraint is shaped like an ellipse, not like a circle.
169  float ellipseMagnitude = finalAcceleration.x*finalAcceleration.x*ellipseSqrFactorX + finalAcceleration.y*finalAcceleration.y*ellipseSqrFactorY;
170  if (ellipseMagnitude > 1.0f) finalAcceleration /= Mathf.Sqrt(ellipseMagnitude);
171 
172  return VectorMath.ComplexMultiply(finalAcceleration, forwardsVector.ToPFV2()).ToUnityV2();
173  }
174  }
