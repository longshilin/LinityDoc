124  public override void Initialize()
125  {
126  base.Initialize();
127  if (_state.TryChange(State.Initial, State.Open))
128  {
129  if (_openingEventHandler != null)
130  {
131  _openingEventHandler(new ClusterOpeningEvent(ClusterId, Settings));
132  }
133 
134  var stopwatch = Stopwatch.StartNew();
135  MonitorServersAsync().ConfigureAwait(false);
136  // We lock here even though AddServer locks. Monitors
137  // are re-entrant such that this won't cause problems,
138  // but could prevent issues of conflicting reports
139  // from servers that are quick to respond.
140  var clusterDescription = Description.WithType(Settings.ConnectionMode.ToClusterType());
141  var newServers = new List<IClusterableServer>();
142  lock (_serversLock)
143  {
144  foreach (var endPoint in Settings.EndPoints)
145  {
146  clusterDescription = EnsureServer(clusterDescription, endPoint, newServers);
147  }
148  }
149 
150  stopwatch.Stop();
151 
152  UpdateClusterDescription(clusterDescription);
153 
154  foreach (var server in newServers)
155  {
156  server.Initialize();
157  }
158 
159  if (_openedEventHandler != null)
160  {
161  _openedEventHandler(new ClusterOpenedEvent(ClusterId, Settings, stopwatch.Elapsed));

166  private bool IsServerValidForCluster(ClusterType clusterType, ClusterConnectionMode connectionMode, ServerType serverType)
167  {
168  switch (clusterType)
169  {
170  case ClusterType.ReplicaSet:
171  return serverType.IsReplicaSetMember();
172 
173  case ClusterType.Sharded:
174  return serverType == ServerType.ShardRouter;
175 
176  case ClusterType.Unknown:
177  switch (connectionMode)
178  {
179  case ClusterConnectionMode.Automatic:
180  return serverType.IsReplicaSetMember() || serverType == ServerType.ShardRouter;
181 
182  default:
183  throw new MongoInternalException("Unexpected connection mode.");
184  }
185 
186  default:
187  throw new MongoInternalException("Unexpected cluster type.");
188  }

191  protected override void RequestHeartbeat()
192  {
193  List<IClusterableServer> servers;
194  lock (_serversLock)
195  {
196  servers = _servers.ToList();
197  }
198 
199  foreach (var server in servers)
200  {
201  if (server.IsInitialized)
202  {
203  try
204  {
205  server.RequestHeartbeat();
206  }
207  catch (ObjectDisposedException)
208  {
209  // There is a possible race condition here
210  // due to the fact that we are working
211  // with the server outside of the lock,
212  // meaning another thread could remove
213  // the server and dispose of it before
214  // we invoke the method.
215  }
216  }

220  private async Task MonitorServersAsync()
221  {
222  var monitorServersCancellationToken = _monitorServersCancellationTokenSource.Token;
223  while (!monitorServersCancellationToken.IsCancellationRequested)
224  {
225  try
226  {
227  var eventArgs = await _serverDescriptionChangedQueue.DequeueAsync(monitorServersCancellationToken).ConfigureAwait(false); // TODO: add timeout and cancellationToken to DequeueAsync
228  ProcessServerDescriptionChanged(eventArgs);
229  }
230  catch (OperationCanceledException) when (monitorServersCancellationToken.IsCancellationRequested)
231  {
232  // ignore OperationCanceledException when monitor servers cancellation is requested
233  }
234  catch (Exception unexpectedException)
235  {
236  // if we catch an exception here it's because of a bug in the driver
237 
238  var handler = _sdamInformationEventHandler;
239  if (handler != null)
240  {
241  try
242  {
243  handler.Invoke(new SdamInformationEvent(() =>
244  string.Format(
245  "Unexpected exception in MultiServerCluster.MonitorServersAsync: {0}",
246  unexpectedException.ToString())));
247  }
248  catch
249  {
250  // ignore any exceptions thrown by the handler (note: event handlers aren't supposed to throw exceptions)
251  }
252  }
253 
254  // TODO: should we reset the cluster state in some way? (the state is undefined since an unexpected exception was thrown)
255  }
256  }

264  private void ProcessServerDescriptionChanged(ServerDescriptionChangedEventArgs args)
265  {
266  var newServerDescription = args.NewServerDescription;
267  var newClusterDescription = Description;
268 
269  if (!_servers.Any(x => EndPointHelper.Equals(x.EndPoint, newServerDescription.EndPoint)))
270  {
271  return;
272  }
273 
274  var newServers = new List<IClusterableServer>();
275  if (newServerDescription.State == ServerState.Disconnected)
276  {
277  newClusterDescription = newClusterDescription.WithServerDescription(newServerDescription);
278  }
279  else
280  {
281  if (IsServerValidForCluster(newClusterDescription.Type, Settings.ConnectionMode, newServerDescription.Type))
282  {
283  if (newClusterDescription.Type == ClusterType.Unknown)
284  {
285  newClusterDescription = newClusterDescription.WithType(newServerDescription.Type.ToClusterType());
286  }
287 
288  switch (newClusterDescription.Type)
289  {
290  case ClusterType.ReplicaSet:
291  newClusterDescription = ProcessReplicaSetChange(newClusterDescription, args, newServers);
292  break;
293 
294  case ClusterType.Sharded:
295  newClusterDescription = ProcessShardedChange(newClusterDescription, args);
296  break;
297 
298  default:
299  throw new MongoInternalException("Unexpected cluster type.");
300  }
301  }
302  else
303  {
304  newClusterDescription = newClusterDescription.WithoutServerDescription(newServerDescription.EndPoint);
305  }
306  }
307 
308  UpdateClusterDescription(newClusterDescription);
309 
310  foreach (var server in newServers)
311  {
312  server.Initialize();
313  }
314  }

316  private ClusterDescription ProcessReplicaSetChange(ClusterDescription clusterDescription, ServerDescriptionChangedEventArgs args, List<IClusterableServer> newServers)
317  {
318  if (!args.NewServerDescription.Type.IsReplicaSetMember())
319  {
320  return RemoveServer(clusterDescription, args.NewServerDescription.EndPoint, string.Format("Server is a {0}, not a replica set member.", args.NewServerDescription.Type));
321  }
322 
323  if (args.NewServerDescription.Type == ServerType.ReplicaSetGhost)
324  {
325  return clusterDescription.WithServerDescription(args.NewServerDescription);
326  }
327 
328  if (_replicaSetName == null)
329  {
330  _replicaSetName = args.NewServerDescription.ReplicaSetConfig.Name;
331  }
332 
333  if (_replicaSetName != args.NewServerDescription.ReplicaSetConfig.Name)
334  {
335  return RemoveServer(clusterDescription, args.NewServerDescription.EndPoint, string.Format("Server was a member of the '{0}' replica set, but should be '{1}'.", args.NewServerDescription.ReplicaSetConfig.Name, _replicaSetName));
336  }
337 
338  clusterDescription = clusterDescription.WithServerDescription(args.NewServerDescription);
339  clusterDescription = EnsureServers(clusterDescription, args.NewServerDescription, newServers);
340 
341  if (args.NewServerDescription.CanonicalEndPoint != null &&
342  !EndPointHelper.Equals(args.NewServerDescription.CanonicalEndPoint, args.NewServerDescription.EndPoint))
343  {
344  return RemoveServer(clusterDescription, args.NewServerDescription.EndPoint, "CanonicalEndPoint is different than seed list EndPoint.");
345  }
346 
347  if (args.NewServerDescription.Type == ServerType.ReplicaSetPrimary)
348  {
349  if (args.NewServerDescription.ReplicaSetConfig.Version != null)
350  {
351  bool isCurrentPrimaryStale = true;
352  if (_maxElectionInfo != null)
353  {
354  isCurrentPrimaryStale = _maxElectionInfo.IsStale(args.NewServerDescription.ReplicaSetConfig.Version.Value, args.NewServerDescription.ElectionId);
355  var isReportedPrimaryStale = _maxElectionInfo.IsFresher(
356  args.NewServerDescription.ReplicaSetConfig.Version.Value,
357  args.NewServerDescription.ElectionId);
358 
359 
360  if (isReportedPrimaryStale && args.NewServerDescription.ElectionId != null)
361  {
362  // we only invalidate the "newly" reported stale primary if electionId was used.
363  lock (_serversLock)
364  {
365  var server = _servers.SingleOrDefault(x => EndPointHelper.Equals(args.NewServerDescription.EndPoint, x.EndPoint));
366  server.Invalidate();
367 
368  _sdamInformationEventHandler?.Invoke(new SdamInformationEvent(() =>
369  string.Format(
370  @"Invalidating server: Setting ServerType to ""Unknown"" for {0} because it " +
371  @"claimed to be the replica set primary for replica set ""{1}"" but sent a " +
372  @"(setVersion, electionId) tuple of ({2}, {3}) that was less than than the " +
373  @"largest tuple seen, (maxSetVersion, maxElectionId), of ({4}, {5}).",
374  args.NewServerDescription.EndPoint,
375  args.NewServerDescription.ReplicaSetConfig.Name,
376  args.NewServerDescription.ReplicaSetConfig.Version,
377  args.NewServerDescription.ElectionId,
378  _maxElectionInfo.SetVersion,
379  _maxElectionInfo.ElectionId)));
380 
381  return clusterDescription.WithServerDescription(
382  new ServerDescription(server.ServerId, server.EndPoint));
383  }
384  }
385  }
386 
387  if (isCurrentPrimaryStale)
388  {
389  if (_maxElectionInfo == null)
390  {
391  _sdamInformationEventHandler?.Invoke(new SdamInformationEvent(() =>
392  string.Format(
393  @"Initializing (maxSetVersion, maxElectionId): Saving tuple " +
394  @"(setVersion, electionId) of ({0}, {1}) as (maxSetVersion, maxElectionId) for " +
395  @"replica set ""{2}"" because replica set primary {3} sent ({0}, {1}), the first " +
396  @"(setVersion, electionId) tuple ever seen for replica set ""{4}"".",
397  args.NewServerDescription.ReplicaSetConfig.Version,
398  args.NewServerDescription.ElectionId,
399  args.NewServerDescription.ReplicaSetConfig.Name,
400  args.NewServerDescription.EndPoint,
401  args.NewServerDescription.ReplicaSetConfig.Name)));
402  }
403  else
404  {
405  if (_maxElectionInfo.SetVersion < args.NewServerDescription.ReplicaSetConfig.Version.Value)
406  {
407  _sdamInformationEventHandler?.Invoke(new SdamInformationEvent(() =>
408  string.Format(
409  @"Updating stale setVersion: Updating the current " +
410  @"(maxSetVersion, maxElectionId) tuple from ({0}, {1}) to ({2}, {3}) for " +
411  @"replica set ""{4}"" because replica set primary {5} sent ({6}, {7})—a larger " +
412  @"(setVersion, electionId) tuple then the saved tuple, ({0}, {1}).",
413  _maxElectionInfo.SetVersion,
414  _maxElectionInfo.ElectionId,
415  args.NewServerDescription.ReplicaSetConfig.Version,
416  args.NewServerDescription.ElectionId,
417  args.NewServerDescription.ReplicaSetConfig.Name,
418  args.NewServerDescription.EndPoint,
419  args.NewServerDescription.ReplicaSetConfig.Version,
420  args.NewServerDescription.ElectionId))) ;
421  }
422  else // current primary is stale & setVersion is not stale ⇒ the electionId must be stale
423  {
424  _sdamInformationEventHandler?.Invoke(new SdamInformationEvent(() =>
425  string.Format(
426  @"Updating stale electionId: Updating the current " +
427  @"(maxSetVersion, maxElectionId) tuple from ({0}, {1}) to ({2}, {3}) for " +
428  @"replica set ""{4}"" because replica set primary {5} sent ({6}, {7})—" +
429  @"a larger (setVersion, electionId) tuple than the saved tuple, ({0}, {1}).",
430  _maxElectionInfo.SetVersion,
431  _maxElectionInfo.ElectionId,
432  args.NewServerDescription.ReplicaSetConfig.Version,
433  args.NewServerDescription.ElectionId,
434  args.NewServerDescription.ReplicaSetConfig.Name,
435  args.NewServerDescription.EndPoint,
436  args.NewServerDescription.ReplicaSetConfig.Version,
437  args.NewServerDescription.ElectionId)));
438  }
439  }
440 
441  _maxElectionInfo = new ElectionInfo(
442  args.NewServerDescription.ReplicaSetConfig.Version.Value,
443  args.NewServerDescription.ElectionId);
444  }
445  }
446 
447  var currentPrimaryEndPoints = clusterDescription.Servers
448  .Where(x => x.Type == ServerType.ReplicaSetPrimary)
449  .Where(x => !EndPointHelper.Equals(x.EndPoint, args.NewServerDescription.EndPoint))
450  .Select(x => x.EndPoint)
451  .ToList();
452 
453  if (currentPrimaryEndPoints.Count > 0)
454  {
455  lock (_serversLock)
456  {
457  var currentPrimaries = _servers.Where(x => EndPointHelper.Contains(currentPrimaryEndPoints, x.EndPoint));
458  foreach (var currentPrimary in currentPrimaries)
459  {
460  // kick off the server to invalidate itself
461  currentPrimary.Invalidate();
462  // set it to disconnected in the cluster
463  clusterDescription = clusterDescription.WithServerDescription(
464  new ServerDescription(currentPrimary.ServerId, currentPrimary.EndPoint));
465  }
466  }
467  }
468  }
469 
470  return clusterDescription;
471  }

483  private ClusterDescription EnsureServer(ClusterDescription clusterDescription, EndPoint endPoint, List<IClusterableServer> newServers)
484  {
485  if (_state.Value == State.Disposed)
486  {
487  return clusterDescription;
488  }
489 
490  IClusterableServer server;
491  Stopwatch stopwatch = new Stopwatch();
492  lock (_serversLock)
493  {
494  if (_servers.Any(n => EndPointHelper.Equals(n.EndPoint, endPoint)))
495  {
496  return clusterDescription;
497  }
498 
499  if (_addingServerEventHandler != null)
500  {
501  _addingServerEventHandler(new ClusterAddingServerEvent(ClusterId, endPoint));
502  }
503 
504  stopwatch.Start();
505  server = CreateServer(endPoint);
506  server.DescriptionChanged += ServerDescriptionChangedHandler;
507  _servers.Add(server);
508  newServers.Add(server);
509  }
510 
511  clusterDescription = clusterDescription.WithServerDescription(server.Description);
512  stopwatch.Stop();
513 
514  if (_addedServerEventHandler != null)
515  {
516  _addedServerEventHandler(new ClusterAddedServerEvent(server.ServerId, stopwatch.Elapsed));
517  }
518 
519  return clusterDescription;
520  }

522  private ClusterDescription EnsureServers(ClusterDescription clusterDescription, ServerDescription serverDescription, List<IClusterableServer> newServers)
523  {
524  if (serverDescription.Type == ServerType.ReplicaSetPrimary ||
525  !clusterDescription.Servers.Any(x => x.Type == ServerType.ReplicaSetPrimary))
526  {
527  foreach (var endPoint in serverDescription.ReplicaSetConfig.Members)
528  {
529  clusterDescription = EnsureServer(clusterDescription, endPoint, newServers);
530  }
531  }
532 
533  if (serverDescription.Type == ServerType.ReplicaSetPrimary)
534  {
535  var requiredEndPoints = serverDescription.ReplicaSetConfig.Members;
536  var extraEndPoints = clusterDescription.Servers.Where(x => !EndPointHelper.Contains(requiredEndPoints, x.EndPoint)).Select(x => x.EndPoint);
537  foreach (var endPoint in extraEndPoints)
538  {
539  clusterDescription = RemoveServer(clusterDescription, endPoint, "Server is not in the host list of the primary.");
540  }
541  }
542 
543  return clusterDescription;
544  }

546  private ClusterDescription RemoveServer(ClusterDescription clusterDescription, EndPoint endPoint, string reason)
547  {
548  IClusterableServer server;
549  var stopwatch = new Stopwatch();
550  lock (_serversLock)
551  {
552  server = _servers.SingleOrDefault(x => EndPointHelper.Equals(x.EndPoint, endPoint));
553  if (server == null)
554  {
555  return clusterDescription;
556  }
557 
558  if (_removingServerEventHandler != null)
559  {
560  _removingServerEventHandler(new ClusterRemovingServerEvent(server.ServerId, reason));
561  }
562 
563  stopwatch.Start();
564  _servers.Remove(server);
565  }
566 
567  server.DescriptionChanged -= ServerDescriptionChangedHandler;
568  server.Dispose();
569  stopwatch.Stop();
570 
571  if (_removedServerEventHandler != null)
572  {
573  _removedServerEventHandler(new ClusterRemovedServerEvent(server.ServerId, reason, stopwatch.Elapsed));
574  }
575 
576  return clusterDescription.WithoutServerDescription(endPoint);
577  }

619  public bool IsFresher(int setVersion, ElectionId electionId)
620  {
621  return
622  _setVersion > setVersion ||
623  _setVersion == setVersion && _electionId != null && _electionId.CompareTo(electionId) > 0;
624  }

626  public bool IsStale(int setVersion, ElectionId electionId)
627  {
628  if (_setVersion < setVersion)
629  {
630  return true;
631  }
632  if (_setVersion > setVersion)
633  {
634  return false;
635  }
636  // Now it must be that _setVersion == setVersion
637  if (_electionId == null)
638  {
639  return true;
640  }
641 
642  return _electionId.CompareTo(electionId) < 0;
643 
644  /* above is equivalent to:
645  * return
646  * _setVersion < setVersion
647  * || _setVersion == setVersion && (_electionId == null || _electionId.CompareTo(electionId) < 0); */
648  }
649  }
