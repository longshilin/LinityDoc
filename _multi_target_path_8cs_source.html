385  void RecalculateHTarget (bool firstTime) {
386  // When pathsForAll is false
387  // then no heuristic should be used
388  if (!pathsForAll) {
389  heuristic = Heuristic.None;
390  heuristicScale = 0.0F;
391  return;
392  }
393 
394  // Calculate a new hTarget and rebuild the open list if necessary
395  // Rebuilding the open list is necessary when the H score for nodes changes
396  switch (heuristicMode) {
397  case HeuristicMode.None:
398  heuristic = Heuristic.None;
399  heuristicScale = 0F;
400  break;
401  case HeuristicMode.Average:
402  if (!firstTime) return;
403 
404  // No break
405  // The first time the implementation
406  // for Average and MovingAverage is identical
407  // so we just use fallthrough
408  goto case HeuristicMode.MovingAverage;
409  case HeuristicMode.MovingAverage:
410 
411  // Pick the average position of all nodes that have not been found yet
412  var avg = Vector3.zero;
413  int count = 0;
414  for (int j = 0; j < targetPoints.Length; j++) {
415  if (!targetsFound[j]) {
416  avg += (Vector3)targetNodes[j].position;
417  count++;
418  }
419  }
420 
421  // Should use asserts, but they were first added in Unity 5.1
422  // so I cannot use them because I want to keep compatibility with 4.6
423  // (as of 2015)
424  if (count == 0) throw new System.Exception("Should not happen");
425 
426  avg /= count;
427  hTarget = (Int3)avg;
428  break;
429  case HeuristicMode.Midpoint:
430  if (!firstTime) return;
431 
432  // No break
433  // The first time the implementation
434  // for Midpoint and MovingMidpoint is identical
435  // so we just use fallthrough
436  goto case HeuristicMode.MovingMidpoint;
437  case HeuristicMode.MovingMidpoint:
438 
439  Vector3 min = Vector3.zero;
440  Vector3 max = Vector3.zero;
441  bool set = false;
442 
443  // Pick the median of all points that have
444  // not been found yet
445  for (int j = 0; j < targetPoints.Length; j++) {
446  if (!targetsFound[j]) {
447  if (!set) {
448  min = (Vector3)targetNodes[j].position;
449  max = (Vector3)targetNodes[j].position;
450  set = true;
451  } else {
452  min = Vector3.Min((Vector3)targetNodes[j].position, min);
453  max = Vector3.Max((Vector3)targetNodes[j].position, max);
454  }
455  }
456  }
457 
458  var midpoint = (Int3)((min+max)*0.5F);
459  hTarget = midpoint;
460  break;
461  case HeuristicMode.Sequential:
462 
463  // The first time the hTarget should always be recalculated
464  // But other times we can skip it if we have not yet found the current target
465  // since then the hTarget would just be set to the same value again
466  if (!firstTime && !targetsFound[sequentialTarget]) {
467  return;
468  }
469 
470  float dist = 0;
471 
472  // Pick the target which is furthest away and has not been found yet
473  for (int j = 0; j < targetPoints.Length; j++) {
474  if (!targetsFound[j]) {
475  float d = (targetNodes[j].position-startNode.position).sqrMagnitude;
476  if (d > dist) {
477  dist = d;
478  hTarget = (Int3)targetPoints[j];
479  sequentialTarget = j;
480  }
481  }
482  }
483  break;
484  }
485 
486  // Rebuild the open list since all the H scores have changed
487  // However the first time we can skip this since
488  // no nodes are added to the heap yet
489  if (!firstTime) {
490  RebuildOpenList();
491  }
492  }

557  protected override void CalculateStep (long targetTick) {
558  int counter = 0;
559 
560  // Continue to search as long as we haven't encountered an error and we haven't found the target
561  while (CompleteState == PathCompleteState.NotCalculated) {
562  // @Performance Just for debug info
563  searchedNodes++;
564 
565  // The node might be the target node for one of the paths
566  if (currentR.flag1) {
567  // Close the current node, if the current node is the target node then the path is finnished
568  for (int i = 0; i < targetNodes.Length; i++) {
569  if (!targetsFound[i] && currentR.node == targetNodes[i]) {
570  FoundTarget(currentR, i);
571  if (CompleteState != PathCompleteState.NotCalculated) {
572  break;
573  }
574  }
575  }
576 
577  if (targetNodeCount <= 0) {
578  CompleteState = PathCompleteState.Complete;
579  break;
580  }
581  }
582 
583  // Loop through all walkable neighbours of the node and add them to the open list.
584  currentR.node.Open(this, currentR, pathHandler);
585 
586  // Any nodes left to search?
587  if (pathHandler.heap.isEmpty) {
588  CompleteState = PathCompleteState.Complete;
589  break;
590  }
591 
592  // Select the node with the lowest F score and remove it from the open list
593  currentR = pathHandler.heap.Remove();
594 
595  // Check for time every 500 nodes, roughly every 0.5 ms usually
596  if (counter > 500) {
597  // Have we exceded the maxFrameTime, if so we should wait one frame before continuing the search since we don't want the game to lag
598  if (System.DateTime.UtcNow.Ticks >= targetTick) {
599  // Return instead of yield'ing, a separate function handles the yield (CalculatePaths)
600  return;
601  }
602 
603  counter = 0;
604  }
605 
606  counter++;
607  }
