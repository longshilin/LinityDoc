58  public static Win32Exception CreateException(long errorCode, string defaultMessage)
59  {
60  string message = defaultMessage;
61  switch (errorCode)
62  {
63  case SEC_E_BUFFER_TOO_SMALL:
64  message = "The message buffer is too small. Used with the Digest SSP.";
65  break;
66  case SEC_E_CONTEXT_EXPIRED:
67  message = "The application is referencing a context that has already been closed.";
68  break;
69  case SEC_E_CRYPTO_SYSTEM_INVALID:
70  message = "The cipher chosen for the security context is not supported. Used with the Digest SSP.";
71  break;
72  case SEC_E_INCOMPLETE_MESSAGE:
73  message = "The data in the input buffer is incomplete.";
74  break;
75  case SEC_E_INSUFFICENT_MEMORY:
76  message = "There is not enough memory available to complete the requested action.";
77  break;
78  case SEC_E_INTERNAL_ERROR:
79  message = "An error occurred that did not map to an SSPI error code.";
80  break;
81  case SEC_E_INVALID_HANDLE:
82  message = "The handle passed to the function is not valid.";
83  break;
84  case SEC_E_INVALID_TOKEN:
85  message = "The input token is malformed . Possible causes include a token corrupted in transit, a token of incorrect size, and a token passed into the wrong security package. This last condition can happen if the client and server did not negotiate the proper security package.";
86  break;
87  case SEC_E_LOGON_DENIED:
88  message = "The logon failed.";
89  break;
90  case SEC_E_MESSAGE_ALTERED:
91  message = "The message has been altered. Used with the Digest and Schannel SSPs.";
92  break;
93  case SEC_E_NO_AUTHENTICATING_AUTHORITY:
94  message = "No authority could be contacted for authentication. The domain name of the authenticating party could be wrong, the domain could be unreachable, or there might have been a trust relationship failure.";
95  break;
96  case SEC_E_NO_CREDENTIALS:
97  message = "No credentials are available in the security package.";
98  break;
99  case SEC_E_OUT_OF_SEQUENCE:
100  message = "The message was not received in the correct sequence.";
101  break;
102  case SEC_E_QOP_NOT_SUPPORTED:
103  message = "Neither confidentiality nor integrity are supported by the security context. Used with the Digest SSP.";
104  break;
105  case SEC_E_SECPKG_NOT_FOUND:
106  message = "The requested security package does not exist.";
107  break;
108  case SEC_E_TARGET_UNKNOWN:
109  message = "The target was not recognized.";
110  break;
111  case SEC_E_UNKNOWN_CREDENTIALS:
112  message = "The credentials supplied to the package were not recognized.";
113  break;
114  case SEC_I_CONTEXT_EXPIRED:
115  message = "The message sender has finished using the connection and has initiated a shutdown.";
116  break;
117  case SEC_I_RENEGOTIATE:
118  message = "The remote party requires a new handshake sequence or the application has just initiated a shutdown.";
119  break;
120  }
121 
122  return new Win32Exception(errorCode, message);
123  }
