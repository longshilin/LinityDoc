298  public override NNInfoInternal GetNearestForce (Vector3 position, NNConstraint constraint) {
299  if (tiles == null) return new NNInfoInternal();
300 
301  var tileCoords = GetTileCoordinates(position);
302 
303  // Clamp to graph borders
304  tileCoords.x = Mathf.Clamp(tileCoords.x, 0, tileXCount-1);
305  tileCoords.y = Mathf.Clamp(tileCoords.y, 0, tileZCount-1);
306 
307  int wmax = Math.Max(tileXCount, tileZCount);
308 
309  var best = new NNInfoInternal();
310  float bestDistance = float.PositiveInfinity;
311 
312  bool xzSearch = nearestSearchOnlyXZ || (constraint != null && constraint.distanceXZ);
313 
314  // Search outwards in a diamond pattern from the closest tile
315  // 2
316  // 2 1 2
317  // 2 1 0 1 2 etc.
318  // 2 1 2
319  // 2
320  for (int w = 0; w < wmax; w++) {
321  // Stop the loop when we can guarantee that no nodes will be closer than the ones we have already searched
322  if (bestDistance < (w-2)*Math.Max(TileWorldSizeX, TileWorldSizeX)) break;
323 
324  int zmax = Math.Min(w+tileCoords.y +1, tileZCount);
325  for (int z = Math.Max(-w+tileCoords.y, 0); z < zmax; z++) {
326  // Solve for z such that abs(x-tx) + abs(z-tx) == w
327  // Delta X coordinate
328  int originalDx = Math.Abs(w - Math.Abs(z-tileCoords.y));
329  var dx = originalDx;
330  // Solution is dx + tx and -dx + tx
331  // This loop will first check +dx and then -dx
332  // If dx happens to be zero, then it will not run twice
333  do {
334  // Absolute x coordinate
335  int x = -dx + tileCoords.x;
336  if (x >= 0 && x < tileXCount) {
337  NavmeshTile tile = tiles[x + z*tileXCount];
338 
339  if (tile != null) {
340  if (xzSearch) {
341  best = tile.bbTree.QueryClosestXZ(position, constraint, ref bestDistance, best);
342  } else {
343  best = tile.bbTree.QueryClosest(position, constraint, ref bestDistance, best);
344  }
345  }
346  }
347 
348  dx = -dx;
349  } while (dx != originalDx);
350  }
351  }
352 
353  best.node = best.constrainedNode;
354  best.constrainedNode = null;
355  best.clampedPosition = best.constClampedPosition;
356 
357  return best;
358  }

376  nodeRefs.Clear();

378  // Build node neighbours

381

383

386  // In that case, duplicate keys might be added and thus an exception will be thrown
387  // It is safe to ignore the second edge though... I think (only found one case where this happens)
388  var key = new Int2(node.GetVertexIndex(a), node.GetVertexIndex((a+1) % av));
389  if (!nodeRefs.ContainsKey(key)) {
390  nodeRefs.Add(key, i);

396  TriangleMeshNode node = nodes[i];
397

401 
402  for (int a = 0; a < av; a++) {
403  int first = node.GetVertexIndex(a);
404  int second = node.GetVertexIndex((a+1) % av);
405  int connNode;
406 
407  if (nodeRefs.TryGetValue(new Int2(second, first), out connNode)) {
408  TriangleMeshNode other = nodes[connNode];
409 
410  int bv = other.GetVertexCount();
411 
412  for (int b = 0; b < bv; b++) {
414  if (other.GetVertexIndex(b) == second && other.GetVertexIndex((b+1) % bv) == first) {
415  connections.Add(new Connection(
416  other,
417  (uint)(node.position - other.position).costMagnitude,
418  (byte)a
419  ));
420  break;
421  }
422  }
423  }
424  }

427  }

437  public void ConnectTiles (NavmeshTile tile1, NavmeshTile tile2) {
438  if (tile1 == null || tile2 == null) return;
439 
440  if (tile1.nodes == null) throw new System.ArgumentException("tile1 does not contain any nodes");
441  if (tile2.nodes == null) throw new System.ArgumentException("tile2 does not contain any nodes");
442 
443  int t1x = Mathf.Clamp(tile2.x, tile1.x, tile1.x+tile1.w-1);
444  int t2x = Mathf.Clamp(tile1.x, tile2.x, tile2.x+tile2.w-1);
445  int t1z = Mathf.Clamp(tile2.z, tile1.z, tile1.z+tile1.d-1);
446  int t2z = Mathf.Clamp(tile1.z, tile2.z, tile2.z+tile2.d-1);
447 
448  int coord, altcoord;
449  int t1coord, t2coord;
450 
451  float tileWorldSize;
452 
453  // Figure out which side that is shared between the two tiles
454  // and what coordinate index is fixed along that edge (x or z)
455  if (t1x == t2x) {
456  coord = 2;
457  altcoord = 0;
458  t1coord = t1z;
459  t2coord = t2z;
460  tileWorldSize = TileWorldSizeZ;
461  } else if (t1z == t2z) {
462  coord = 0;
463  altcoord = 2;
464  t1coord = t1x;
465  t2coord = t2x;
466  tileWorldSize = TileWorldSizeX;
467  } else {
468  throw new System.ArgumentException("Tiles are not adjacent (neither x or z coordinates match)");
469  }
470 
471  if (Math.Abs(t1coord-t2coord) != 1) {
472  throw new System.ArgumentException("Tiles are not adjacent (tile coordinates must differ by exactly 1. Got '" + t1coord + "' and '" + t2coord + "')");
473  }
474 
475  // Midpoint between the two tiles
476  int midpoint = (int)Math.Round((Math.Max(t1coord, t2coord) * tileWorldSize) * Int3.Precision);
477 
478  #if ASTARDEBUG
479  Vector3 v1 = new Vector3(-100, 0, -100);
480  Vector3 v2 = new Vector3(100, 0, 100);
481  v1[coord] = midpoint*Int3.PrecisionFactor;
482  v2[coord] = midpoint*Int3.PrecisionFactor;
483 
484  Debug.DrawLine(v1, v2, Color.magenta);
485  #endif
486 
487  TriangleMeshNode[] nodes1 = tile1.nodes;
488  TriangleMeshNode[] nodes2 = tile2.nodes;
489 
490  // Find adjacent nodes on the border between the tiles
491  for (int i = 0; i < nodes1.Length; i++) {
492  TriangleMeshNode nodeA = nodes1[i];
493  int aVertexCount = nodeA.GetVertexCount();
494 
495  // Loop through all *sides* of the node
496  for (int a = 0; a < aVertexCount; a++) {
497  // Vertices that the segment consists of
498  Int3 aVertex1 = nodeA.GetVertexInGraphSpace(a);
499  Int3 aVertex2 = nodeA.GetVertexInGraphSpace((a+1) % aVertexCount);
500 
501  // Check if it is really close to the tile border
502  if (Math.Abs(aVertex1[coord] - midpoint) < 2 && Math.Abs(aVertex2[coord] - midpoint) < 2) {
503  int minalt = Math.Min(aVertex1[altcoord], aVertex2[altcoord]);
504  int maxalt = Math.Max(aVertex1[altcoord], aVertex2[altcoord]);
505 
506  // Degenerate edge
507  if (minalt == maxalt) continue;
508 
509  for (int j = 0; j < nodes2.Length; j++) {
510  TriangleMeshNode nodeB = nodes2[j];
511  int bVertexCount = nodeB.GetVertexCount();
512  for (int b = 0; b < bVertexCount; b++) {
513  Int3 bVertex1 = nodeB.GetVertexInGraphSpace(b);
514  Int3 bVertex2 = nodeB.GetVertexInGraphSpace((b+1) % aVertexCount);
515  if (Math.Abs(bVertex1[coord] - midpoint) < 2 && Math.Abs(bVertex2[coord] - midpoint) < 2) {
516  int minalt2 = Math.Min(bVertex1[altcoord], bVertex2[altcoord]);
517  int maxalt2 = Math.Max(bVertex1[altcoord], bVertex2[altcoord]);
518 
519  // Degenerate edge
520  if (minalt2 == maxalt2) continue;
521 
522  if (maxalt > minalt2 && minalt < maxalt2) {
523  // The two nodes seem to be adjacent
524 
525  // Test shortest distance between the segments (first test if they are equal since that is much faster and pretty common)
526  if ((aVertex1 == bVertex1 && aVertex2 == bVertex2) || (aVertex1 == bVertex2 && aVertex2 == bVertex1) ||
527  VectorMath.SqrDistanceSegmentSegment((Vector3)aVertex1, (Vector3)aVertex2, (Vector3)bVertex1, (Vector3)bVertex2) < MaxTileConnectionEdgeDistance*MaxTileConnectionEdgeDistance) {
528  uint cost = (uint)(nodeA.position - nodeB.position).costMagnitude;
529 
530  nodeA.AddConnection(nodeB, cost, a);
531  nodeB.AddConnection(nodeA, cost, b);
532  }
533  }
534  }
535  }
536  }
537  }
538  }

554  void PrepareNodeRecycling (int x, int z, Int3[] verts, int[] tris, TriangleMeshNode[] recycledNodeBuffer) {
555  NavmeshTile tile = GetTile(x, z);
556 
557  if (tile == null || tile.nodes.Length == 0) return;
558  var nodes = tile.nodes;
559  var recycling = nodeRecyclingHashBuffer;
560  for (int i = 0, j = 0; i < tris.Length; i += 3, j++) {
561  recycling[verts[tris[i+0]].GetHashCode() + verts[tris[i+1]].GetHashCode() + verts[tris[i+2]].GetHashCode()] = j;
562  }
563  var connectionsToKeep = ListPool<Connection>.Claim();
564 
565  for (int i = 0; i < nodes.Length; i++) {
566  var node = nodes[i];
567  Int3 v0, v1, v2;
568  node.GetVerticesInGraphSpace(out v0, out v1, out v2);
569  var hash = v0.GetHashCode() + v1.GetHashCode() + v2.GetHashCode();
570  int newNodeIndex;
571  if (recycling.TryGetValue(hash, out newNodeIndex)) {
572  // Technically we should check for a cyclic permutations of the vertices (e.g node a,b,c could become node b,c,a)
573  // but in almost all cases the vertices will keep the same order. Allocating one or two extra nodes isn't such a big deal.
574  if (verts[tris[3*newNodeIndex+0]] == v0 && verts[tris[3*newNodeIndex+1]] == v1 && verts[tris[3*newNodeIndex+2]] == v2) {
575  recycledNodeBuffer[newNodeIndex] = node;
576  // Remove the node from the tile
577  nodes[i] = null;
578  // Only keep connections to nodes on other graphs
579  // Usually there are no connections to nodes to other graphs and this is faster than removing all connections them one by one
580  for (int j = 0; j < node.connections.Length; j++) {
581  if (node.connections[j].node.GraphIndex != node.GraphIndex) {
582  connectionsToKeep.Add(node.connections[j]);
583  }
584  }
585  ArrayPool<Connection>.Release(ref node.connections, true);
586  if (connectionsToKeep.Count > 0) {
587  node.connections = connectionsToKeep.ToArrayFromPool();
588  connectionsToKeep.Clear();
589  }
590  }
591  }
592  }
593 
594  recycling.Clear();
595  ListPool<Connection>.Release(ref connectionsToKeep);
596  }

610  node.Walkable = true;

613  node.GraphIndex = graphIndex;

707  static readonly byte[] LinecastShapeEdgeLookup;
708 
709  static NavmeshBase () {
710  // Want want to figure out which side of a triangle that a ray exists using.
711  // There are only 3*3*3 = 27 different options for the [left/right/colinear] options for the 3 vertices of a triangle.
712  // So we can precompute the result to improve the performance of linecasts.
713  // For simplicity we reserve 2 bits for each side which means that we have 4*4*4 = 64 entries in the lookup table.
714  LinecastShapeEdgeLookup = new byte[64];
715  Side[] sideOfLine = new Side[3];
716  for (int i = 0; i < LinecastShapeEdgeLookup.Length; i++) {
717  sideOfLine[0] = (Side)((i >> 0) & 0x3);
718  sideOfLine[1] = (Side)((i >> 2) & 0x3);
719  sideOfLine[2] = (Side)((i >> 4) & 0x3);
720  LinecastShapeEdgeLookup[i] = 0xFF;
721  // Value 3 is an invalid value. So we just skip it.
722  if (sideOfLine[0] != (Side)3 && sideOfLine[1] != (Side)3 && sideOfLine[2] != (Side)3) {
723  // Figure out the side of the triangle that the line exits.
724  // In case the line passes through one of the vertices of the triangle
725  // there may be multiple alternatives. In that case pick the edge
726  // which contains the fewest vertices that lie on the line.
727  // This prevents a potential infinite loop when a linecast is done colinear
728  // to the edge of a triangle.
729  int bestBadness = int.MaxValue;
730  for (int j = 0; j < 3; j++) {
731  if ((sideOfLine[j] == Side.Left || sideOfLine[j] == Side.Colinear) && (sideOfLine[(j+1)%3] == Side.Right || sideOfLine[(j+1)%3] == Side.Colinear)) {
732  var badness = (sideOfLine[j] == Side.Colinear ? 1 : 0) + (sideOfLine[(j+1)%3] == Side.Colinear ? 1 : 0);
733  if (badness < bestBadness) {
734  LinecastShapeEdgeLookup[i] = (byte)j;
735  bestBadness = badness;
736  }
737  }
738  }
739  }
740  }
741  }

792  var i3endInGraphSpace = (Int3)endInGraphSpace;
793 
794  // Fast early out check
795  if (i3originInGraphSpace == i3endInGraphSpace) {
796  hit.point = hit.origin;
797  hit.node = node;
798  return false;
799  }
800 
801  int counter = 0;
802  while (true) {
803  counter++;
804  if (counter > 2000) {
805 #if !SERVER
806  UnityEngine.Debug.LogError("Linecast was stuck in infinite loop. Breaking.");
807 #endif
808  return true;
809  }
810 
811  if (trace != null) trace.Add(node);
812 
813  Int3 a0, a1, a2;
814  node.GetVerticesInGraphSpace(out a0, out a1, out a2);
815  int sideOfLine = (byte)VectorMath.SideXZ(i3originInGraphSpace, i3endInGraphSpace, a0);
816  sideOfLine |= (byte)VectorMath.SideXZ(i3originInGraphSpace, i3endInGraphSpace, a1) << 2;
817  sideOfLine |= (byte)VectorMath.SideXZ(i3originInGraphSpace, i3endInGraphSpace, a2) << 4;
818  // Use a lookup table to figure out which side of this triangle that the ray exits
819  int shapeEdgeA = (int)LinecastShapeEdgeLookup[sideOfLine];
820  // The edge consists of the vertex with index 'sharedEdgeA' and the next vertex after that (index '(sharedEdgeA+1)%3')
821 
822  var sideNodeExit = VectorMath.SideXZ(shapeEdgeA == 0 ? a0 : (shapeEdgeA == 1 ? a1 : a2), shapeEdgeA == 0 ? a1 : (shapeEdgeA == 1 ? a2 : a0), i3endInGraphSpace);
823  if (sideNodeExit != Side.Left) {
824  // Ray stops before it leaves the current node.
825  // The endpoint must be inside the current node.
826  hit.point = end;
827  hit.node = node;
828  return false;
829  }
830 
831  if (shapeEdgeA == 0xFF) {
832  // Line does not intersect node at all?
833  // This may theoretically happen if the origin was not properly snapped to the inside of the triangle, but is instead a tiny distance outside the node.
834 #if !SERVER
835  UnityEngine.Debug.LogError("Line does not intersect node at all");
836 #endif
837  hit.node = node;
838  hit.point = hit.tangentOrigin = hit.origin;
839  return true;
840  } else {
841  bool success = false;
842  var nodeConnections = node.connections;
843  for (int i = 0; i < nodeConnections.Length; i++) {
844  if (nodeConnections[i].shapeEdge == shapeEdgeA) {
845  // This might be the next node that we enter
846 
847  var neighbour = nodeConnections[i].node as TriangleMeshNode;
848  if (neighbour == null || !neighbour.Walkable) continue;
849 
850  var neighbourConnections = neighbour.connections;
851  int shapeEdgeB = -1;
852  for (int j = 0; j < neighbourConnections.Length; j++) {
853  if (neighbourConnections[j].node == node) {
854  shapeEdgeB = neighbourConnections[j].shapeEdge;
855  break;
856  }
857  }
858 
859  if (shapeEdgeB == -1) {
860  // Connection was mono-directional!
861  // This shouldn't normally not happen on navmeshes happen on navmeshes (when the shapeEdge matches at least) unless a user has done something strange to the navmesh.
862  continue;
863  }
864 
865  var side1 = VectorMath.SideXZ(i3originInGraphSpace, i3endInGraphSpace, neighbour.GetVertexInGraphSpace(shapeEdgeB));
866  var side2 = VectorMath.SideXZ(i3originInGraphSpace, i3endInGraphSpace, neighbour.GetVertexInGraphSpace((shapeEdgeB+1) % 3));
867 
868  // Check if the line enters this edge
869  success = (side1 == Side.Right || side1 == Side.Colinear) && (side2 == Side.Left || side2 == Side.Colinear);
870 
871  if (!success) continue;
872 
873  // Ray has entered the neighbouring node.
874  // After the first node, it is possible to prove the loop invariant that shapeEdgeA will *never* end up as -1 (checked above)
875  // Since side = Colinear acts essentially as a wildcard. side1 and side2 can be the most restricted if they are side1=right, side2=left.
876  // Then when we get to the next node we know that the sideOfLine array is either [*, Right, Left], [Left, *, Right] or [Right, Left, *], where * is unknown.
877  // We are looking for the sequence [Left, Right] (possibly including Colinear as wildcard). We will always find this sequence regardless of the value of *.
878  node = neighbour;
879  break;
880  }
881  }
882 
883  if (!success) {
884  // Node did not enter any neighbours
885  // It must have hit the border of the navmesh
886  var hitEdgeStartInGraphSpace = (Vector3)(shapeEdgeA == 0 ? a0 : (shapeEdgeA == 1 ? a1 : a2));
887  var hitEdgeEndInGraphSpace = (Vector3)(shapeEdgeA == 0 ? a1 : (shapeEdgeA == 1 ? a2 : a0));
888  var intersectionInGraphSpace = VectorMath.LineIntersectionPointXZ(hitEdgeStartInGraphSpace, hitEdgeEndInGraphSpace, (Vector3)i3originInGraphSpace, (Vector3)i3endInGraphSpace);
889  hit.point = graph.transform.Transform(intersectionInGraphSpace);
890  hit.node = node;
891  var hitEdgeStart = graph.transform.Transform(hitEdgeStartInGraphSpace);
892  var hitEdgeEnd = graph.transform.Transform(hitEdgeEndInGraphSpace);
893  hit.tangent = hitEdgeEnd - hitEdgeStart;
894  hit.tangentOrigin = hitEdgeStart;
895  return true;
896  }
897  }
898  }
