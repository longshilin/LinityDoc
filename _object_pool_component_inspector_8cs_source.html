20  [CustomEditor(typeof(ObjectPoolComponent))]
21  internal sealed class ObjectPoolComponentInspector : GameFrameworkInspector
22  {
23  private readonly HashSet<string> m_OpenedItems = new HashSet<string>();
24 
25  public override void OnInspectorGUI()
26  {
27  base.OnInspectorGUI();
28 
29  if (!EditorApplication.isPlaying)
30  {
31  EditorGUILayout.HelpBox("Available during runtime only.", MessageType.Info);
32  return;
33  }
34 
35  ObjectPoolComponent t = (ObjectPoolComponent)target;
36 
37  if (IsPrefabInHierarchy(t.gameObject))
38  {
39  EditorGUILayout.LabelField("Object Pool Count", t.Count.ToString());
40 
41  ObjectPoolBase[] objectPools = t.GetAllObjectPools(true);
42  foreach (ObjectPoolBase objectPool in objectPools)
43  {
44  DrawObjectPool(objectPool);
45  }
46  }
47 
48  Repaint();
49  }
50 
51  private void OnEnable()
52  {
53  }
54 
55  private void DrawObjectPool(ObjectPoolBase objectPool)
56  {
57  bool lastState = m_OpenedItems.Contains(objectPool.FullName);
58  bool currentState = EditorGUILayout.Foldout(lastState, objectPool.FullName);
59  if (currentState != lastState)
60  {
61  if (currentState)
62  {
63  m_OpenedItems.Add(objectPool.FullName);
64  }
65  else
66  {
67  m_OpenedItems.Remove(objectPool.FullName);
68  }
69  }
70 
71  if (currentState)
72  {
73  EditorGUILayout.BeginVertical("box");
74  {
75  EditorGUILayout.LabelField("Name", objectPool.Name);
76  EditorGUILayout.LabelField("Type", objectPool.ObjectType.FullName);
77  EditorGUILayout.LabelField("Auto Release Interval", objectPool.AutoReleaseInterval.ToString());
78  EditorGUILayout.LabelField("Capacity", objectPool.Capacity.ToString());
79  EditorGUILayout.LabelField("Used Count", objectPool.Count.ToString());
80  EditorGUILayout.LabelField("Can Release Count", objectPool.CanReleaseCount.ToString());
81  EditorGUILayout.LabelField("Expire Time", objectPool.ExpireTime.ToString());
82  EditorGUILayout.LabelField("Priority", objectPool.Priority.ToString());
83  ObjectInfo[] objectInfos = objectPool.GetAllObjectInfos();
84  if (objectInfos.Length > 0)
85  {
86  EditorGUILayout.LabelField("Name", objectPool.AllowMultiSpawn ? "Locked\tCount\tFlag\tPriority\tLast Use Time" : "Locked\tIn Use\tFlag\tPriority\tLast Use Time");
87  foreach (ObjectInfo objectInfo in objectInfos)
88  {
89  EditorGUILayout.LabelField(string.IsNullOrEmpty(objectInfo.Name) ? "<None>" : objectInfo.Name, Utility.Text.Format("{0}\t{1}\t{2}\t{3}\t{4}", objectInfo.Locked.ToString(), objectPool.AllowMultiSpawn ? objectInfo.SpawnCount.ToString() : objectInfo.IsInUse.ToString(), objectInfo.CustomCanReleaseFlag.ToString(), objectInfo.Priority.ToString(), objectInfo.LastUseTime.ToLocalTime().ToString("yyyy-MM-dd HH:mm:ss")));
90  }
91 
92  if (GUILayout.Button("Release"))
93  {
94  objectPool.Release();
95  }
96 
97  if (GUILayout.Button("Release All Unused"))
98  {
99  objectPool.ReleaseAllUnused();
100  }
101 
102  if (GUILayout.Button("Export CSV Data"))
103  {
104  string exportFileName = EditorUtility.SaveFilePanel("Export CSV Data", string.Empty, Utility.Text.Format("Object Pool Data - {0}.csv", objectPool.Name), string.Empty);
105  if (!string.IsNullOrEmpty(exportFileName))
106  {
107  try
108  {
109  int index = 0;
110  string[] data = new string[objectInfos.Length + 1];
111  data[index++] = Utility.Text.Format("Name,Locked,{0},Custom Can Release Flag,Priority,Last Use Time", objectPool.AllowMultiSpawn ? "Count" : "In Use");
112  foreach (ObjectInfo objectInfo in objectInfos)
113  {
114  data[index++] = Utility.Text.Format("{0},{1},{2},{3},{4},{5}", objectInfo.Name, objectInfo.Locked.ToString(), objectPool.AllowMultiSpawn ? objectInfo.SpawnCount.ToString() : objectInfo.IsInUse.ToString(), objectInfo.CustomCanReleaseFlag.ToString(), objectInfo.Priority.ToString(), objectInfo.LastUseTime.ToLocalTime().ToString("yyyy-MM-dd HH:mm:ss"));
115  }
116 
117  File.WriteAllLines(exportFileName, data, Encoding.UTF8);
118  Debug.Log(Utility.Text.Format("Export object pool CSV data to '{0}' success.", exportFileName));
119  }
120  catch (Exception exception)
121  {
122  Debug.LogError(Utility.Text.Format("Export object pool CSV data to '{0}' failure, exception is '{1}'.", exportFileName, exception.ToString()));
123  }
124  }
125  }
126  }
127  else
128  {
129  GUILayout.Label("Object Pool is Empty ...");
130  }
131  }
132  EditorGUILayout.EndVertical();
133 
134  EditorGUILayout.Separator();
135  }
136  }
137  }
138 }
