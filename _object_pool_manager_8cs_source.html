330  public ObjectPoolBase[] GetAllObjectPools(bool sort)
331  {
332  if (sort)
333  {
334  List<ObjectPoolBase> results = new List<ObjectPoolBase>();
335  foreach (KeyValuePair<TypeNamePair, ObjectPoolBase> objectPool in m_ObjectPools)
336  {
337  results.Add(objectPool.Value);
338  }
339 
340  results.Sort(m_ObjectPoolComparer);
341  return results.ToArray();
342  }
343  else
344  {
345  int index = 0;
346  ObjectPoolBase[] results = new ObjectPoolBase[m_ObjectPools.Count];
347  foreach (KeyValuePair<TypeNamePair, ObjectPoolBase> objectPool in m_ObjectPools)
348  {
349  results[index++] = objectPool.Value;
350  }
351 
352  return results;
353  }
