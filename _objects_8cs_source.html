801  // Note that QuoteStringIfNeeded does its own escaping. Double-escaping with quotes is

804  static string EscapeWithQuotesIfNeeded(string name, string value)
805  {
806  if (name != "LIBRARY_SEARCH_PATHS" && name != "FRAMEWORK_SEARCH_PATHS")
807  return value;
808  if (!value.Contains(" "))
809  return value;
810  if (value.First() == '\"' && value.Last() == '\"')
811  return value;
