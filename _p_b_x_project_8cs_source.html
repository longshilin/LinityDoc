621  // Allow user to add a Capability
622  public bool AddCapability(string targetGuid, PBXCapabilityType capability, string entitlementsFilePath = null, bool addOptionalFramework = false)
623  {
624  // If the capability requires entitlements then you have to provide the name of it or we don't add the capability.
625  if (capability.requiresEntitlements && entitlementsFilePath == "")
626  {
627  throw new Exception("Couldn't add the Xcode Capability " + capability.id + " to the PBXProject file because this capability requires an entitlement file.");
628  }
629 
630  var p = project.project;
631 
632  // If an entitlement with a different name was added for another capability
633  // we don't add this capacity.
634  if (p.entitlementsFile != null && entitlementsFilePath != null && p.entitlementsFile != entitlementsFilePath)
635  {
636  if (p.capabilities.Count > 0)
637  {
638  //throw new WarningException("Attention, it seems that you have multiple entitlements file. Only one will be added the Project : " + p.entitlementsFile);
639  throw new Exception("Attention, it seems that you have multiple entitlements file. Only one will be added the Project : " + p.entitlementsFile);
640  }
641 
642  return false;
643  }
644 
645  // Add the capability only if it doesn't already exist.
646  if (p.capabilities.Contains(new PBXCapabilityType.TargetCapabilityPair(targetGuid, capability)))
647  {
648  //throw new WarningException("This capability has already been added. Method ignored");
649  }
650 
651  p.capabilities.Add(new PBXCapabilityType.TargetCapabilityPair(targetGuid, capability));
652 
653  // Add the required framework.
654  if (capability.framework != "" && !capability.optionalFramework ||
655  (capability.framework != "" && capability.optionalFramework && addOptionalFramework))
656  {
657  AddFrameworkToProject(targetGuid, capability.framework, false);
658  }
659 
660  // Finally add the entitlement code signing if it wasn't added before.
661  if (entitlementsFilePath != null && p.entitlementsFile == null)
662  {
663  p.entitlementsFile = entitlementsFilePath;
664  AddFileImpl(entitlementsFilePath, entitlementsFilePath, PBXSourceTree.Source, false);
665  SetBuildProperty(targetGuid, "CODE_SIGN_ENTITLEMENTS", PBXPath.FixSlashes(entitlementsFilePath));
666  }
667 
668  return true;
669  }
