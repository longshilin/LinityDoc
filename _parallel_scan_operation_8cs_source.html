156  private IReadOnlyList<IAsyncCursor<TDocument>> CreateCursors(IChannelSourceHandle channelSource, BsonDocument command, BsonDocument result)
157  {
158  var cursors = new List<AsyncCursor<TDocument>>();
159 
160  using (var getMoreChannelSource = new ChannelSourceHandle(new ServerChannelSource(channelSource.Server, channelSource.Session.Fork())))
161  {
162  foreach (var cursorDocument in result["cursors"].AsBsonArray.Select(v => v["cursor"].AsBsonDocument))
163  {
164  var cursorId = cursorDocument["id"].ToInt64();
165  var firstBatch = cursorDocument["firstBatch"].AsBsonArray.Select(v =>
166  {
167  var bsonDocument = (BsonDocument)v;
168  using (var reader = new BsonDocumentReader(bsonDocument))
169  {
170  var context = BsonDeserializationContext.CreateRoot(reader);
171  var document = _serializer.Deserialize(context);
172  return document;
173  }
174  })
175  .ToList();
176 
177  var cursor = new AsyncCursor<TDocument>(
178  getMoreChannelSource.Fork(),
179  _collectionNamespace,
180  command,
181  firstBatch,
182  cursorId,
183  _batchSize ?? 0,
184  0, // limit
185  _serializer,
186  _messageEncoderSettings);
187 
188  cursors.Add(cursor);
189  }
190  }
191 
192  return cursors;
193  }

196  public IReadOnlyList<IAsyncCursor<TDocument>> Execute(IReadBinding binding, CancellationToken cancellationToken)
197  {
198  Ensure.IsNotNull(binding, nameof(binding));
199 
200  using (EventContext.BeginOperation())
201  using (var channelSource = binding.GetReadChannelSource(cancellationToken))
202  using (var channel = channelSource.GetChannel(cancellationToken))
203  using (var channelBinding = new ChannelReadBinding(channelSource.Server, channel, binding.ReadPreference, binding.Session.Fork()))
204  {
205  var operation = CreateOperation(channel, channelBinding);
206  var result = operation.Execute(channelBinding, cancellationToken);
207  return CreateCursors(channelSource, operation.Command, result);
208  }
