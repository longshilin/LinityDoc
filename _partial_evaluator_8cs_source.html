24  internal sealed class PartialEvaluator : ExpressionVisitor
25  {
26  public static Expression Evaluate(Expression node)
27  {
28  var candidates = Nominator.Nominate(node);
29  var evaluator = new PartialEvaluator(candidates);
30  return evaluator.Visit(node);
31  }

33  private readonly HashSet<Expression> _candidates;
34 
35  private PartialEvaluator(HashSet<Expression> candidates)
36  {
37  _candidates = candidates;
38  }
39 
40  public override Expression Visit(Expression node)
41  {
42  if (node == null)
43  {
44  return null;
45  }
46  if (!_candidates.Contains(node))
47  {
48  return base.Visit(node);
49  }
50 
51  var evaluated = EvaluateSubtree(node);
52 
53  if (evaluated != node)
54  {
55  return PartialEvaluator.Evaluate(evaluated);
56  }
57 
58  return evaluated;
59  }

61  private Expression EvaluateSubtree(Expression subtree)
62  {
63  subtree = ReflectionEvaluator.Evaluate(subtree);
64 
65  if (subtree.NodeType == ExpressionType.Constant)
66  {
67  // we don't want to partially evaluate constants...
68  var constant = (ConstantExpression)subtree;
69  var queryableValue = constant.Value as IQueryable;
70  if (queryableValue != null && queryableValue.Expression != constant)
71  {
72  return queryableValue.Expression;
73  }
74 
75  return subtree;
76  }
77 
78  Expression<Func<object>> lambda = Expression.Lambda<Func<object>>(Expression.Convert(subtree, typeof(object)));
79  var compiledLambda = lambda.Compile();
80  return Expression.Constant(compiledLambda(), subtree.Type);
81  }

83  private class Nominator : ExpressionVisitor
84  {
85  public static HashSet<Expression> Nominate(Expression node)
86  {
87  var nominator = new Nominator();
88  nominator.Visit(node);
89  return nominator._candidates;
90  }

92  private HashSet<Expression> _candidates;
93  private bool _isBlocked;
94 
95  private Nominator()
96  {
97  _candidates = new HashSet<Expression>();
98  }
99 
100  public override Expression Visit(Expression node)
101  {
102  if (node == null)
103  {
104  return base.Visit(node);
105  }
106 
107  bool _oldIsBlocked = _isBlocked;
108  _isBlocked = false;
109 
110  var visited = base.Visit(node);
111 
112  if (!_isBlocked)
113  {
114  _candidates.Add(node);
115  }
116 
117  _isBlocked |= _oldIsBlocked;
118  return visited;
119  }

157  protected override Expression VisitMethodCall(MethodCallExpression node)
158  {
159  if (IsQueryableExpression(node.Object))
160  {
161  _isBlocked = true;
162  }
163 
164  for (int i = 0; i < node.Arguments.Count && !_isBlocked; i++)
165  {
166  if (IsQueryableExpression(node.Arguments[i]))
167  {
168  _isBlocked = true;
169  }
170  }
171 
172  node = (MethodCallExpression)base.VisitMethodCall(node);
173  if (node.Object == null && node.Method.DeclaringType == typeof(LinqExtensions) &&
174  node.Method.Name == "Inject")
175  {
176  _isBlocked = true;
177  }
178 
179  return node;
180  }
181 
182  protected override Expression VisitParameter(ParameterExpression node)
183  {
184  _isBlocked = true;
185  return base.VisitParameter(node);
186  }

206  protected override Expression VisitBinary(BinaryExpression node)
207  {
208  if (node.NodeType == ExpressionType.ArrayIndex)
209  {
210  var left = Visit(node.Left);
211  var right = Visit(node.Right);
212 
213  if (left.NodeType == ExpressionType.Constant && right.NodeType == ExpressionType.Constant)
214  {
215  var array = (Array)((ConstantExpression)left).Value;
216  var index = (int)((ConstantExpression)right).Value;
217  return Expression.Constant(
218  array.GetValue(index),
219  node.Type);
220  }
221  }
222 
223  return node;
224  }
225 
226  protected override Expression VisitMember(MemberExpression node)
227  {
228  if (node.Expression != null)
229  {
230  // This will handle the situations where we have a local capture variable.
231 
232  var field = node.Member as FieldInfo;
233  if (field != null)
234  {
235  var obj = Visit(node.Expression);
236  if (obj.NodeType == ExpressionType.Constant)
237  {
238  var value = field.GetValue(((ConstantExpression)obj).Value);
239  return Expression.Constant(value, node.Type);
240  }
241  }
242 
243  var property = node.Member as PropertyInfo;
244  if (property != null)
245  {
246  var obj = Visit(node.Expression);
247  if (obj.NodeType == ExpressionType.Constant)
248  {
249  var value = property.GetValue(((ConstantExpression)obj).Value);
250  return Expression.Constant(value, node.Type);
251  }
252  }
253  }
254 
255  return node;
256  }
257 
258  protected override Expression VisitMethodCall(MethodCallExpression node)
259  {
260  var obj = Visit(node.Object);
261 
262  if (obj == null || obj.NodeType == ExpressionType.Constant)
263  {
264  var arguments = new object[node.Arguments.Count];
265  for (int i = 0; i < node.Arguments.Count; i++)
266  {
267  var argument = Visit(node.Arguments[i]);
268  if (argument.NodeType != ExpressionType.Constant)
269  {
270  return node;
271  }
272 
273  arguments[i] = ((ConstantExpression)argument).Value;
274  }
275 
276  var value = node.Method.Invoke(
277  obj == null ? null : ((ConstantExpression)obj).Value,
278  arguments);
279 
280  return Expression.Constant(value, node.Type);
281  }
282 
283  return node;
284  }
285  }
