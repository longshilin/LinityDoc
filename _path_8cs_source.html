371  internal uint CalculateHScore (GraphNode node) {
372  uint h;
373 
374  switch (heuristic) {
375  case Heuristic.Euclidean:
376  h = (uint)(((GetHTarget() - node.position).costMagnitude)*heuristicScale);
377  // Inlining this check and the return
378  // for each case saves an extra jump.
379  // This code is pretty hot
380  if (hTargetNode != null) {
381  h = System.Math.Max(h, PathFindHelper.euclideanEmbedding.GetHeuristic(node.NodeIndex, hTargetNode.NodeIndex));
382  }
383  return h;
384  case Heuristic.Manhattan:
385  Int3 p2 = node.position;
386  h = (uint)((System.Math.Abs(hTarget.x-p2.x) + System.Math.Abs(hTarget.y-p2.y) + System.Math.Abs(hTarget.z-p2.z))*heuristicScale);
387  if (hTargetNode != null) {
388  h = System.Math.Max(h, PathFindHelper.GetConfig().euclideanEmbedding.GetHeuristic(node.NodeIndex, hTargetNode.NodeIndex));
389  }
390  return h;
391  case Heuristic.DiagonalManhattan:
392  Int3 p = GetHTarget() - node.position;
393  p.x = System.Math.Abs(p.x);
394  p.y = System.Math.Abs(p.y);
395  p.z = System.Math.Abs(p.z);
396  int diag = System.Math.Min(p.x, p.z);
397  int diag2 = System.Math.Max(p.x, p.z);
398  h = (uint)((((14*diag)/10) + (diag2-diag) + p.y) * heuristicScale);
399  if (hTargetNode != null) {
400  h = System.Math.Max(h, PathFindHelper.euclideanEmbedding.GetHeuristic(node.NodeIndex, hTargetNode.NodeIndex));
401  }
402  return h;
403  }
404  return 0U;
405  }

670  public void Release (System.Object o, bool silent = false) {
671  if (o == null) throw new System.ArgumentNullException("o");

674  // Need to use ReferenceEquals because it might be called from another thread

704  protected virtual void Trace (PathNode from) {
705  // Current node we are processing
706  PathNode c = from;
707  int count = 0;
708 
709  while (c != null) {
710  c = c.parent;
711  count++;
712  if (count > 2048) {
713 #if !SERVER
714  UnityEngine.Debug.LogWarning("Infinite loop? >2048 node path. Remove this message if you really have that long paths (Path.cs, Trace method)");
715 #endif
716  break;
717  }
718  }
719 
720 
721  if (path.Capacity < count) path.Capacity = count;
722  if (vectorPath.Capacity < count) vectorPath.Capacity = count;
723 
724 
725  c = from;
726 
727  for (int i = 0; i < count; i++) {
728  path.Add(c.node);
729  c = c.parent;
730  }
731 
732  // Reverse
733  int half = count/2;
734  for (int i = 0; i < half; i++) {
735  var tmp = path[i];
736  path[i] = path[count-i-1];
737  path[count - i - 1] = tmp;
738  }
739 
740  for (int i = 0; i < count; i++) {
741  vectorPath.Add((Vector3)path[i].position);
742  }
743  }
