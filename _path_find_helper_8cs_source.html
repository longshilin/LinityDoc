124  public static NNInfo GetNearest(Vector3 position, NNConstraint constraint, GraphNode hint)
125  {
126  // Cache property lookup
127  var localGraphs = GetConfig().graphs;
128 
129  float minDist = float.PositiveInfinity;
130  NNInfoInternal nearestNode = new NNInfoInternal();
131  int nearestGraph = -1;
132 
133  if (localGraphs != null)
134  {
135  for (int i = 0; i < localGraphs.Length; i++)
136  {
137  NavGraph graph = localGraphs[i];
138 
139  // Check if this graph should be searched
140  if (graph == null || !constraint.SuitableGraph(i, graph))
141  {
142  continue;
143  }
144 
145  NNInfoInternal nnInfo;
146  if (GetConfig().fullGetNearestSearch)
147  {
148  // Slower nearest node search
149  // this will try to find a node which is suitable according to the constraint
150  nnInfo = graph.GetNearestForce(position, constraint);
151  }
152  else
153  {
154  // Fast nearest node search
155  // just find a node close to the position without using the constraint that much
156  // (unless that comes essentially 'for free')
157  nnInfo = graph.GetNearest(position, constraint);
158  }
159 
160  GraphNode node = nnInfo.node;
161 
162  // No node found in this graph
163  if (node == null)
164  {
165  continue;
166  }
167 
168  // Distance to the closest point on the node from the requested position
169  float dist = ((Vector3) nnInfo.clampedPosition - position).magnitude;
170 
171  if (GetConfig().prioritizeGraphs && dist < GetConfig().prioritizeGraphsLimit)
172  {
173  // The node is close enough, choose this graph and discard all others
174  minDist = dist;
175  nearestNode = nnInfo;
176  nearestGraph = i;
177  break;
178  }
179  else
180  {
181  // Choose the best node found so far
182  if (dist < minDist)
183  {
184  minDist = dist;
185  nearestNode = nnInfo;
186  nearestGraph = i;
187  }
188  }
189  }
190  }
191 
192  // No matches found
193  if (nearestGraph == -1)
194  {
195  return new NNInfo();
196  }
197 
198  // Check if a constrained node has already been set
199  if (nearestNode.constrainedNode != null)
200  {
201  nearestNode.node = nearestNode.constrainedNode;
202  nearestNode.clampedPosition = nearestNode.constClampedPosition;
203  }
204 
205  if (!GetConfig().fullGetNearestSearch && nearestNode.node != null && !constraint.Suitable(nearestNode.node))
206  {
207  // Otherwise, perform a check to force the graphs to check for a suitable node
208  NNInfoInternal nnInfo = localGraphs[nearestGraph].GetNearestForce(position, constraint);
209 
210  if (nnInfo.node != null)
211  {
212  nearestNode = nnInfo;
213  }
214  }
215 
216  if (!constraint.Suitable(nearestNode.node) || (constraint.constrainDistance &&
217  (nearestNode.clampedPosition - position).sqrMagnitude > GetConfig().maxNearestNodeDistanceSqr))
218  {
219  return new NNInfo();
220  }
221 
222  // Convert to NNInfo which doesn't have all the internal fields
223  return new NNInfo(nearestNode);
224  }
