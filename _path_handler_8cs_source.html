159  public void InitializeNode (GraphNode node) {
160  //Get the index of the node
161  int ind = node.NodeIndex;
162 
163  if (ind >= nodes.Length) {
164  // Grow by a factor of 2
165  PathNode[] newNodes = new PathNode[System.Math.Max(128, nodes.Length*2)];
166  nodes.CopyTo(newNodes, 0);
167  // Initialize all PathNode instances at once
168  // It is important that we do this here and don't for example leave the entries as NULL and initialize
169  // them lazily. By allocating them all at once we are much more likely to allocate the PathNodes close
170  // to each other in memory (most systems use some kind of bumb-allocator) and this improves cache locality
171  // and reduces false sharing (which would happen if we allocated PathNodes for the different threads close
172  // to each other). This has been profiled to give around a 4% difference in overall pathfinding performance.
173  for (int i = nodes.Length; i < newNodes.Length; i++) newNodes[i] = new PathNode();
174  nodes = newNodes;
175  }
176 
177  nodes[ind].node = node;
178  }
