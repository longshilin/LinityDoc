19  public static void FunnelModify (Path p) {
20  if (p.path == null || p.path.Count == 0 || p.vectorPath == null || p.vectorPath.Count == 0) {
21  return;
22  }
23 
24  List<Vector3> funnelPath = ListPool<Vector3>.Claim();
25 
26  // Split the path into different parts (separated by custom links)
27  // and run the funnel algorithm on each of them in turn
28  var parts = Funnel.SplitIntoParts(p);
29 
30  if (parts.Count == 0) {
31  // As a really special case, it might happen that the path contained only a single node
32  // and that node was part of a custom link (e.g added by the NodeLink2 component).
33  // In that case the SplitIntoParts method will not know what to do with it because it is
34  // neither a link (as only 1 of the 2 nodes of the link was part of the path) nor a normal
35  // path part. So it will skip it. This will cause it to return an empty list.
36  // In that case we want to simply keep the original path, which is just a single point.
37  return;
38  }
39 
40  for (int i = 0; i < parts.Count; i++) {
41  var part = parts[i];
42  if (!part.isLink) {
43  var portals = Funnel.ConstructFunnelPortals(p.path, part);
44  var result = Funnel.Calculate(portals, true, false);
45  funnelPath.AddRange(result);
46  ListPool<Vector3>.Release(ref portals.left);
47  ListPool<Vector3>.Release(ref portals.right);
48  ListPool<Vector3>.Release(ref result);
49  } else {
50  // non-link parts will add the start/end points for the adjacent parts.
51  // So if there is no non-link part before this one, then we need to add the start point of the link
52  // and if there is no non-link part after this one, then we need to add the end point.
53  if (i == 0 || parts[i-1].isLink) {
54  funnelPath.Add(part.startPoint);
55  }
56  if (i == parts.Count - 1 || parts[i+1].isLink) {
57  funnelPath.Add(part.endPoint);
58  }
59  }
60  }
61 
62  ListPool<Funnel.PathPart>.Release(ref parts);
63  // Pool the previous vectorPath
64  ListPool<Vector3>.Release(ref p.vectorPath);
65  p.vectorPath = funnelPath;
66  }
67  }
