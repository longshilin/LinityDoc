154  // public static methods
155  public static void Install()
156  {
157  if (PerformanceCounterCategory.Exists(CategoryName))
158  {
159  return;
160  }
161 
162  var properties = GetCounterProperties();
163  var collection = new CounterCreationDataCollection();
164  foreach (var property in properties)
165  {
166  var attribute = GetCounterAttribute(property);
167  collection.Add(new CounterCreationData(attribute.Name, attribute.Help, attribute.Type));
168  }
169 
170  PerformanceCounterCategory.Create(
171  CategoryName,
172  "Stats for the .NET MongoDB Driver.",
173  PerformanceCounterCategoryType.MultiInstance,
174  collection);
175  }
176 
177  // private static methods
178  private static PerformanceCounterAttribute GetCounterAttribute(PropertyInfo info)
179  {
180  return info.GetCustomAttributes(typeof(PerformanceCounterAttribute), false)
181  .OfType<PerformanceCounterAttribute>()
182  .Single();
183  }
184 
185  private static IEnumerable<PropertyInfo> GetCounterProperties()
186  {
187  return typeof(PerformanceCounterPackage)
188  .GetProperties()
189  .Where(x => x.PropertyType == typeof(PerformanceCounter));
190  }
191 
192  // public methods
193  public void Dispose()
194  {
195  if (!_disposed)
196  {
197  foreach (var property in GetCounterProperties())
198  {
199  var counter = (PerformanceCounter)property.GetValue(this, null);
200  try
201  {
202  counter.RemoveInstance();
203  }
204  catch (NotImplementedException)
205  {
206  // apparently this happens on Mono...
207  }
208  counter.Dispose();
209  property.SetValue(this, null, null);
210  }
211  }
212 
213  _disposed = true;
214  }
215 
216  private void Initialize(string instanceName)
217  {
218  foreach (var property in GetCounterProperties())
219  {
220  var attribute = GetCounterAttribute(property);
221 
222  var counter = new PerformanceCounter(
223  CategoryName,
224  attribute.Name,
225  instanceName,
226  false);
227 
228  counter.RawValue = 0; // initialize it to it's default value...
229 
230  property.SetValue(this, counter, null);
231  }
232  }
