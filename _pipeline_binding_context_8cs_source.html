81  public IBsonSerializer GetSerializer(Type type, Expression node)
82  {
83  Ensure.IsNotNull(type, nameof(type));
84 
85  IBsonSerializer serializer;
86  if (node != null && PreviouslyUsedSerializerFinder.TryFindSerializer(node, type, out serializer))
87  {
88  return serializer;
89  }
90  else if (node == null || type != node.Type)
91  {
92  serializer = _serializerRegistry.GetSerializer(type);
93  if (node != null)
94  {
95  var childConfigurableSerializer = serializer as IChildSerializerConfigurable;
96  if (childConfigurableSerializer != null)
97  {
98  var nodeSerializer = GetSerializer(node.Type, node);
99  var deepestChildSerializer = SerializerHelper.GetDeepestChildSerializer(childConfigurableSerializer);
100 
101  if (nodeSerializer.ValueType == deepestChildSerializer.ValueType)
102  {
103  serializer = SerializerHelper.RecursiveConfigureChildSerializer(childConfigurableSerializer, nodeSerializer);
104  }
105  }
106  }
107  }
108  else
109  {
110  serializer = SerializerBuilder.Build(node, _serializerRegistry);
111  }
112 
113  return serializer;
114  }

116  public SerializationExpression BindProjector(ref Expression selector)
117  {
118  var projector = selector as SerializationExpression;
119  if (selector.NodeType == ExpressionType.MemberInit || selector.NodeType == ExpressionType.New)
120  {
121  var serializer = SerializerBuilder.Build(selector, _serializerRegistry);
122 
123  projector = new DocumentExpression(serializer);
124  }
125  else if (projector == null || projector is PipelineExpression || projector is IFieldExpression || projector is ArrayIndexExpression)
126  {
127  var newFieldName = "__fld0";
128  if (projector is IFieldExpression)
129  {
130  // We don't have to do this, but it makes the output a little nicer.
131  newFieldName = ((IFieldExpression)projector).FieldName;
132  }
133 
134  // the output of a $project stage must be a document, so
135  // if this isn't already a serialization expression and it's not
136  // a new expression or member init, then we need to create an
137  // artificial field to project the computation into.
138  var serializer = GetSerializer(selector.Type, selector);
139  selector = new FieldAsDocumentExpression(selector, newFieldName, serializer);
140  projector = new FieldExpression(newFieldName, serializer);
141  }
142 
143  return projector;
144  }
