313  public override RenderedPipelineDefinition<TOutput> Render(IBsonSerializer<TInput> inputSerializer, IBsonSerializerRegistry serializerRegistry)
314  {
315  var pipeline = new List<BsonDocument>();
316 
317  IBsonSerializer currentSerializer = inputSerializer;
318  foreach (var stage in _stages)
319  {
320  var renderedStage = stage.Render(currentSerializer, serializerRegistry);
321  currentSerializer = renderedStage.OutputSerializer;
322  if (renderedStage.Document.ElementCount > 0)
323  {
324  pipeline.Add(renderedStage.Document);
325  }
326  }
327 
328  return new RenderedPipelineDefinition<TOutput>(
329  pipeline,
330  _outputSerializer ?? (currentSerializer as IBsonSerializer<TOutput>) ?? serializerRegistry.GetSerializer<TOutput>());
331  }

333  private static List<IPipelineStageDefinition> VerifyStages(List<IPipelineStageDefinition> stages)
334  {
335  var nextInputType = typeof(TInput);
336  for (int i = 0; i < stages.Count; i++)
337  {
338  if (stages[i].InputType != nextInputType)
339  {
340  var message = string.Format(
341  "The input type to stage[{0}] was expected to be {1}, but was {2}.",
342  i,
343  nextInputType,
344  stages[i].InputType);
345  throw new ArgumentException(message, "stages");
346  }
347 
348  nextInputType = stages[i].OutputType;
349  }
350 
351  if (nextInputType != typeof(TOutput))
352  {
353  var message = string.Format(
354  "The output type to the last stage was expected to be {0}, but was {1}.",
355  nextInputType,
356  stages.Last().OutputType);
357  throw new ArgumentException(message, "stages");
358  }
359 
360  return stages;
361  }

364  internal class OptimizingPipelineDefinition<TInput, TOutput> : PipelineDefinition<TInput, TOutput>
365  {
366  private readonly PipelineDefinition<TInput, TOutput> _wrapped;
367 
368  public OptimizingPipelineDefinition(PipelineDefinition<TInput, TOutput> wrapped)
369  {
370  _wrapped = wrapped;
371  }
372 
374  public override IBsonSerializer<TOutput> OutputSerializer => _wrapped.OutputSerializer;
375 
377  public override IEnumerable<IPipelineStageDefinition> Stages => _wrapped.Stages;
378 
379  public override RenderedPipelineDefinition<TOutput> Render(IBsonSerializer<TInput> inputSerializer, IBsonSerializerRegistry serializerRegistry)
380  {
381  var rendered = _wrapped.Render(inputSerializer, serializerRegistry);
382 
383  // do some combining of $match documents if possible. This is optimized for the
384  // OfType case where we've added a discriminator as a match at the beginning of the pipeline.
385  if (rendered.Documents.Count > 1)
386  {
387  var firstStage = rendered.Documents[0].GetElement(0);
388  var secondStage = rendered.Documents[1].GetElement(0);
389  if (firstStage.Name == "$match" && secondStage.Name == "$match")
390  {
391  var combinedFilter = Builders<BsonDocument>.Filter.And(
392  (BsonDocument)firstStage.Value,
393  (BsonDocument)secondStage.Value);
394  var combinedStage = new BsonDocument("$match", combinedFilter.Render(BsonDocumentSerializer.Instance, serializerRegistry));
395 
396  rendered.Documents[0] = combinedStage;
397  rendered.Documents.RemoveAt(1);
398  }
399  }
400 
401  return rendered;
402  }
403  }
