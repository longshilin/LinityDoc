1114  public sealed class AppendedStagePipelineDefinition<TInput, TIntermediate, TOutput> : PipelineDefinition<TInput, TOutput>
1115  {
1116  private readonly IBsonSerializer<TOutput> _outputSerializer;
1117  private readonly PipelineDefinition<TInput, TIntermediate> _pipeline;
1118  private readonly PipelineStageDefinition<TIntermediate, TOutput> _stage;
1119 
1126  public AppendedStagePipelineDefinition(
1127  PipelineDefinition<TInput, TIntermediate> pipeline,
1128  PipelineStageDefinition<TIntermediate, TOutput> stage,
1129  IBsonSerializer<TOutput> outputSerializer = null)
1130  {
1131  _pipeline = Ensure.IsNotNull(pipeline, nameof(pipeline));
1132  _stage = Ensure.IsNotNull(stage, nameof(stage));
1133  _outputSerializer = outputSerializer; // can be null
1134  }
1135 
1137  public override IBsonSerializer<TOutput> OutputSerializer => _outputSerializer;
1138 
1140  public override IEnumerable<IPipelineStageDefinition> Stages => _pipeline.Stages.Concat(new[] { _stage });
1141 
1143  public override RenderedPipelineDefinition<TOutput> Render(IBsonSerializer<TInput> inputSerializer, IBsonSerializerRegistry serializerRegistry)
1144  {
1145  var renderedPipeline = _pipeline.Render(inputSerializer, serializerRegistry);
1146  var renderedStage = _stage.Render(renderedPipeline.OutputSerializer, serializerRegistry);
1147  var documents = renderedPipeline.Documents.Concat(new[] { renderedStage.Document });
1148  var outputSerializer = _outputSerializer ?? renderedStage.OutputSerializer;
1149  return new RenderedPipelineDefinition<TOutput>(documents, outputSerializer);
1150  }
