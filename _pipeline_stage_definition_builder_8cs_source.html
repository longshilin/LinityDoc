42  public static PipelineStageDefinition<TInput, AggregateBucketResult<TValue>> Bucket<TInput, TValue>(
43  AggregateExpressionDefinition<TInput, TValue> groupBy,
44  IEnumerable<TValue> boundaries,
45  AggregateBucketOptions<TValue> options = null)
46  {
47  Ensure.IsNotNull(groupBy, nameof(groupBy));
48  Ensure.IsNotNull(boundaries, nameof(boundaries));
49 
50  const string operatorName = "$bucket";
51  var stage = new DelegatedPipelineStageDefinition<TInput, AggregateBucketResult<TValue>>(
52  operatorName,
53  (s, sr) =>
54  {
55  var valueSerializer = sr.GetSerializer<TValue>();
56  var renderedGroupBy = groupBy.Render(s, sr);
57  var serializedBoundaries = boundaries.Select(b => valueSerializer.ToBsonValue(b));
58  var serializedDefaultBucket = options != null && options.DefaultBucket.HasValue ? valueSerializer.ToBsonValue(options.DefaultBucket.Value) : null;
59  var document = new BsonDocument
60  {
61  { operatorName, new BsonDocument
62  {
63  { "groupBy", renderedGroupBy },
64  { "boundaries", new BsonArray(serializedBoundaries) },
65  { "default", serializedDefaultBucket, serializedDefaultBucket != null }
66  }
67  }
68  };
69  return new RenderedPipelineStageDefinition<AggregateBucketResult<TValue>>(
70  operatorName,
71  document,
72  sr.GetSerializer<AggregateBucketResult<TValue>>());
73  });
74 
75  return stage;
76  }

476  public static PipelineStageDefinition<TInput, TOutput> GraphLookup<TInput, TFrom, TConnectFrom, TConnectTo, TStartWith, TAsElement, TAs, TOutput>(
477  IMongoCollection<TFrom> from,
478  FieldDefinition<TFrom, TConnectFrom> connectFromField,
479  FieldDefinition<TFrom, TConnectTo> connectToField,
480  AggregateExpressionDefinition<TInput, TStartWith> startWith,
481  FieldDefinition<TOutput, TAs> @as,
482  FieldDefinition<TAsElement, int> depthField,
483  AggregateGraphLookupOptions<TFrom, TAsElement, TOutput> options = null)
484  where TAs : IEnumerable<TAsElement>
485  {
486  Ensure.IsNotNull(from, nameof(from));
487  Ensure.IsNotNull(connectFromField, nameof(connectFromField));
488  Ensure.IsNotNull(connectToField, nameof(connectToField));
489  Ensure.IsNotNull(startWith, nameof(startWith));
490  Ensure.IsNotNull(@as, nameof(@as));
491  Ensure.That(IsTConnectToOrEnumerableTConnectTo<TConnectFrom, TConnectTo>(), "TConnectFrom must be either TConnectTo or a type that implements IEnumerable<TConnectTo>.", nameof(TConnectFrom));
492  Ensure.That(IsTConnectToOrEnumerableTConnectTo<TStartWith, TConnectTo>(), "TStartWith must be either TConnectTo or a type that implements IEnumerable<TConnectTo>.", nameof(TStartWith));
493 
494  const string operatorName = "$graphLookup";
495  var stage = new DelegatedPipelineStageDefinition<TInput, TOutput>(
496  operatorName,
497  (s, sr) =>
498  {
499  var inputSerializer = s;
500  var outputSerializer = options?.OutputSerializer ?? sr.GetSerializer<TOutput>();
501  var fromSerializer = options?.FromSerializer ?? sr.GetSerializer<TFrom>();
502  var asElementSerializer = options?.AsElementSerializer ?? sr.GetSerializer<TAsElement>();
503  var renderedConnectFromField = connectFromField.Render(fromSerializer, sr);
504  var renderedConnectToField = connectToField.Render(fromSerializer, sr);
505  var renderedStartWith = startWith.Render(inputSerializer, sr);
506  var renderedAs = @as.Render(outputSerializer, sr);
507  var renderedDepthField = depthField?.Render(asElementSerializer, sr);
508  var renderedRestrictSearchWithMatch = options?.RestrictSearchWithMatch?.Render(fromSerializer, sr);
509  var document = new BsonDocument
510  {
511  { operatorName, new BsonDocument
512  {
513  { "from", from.CollectionNamespace.CollectionName },
514  { "connectFromField", renderedConnectFromField.FieldName },
515  { "connectToField", renderedConnectToField.FieldName },
516  { "startWith", renderedStartWith },
517  { "as", renderedAs.FieldName },
518  { "depthField", () => renderedDepthField.FieldName, renderedDepthField != null },
519  { "maxDepth", () => options.MaxDepth.Value, options != null && options.MaxDepth.HasValue },
520  { "restrictSearchWithMatch", renderedRestrictSearchWithMatch, renderedRestrictSearchWithMatch != null }
521  }
522  }
523  };
524  return new RenderedPipelineStageDefinition<TOutput>(operatorName, document, outputSerializer);
525  });
526 
527  return stage;
528  }
