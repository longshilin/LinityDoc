95  private void OnWebRequestSuccess(object sender, GameEventArgs e)
96  {
97  WebRequestSuccessEventArgs ne = (WebRequestSuccessEventArgs)e;
98  if (ne.UserData != this)
99  {
100  return;
101  }
102 
103  // 解析版本信息
104  byte[] versionInfoBytes = ne.GetWebResponseBytes();
105  string versionInfoString = Utility.Converter.GetString(versionInfoBytes);
106  m_VersionInfo = Utility.Json.ToObject<VersionInfo>(versionInfoString);
107  if (m_VersionInfo == null)
108  {
109  Log.Error("Parse VersionInfo failure.");
110  return;
111  }
112 
113  Log.Info("Latest game version is '{0} ({1})', local game version is '{2} ({3})'.", m_VersionInfo.LatestGameVersion, m_VersionInfo.InternalGameVersion.ToString(), Version.GameVersion, Version.InternalGameVersion.ToString());
114 
115  if (m_VersionInfo.ForceUpdateGame)
116  {
117  // 需要强制更新游戏应用
118  GameEntry.UI.OpenDialog(new DialogParams
119  {
120  Mode = 2,
121  Title = GameEntry.Localization.GetString("ForceUpdate.Title"),
122  Message = GameEntry.Localization.GetString("ForceUpdate.Message"),
123  ConfirmText = GameEntry.Localization.GetString("ForceUpdate.UpdateButton"),
124  OnClickConfirm = GotoUpdateApp,
125  CancelText = GameEntry.Localization.GetString("ForceUpdate.QuitButton"),
126  OnClickCancel = delegate (object userData) { UnityGameFramework.Runtime.GameEntry.Shutdown(ShutdownType.Quit); },
127  });
128 
129  return;
130  }
131 
132  // 设置资源更新下载地址
133  GameEntry.Resource.UpdatePrefixUri = Utility.Path.GetRegularPath(m_VersionInfo.UpdatePrefixUri);
134 
135  Log.Info("UpdatePrefixUri: {0}", GameEntry.Resource.UpdatePrefixUri);
136  m_CheckVersionComplete = true;
137  m_NeedUpdateVersion = GameEntry.Resource.CheckVersionList(m_VersionInfo.InternalResourceVersion) == CheckVersionListResult.NeedUpdate;
138  }
139 
140  private void OnWebRequestFailure(object sender, GameEventArgs e)
141  {
142  WebRequestFailureEventArgs ne = (WebRequestFailureEventArgs)e;
143  if (ne.UserData != this)
144  {
145  return;
146  }
147 
148  Log.Warning("Check version failure, error message is '{0}'.", ne.ErrorMessage);
149  }
150  }
