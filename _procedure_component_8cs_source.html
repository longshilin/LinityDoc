73  for (int i = 0; i < m_AvailableProcedureTypeNames.Length; i++)
74  {
75  Type procedureType = Utility.Assembly.GetType(m_AvailableProcedureTypeNames[i]);

79  yield break;

83  if (procedures[i] == null)

85  Log.Error("Can not create procedure instance '{0}'.", m_AvailableProcedureTypeNames[i]);
86  yield break;
