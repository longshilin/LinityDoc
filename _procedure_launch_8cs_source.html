54  private void InitLanguageSettings()
55  {
56  if (GameEntry.Base.EditorResourceMode && GameEntry.Base.EditorLanguage != Language.Unspecified)
57  {
58  // 编辑器资源模式直接使用 Inspector 上设置的语言
59  return;
60  }
61 
62  Language language = GameEntry.Localization.Language;
63  if (GameEntry.Setting.HasSetting(Constant.Setting.Language))
64  {
65  try
66  {
67  string languageString = GameEntry.Setting.GetString(Constant.Setting.Language);
68  language = (Language)Enum.Parse(typeof(Language), languageString);
69  }
70  catch
71  {
72  }
73  }
74 
75  if (language != Language.English
76  && language != Language.ChineseSimplified
77  && language != Language.ChineseTraditional
78  && language != Language.Korean)
79  {
80  // 若是暂不支持的语言，则使用英语
81  language = Language.English;
82 
83  GameEntry.Setting.SetString(Constant.Setting.Language, language.ToString());
84  GameEntry.Setting.Save();
85  }
86 
87  GameEntry.Localization.Language = language;
88  Log.Info("Init language settings complete, current language is '{0}'.", language.ToString());
89  }
90 
91  private void InitCurrentVariant()
92  {
93  if (GameEntry.Base.EditorResourceMode)
94  {
95  // 编辑器资源模式不使用 AssetBundle，也就没有变体了
96  return;
97  }
98 
99  string currentVariant = null;
100  switch (GameEntry.Localization.Language)
101  {
102  case Language.English:
103  currentVariant = "en-us";
104  break;
105 
106  case Language.ChineseSimplified:
107  currentVariant = "zh-cn";
108  break;
109 
110  case Language.ChineseTraditional:
111  currentVariant = "zh-tw";
112  break;
113 
114  case Language.Korean:
115  currentVariant = "ko-kr";
116  break;
117 
118  default:
119  currentVariant = "zh-cn";
120  break;
121  }
122 
123  GameEntry.Resource.SetCurrentVariant(currentVariant);
124  Log.Info("Init current variant complete.");
125  }
126 
127  private void InitSoundSettings()
128  {
129  GameEntry.Sound.Mute("Music", GameEntry.Setting.GetBool(Constant.Setting.MusicMuted, false));
130  GameEntry.Sound.SetVolume("Music", GameEntry.Setting.GetFloat(Constant.Setting.MusicVolume, 0.3f));
131  GameEntry.Sound.Mute("Sound", GameEntry.Setting.GetBool(Constant.Setting.SoundMuted, false));
132  GameEntry.Sound.SetVolume("Sound", GameEntry.Setting.GetFloat(Constant.Setting.SoundVolume, 1f));
133  GameEntry.Sound.Mute("UISound", GameEntry.Setting.GetBool(Constant.Setting.UISoundMuted, false));
134  GameEntry.Sound.SetVolume("UISound", GameEntry.Setting.GetFloat(Constant.Setting.UISoundVolume, 1f));
135  Log.Info("Init sound settings complete.");
136  }
137  }
