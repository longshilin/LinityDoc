71  protected override void OnUpdate(ProcedureOwner procedureOwner, float elapseSeconds, float realElapseSeconds)
72  {

74 
75  foreach (KeyValuePair<string, bool> loadedFlag in m_LoadedFlag)

132  private void LoadFont(string fontName)
133  {
134  m_LoadedFlag.Add(GameFramework.Utility.Text.Format("Font.{0}", fontName), false);
135  GameEntry.Resource.LoadAsset(AssetUtility.GetFontAsset(fontName), Constant.AssetPriority.FontAsset, new LoadAssetCallbacks(
136  (assetName, asset, duration, userData) =>
137  {
138  m_LoadedFlag[GameFramework.Utility.Text.Format("Font.{0}", fontName)] = true;
139  UGuiForm.SetMainFont((Font) asset);
140  Log.Info("Load font '{0}' OK.", fontName);
141  },
142  (assetName, status, errorMessage, userData) => { Log.Error("Can not load font '{0}' from '{1}' with error message '{2}'.", fontName, assetName, errorMessage); }));
143  }
