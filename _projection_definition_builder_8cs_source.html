378  internal sealed class CombinedProjectionDefinition<TSource> : ProjectionDefinition<TSource>
379  {
380  private readonly List<ProjectionDefinition<TSource>> _projections;
381 
382  public CombinedProjectionDefinition(IEnumerable<ProjectionDefinition<TSource>> projections)
383  {
384  _projections = Ensure.IsNotNull(projections, nameof(projections)).ToList();
385  }
386 
387  public override BsonDocument Render(IBsonSerializer<TSource> sourceSerializer, IBsonSerializerRegistry serializerRegistry)
388  {
389  var document = new BsonDocument();
390 
391  foreach (var projection in _projections)
392  {
393  var renderedProjection = projection.Render(sourceSerializer, serializerRegistry);
394 
395  foreach (var element in renderedProjection.Elements)
396  {
397  // last one wins
398  document.Remove(element.Name);
399  document.Add(element);
400  }
401  }
402 
403  return document;
404  }
405  }

407  internal sealed class ElementMatchProjectionDefinition<TSource, TItem> : ProjectionDefinition<TSource>
408  {
409  private readonly FieldDefinition<TSource> _field;
410  private readonly FilterDefinition<TItem> _filter;
411 
412  public ElementMatchProjectionDefinition(FieldDefinition<TSource> field, FilterDefinition<TItem> filter)
413  {
414  _field = Ensure.IsNotNull(field, nameof(field));
415  _filter = filter;
416  }
417 
418  public override BsonDocument Render(IBsonSerializer<TSource> sourceSerializer, IBsonSerializerRegistry serializerRegistry)
419  {
420  var renderedField = _field.Render(sourceSerializer, serializerRegistry);
421 
422  IBsonSerializer<TItem> itemSerializer;
423  if (renderedField.FieldSerializer != null)
424  {
425  var arraySerializer = renderedField.FieldSerializer as IBsonArraySerializer;
426  BsonSerializationInfo itemSerializationInfo;
427  if (arraySerializer == null || !arraySerializer.TryGetItemSerializationInfo(out itemSerializationInfo))
428  {
429  var message = string.Format("The serializer for field '{0}' must implement IBsonArraySerializer and provide item serialization info.", renderedField.FieldName);
430  throw new InvalidOperationException(message);
431  }
432  itemSerializer = (IBsonSerializer<TItem>)itemSerializationInfo.Serializer;
433  }
434  else
435  {
436  itemSerializer = serializerRegistry.GetSerializer<TItem>();
437  }
438 
439  var renderedFilter = _filter.Render(itemSerializer, serializerRegistry);
440 
441  return new BsonDocument(renderedField.FieldName, new BsonDocument("$elemMatch", renderedFilter));
442  }
443  }
