82  internal QueryMessage ReadMessage<TDocument>(IBsonSerializer<TDocument> serializer)
83  where TDocument : BsonDocument
84  {
85  var binaryReader = CreateBinaryReader();
86  var stream = binaryReader.BsonStream;
87  var startPosition = stream.Position;
88 
89  var messageSize = stream.ReadInt32();
90  var requestId = stream.ReadInt32();
91  stream.ReadInt32(); // responseTo
92  stream.ReadInt32(); // opcode
93  var flags = (QueryFlags)stream.ReadInt32();
94  var fullCollectionName = stream.ReadCString(Encoding);
95  var skip = stream.ReadInt32();
96  var batchSize = stream.ReadInt32();
97  var context = BsonDeserializationContext.CreateRoot(binaryReader);
98  var query = serializer.Deserialize(context);
99  BsonDocument fields = null;
100  if (stream.Position < startPosition + messageSize)
101  {
102  fields = serializer.Deserialize(context);
103  }
104 
105  var awaitData = (flags & QueryFlags.AwaitData) == QueryFlags.AwaitData;
106  var slaveOk = (flags & QueryFlags.SlaveOk) == QueryFlags.SlaveOk;
107  var partialOk = (flags & QueryFlags.Partial) == QueryFlags.Partial;
108  var noCursorTimeout = (flags & QueryFlags.NoCursorTimeout) == QueryFlags.NoCursorTimeout;
109  var oplogReplay = (flags & QueryFlags.OplogReplay) == QueryFlags.OplogReplay;
110  var tailableCursor = (flags & QueryFlags.TailableCursor) == QueryFlags.TailableCursor;
111 
112  return new QueryMessage(
113  requestId,
114  CollectionNamespace.FromFullName(fullCollectionName),
115  query,
116  fields,
117  NoOpElementNameValidator.Instance,
118  skip,
119  batchSize,
120  slaveOk,
121  partialOk,
122  noCursorTimeout,
123  oplogReplay,
124  tailableCursor,
125  awaitData);
126  }

132  public void WriteMessage(QueryMessage message)
133  {
134  Ensure.IsNotNull(message, nameof(message));
135 
136  var binaryWriter = CreateBinaryWriter();
137  var stream = binaryWriter.BsonStream;
138  var messageStartPosition = stream.Position;
139 
140  stream.WriteInt32(0); // messageSize
141  stream.WriteInt32(message.RequestId);
142  stream.WriteInt32(0); // responseTo
143  stream.WriteInt32((int)Opcode.Query);
144  stream.WriteInt32((int)BuildQueryFlags(message));
145  stream.WriteCString(message.CollectionNamespace.FullName);
146  stream.WriteInt32(message.Skip);
147  stream.WriteInt32(message.BatchSize);
148  WriteQuery(binaryWriter, message.Query, message.QueryValidator);
149  WriteOptionalFields(binaryWriter, message.Fields);
150  stream.BackpatchSize(messageStartPosition);
151 
152  message.PostWriteAction?.Invoke(new PostProcessor(message, stream, messageStartPosition));
153  }

165  private void WriteQuery(BsonBinaryWriter binaryWriter, BsonDocument query, IElementNameValidator queryValidator)
166  {
167  var maxWireDocumentSize = MaxWireDocumentSize ?? MaxDocumentSize ?? binaryWriter.Settings.MaxDocumentSize;
168 
169  binaryWriter.PushSettings(s => ((BsonBinaryWriterSettings)s).MaxDocumentSize = maxWireDocumentSize);
170  binaryWriter.PushElementNameValidator(queryValidator);
171  try
172  {
173  var context = BsonSerializationContext.CreateRoot(binaryWriter);
174  BsonDocumentSerializer.Instance.Serialize(context, query ?? new BsonDocument());
175  }
176  finally
177  {
178  binaryWriter.PopElementNameValidator();
179  binaryWriter.PopSettings();
180  }
181  }

218  private void ChangeWFrom0To1()
219  {
220  var wPosition = FindWPosition();
221  _stream.Position = wPosition;
222  var w = _stream.ReadInt32();
223  if (w != 0)
224  {
225  throw new InvalidOperationException("w was not 0.");
226  }
227  _stream.Position = wPosition;
228  _stream.WriteInt32(1);
229  }

231  private long FindWPosition()
232  {
233  _stream.Position = _messageStartPosition + 20;
234  _stream.SkipCString();
235  _stream.Position += 8;
236 
237  using (var reader = new BsonBinaryReader(_stream))
238  {
239  return FindWPosition(reader, false);
240  }
241  }
242 
243  private long FindWPosition(BsonBinaryReader reader, bool alreadyUnwrapped)
244  {
245  reader.ReadStartDocument();
246  while (reader.ReadBsonType() != 0)
247  {
248  var name = reader.ReadName();
249  if (name == "writeConcern")
250  {
251  reader.ReadStartDocument();
252  while (reader.ReadBsonType() != 0)
253  {
254  if (reader.ReadName() == "w")
255  {
256  if (reader.CurrentBsonType == BsonType.Int32)
257  {
258  return _stream.Position;
259  }
260  else
261  {
262  goto notFound;
263  }
264  }
265  else
266  {
267  reader.SkipValue();
268  }
269  }
270  goto notFound;
271  }
272  else if (name == "$query" && !alreadyUnwrapped)
273  {
274  return FindWPosition(reader, true);
275  }
276 
277  reader.SkipValue();
278  }
279 
280  notFound:
281  throw new InvalidOperationException("{ w : <Int32> } not found.");
282  }
