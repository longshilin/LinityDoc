48  public QueryMessage ReadMessage()
49  {
50  var jsonReader = CreateJsonReader();
51  var messageContext = BsonDeserializationContext.CreateRoot(jsonReader);
52  var messageDocument = BsonDocumentSerializer.Instance.Deserialize(messageContext);
53 
54  var opcode = messageDocument["opcode"].AsString;
55  if (opcode != "query")
56  {
57  throw new FormatException("Opcode is not query.");
58  }
59 
60  var requestId = messageDocument["requestId"].ToInt32();
61  var databaseName = messageDocument["database"].AsString;
62  var collectionName = messageDocument["collection"].AsString;
63  var query = messageDocument["query"].AsBsonDocument;
64  var fields = (BsonDocument)messageDocument.GetValue("fields", null);
65  var skip = messageDocument.GetValue("skip", 0).ToInt32();
66  var batchSize = messageDocument.GetValue("batchSize", 0).ToInt32();
67  var slaveOk = messageDocument.GetValue("slaveOk", false).ToBoolean();
68  var partialOk = messageDocument.GetValue("partialOk", false).ToBoolean();
69  var noCursorTimeout = messageDocument.GetValue("noCursorTimeout", false).ToBoolean();
70  var oplogReplay = messageDocument.GetValue("oplogReplay", false).ToBoolean();
71  var tailableCursor = messageDocument.GetValue("tailableCursor", false).ToBoolean();
72  var awaitData = messageDocument.GetValue("awaitData", false).ToBoolean();
73 
74  return new QueryMessage(
75  requestId,
76  new CollectionNamespace(databaseName, collectionName),
77  query,
78  fields,
79  NoOpElementNameValidator.Instance,
80  skip,
81  batchSize,
82  slaveOk,
83  partialOk,
84  noCursorTimeout,
85  oplogReplay,
86  tailableCursor,
87  awaitData);
88  }

94  public void WriteMessage(QueryMessage message)
95  {
96  Ensure.IsNotNull(message, nameof(message));
97 
98  var messageDocument = new BsonDocument
99  {
100  { "opcode", "query" },
101  { "requestId", message.RequestId },
102  { "database", message.CollectionNamespace.DatabaseNamespace.DatabaseName },
103  { "collection", message.CollectionNamespace.CollectionName },
104  { "fields", message.Fields, message.Fields != null },
105  { "skip", message.Skip, message.Skip != 0 },
106  { "batchSize", message.BatchSize, message.BatchSize != 0 },
107  { "slaveOk", true, message.SlaveOk },
108  { "partialOk", true, message.PartialOk },
109  { "noCursorTimeout", true, message.NoCursorTimeout },
110  { "oplogReplay", true, message.OplogReplay },
111  { "tailableCursor", true, message.TailableCursor },
112  { "awaitData", true, message.AwaitData },
113  { "query", message.Query }
114  };
115 
116  var jsonWriter = CreateJsonWriter();
117  var messageContext = BsonSerializationContext.CreateRoot(jsonWriter);
118  BsonDocumentSerializer.Instance.Serialize(messageContext, messageDocument);
119  }
