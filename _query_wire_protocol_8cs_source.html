117  private CursorBatch<TDocument> ProcessReply(ConnectionId connectionId, ReplyMessage<TDocument> reply)
118  {
119  if (reply.QueryFailure)
120  {
121  var response = reply.QueryFailureDocument;
122 
123  var notPrimaryOrNodeIsRecoveringException = ExceptionMapper.MapNotPrimaryOrNodeIsRecovering(connectionId, _query, response, "$err");
124  if (notPrimaryOrNodeIsRecoveringException != null)
125  {
126  throw notPrimaryOrNodeIsRecoveringException;
127  }
128 
129  var mappedException = ExceptionMapper.Map(connectionId, response);
130  if (mappedException != null)
131  {
132  throw mappedException;
133  }
134 
135  var message = string.Format("QueryFailure flag was true (response was {0}).", response.ToJson());
136  throw new MongoQueryException(connectionId, message, _query, response);
137  }
138 
139  return new CursorBatch<TDocument>(reply.CursorId, reply.Documents);
140  }
