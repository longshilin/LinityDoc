268  internal float InsertAgentNeighbour (Agent agent, float rangeSq) {
269  // Check if this agent collides with the other agent
270  if (this == agent || (agent.layer & collidesWith) == 0) return rangeSq;
271 
272  // 2D distance
273  float dist = (agent.position - position).sqrMagnitude;
274 
275  if (dist < rangeSq) {
276  if (neighbours.Count < maxNeighbours) {
277  neighbours.Add(null);
278  neighbourDists.Add(float.PositiveInfinity);
279  }
280 
281  // Insert the agent into the ordered list of neighbours
282  int i = neighbours.Count-1;
283  if (dist < neighbourDists[i]) {
284  while (i != 0 && dist < neighbourDists[i-1]) {
285  neighbours[i] = neighbours[i-1];
286  neighbourDists[i] = neighbourDists[i-1];
287  i--;
288  }
289  neighbours[i] = agent;
290  neighbourDists[i] = dist;
291  }
292 
293  if (neighbours.Count == maxNeighbours) {
294  rangeSq = neighbourDists[neighbourDists.Count-1];
295  }
296  }
297  return rangeSq;
298  }

371  public VO (Vector2 center, Vector2 offset, float radius, float inverseDt, float inverseDeltaTime) {
372  // Adjusted so that a parameter weightFactor of 1 will be the default ("natural") weight factor
373  this.weightFactor = 1;
374  weightBonus = 0;
375 
376  //this.radius = radius;
377  Vector2 globalCenter;
378 
379  circleCenter = center*inverseDt + offset;
380 
381  this.weightFactor = 4*Mathf.Exp(-Sqr(center.sqrMagnitude/(radius*radius))) + 1;
382  // Collision?
383  if (center.magnitude < radius) {
384  colliding = true;
385 
386  // 0.001 is there to make sure lin1.magnitude is not so small that the normalization
387  // below will return Vector2.zero as that will make the VO invalid and it will be ignored.
388  line1 = center.normalized * (center.magnitude - radius - 0.001f) * 0.3f * inverseDeltaTime;
389  dir1 = new Vector2(line1.y, -line1.x).normalized;
390  line1 += offset;
391 
392  cutoffDir = Vector2.zero;
393  cutoffLine = Vector2.zero;
394  dir2 = Vector2.zero;
395  line2 = Vector2.zero;
396  this.radius = 0;
397  } else {
398  colliding = false;
399 
400  center *= inverseDt;
401  radius *= inverseDt;
402  globalCenter = center+offset;
403 
404  // 0.001 is there to make sure cutoffDistance is not so small that the normalization
405  // below will return Vector2.zero as that will make the VO invalid and it will be ignored.
406  var cutoffDistance = center.magnitude - radius + 0.001f;
407 
408  cutoffLine = center.normalized * cutoffDistance;
409  cutoffDir = new Vector2(-cutoffLine.y, cutoffLine.x).normalized;
410  cutoffLine += offset;
411 
412  float alpha = Mathf.Atan2(-center.y, -center.x);
413 
414  float delta = Mathf.Abs(Mathf.Acos(radius/center.magnitude));
415 
416  this.radius = radius;
417 
418  // Bounding Lines
419 
420  // Point on circle
421  line1 = new Vector2(Mathf.Cos(alpha+delta), Mathf.Sin(alpha+delta));
422  // Vector tangent to circle which is the correct line tangent
423  // Note that this vector is normalized
424  dir1 = new Vector2(line1.y, -line1.x);
425 
426  // Point on circle
427  line2 = new Vector2(Mathf.Cos(alpha-delta), Mathf.Sin(alpha-delta));
428  // Vector tangent to circle which is the correct line tangent
429  // Note that this vector is normalized
430  dir2 = new Vector2(line2.y, -line2.x);
431 
432  line1 = line1 * radius + globalCenter;
433  line2 = line2 * radius + globalCenter;
434  }

436  segmentStart = Vector2.zero;
437  segmentEnd = Vector2.zero;
438  segment = false;

444  public static VO SegmentObstacle (Vector2 segmentStart, Vector2 segmentEnd, Vector2 offset, float radius, float inverseDt, float inverseDeltaTime) {
445  var vo = new VO();
446 
447  // Adjusted so that a parameter weightFactor of 1 will be the default ("natural") weight factor
448  vo.weightFactor = 1;
449  // Just higher than anything else
450  vo.weightBonus = Mathf.Max(radius, 1)*40;
451 
452  var closestOnSegment = VectorMath.ClosestPointOnSegment(segmentStart.ToPFV2(), segmentEnd.ToPFV2(), Vector2.zero.ToPFV2());
453 
454  // Collision?
455  if (closestOnSegment.magnitude <= radius) {
456  vo.colliding = true;
457 
458  vo.line1 = closestOnSegment.normalized.ToUnityV3() * (closestOnSegment.magnitude - radius) * 0.3f * inverseDeltaTime;
459  vo.dir1 = new Vector2(vo.line1.y, -vo.line1.x).normalized;
460  vo.line1 += offset;
461 
462  vo.cutoffDir = Vector2.zero;
463  vo.cutoffLine = Vector2.zero;
464  vo.dir2 = Vector2.zero;
465  vo.line2 = Vector2.zero;
466  vo.radius = 0;
467 
468  vo.segmentStart = Vector2.zero;
469  vo.segmentEnd = Vector2.zero;
470  vo.segment = false;
471  } else {
472  vo.colliding = false;
473 
474  segmentStart *= inverseDt;
475  segmentEnd *= inverseDt;
476  radius *= inverseDt;
477 
478  var cutoffTangent = (segmentEnd - segmentStart).normalized;
479  vo.cutoffDir = cutoffTangent;
480  vo.cutoffLine = segmentStart + new Vector2(-cutoffTangent.y, cutoffTangent.x) * radius;
481  vo.cutoffLine += offset;
482 
483  // See documentation for details
484  // The call to Max is just to prevent floating point errors causing NaNs to appear
485  var startSqrMagnitude = segmentStart.sqrMagnitude;
486  var normal1 = -VectorMath.ComplexMultiply(segmentStart, new Vector2(radius, Mathf.Sqrt(Mathf.Max(0, startSqrMagnitude - radius*radius)))) / startSqrMagnitude;
487  var endSqrMagnitude = segmentEnd.sqrMagnitude;
488  var normal2 = -VectorMath.ComplexMultiply(segmentEnd, new Vector2(radius, -Mathf.Sqrt(Mathf.Max(0, endSqrMagnitude - radius*radius)))) / endSqrMagnitude;
489 
490  vo.line1 = segmentStart + normal1.ToUnityV2() * radius + offset;
491  vo.line2 = segmentEnd + normal2.ToUnityV2() * radius + offset;
492 
493  // Note that the normals are already normalized
494  vo.dir1 = new Vector2(normal1.y, -normal1.x);
495  vo.dir2 = new Vector2(normal2.y, -normal2.x);
496 
497  vo.segmentStart = segmentStart;
498  vo.segmentEnd = segmentEnd;
499  vo.radius = radius;
500  vo.segment = true;
501  }
502 
503  return vo;
504  }

543  public Vector2 Gradient (Vector2 p, out float weight) {
544  if (colliding) {
545  // Calculate double signed area of the triangle consisting of the points
546  // {line1, line1+dir1, p}
547  float l1 = SignedDistanceFromLine(line1, dir1, p);
548 
549  // Serves as a check for which side of the line the point p is
550  if (l1 >= 0) {
551  weight = l1;
552  return new Vector2(-dir1.y, dir1.x);
553  } else {
554  weight = 0;
555  return new Vector2(0, 0);
556  }
557  }
558 
559  float det3 = SignedDistanceFromLine(cutoffLine, cutoffDir, p);
560  if (det3 <= 0) {
561  weight = 0;
562  return Vector2.zero;
563  } else {
564  // Signed distances to the two edges along the sides of the VO
565  float det1 = SignedDistanceFromLine(line1, dir1, p);
566  float det2 = SignedDistanceFromLine(line2, dir2, p);
567  if (det1 >= 0 && det2 >= 0) {
568  // We are inside both of the half planes
569  // (all three if we count the cutoff line)
570  // and thus inside the forbidden region in velocity space
571 
572  // Actually the negative gradient because we want the
573  // direction where it slopes the most downwards, not upwards
574  Vector2 gradient;
575 
576  // Check if we are in the semicircle region near the cap of the VO
577  if (Vector2.Dot(p - line1, dir1) > 0 && Vector2.Dot(p - line2, dir2) < 0) {
578  if (segment) {
579  // This part will only be reached for line obstacles (i.e not other agents)
580  if (det3 < radius) {
581  Vector3 closestPointOnLine = VectorMath.ClosestPointOnSegment(segmentStart.ToPFV2(), segmentEnd.ToPFV2(), p.ToPFV2());
582  var dirFromCenter = p.ToPFV2() - closestPointOnLine.ToV2();
583  float distToCenter;
584  gradient = VectorMath.Normalize(dirFromCenter, out distToCenter);
585  // The weight is the distance to the edge
586  weight = radius - distToCenter;
587  return gradient;
588  }
589  } else {
590  var dirFromCenter = p - circleCenter;
591  float distToCenter;
592  gradient = VectorMath.Normalize(dirFromCenter, out distToCenter);
593  // The weight is the distance to the edge
594  weight = radius - distToCenter;
595  return gradient;
596  }
597  }
598 
599  if (segment && det3 < det1 && det3 < det2) {
600  weight = det3;
601  gradient = new Vector2(-cutoffDir.y, cutoffDir.x);
602  return gradient;
603  }
604 
605  // Just move towards the closest edge
606  // The weight is the distance to the edge
607  if (det1 < det2) {
608  weight = det1;
609  gradient = new Vector2(-dir1.y, dir1.x);
610  } else {
611  weight = det2;
612  gradient = new Vector2(-dir2.y, dir2.x);
613  }
614 
615  return gradient;
616  }
617 
618  weight = 0;
619  return Vector2.zero;
620  }
621  }

655  }
656 
657  if (locked) {
658  calculatedSpeed = 0;

674  // No need to run optimizer because this will be the global minima.
675  // This is also a special case in which we can set the

677  // instead of calculating a point based on a calculated velocity
678  // which is an important difference when the agent is very close
679  // to the target point
680  // TODO: Not actually guaranteed to be global minima if desiredTargetPointInVelocitySpace.magnitude < desiredSpeed

707  void GenerateObstacleVOs (VOBuffer vos) {
708  var range = maxSpeed * obstacleTimeHorizon;
709 
710  // Iterate through all obstacles that we might need to avoid
711  for (int i = 0; i < simulator.obstacles.Count; i++) {
712  var obstacle = simulator.obstacles[i];
713  var vertex = obstacle;
714  // Iterate through all edges (defined by vertex and vertex.dir) in the obstacle
715  do {
716  // Ignore the edge if the agent should not collide with it
717  if (vertex.ignore || (vertex.layer & collidesWith) == 0) {
718  vertex = vertex.next;
719  continue;
720  }
721 
722  // Start and end points of the current segment
723  float elevation1, elevation2;
724  var p1 = To2D(vertex.position, out elevation1);
725  var p2 = To2D(vertex.next.position, out elevation2);
726 
727  Vector2 dir = (p2 - p1).normalized;
728 
729  // Signed distance from the line (not segment, lines are infinite)
730  // TODO: Can be optimized
731  float dist = VO.SignedDistanceFromLine(p1, dir, position);
732 
733  if (dist >= -0.01f && dist < range) {
734  float factorAlongSegment = Vector2.Dot(position - p1, p2 - p1) / (p2 - p1).sqrMagnitude;
735 
736  // Calculate the elevation (y) coordinate of the point on the segment closest to the agent
737  var segmentY = Mathf.Lerp(elevation1, elevation2, factorAlongSegment);
738 
739  // Calculate distance from the segment (not line)
740  var sqrDistToSegment = (Vector2.Lerp(p1, p2, factorAlongSegment) - position).sqrMagnitude;
741 
742  // Ignore the segment if it is too far away
743  // or the agent is too high up (or too far down) on the elevation axis (usually y axis) to avoid it.
744  // If the XY plane is used then all elevation checks are disabled
745  if (sqrDistToSegment < range*range && (simulator.movementPlane == MovementPlane.XY || (elevationCoordinate <= segmentY + vertex.height && elevationCoordinate+height >= segmentY))) {
746  vos.Add(VO.SegmentObstacle(p2 - position, p1 - position, Vector2.zero, radius * 0.01f, 1f / ObstacleTimeHorizon, 1f / simulator.DeltaTime));
747  }
748  }
749 
750  vertex = vertex.next;
751  } while (vertex != obstacle && vertex != null && vertex.next != null);
752  }

755  void GenerateNeighbourAgentVOs (VOBuffer vos) {
756  float inverseAgentTimeHorizon = 1.0f/agentTimeHorizon;
757 
758  // The RVO algorithm assumes we will continue to
759  // move in roughly the same direction
760  Vector2 optimalVelocity = currentVelocity;
761 
762  for (int o = 0; o < neighbours.Count; o++) {
763  Agent other = neighbours[o];
764 
765  // Don't avoid ourselves
766  if (other == this)
767  continue;
768 
769  // Interval along the y axis in which the agents overlap
770  float maxY = System.Math.Min(elevationCoordinate + height, other.elevationCoordinate + other.height);
771  float minY = System.Math.Max(elevationCoordinate, other.elevationCoordinate);
772 
773  // The agents cannot collide since they are on different y-levels
774  if (maxY - minY < 0) {
775  continue;
776  }
777 
778  float totalRadius = radius + other.radius;
779 
780  // Describes a circle on the border of the VO
781  Vector2 voBoundingOrigin = other.position - position;
782 
783  float avoidanceStrength;
784  if (other.locked || other.manuallyControlled) {
785  avoidanceStrength = 1;
786  } else if (other.Priority > 0.00001f || Priority > 0.00001f) {
787  avoidanceStrength = other.Priority / (Priority + other.Priority);
788  } else {
789  // Both this agent's priority and the other agent's priority is zero or negative
790  // Assume they have the same priority
791  avoidanceStrength = 0.5f;
792  }
793 
794  // We assume that the other agent will continue to move with roughly the same velocity if the priorities for the agents are similar.
795  // If the other agent has a higher priority than this agent (avoidanceStrength > 0.5) then we will assume it will move more along its
796  // desired velocity. This will have the effect of other agents trying to clear a path for where a high priority agent wants to go.
797  // If this is not done then even high priority agents can get stuck when it is really crowded and they have had to slow down.
798  Vector2 otherOptimalVelocity = Vector2.Lerp(other.currentVelocity, other.desiredVelocity, 2*avoidanceStrength - 1);
799 
800  var voCenter = Vector2.Lerp(optimalVelocity, otherOptimalVelocity, avoidanceStrength);
801 
802  vos.Add(new VO(voBoundingOrigin, voCenter, totalRadius, inverseAgentTimeHorizon, 1 / simulator.DeltaTime));
803  if (DebugDraw)
804  DrawVO(position + voBoundingOrigin * inverseAgentTimeHorizon + voCenter, totalRadius * inverseAgentTimeHorizon, position + voCenter);
805  }

844  static bool BiasDesiredVelocity (VOBuffer vos, ref Vector2 desiredVelocity, ref Vector2 targetPointInVelocitySpace, float maxBiasRadians) {
845  var desiredVelocityMagn = desiredVelocity.magnitude;
846  var maxValue = 0f;
847 
848  for (int i = 0; i < vos.length; i++) {
849  float value;
850  // The value is approximately the distance to the edge of the VO
851  // so taking the maximum will give us the distance to the edge of the VO
852  // which the desired velocity is furthest inside
853  vos.buffer[i].Gradient(desiredVelocity, out value);
854  maxValue = Mathf.Max(maxValue, value);
855  }
856 
857  // Check if the agent was inside any VO
858  var inside = maxValue > 0;
859 
860  // Avoid division by zero below
861  if (desiredVelocityMagn < 0.001f) {
862  return inside;
863  }
864 
865  // Rotate the desired velocity clockwise (to the right) at most maxBiasRadians number of radians
866  // Assuming maxBiasRadians is small, we can just move it instead and it will give approximately the same effect
867  // See https://en.wikipedia.org/wiki/Small-angle_approximation
868  var angle = Mathf.Min(maxBiasRadians, maxValue / desiredVelocityMagn);
869  desiredVelocity += new Vector2(desiredVelocity.y, -desiredVelocity.x) * angle;
870  targetPointInVelocitySpace += new Vector2(targetPointInVelocitySpace.y, -targetPointInVelocitySpace.x) * angle;
871  return inside;
872  }

875  Vector2 EvaluateGradient (VOBuffer vos, Vector2 p, out float value) {
876  Vector2 gradient = Vector2.zero;
877 
878  value = 0;
879 
880  // Avoid other agents
881  for (int i = 0; i < vos.length; i++) {
882  float w;
883  var grad = vos.buffer[i].ScaledGradient(p, out w);
884  if (w > value) {
885  value = w;
886  gradient = grad;
887  }
888  }
889 
890  // Move closer to the desired velocity
891  var dirToDesiredVelocity = desiredVelocity - p;
892  var distToDesiredVelocity = dirToDesiredVelocity.magnitude;
893  if (distToDesiredVelocity > 0.0001f) {
894  gradient += dirToDesiredVelocity * (DesiredVelocityWeight/distToDesiredVelocity);
895  value += distToDesiredVelocity * DesiredVelocityWeight;
896  }
897 
898  // Prefer speeds lower or equal to the desired speed
899  // and avoid speeds greater than the max speed
900  var sqrSpeed = p.sqrMagnitude;
901  if (sqrSpeed > desiredSpeed*desiredSpeed) {
902  var speed = Mathf.Sqrt(sqrSpeed);
903 
904  if (speed > maxSpeed) {
905  const float MaxSpeedWeight = 3;
906  value += MaxSpeedWeight * (speed - maxSpeed);
907  gradient -= MaxSpeedWeight * (p/speed);
908  }
909 
910  // Scale needs to be strictly greater than DesiredVelocityWeight
911  // otherwise the agent will not prefer the desired speed over
912  // the maximum speed
913  float scale = 2*DesiredVelocityWeight;
914  value += scale * (speed - desiredSpeed);
915  gradient -= scale * (p/speed);
916  }
917 
918  return gradient;
919  }

926  Vector2 Trace (VOBuffer vos, Vector2 p, out float score) {

928  float stepSize = Mathf.Max(radius, 0.2f * desiredSpeed);

938  float step = 1.0f - (s/(float)MaxIterations);

940

943
