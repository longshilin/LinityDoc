698  public void Update () {
699  // Initialize last step
700  if (lastStep < 0) {
701  lastStep = Time.time;
702  deltaTime = DesiredDeltaTime;
703  }
704 
705  if (Time.time - lastStep >= DesiredDeltaTime) {
706  deltaTime = Time.time - lastStep;
707  lastStep = Time.time;
708 
709  // Prevent a zero delta time
710  deltaTime = System.Math.Max(deltaTime, 1.0f/2000f);
711 
712  if (Multithreading) {
713  // Make sure the threads have completed their tasks
714  // Otherwise block until they have
715  if (doubleBuffering) {
716  for (int i = 0; i < workers.Length; i++) workers[i].WaitOne();
717  for (int i = 0; i < agents.Count; i++) agents[i].PostCalculation();
718  }
719 
720  PreCalculation();
721  CleanAndUpdateObstaclesIfNecessary();
722  BuildQuadtree();
723 
724  for (int i = 0; i < workers.Length; i++) {
725  workers[i].start = i*agents.Count / workers.Length;
726  workers[i].end = (i+1)*agents.Count / workers.Length;
727  }
728 
729  // BufferSwitch
730  for (int i = 0; i < workers.Length; i++) workers[i].Execute(1);
731  for (int i = 0; i < workers.Length; i++) workers[i].WaitOne();
732 
733  // Calculate New Velocity
734  for (int i = 0; i < workers.Length; i++) workers[i].Execute(0);
735 
736  // Make sure the threads have completed their tasks
737  // Otherwise block until they have
738  if (!doubleBuffering) {
739  for (int i = 0; i < workers.Length; i++) workers[i].WaitOne();
740  for (int i = 0; i < agents.Count; i++) agents[i].PostCalculation();
741  }
742  } else {
743  PreCalculation();
744  CleanAndUpdateObstaclesIfNecessary();
745  BuildQuadtree();
746 
747  for (int i = 0; i < agents.Count; i++) {
748  agents[i].BufferSwitch();
749  }
750 
751  for (int i = 0; i < agents.Count; i++) {
752  agents[i].CalculateNeighbours();
753  agents[i].CalculateVelocity(coroutineWorkerContext);
754  }
755 
756  for (int i = 0; i < agents.Count; i++) agents[i].PostCalculation();
757  }
