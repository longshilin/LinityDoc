178  public void QueryRec (int i, Rect r) {
179  // Determine the radius that we need to search to take all agents into account
180  // Note: the second agentRadius usage should actually be the radius of the other agents, not this agent
181  // but for performance reasons and for simplicity we assume that agents have approximately the same radius.
182  // Thus an agent with a very small radius may in some cases detect an agent with a very large radius too late
183  // however this effect should be minor.
184  var radius = System.Math.Min(System.Math.Max((nodes[i].maxSpeed + speed)*timeHorizon, agentRadius) + agentRadius, maxRadius);
185 
186  if (nodes[i].child00 == i) {
187  // Leaf node
188  for (Agent a = nodes[i].linkedList; a != null; a = a.next) {
189  float v = agent.InsertAgentNeighbour(a, radius*radius);
190  // Limit the search if the agent has hit the max number of nearby agents threshold
191  if (v < maxRadius*maxRadius) {
192  maxRadius = Mathf.Sqrt(v);
193  }
194  }
195  } else {
196  // Not a leaf node
197  Vector2 c = r.center;
198  if (p.x-radius < c.x) {
199  if (p.y-radius < c.y) {
200  QueryRec(nodes[i].child00, Rect.MinMaxRect(r.xMin, r.yMin, c.x, c.y));
201  radius = System.Math.Min(radius, maxRadius);
202  }
203  if (p.y+radius > c.y) {
204  QueryRec(nodes[i].child01, Rect.MinMaxRect(r.xMin, c.y, c.x, r.yMax));
205  radius = System.Math.Min(radius, maxRadius);
206  }
207  }
208 
209  if (p.x+radius > c.x) {
210  if (p.y-radius < c.y) {
211  QueryRec(nodes[i].child10, Rect.MinMaxRect(c.x, r.yMin, r.xMax, c.y));
212  radius = System.Math.Min(radius, maxRadius);
213  }
214  if (p.y+radius > c.y) {
215  QueryRec(nodes[i].child11, Rect.MinMaxRect(c.x, c.y, r.xMax, r.yMax));
216  }
217  }
218  }
219  }
