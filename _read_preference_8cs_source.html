102  public static ReadPreference FromBsonDocument(BsonDocument document)
103  {
104  ReadPreferenceMode mode = ReadPreferenceMode.Primary;
105 
106  foreach (var element in document)
107  {
108  switch (element.Name)
109  {
110  case "mode":
111  mode = (ReadPreferenceMode)Enum.Parse(typeof(ReadPreferenceMode), element.Value.AsString, ignoreCase: true);
112  break;
113 
114  default:
115  throw new ArgumentException($"Invalid element in ReadConcern document: {element.Name}.");
116  }
117  }
118 
119  return new ReadPreference(mode);
120  }

135  public ReadPreference(
136  ReadPreferenceMode mode,

138  TimeSpan? maxStaleness = null)
139  {
140  var tagSetsArray = tagSets == null ? __emptyTagSetsArray : tagSets.ToArray();
141  if (tagSetsArray.Length > 0)
142  {
143  Ensure.That(mode != ReadPreferenceMode.Primary, "TagSets cannot be used with ReadPreferenceMode Primary.", nameof(tagSets));
144  }
145 
146  if (maxStaleness.HasValue)
147  {
148  Ensure.IsInfiniteOrGreaterThanZero(maxStaleness.Value, nameof(maxStaleness));
149  if (maxStaleness.Value > TimeSpan.Zero)
150  {
151  Ensure.That(maxStaleness.Value.Ticks % TimeSpan.TicksPerMillisecond == 0, "MaxStaleness must not have fractional seconds.", nameof(maxStaleness));
152  }
153  Ensure.That(mode != ReadPreferenceMode.Primary, "MaxStaleness cannot be used with ReadPreferenceMode Primary.", nameof(maxStaleness));
154  }
155 
156  _mode = mode;
157  _tagSets = tagSetsArray;
158  _maxStaleness = maxStaleness;
