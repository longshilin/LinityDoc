197  public static void ExportToFile (RecastGraph target) {
198  //INavmesh graph = (INavmesh)target;
199  if (target == null) return;
200 
201  NavmeshTile[] tiles = target.GetTiles();
202 
203  if (tiles == null) {
204  if (EditorUtility.DisplayDialog("Scan graph before exporting?", "The graph does not contain any mesh data. Do you want to scan it?", "Ok", "Cancel")) {
205  AstarPathEditor.MenuScan();
206  tiles = target.GetTiles();
207  if (tiles == null) return;
208  } else {
209  return;
210  }
211  }
212 
213  string path = EditorUtility.SaveFilePanel("Export .obj", "", "navmesh.obj", "obj");
214  if (path == "") return;
215 
216  //Generate .obj
217  var sb = new System.Text.StringBuilder();
218 
219  string name = System.IO.Path.GetFileNameWithoutExtension(path);
220 
221  sb.Append("g ").Append(name).AppendLine();
222 
223  //Vertices start from 1
224  int vCount = 1;
225 
226  //Define single texture coordinate to zero
227  sb.Append("vt 0 0\n");
228 
229  for (int t = 0; t < tiles.Length; t++) {
230  NavmeshTile tile = tiles[t];
231 
232  if (tile == null) continue;
233 
234  Int3[] vertices = tile.verts;
235 
236  //Write vertices
237  for (int i = 0; i < vertices.Length; i++) {
238  var v = (Vector3)vertices[i];
239  sb.Append(string.Format("v {0} {1} {2}\n", -v.x, v.y, v.z));
240  }
241 
242  //Write triangles
243  TriangleMeshNode[] nodes = tile.nodes;
244  for (int i = 0; i < nodes.Length; i++) {
245  TriangleMeshNode node = nodes[i];
246  if (node == null) {
247  Debug.LogError("Node was null or no TriangleMeshNode. Critical error. Graph type " + target.GetType().Name);
248  return;
249  }
250  if (node.GetVertexArrayIndex(0) < 0 || node.GetVertexArrayIndex(0) >= vertices.Length) throw new System.Exception("ERR");
251 
252  sb.Append(string.Format("f {0}/1 {1}/1 {2}/1\n", (node.GetVertexArrayIndex(0) + vCount), (node.GetVertexArrayIndex(1) + vCount), (node.GetVertexArrayIndex(2) + vCount)));
253  }
254 
255  vCount += vertices.Length;
256  }
257 
258  string obj = sb.ToString();
259 
260  using (var sw = new System.IO.StreamWriter(path))
261  {
262  sw.Write(obj);
263  }
264  }
