196  void GenerateTerrainChunks (Terrain terrain, Bounds bounds, float desiredChunkSize, List<RasterizationMesh> result) {
197  var terrainData = terrain.terrainData;
198 
199  if (terrainData == null)
200  throw new System.ArgumentException("Terrain contains no terrain data");
201 
202  Vector3 offset = terrain.GetPosition();
203  Vector3 center = offset + terrainData.size * 0.5F;
204 
205  // Figure out the bounds of the terrain in world space
206  var terrainBounds = new Bounds(center, terrainData.size);
207 
208  // Only include terrains which intersects the graph
209  if (!terrainBounds.Intersects(bounds))
210  return;
211 
212  // Original heightmap size
213  int heightmapWidth = terrainData.heightmapResolution;
214  int heightmapDepth = terrainData.heightmapResolution;
215 
216  // Sample the terrain heightmap
217  float[, ] heights = terrainData.GetHeights(0, 0, heightmapWidth, heightmapDepth);
218 
219  Vector3 sampleSize = terrainData.heightmapScale;
220  sampleSize.y = terrainData.size.y;
221 
222  // Make chunks at least 12 quads wide
223  // since too small chunks just decreases performance due
224  // to the overhead of checking for bounds and similar things
225  const int MinChunkSize = 12;
226 
227  // Find the number of samples along each edge that corresponds to a world size of desiredChunkSize
228  // Then round up to the nearest multiple of terrainSampleSize
229  var chunkSizeAlongX = Mathf.CeilToInt(Mathf.Max(desiredChunkSize / (sampleSize.x * terrainSampleSize), MinChunkSize)) * terrainSampleSize;
230  var chunkSizeAlongZ = Mathf.CeilToInt(Mathf.Max(desiredChunkSize / (sampleSize.z * terrainSampleSize), MinChunkSize)) * terrainSampleSize;
231 
232  for (int z = 0; z < heightmapDepth; z += chunkSizeAlongZ) {
233  for (int x = 0; x < heightmapWidth; x += chunkSizeAlongX) {
234  var width = Mathf.Min(chunkSizeAlongX, heightmapWidth - x);
235  var depth = Mathf.Min(chunkSizeAlongZ, heightmapDepth - z);
236  var chunkMin = offset + new Vector3(z * sampleSize.x, 0, x * sampleSize.z);
237  var chunkMax = offset + new Vector3((z + depth) * sampleSize.x, sampleSize.y, (x + width) * sampleSize.z);
238  var chunkBounds = new Bounds();
239  chunkBounds.SetMinMax(chunkMin, chunkMax);
240 
241  // Skip chunks that are not inside the desired bounds
242  if (chunkBounds.Intersects(bounds)) {
243  var chunk = GenerateHeightmapChunk(heights, sampleSize, offset, x, z, width, depth, terrainSampleSize);
244  result.Add(chunk);
245  }
246  }
247  }
248  }

256  RasterizationMesh GenerateHeightmapChunk (float[, ] heights, Vector3 sampleSize, Vector3 offset, int x0, int z0, int width, int depth, int stride) {
257  // Downsample to a smaller mesh (full resolution will take a long time to rasterize)
258  // Round up the width to the nearest multiple of terrainSampleSize and then add 1
259  // (off by one because there are vertices at the edge of the mesh)
260  int resultWidth = CeilDivision(width, terrainSampleSize) + 1;
261  int resultDepth = CeilDivision(depth, terrainSampleSize) + 1;
262 
263  var heightmapWidth = heights.GetLength(0);
264  var heightmapDepth = heights.GetLength(1);
265 
266  // Create a mesh from the heightmap
267  var numVerts = resultWidth * resultDepth;
268  var terrainVertices = ArrayPool<Vector3>.Claim(numVerts);
269 
270  // Create lots of vertices
271  for (int z = 0; z < resultDepth; z++) {
272  for (int x = 0; x < resultWidth; x++) {
273  int sampleX = Math.Min(x0 + x*stride, heightmapWidth-1);
274  int sampleZ = Math.Min(z0 + z*stride, heightmapDepth-1);
275 
276  terrainVertices[z*resultWidth + x] = new Vector3(sampleZ * sampleSize.x, heights[sampleX, sampleZ]*sampleSize.y, sampleX * sampleSize.z) + offset;
277  }
278  }
279 
280  // Create the mesh by creating triangles in a grid like pattern
281  int numTris = (resultWidth-1)*(resultDepth-1)*2*3;
282  var tris = ArrayPool<int>.Claim(numTris);
283  int triangleIndex = 0;
284  for (int z = 0; z < resultDepth-1; z++) {
285  for (int x = 0; x < resultWidth-1; x++) {
286  tris[triangleIndex] = z*resultWidth + x;
287  tris[triangleIndex+1] = z*resultWidth + x+1;
288  tris[triangleIndex+2] = (z+1)*resultWidth + x+1;
289  triangleIndex += 3;
290  tris[triangleIndex] = z*resultWidth + x;
291  tris[triangleIndex+1] = (z+1)*resultWidth + x+1;
292  tris[triangleIndex+2] = (z+1)*resultWidth + x;
293  triangleIndex += 3;
294  }
295  }
296 
297 #if ASTARDEBUG
298  var color = AstarMath.IntToColor(x0 + 7 * z0, 0.7f);
299  for (int i = 0; i < numTris; i += 3) {
300  Debug.DrawLine(terrainVertices[tris[i]], terrainVertices[tris[i+1]], color, 40);
301  Debug.DrawLine(terrainVertices[tris[i+1]], terrainVertices[tris[i+2]], color, 40);
302  Debug.DrawLine(terrainVertices[tris[i+2]], terrainVertices[tris[i]], color, 40);
303  }
304 #endif
305 
306  var mesh = new RasterizationMesh(terrainVertices, tris, new Bounds());
307  mesh.numVertices = numVerts;
308  mesh.numTriangles = numTris;
309  mesh.pool = true;
310  // Could probably calculate these bounds in a faster way
311  mesh.RecalculateBounds();
312  return mesh;
313  }

439  RasterizationMesh RasterizeCollider (Collider col, Matrix4x4 localToWorldMatrix) {
440  RasterizationMesh result = null;
441 
442  if (col is BoxCollider) {
443  result = RasterizeBoxCollider(col as BoxCollider, localToWorldMatrix);
444  } else if (col is SphereCollider || col is CapsuleCollider) {
445  var scollider = col as SphereCollider;
446  var ccollider = col as CapsuleCollider;
447 
448  float radius = (scollider != null ? scollider.radius : ccollider.radius);
449  float height = scollider != null ? 0 : (ccollider.height*0.5f/radius) - 1;
450  Quaternion rot = Quaternion.identity;
451  // Capsule colliders can be aligned along the X, Y or Z axis
452  if (ccollider != null) rot = Quaternion.Euler(ccollider.direction == 2 ? 90 : 0, 0, ccollider.direction == 0 ? 90 : 0);
453  Matrix4x4 matrix = Matrix4x4.TRS(scollider != null ? scollider.center : ccollider.center, rot, Vector3.one*radius);
454 
455  matrix = localToWorldMatrix * matrix;
456 
457  result = RasterizeCapsuleCollider(radius, height, col.bounds, matrix);
458  } else if (col is MeshCollider) {
459  var collider = col as MeshCollider;
460 
461  if (collider.sharedMesh != null) {
462  result = new RasterizationMesh(collider.sharedMesh.vertices, collider.sharedMesh.triangles, collider.bounds, localToWorldMatrix);
463  }
464  }
465 
466  #if ASTARDEBUG
467  for (int i = 0; i < result.triangles.Length; i += 3) {
468  Debug.DrawLine(result.matrix.MultiplyPoint3x4(result.vertices[result.triangles[i]]), result.matrix.MultiplyPoint3x4(result.vertices[result.triangles[i+1]]), Color.yellow);
469  Debug.DrawLine(result.matrix.MultiplyPoint3x4(result.vertices[result.triangles[i+2]]), result.matrix.MultiplyPoint3x4(result.vertices[result.triangles[i+1]]), Color.yellow);
470  Debug.DrawLine(result.matrix.MultiplyPoint3x4(result.vertices[result.triangles[i]]), result.matrix.MultiplyPoint3x4(result.vertices[result.triangles[i+2]]), Color.yellow);

485  RasterizationMesh RasterizeCapsuleCollider (float radius, float height, Bounds bounds, Matrix4x4 localToWorldMatrix) {
486  // Calculate the number of rows to use
487  // grows as sqrt(x) to the radius of the sphere/capsule which I have found works quite well
488  int rows = Mathf.Max(4, Mathf.RoundToInt(colliderRasterizeDetail*Mathf.Sqrt(localToWorldMatrix.MultiplyVector(Vector3.one).magnitude)));
489 
490  if (rows > 100) {
491  Debug.LogWarning("Very large detail for some collider meshes. Consider decreasing Collider Rasterize Detail (RecastGraph)");
492  }
493 
494  int cols = rows;
495 
496  Vector3[] verts;
497  int[] trisArr;
498 
499  // Check if we have already calculated a similar capsule
500  CapsuleCache cached = null;
501  for (int i = 0; i < capsuleCache.Count; i++) {
502  CapsuleCache c = capsuleCache[i];
503  if (c.rows == rows && Mathf.Approximately(c.height, height)) {
504  cached = c;
505  }
506  }
507 
508  if (cached == null) {
509  // Generate a sphere/capsule mesh
510 
511  verts = new Vector3[(rows)*cols + 2];
512 
513  var tris = new List<int>();
514  verts[verts.Length-1] = Vector3.up;
515 
516  for (int r = 0; r < rows; r++) {
517  for (int c = 0; c < cols; c++) {
518  verts[c + r*cols] = new Vector3(Mathf.Cos(c*Mathf.PI*2/cols)*Mathf.Sin((r*Mathf.PI/(rows-1))), Mathf.Cos((r*Mathf.PI/(rows-1))) + (r < rows/2 ? height : -height), Mathf.Sin(c*Mathf.PI*2/cols)*Mathf.Sin((r*Mathf.PI/(rows-1))));
519  }
520  }
521 
522  verts[verts.Length-2] = Vector3.down;
523 
524  for (int i = 0, j = cols-1; i < cols; j = i++) {
525  tris.Add(verts.Length-1);
526  tris.Add(0*cols + j);
527  tris.Add(0*cols + i);
528  }
529 
530  for (int r = 1; r < rows; r++) {
531  for (int i = 0, j = cols-1; i < cols; j = i++) {
532  tris.Add(r*cols + i);
533  tris.Add(r*cols + j);
534  tris.Add((r-1)*cols + i);
535 
536  tris.Add((r-1)*cols + j);
537  tris.Add((r-1)*cols + i);
538  tris.Add(r*cols + j);
539  }
540  }
541 
542  for (int i = 0, j = cols-1; i < cols; j = i++) {
543  tris.Add(verts.Length-2);
544  tris.Add((rows-1)*cols + j);
545  tris.Add((rows-1)*cols + i);
546  }
547 
548  // Add calculated mesh to the cache
549  cached = new CapsuleCache();
550  cached.rows = rows;
551  cached.height = height;
552  cached.verts = verts;
553  cached.tris = tris.ToArray();
554  capsuleCache.Add(cached);
555  }
556 
557  // Read from cache
558  verts = cached.verts;
559  trisArr = cached.tris;
560 
561  return new RasterizationMesh(verts, trisArr, bounds, localToWorldMatrix);
562  }
563  }
