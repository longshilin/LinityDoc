227  [SuppressMessage("Microsoft.Usage", "CA1816:CallGCSuppressFinalizeCorrectly",
228  Justification = "We have different disposal semantics, so SuppressFinalize is in a different spot.")]
229  protected override void Dispose(bool disposing)
230  {
231  if (Interlocked.CompareExchange(ref this.disposedState, 1, 0) != 0)
232  {
233  string doubleDisposeStack = null;
234  if (this.memoryManager.GenerateCallStacks)
235  {
236  doubleDisposeStack = Environment.StackTrace;
237  }
238 
239  RecyclableMemoryStreamManager.Events.Writer.MemoryStreamDoubleDispose(this.id, this.tag,
240  this.AllocationStack,
241  this.DisposeStack,
242  doubleDisposeStack);
243  return;
244  }
245 
246  RecyclableMemoryStreamManager.Events.Writer.MemoryStreamDisposed(this.id, this.tag);
247 
248  if (this.memoryManager.GenerateCallStacks)
249  {
250  this.DisposeStack = Environment.StackTrace;
251  }
252 
253  if (disposing)
254  {
255  this.memoryManager.ReportStreamDisposed();
256 
257  GC.SuppressFinalize(this);
258  }
259  else
260  {
261  // We're being finalized.
262 
263  RecyclableMemoryStreamManager.Events.Writer.MemoryStreamFinalized(this.id, this.tag, this.AllocationStack);
264 
265 #if !NETSTANDARD1_4
266  if (AppDomain.CurrentDomain.IsFinalizingForUnload())
267  {
268  // If we're being finalized because of a shutdown, don't go any further.
269  // We have no idea what's already been cleaned up. Triggering events may cause
270  // a crash.
271  base.Dispose(disposing);
272  return;
273  }
274 #endif
275 
276  this.memoryManager.ReportStreamFinalized();
277  }
278 
279  this.memoryManager.ReportStreamLength(this.length);
280 
281  if (this.largeBuffer != null)
282  {
283  this.memoryManager.ReturnLargeBuffer(this.largeBuffer, this.tag);
284  }
285 
286  if (this.dirtyBuffers != null)
287  {
288  foreach (var buffer in this.dirtyBuffers)
289  {
290  this.memoryManager.ReturnLargeBuffer(buffer, this.tag);
291  }
292  }
293 
294  this.memoryManager.ReturnBlocks(this.blocks, this.tag);
295  this.blocks.Clear();
296 
297  base.Dispose(disposing);
298  }

426  {

428  }

430  if (this.blocks.Count == 1)
431  {

433  }

436  // it's possible that people will manipulate the buffer directly
437  // and set the length afterward. Capacity sets the expectation
438  // for the size of the buffer.
439  var newBuffer = this.memoryManager.GetLargeBuffer(this.Capacity, this.tag);

443  this.InternalRead(newBuffer, 0, this.length, 0);

449  this.blocks.Clear();

543  public override void Write(byte[] buffer, int offset, int count)
544  {
545  this.CheckDisposed();
546  if (buffer == null)
547  {
548  throw new ArgumentNullException(nameof(buffer));
549  }
550 
551  if (offset < 0)
552  {
553  throw new ArgumentOutOfRangeException(nameof(offset), offset,
554  "Offset must be in the range of 0 - buffer.Length-1");
555  }
556 
557  if (count < 0)
558  {
559  throw new ArgumentOutOfRangeException(nameof(count), count, "count must be non-negative");
560  }
561 
562  if (count + offset > buffer.Length)
563  {
564  throw new ArgumentException("count must be greater than buffer.Length - offset");
565  }
566 
567  int blockSize = this.memoryManager.BlockSize;
568  long end = (long)this.position + count;
569  // Check for overflow
570  if (end > MaxStreamLength)
571  {
572  throw new IOException("Maximum capacity exceeded");
573  }
574 
575  long requiredBuffers = (end + blockSize - 1) / blockSize;
576 
577  if (requiredBuffers * blockSize > MaxStreamLength)
578  {
579  throw new IOException("Maximum capacity exceeded");
580  }
581 
582  this.EnsureCapacity((int)end);
583 
584  if (this.largeBuffer == null)
585  {
586  int bytesRemaining = count;
587  int bytesWritten = 0;
588  var blockAndOffset = this.GetBlockAndRelativeOffset(this.position);
589 
590  while (bytesRemaining > 0)
591  {
592  byte[] currentBlock = this.blocks[blockAndOffset.Block];
593  int remainingInBlock = blockSize - blockAndOffset.Offset;
594  int amountToWriteInBlock = Math.Min(remainingInBlock, bytesRemaining);
595 
596  Buffer.BlockCopy(buffer, offset + bytesWritten, currentBlock, blockAndOffset.Offset,
597  amountToWriteInBlock);
598 
599  bytesRemaining -= amountToWriteInBlock;
600  bytesWritten += amountToWriteInBlock;
601 
602  ++blockAndOffset.Block;
603  blockAndOffset.Offset = 0;
604  }
605  }
606  else
607  {
608  Buffer.BlockCopy(buffer, offset, this.largeBuffer, this.position, count);
609  }
610  this.position = (int)end;
611  this.length = Math.Max(this.position, this.length);
612  }

781  private int InternalRead(byte[] buffer, int offset, int count, int fromPosition)
782  {
783  if (this.length - fromPosition <= 0)
784  {
785  return 0;
786  }
787 
788  int amountToCopy;
789 
790  if (this.largeBuffer == null)
791  {
792  var blockAndOffset = this.GetBlockAndRelativeOffset(fromPosition);
793  int bytesWritten = 0;
794  int bytesRemaining = Math.Min(count, this.length - fromPosition);
795 
796  while (bytesRemaining > 0)
797  {
798  amountToCopy = Math.Min(this.blocks[blockAndOffset.Block].Length - blockAndOffset.Offset,
799  bytesRemaining);
800  Buffer.BlockCopy(this.blocks[blockAndOffset.Block], blockAndOffset.Offset, buffer,
801  bytesWritten + offset, amountToCopy);
802 
803  bytesWritten += amountToCopy;
804  bytesRemaining -= amountToCopy;
805 
806  ++blockAndOffset.Block;
807  blockAndOffset.Offset = 0;
808  }
809  return bytesWritten;
810  }
811  amountToCopy = Math.Min(count, this.length - fromPosition);
812  Buffer.BlockCopy(this.largeBuffer, fromPosition, buffer, offset, amountToCopy);
813  return amountToCopy;
814  }

834  private void EnsureCapacity(int newCapacity)
835  {
836  if (newCapacity > this.memoryManager.MaximumStreamCapacity && this.memoryManager.MaximumStreamCapacity > 0)
837  {
838  RecyclableMemoryStreamManager.Events.Writer.MemoryStreamOverCapacity(newCapacity,
839  this.memoryManager
840  .MaximumStreamCapacity, this.tag,
841  this.AllocationStack);
842  throw new InvalidOperationException("Requested capacity is too large: " + newCapacity + ". Limit is " +
843  this.memoryManager.MaximumStreamCapacity);
844  }
845 
846  if (this.largeBuffer != null)
847  {
848  if (newCapacity > this.largeBuffer.Length)
849  {
850  var newBuffer = this.memoryManager.GetLargeBuffer(newCapacity, this.tag);
851  this.InternalRead(newBuffer, 0, this.length, 0);
852  this.ReleaseLargeBuffer();
853  this.largeBuffer = newBuffer;
854  }
855  }
856  else
857  {
858  while (this.Capacity < newCapacity)
859  {
860  blocks.Add((this.memoryManager.GetBlock()));
861  }
862  }
863  }

869  {

871  {
872  this.memoryManager.ReturnLargeBuffer(this.largeBuffer, this.tag);
873  }

879  this.dirtyBuffers = new List<byte[]>(1);
880  }
881  this.dirtyBuffers.Add(this.largeBuffer);
