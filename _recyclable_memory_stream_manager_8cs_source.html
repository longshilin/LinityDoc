110  public RecyclableMemoryStreamManager(int blockSize, int largeBufferMultiple, int maximumBufferSize)
111  {
112  if (blockSize <= 0)
113  {
114  throw new ArgumentOutOfRangeException(nameof(blockSize), blockSize, "blockSize must be a positive number");
115  }
116 
117  if (largeBufferMultiple <= 0)
118  {
119  throw new ArgumentOutOfRangeException(nameof(largeBufferMultiple),
120  "largeBufferMultiple must be a positive number");
121  }
122 
123  if (maximumBufferSize < blockSize)
124  {
125  throw new ArgumentOutOfRangeException(nameof(maximumBufferSize),
126  "maximumBufferSize must be at least blockSize");
127  }

129  this.blockSize = blockSize;
130  this.largeBufferMultiple = largeBufferMultiple;
131  this.maximumBufferSize = maximumBufferSize;
132 
133  if (!this.IsLargeBufferMultiple(maximumBufferSize))
134  {
135  throw new ArgumentException("maximumBufferSize is not a multiple of largeBufferMultiple",
136  nameof(maximumBufferSize));
137  }
138 
139  this.smallPool = new ConcurrentStack<byte[]>();
140  var numLargePools = maximumBufferSize / largeBufferMultiple;
141 
142  // +1 to store size of bytes in use that are too large to be pooled
143  this.largeBufferInUseSize = new long[numLargePools + 1];
144  this.largeBufferFreeSize = new long[numLargePools];
145 
146  this.largePools = new ConcurrentStack<byte[]>[numLargePools];
147 
148  for (var i = 0; i < this.largePools.Length; ++i)
149  {

151  }

253  internal byte[] GetBlock()
254  {
255  byte[] block;
256  if (!this.smallPool.TryPop(out block))
257  {
258  // We'll add this back to the pool when the stream is disposed
259  // (unless our free pool is too large)
260  block = new byte[this.BlockSize];
261  Events.Writer.MemoryStreamNewBlockCreated(this.smallPoolInUseSize);
262  ReportBlockCreated();
263  }
264  else
265  {
266  Interlocked.Add(ref this.smallPoolFreeSize, -this.BlockSize);
267  }
268 
269  Interlocked.Add(ref this.smallPoolInUseSize, this.BlockSize);
270  return block;
271  }

280  internal byte[] GetLargeBuffer(int requiredSize, string tag)
281  {
282  requiredSize = this.RoundToLargeBufferMultiple(requiredSize);
283 
284  var poolIndex = requiredSize / this.largeBufferMultiple - 1;
285 
286  byte[] buffer;
287  if (poolIndex < this.largePools.Length)
288  {
289  if (!this.largePools[poolIndex].TryPop(out buffer))
290  {
291  buffer = new byte[requiredSize];
292 
293  Events.Writer.MemoryStreamNewLargeBufferCreated(requiredSize, this.LargePoolInUseSize);
294  ReportLargeBufferCreated();
295  }
296  else
297  {
298  Interlocked.Add(ref this.largeBufferFreeSize[poolIndex], -buffer.Length);
299  }
300  }
301  else
302  {
303  // Buffer is too large to pool. They get a new buffer.
304 
305  // We still want to track the size, though, and we've reserved a slot
306  // in the end of the inuse array for nonpooled bytes in use.
307  poolIndex = this.largeBufferInUseSize.Length - 1;
308 
309  // We still want to round up to reduce heap fragmentation.
310  buffer = new byte[requiredSize];
311  string callStack = null;
312  if (this.GenerateCallStacks)
313  {
314  // Grab the stack -- we want to know who requires such large buffers
315  callStack = Environment.StackTrace;
316  }
317  Events.Writer.MemoryStreamNonPooledLargeBufferCreated(requiredSize, tag, callStack);
318  ReportLargeBufferCreated();
319  }
320 
321  Interlocked.Add(ref this.largeBufferInUseSize[poolIndex], buffer.Length);
322 
323  return buffer;
324  }

343  internal void ReturnLargeBuffer(byte[] buffer, string tag)
344  {
345  if (buffer == null)
346  {
347  throw new ArgumentNullException(nameof(buffer));
348  }
349 
350  if (!this.IsLargeBufferMultiple(buffer.Length))
351  {
352  throw new ArgumentException(
353  "buffer did not originate from this memory manager. The size is not a multiple of " +
354  this.LargeBufferMultiple);
355  }
356 
357  var poolIndex = buffer.Length / this.largeBufferMultiple - 1;
358 
359  if (poolIndex < this.largePools.Length)
360  {
361  if ((this.largePools[poolIndex].Count + 1) * buffer.Length <= this.MaximumFreeLargePoolBytes ||
362  this.MaximumFreeLargePoolBytes == 0)
363  {
364  this.largePools[poolIndex].Push(buffer);
365  Interlocked.Add(ref this.largeBufferFreeSize[poolIndex], buffer.Length);
366  }
367  else
368  {
369  Events.Writer.MemoryStreamDiscardBuffer(Events.MemoryStreamBufferType.Large, tag,
370  Events.MemoryStreamDiscardReason.EnoughFree);
371  ReportLargeBufferDiscarded(Events.MemoryStreamDiscardReason.EnoughFree);
372  }
373  }
374  else
375  {
376  // This is a non-poolable buffer, but we still want to track its size for inuse
377  // analysis. We have space in the inuse array for this.
378  poolIndex = this.largeBufferInUseSize.Length - 1;
379 
380  Events.Writer.MemoryStreamDiscardBuffer(Events.MemoryStreamBufferType.Large, tag,
381  Events.MemoryStreamDiscardReason.TooLarge);
382  ReportLargeBufferDiscarded(Events.MemoryStreamDiscardReason.TooLarge);
383  }
384 
385  Interlocked.Add(ref this.largeBufferInUseSize[poolIndex], -buffer.Length);
386 
387  ReportUsageReport(this.smallPoolInUseSize, this.smallPoolFreeSize, this.LargePoolInUseSize,
388  this.LargePoolFreeSize);
389  }

398  internal void ReturnBlocks(ICollection<byte[]> blocks, string tag)
399  {
400  if (blocks == null)
401  {
402  throw new ArgumentNullException(nameof(blocks));
403  }
404 
405  var bytesToReturn = blocks.Count * this.BlockSize;
406  Interlocked.Add(ref this.smallPoolInUseSize, -bytesToReturn);
407 
408  foreach (var block in blocks)
409  {
410  if (block == null || block.Length != this.BlockSize)
411  {
412  throw new ArgumentException("blocks contains buffers that are not BlockSize in length");
413  }
414  }
415 
416  foreach (var block in blocks)
417  {
418  if (this.MaximumFreeSmallPoolBytes == 0 || this.SmallPoolFreeSize < this.MaximumFreeSmallPoolBytes)
419  {
420  Interlocked.Add(ref this.smallPoolFreeSize, this.BlockSize);
421  this.smallPool.Push(block);
422  }
423  else
424  {
425  Events.Writer.MemoryStreamDiscardBuffer(Events.MemoryStreamBufferType.Small, tag,
426  Events.MemoryStreamDiscardReason.EnoughFree);
427  ReportBlockDiscarded();
428  break;
429  }
430  }
431 
432  ReportUsageReport(this.smallPoolInUseSize, this.smallPoolFreeSize, this.LargePoolInUseSize,
433  this.LargePoolFreeSize);
434  }
