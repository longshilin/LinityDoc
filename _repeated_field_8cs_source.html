62  public void AddEntriesFrom(CodedInputStream input, FieldCodec<T> codec)
63  {
64  // TODO: Inline some of the Add code, so we can avoid checking the size on every
65  // iteration.
66  uint tag = input.LastTag;
67  var reader = codec.ValueReader;
68  // Non-nullable value types can be packed or not.
69  if (FieldCodec<T>.IsPackedRepeatedField(tag))
70  {
71  int length = input.ReadLength();
72  if (length > 0)
73  {
74  int oldLimit = input.PushLimit(length);
75  while (!input.ReachedLimit)
76  {
77  Add(reader(input));
78  }
79  input.PopLimit(oldLimit);
80  }
81  // Empty packed field. Odd, but valid - just ignore.
82  }
83  else
84  {
85  // Not packed... (possibly not packable)
86  do
87  {
88  Add(reader(input));
89  } while (input.MaybeConsumeTag(tag));
90  }
91  }

99  public int CalculateSize(FieldCodec<T> codec)
100  {
101  if (count == 0)
102  {
103  return 0;
104  }
105  uint tag = codec.Tag;
106  if (codec.PackedRepeatedField)
107  {
108  int dataSize = CalculatePackedDataSize(codec);
109  return CodedOutputStream.ComputeRawVarint32Size(tag) +
110  CodedOutputStream.ComputeLengthSize(dataSize) +
111  dataSize;
112  }
113  else
114  {
115  var sizeCalculator = codec.ValueSizeCalculator;
116  int size = count * CodedOutputStream.ComputeRawVarint32Size(tag);
117  for (int i = 0; i < count; i++)
118  {
119  size += sizeCalculator(array[i]);
120  }
121  return size;
122  }
123  }
124 
125  private int CalculatePackedDataSize(FieldCodec<T> codec)
126  {
127  int fixedSize = codec.FixedSize;
128  if (fixedSize == 0)
129  {
130  var calculator = codec.ValueSizeCalculator;
131  int tmp = 0;
132  for (int i = 0; i < count; i++)
133  {
134  tmp += calculator(array[i]);
135  }
136  return tmp;
137  }
138  else
139  {
140  return fixedSize * Count;
141  }
142  }

150  public void WriteTo(CodedOutputStream output, FieldCodec<T> codec)
151  {
152  if (count == 0)
153  {
154  return;
155  }
156  var writer = codec.ValueWriter;
157  var tag = codec.Tag;
158  if (codec.PackedRepeatedField)
159  {
160  // Packed primitive type
161  uint size = (uint)CalculatePackedDataSize(codec);
162  output.WriteTag(tag);
163  output.WriteRawVarint32(size);
164  for (int i = 0; i < count; i++)
165  {
166  writer(output, array[i]);
167  }
168  }
169  else
170  {
171  // Not packed: a simple tag/value pair for each value.
172  // Can't use codec.WriteTagAndValue, as that omits default values.
173  for (int i = 0; i < count; i++)
174  {
175  output.WriteTag(tag);
176  writer(output, array[i]);
177  }
178  }
179  }

181  private void EnsureSize(int size)
182  {
183  if (array.Length < size)
184  {
185  size = Math.Max(size, MinArraySize);
186  int newSize = Math.Max(array.Length * 2, size);
187  var tmp = new T[newSize];
188  Array.Copy(array, 0, tmp, 0, array.Length);
189  array = tmp;
190  }
191  }

278  public void AddRange(IEnumerable<T> values)
279  {
280  ProtoPreconditions.CheckNotNull(values, "values");
281 
282  // Optimization 1: If the collection we're adding is already a RepeatedField<T>,
283  // we know the values are valid.
284  var otherRepeatedField = values as RepeatedField<T>;
285  if (otherRepeatedField != null)
286  {
287  EnsureSize(count + otherRepeatedField.count);
288  Array.Copy(otherRepeatedField.array, 0, array, count, otherRepeatedField.count);
289  count += otherRepeatedField.count;
290  return;
291  }
292 
293  // Optimization 2: The collection is an ICollection, so we can expand
294  // just once and ask the collection to copy itself into the array.
295  var collection = values as ICollection;
296  if (collection != null)
297  {
298  var extraCount = collection.Count;
299  // For reference types and nullable value types, we need to check that there are no nulls
300  // present. (This isn't a thread-safe approach, but we don't advertise this is thread-safe.)
301  // We expect the JITter to optimize this test to true/false, so it's effectively conditional
302  // specialization.
303  if (default(T) == null)
304  {
305  // TODO: Measure whether iterating once to check and then letting the collection copy
306  // itself is faster or slower than iterating and adding as we go. For large
307  // collections this will not be great in terms of cache usage... but the optimized
308  // copy may be significantly faster than doing it one at a time.
309  foreach (var item in collection)
310  {
311  if (item == null)
312  {
313  throw new ArgumentException("Sequence contained null element", "values");
314  }
315  }
316  }
317  EnsureSize(count + extraCount);
318  collection.CopyTo(array, count);
319  count += extraCount;
320  return;
321  }
322 
323  // We *could* check for ICollection<T> as well, but very very few collections implement
324  // ICollection<T> but not ICollection. (HashSet<T> does, for one...)
325 
326  // Fall back to a slower path of adding items one at a time.
327  foreach (T item in values)
328  {
329  Add(item);
330  }
331  }
