57  public ReplyMessage<TDocument> ReadMessage()
58  {
59  var binaryReader = CreateBinaryReader();
60  var stream = binaryReader.BsonStream;
61 
62  stream.ReadInt32(); // messageSize
63  var requestId = stream.ReadInt32();
64  var responseTo = stream.ReadInt32();
65  stream.ReadInt32(); // opcode
66  var flags = (ResponseFlags)stream.ReadInt32();
67  var cursorId = stream.ReadInt64();
68  var startingFrom = stream.ReadInt32();
69  var numberReturned = stream.ReadInt32();
70  List<TDocument> documents = null;
71  BsonDocument queryFailureDocument = null;
72 
73  var awaitCapable = (flags & ResponseFlags.AwaitCapable) == ResponseFlags.AwaitCapable;
74  var cursorNotFound = (flags & ResponseFlags.CursorNotFound) == ResponseFlags.CursorNotFound;
75  var queryFailure = (flags & ResponseFlags.QueryFailure) == ResponseFlags.QueryFailure;
76 
77  if (queryFailure)
78  {
79  var context = BsonDeserializationContext.CreateRoot(binaryReader);
80  queryFailureDocument = BsonDocumentSerializer.Instance.Deserialize(context);
81  }
82  else
83  {
84  documents = new List<TDocument>();
85  for (var i = 0; i < numberReturned; i++)
86  {
87  var allowDuplicateElementNames = typeof(TDocument) == typeof(BsonDocument);
88  var context = BsonDeserializationContext.CreateRoot(binaryReader, builder =>
89  {
90  builder.AllowDuplicateElementNames = allowDuplicateElementNames;
91  });
92  documents.Add(_serializer.Deserialize(context));
93  }
94  }
95 
96  return new ReplyMessage<TDocument>(
97  awaitCapable,
98  cursorId,
99  cursorNotFound,
100  documents,
101  numberReturned,
102  queryFailure,
103  queryFailureDocument,
104  requestId,
105  responseTo,
106  _serializer,
107  startingFrom);
108  }

114  public void WriteMessage(ReplyMessage<TDocument> message)
115  {
116  Ensure.IsNotNull(message, nameof(message));
117 
118  var binaryWriter = CreateBinaryWriter();
119  var stream = binaryWriter.BsonStream;
120  var startPosition = stream.Position;
121 
122  stream.WriteInt32(0); // messageSize
123  stream.WriteInt32(message.RequestId);
124  stream.WriteInt32(message.ResponseTo);
125  stream.WriteInt32((int)Opcode.Reply);
126 
127  var flags = ResponseFlags.None;
128  if (message.AwaitCapable)
129  {
130  flags |= ResponseFlags.AwaitCapable;
131  }
132  if (message.QueryFailure)
133  {
134  flags |= ResponseFlags.QueryFailure;
135  }
136  if (message.CursorNotFound)
137  {
138  flags |= ResponseFlags.CursorNotFound;
139  }
140  stream.WriteInt32((int)flags);
141 
142  stream.WriteInt64(message.CursorId);
143  stream.WriteInt32(message.StartingFrom);
144  stream.WriteInt32(message.NumberReturned);
145  if (message.QueryFailure)
146  {
147  var context = BsonSerializationContext.CreateRoot(binaryWriter);
148  _serializer.Serialize(context, message.QueryFailureDocument);
149  }
150  else
151  {
152  foreach (var doc in message.Documents)
153  {
154  var context = BsonSerializationContext.CreateRoot(binaryWriter);
155  _serializer.Serialize(context, doc);
156  }
157  }
158  stream.BackpatchSize(startPosition);
159  }
