90  public void Analyze()
91  {
92  m_DependencyDatas.Clear();
93  m_ScatteredAssets.Clear();
94  m_CircularDependencyDatas.Clear();
95  m_AnalyzedStamps.Clear();
96 
97  HashSet<string> scriptAssetNames = GetFilteredAssetNames("t:Script");
98  Asset[] assets = m_ResourceCollection.GetAssets();
99  int count = assets.Length;
100  for (int i = 0; i < count; i++)
101  {
102  if (OnAnalyzingAsset != null)
103  {
104  OnAnalyzingAsset(i, count);
105  }
106 
107  string assetName = assets[i].Name;
108  if (string.IsNullOrEmpty(assetName))
109  {
110  Debug.LogWarning(Utility.Text.Format("Can not find asset by guid '{0}'.", assets[i].Guid));
111  continue;
112  }
113 
114  DependencyData dependencyData = new DependencyData();
115  AnalyzeAsset(assetName, assets[i], dependencyData, scriptAssetNames);
116  dependencyData.RefreshData();
117  m_DependencyDatas.Add(assetName, dependencyData);
118  }
119 
120  foreach (List<Asset> scatteredAsset in m_ScatteredAssets.Values)
121  {
122  scatteredAsset.Sort((a, b) => a.Name.CompareTo(b.Name));
123  }
124 
125  m_CircularDependencyDatas.AddRange(new CircularDependencyChecker(m_AnalyzedStamps.ToArray()).Check());
126 
127  if (OnAnalyzeCompleted != null)
128  {
129  OnAnalyzeCompleted();
130  }
131  }

133  private void AnalyzeAsset(string assetName, Asset hostAsset, DependencyData dependencyData, HashSet<string> scriptAssetNames)
134  {
135  string[] dependencyAssetNames = AssetDatabase.GetDependencies(assetName, false);
136  foreach (string dependencyAssetName in dependencyAssetNames)
137  {
138  if (scriptAssetNames.Contains(dependencyAssetName))
139  {
140  continue;
141  }
142 
143  if (dependencyAssetName == assetName)
144  {
145  continue;
146  }
147 
148  if (dependencyAssetName.EndsWith(".unity", StringComparison.Ordinal))
149  {
150  // 忽略对场景的依赖
151  continue;
152  }
153 
154  Stamp stamp = new Stamp(hostAsset.Name, dependencyAssetName);
155  if (m_AnalyzedStamps.Contains(stamp))
156  {
157  continue;
158  }
159 
160  m_AnalyzedStamps.Add(stamp);
161 
162  string guid = AssetDatabase.AssetPathToGUID(dependencyAssetName);
163  if (string.IsNullOrEmpty(guid))
164  {
165  Debug.LogWarning(Utility.Text.Format("Can not find guid by asset '{0}'.", dependencyAssetName));
166  continue;
167  }
168 
169  Asset asset = m_ResourceCollection.GetAsset(guid);
170  if (asset != null)
171  {
172  dependencyData.AddDependencyAsset(asset);
173  }
174  else
175  {
176  dependencyData.AddScatteredDependencyAsset(dependencyAssetName);
177 
178  List<Asset> scatteredAssets = null;
179  if (!m_ScatteredAssets.TryGetValue(dependencyAssetName, out scatteredAssets))
180  {
181  scatteredAssets = new List<Asset>();
182  m_ScatteredAssets.Add(dependencyAssetName, scatteredAssets);
183  }
184 
185  scatteredAssets.Add(hostAsset);
186 
187  AnalyzeAsset(dependencyAssetName, hostAsset, dependencyData, scriptAssetNames);
188  }
189  }
190  }

202  public string[] GetAssetNames(AssetsOrder order, string filter)
203  {
204  HashSet<string> filteredAssetNames = GetFilteredAssetNames(filter);
205  IEnumerable<KeyValuePair<string, DependencyData>> filteredResult = m_DependencyDatas.Where(pair => filteredAssetNames.Contains(pair.Key));
206  IEnumerable<KeyValuePair<string, DependencyData>> orderedResult = null;
207  switch (order)
208  {
209  case AssetsOrder.AssetNameAsc:
210  orderedResult = filteredResult.OrderBy(pair => pair.Key);
211  break;
212 
213  case AssetsOrder.AssetNameDesc:
214  orderedResult = filteredResult.OrderByDescending(pair => pair.Key);
215  break;
216 
217  case AssetsOrder.DependencyResourceCountAsc:
218  orderedResult = filteredResult.OrderBy(pair => pair.Value.DependencyResourceCount);
219  break;
220 
221  case AssetsOrder.DependencyResourceCountDesc:
222  orderedResult = filteredResult.OrderByDescending(pair => pair.Value.DependencyResourceCount);
223  break;
224 
225  case AssetsOrder.DependencyAssetCountAsc:
226  orderedResult = filteredResult.OrderBy(pair => pair.Value.DependencyAssetCount);
227  break;
228 
229  case AssetsOrder.DependencyAssetCountDesc:
230  orderedResult = filteredResult.OrderByDescending(pair => pair.Value.DependencyAssetCount);
231  break;
232 
233  case AssetsOrder.ScatteredDependencyAssetCountAsc:
234  orderedResult = filteredResult.OrderBy(pair => pair.Value.ScatteredDependencyAssetCount);
235  break;
236 
237  case AssetsOrder.ScatteredDependencyAssetCountDesc:
238  orderedResult = filteredResult.OrderByDescending(pair => pair.Value.ScatteredDependencyAssetCount);
239  break;
240 
241  default:
242  orderedResult = filteredResult;
243  break;
244  }
245 
246  return orderedResult.Select(pair => pair.Key).ToArray();
247  }
