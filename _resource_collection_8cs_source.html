126  }
127 
128  xmlNodeList = xmlAssets.ChildNodes;
129  count = xmlNodeList.Count;
130  for (int i = 0; i < count; i++)
131  {
132  if (OnLoadingAsset != null)
133  {
134  OnLoadingAsset(i, count);
135  }

137  xmlNode = xmlNodeList.Item(i);
138  if (xmlNode.Name != "Asset")

472  {

490  if (string.IsNullOrEmpty(assetName))
491  {

493  }
494 
495  Asset[] assetsInResource = resource.GetAssets();

497  {
498  if (assetInResource.Name == assetName)

504  {

507  }

511  {
512  return false;
513  }

574  private bool IsAvailableResourceName(string name, string variant, Resource current)
575  {
576  Resource found = GetResource(name, variant);
577  if (found != null && found != current)
578  {
579  return false;
580  }
581 
582  string[] foundPathNames = name.Split('/');
583  foreach (Resource resource in m_Resources.Values)
584  {
585  if (current != null && resource == current)
586  {
587  continue;
588  }
589 
590  if (resource.Name == name)
591  {
592  if (resource.Variant == null && variant != null)
593  {
594  return false;
595  }
596 
597  if (resource.Variant != null && variant == null)
598  {
599  return false;
600  }
601  }
602 
603  if (resource.Name.Length > name.Length
604  && resource.Name.IndexOf(name, StringComparison.CurrentCultureIgnoreCase) == 0
605  && resource.Name[name.Length] == '/')
606  {
607  return false;
608  }
609 
610  if (name.Length > resource.Name.Length
611  && name.IndexOf(resource.Name, StringComparison.CurrentCultureIgnoreCase) == 0
612  && name[resource.Name.Length] == '/')
613  {
614  return false;
615  }
616 
617  string[] pathNames = resource.Name.Split('/');
618  for (int i = 0; i < foundPathNames.Length - 1 && i < pathNames.Length - 1; i++)
619  {
