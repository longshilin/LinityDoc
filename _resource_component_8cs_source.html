591  private void Start()
592  {
593  BaseComponent baseComponent = GameEntry.GetComponent<BaseComponent>();
594  if (baseComponent == null)
595  {
596  Log.Fatal("Base component is invalid.");
597  return;
598  }
599 
600  m_EventComponent = GameEntry.GetComponent<EventComponent>();
601  if (m_EventComponent == null)
602  {
603  Log.Fatal("Event component is invalid.");
604  return;
605  }
606 
607  m_EditorResourceMode = baseComponent.EditorResourceMode;
608  m_ResourceManager = m_EditorResourceMode ? baseComponent.EditorResourceHelper : GameFrameworkEntry.GetModule<IResourceManager>();
609  if (m_ResourceManager == null)
610  {
611  Log.Fatal("Resource manager is invalid.");
612  return;
613  }
614 
615  m_ResourceManager.ResourceApplySuccess += OnResourceApplySuccess;
616  m_ResourceManager.ResourceApplyFailure += OnResourceApplyFailure;
617  m_ResourceManager.ResourceUpdateStart += OnResourceUpdateStart;
618  m_ResourceManager.ResourceUpdateChanged += OnResourceUpdateChanged;
619  m_ResourceManager.ResourceUpdateSuccess += OnResourceUpdateSuccess;
620  m_ResourceManager.ResourceUpdateFailure += OnResourceUpdateFailure;
621  m_ResourceManager.ResourceUpdateAllComplete += OnResourceUpdateAllComplete;
622 
623  m_ResourceManager.SetReadOnlyPath(Application.streamingAssetsPath);
624  if (m_ReadWritePathType == ReadWritePathType.TemporaryCache)
625  {
626  m_ResourceManager.SetReadWritePath(Application.temporaryCachePath);
627  }
628  else
629  {
630  if (m_ReadWritePathType == ReadWritePathType.Unspecified)
631  {
632  m_ReadWritePathType = ReadWritePathType.PersistentData;
633  }
634 
635  m_ResourceManager.SetReadWritePath(Application.persistentDataPath);
636  }
637 
638  if (m_EditorResourceMode)
639  {
640  return;
641  }
642 
643  SetResourceMode(m_ResourceMode);
644  m_ResourceManager.SetObjectPoolManager(GameFrameworkEntry.GetModule<IObjectPoolManager>());
645  m_ResourceManager.SetFileSystemManager(GameFrameworkEntry.GetModule<IFileSystemManager>());
646  m_ResourceManager.SetDownloadManager(GameFrameworkEntry.GetModule<IDownloadManager>());
647  m_ResourceManager.AssetAutoReleaseInterval = m_AssetAutoReleaseInterval;
648  m_ResourceManager.AssetCapacity = m_AssetCapacity;
649  m_ResourceManager.AssetExpireTime = m_AssetExpireTime;
650  m_ResourceManager.AssetPriority = m_AssetPriority;
651  m_ResourceManager.ResourceAutoReleaseInterval = m_ResourceAutoReleaseInterval;
652  m_ResourceManager.ResourceCapacity = m_ResourceCapacity;
653  m_ResourceManager.ResourceExpireTime = m_ResourceExpireTime;
654  m_ResourceManager.ResourcePriority = m_ResourcePriority;
655  if (m_ResourceMode == ResourceMode.Updatable || m_ResourceMode == ResourceMode.UpdatableWhilePlaying)
656  {
657  m_ResourceManager.UpdatePrefixUri = m_UpdatePrefixUri;
658  m_ResourceManager.GenerateReadWriteVersionListLength = m_GenerateReadWriteVersionListLength;
659  m_ResourceManager.UpdateRetryCount = m_UpdateRetryCount;
660  }
661 
662  m_ResourceHelper = Helper.CreateHelper(m_ResourceHelperTypeName, m_CustomResourceHelper);
663  if (m_ResourceHelper == null)
664  {
665  Log.Error("Can not create resource helper.");
666  return;
667  }
668 
669  m_ResourceHelper.name = "Resource Helper";
670  Transform transform = m_ResourceHelper.transform;
671  transform.SetParent(this.transform);
672  transform.localScale = Vector3.one;
673 
674  m_ResourceManager.SetResourceHelper(m_ResourceHelper);
675 
676  if (m_InstanceRoot == null)
677  {
678  m_InstanceRoot = new GameObject("Load Resource Agent Instances").transform;
679  m_InstanceRoot.SetParent(gameObject.transform);
680  m_InstanceRoot.localScale = Vector3.one;
681  }
682 
683  for (int i = 0; i < m_LoadResourceAgentHelperCount; i++)
684  {
685  AddLoadResourceAgentHelper(i);
686  }
687  }

689  private void Update()
690  {
691  m_LastUnloadUnusedAssetsOperationElapseSeconds += Time.unscaledDeltaTime;
692  if (m_AsyncOperation == null && (m_ForceUnloadUnusedAssets || m_LastUnloadUnusedAssetsOperationElapseSeconds >= m_MaxUnloadUnusedAssetsInterval || m_PreorderUnloadUnusedAssets && m_LastUnloadUnusedAssetsOperationElapseSeconds >= m_MinUnloadUnusedAssetsInterval))
693  {
694  Log.Info("Unload unused assets...");
695  m_ForceUnloadUnusedAssets = false;
696  m_PreorderUnloadUnusedAssets = false;
697  m_LastUnloadUnusedAssetsOperationElapseSeconds = 0f;
698  m_AsyncOperation = Resources.UnloadUnusedAssets();
699  }
700 
701  if (m_AsyncOperation != null && m_AsyncOperation.isDone)
702  {
703  m_AsyncOperation = null;
704  if (m_PerformGCCollect)
705  {
706  Log.Info("GC.Collect...");
707  m_PerformGCCollect = false;
708  GC.Collect();
709  }
710  }
711  }

717  public void SetResourceMode(ResourceMode resourceMode)
718  {
719  m_ResourceManager.SetResourceMode(resourceMode);
720  switch (resourceMode)
721  {
722  case ResourceMode.Package:
723  m_ResourceManager.PackageVersionListSerializer.RegisterDeserializeCallback(0, BuiltinVersionListSerializer.PackageVersionListDeserializeCallback_V0);
724  m_ResourceManager.PackageVersionListSerializer.RegisterDeserializeCallback(1, BuiltinVersionListSerializer.PackageVersionListDeserializeCallback_V1);
725  m_ResourceManager.PackageVersionListSerializer.RegisterDeserializeCallback(2, BuiltinVersionListSerializer.PackageVersionListDeserializeCallback_V2);
726  break;
727 
728  case ResourceMode.Updatable:
729  case ResourceMode.UpdatableWhilePlaying:
730  m_ResourceManager.UpdatableVersionListSerializer.RegisterDeserializeCallback(0, BuiltinVersionListSerializer.UpdatableVersionListDeserializeCallback_V0);
731  m_ResourceManager.UpdatableVersionListSerializer.RegisterDeserializeCallback(1, BuiltinVersionListSerializer.UpdatableVersionListDeserializeCallback_V1);
732  m_ResourceManager.UpdatableVersionListSerializer.RegisterDeserializeCallback(2, BuiltinVersionListSerializer.UpdatableVersionListDeserializeCallback_V2);
733 
734  m_ResourceManager.UpdatableVersionListSerializer.RegisterTryGetValueCallback(0, BuiltinVersionListSerializer.UpdatableVersionListTryGetValueCallback_V0);
735  m_ResourceManager.UpdatableVersionListSerializer.RegisterTryGetValueCallback(1, BuiltinVersionListSerializer.UpdatableVersionListTryGetValueCallback_V1_V2);
736  m_ResourceManager.UpdatableVersionListSerializer.RegisterTryGetValueCallback(2, BuiltinVersionListSerializer.UpdatableVersionListTryGetValueCallback_V1_V2);
737 
738  m_ResourceManager.ReadOnlyVersionListSerializer.RegisterDeserializeCallback(0, BuiltinVersionListSerializer.LocalVersionListDeserializeCallback_V0);
739  m_ResourceManager.ReadOnlyVersionListSerializer.RegisterDeserializeCallback(1, BuiltinVersionListSerializer.LocalVersionListDeserializeCallback_V1);
740  m_ResourceManager.ReadOnlyVersionListSerializer.RegisterDeserializeCallback(2, BuiltinVersionListSerializer.LocalVersionListDeserializeCallback_V2);
741 
742  m_ResourceManager.ReadWriteVersionListSerializer.RegisterSerializeCallback(0, BuiltinVersionListSerializer.LocalVersionListSerializeCallback_V0);
743  m_ResourceManager.ReadWriteVersionListSerializer.RegisterSerializeCallback(1, BuiltinVersionListSerializer.LocalVersionListSerializeCallback_V1);
744  m_ResourceManager.ReadWriteVersionListSerializer.RegisterSerializeCallback(2, BuiltinVersionListSerializer.LocalVersionListSerializeCallback_V2);
745 
746  m_ResourceManager.ReadWriteVersionListSerializer.RegisterDeserializeCallback(0, BuiltinVersionListSerializer.LocalVersionListDeserializeCallback_V0);
747  m_ResourceManager.ReadWriteVersionListSerializer.RegisterDeserializeCallback(1, BuiltinVersionListSerializer.LocalVersionListDeserializeCallback_V1);
