215  public bool Load()
216  {
217  if (!File.Exists(m_ConfigurationPath))
218  {
219  return false;
220  }
221 
222  try
223  {
224  XmlDocument xmlDocument = new XmlDocument();
225  xmlDocument.Load(m_ConfigurationPath);
226  XmlNode xmlRoot = xmlDocument.SelectSingleNode("UnityGameFramework");
227  XmlNode xmlEditor = xmlRoot.SelectSingleNode("ResourceEditor");
228  XmlNode xmlSettings = xmlEditor.SelectSingleNode("Settings");
229 
230  XmlNodeList xmlNodeList = null;
231  XmlNode xmlNode = null;
232 
233  xmlNodeList = xmlSettings.ChildNodes;
234  for (int i = 0; i < xmlNodeList.Count; i++)
235  {
236  xmlNode = xmlNodeList.Item(i);
237  switch (xmlNode.Name)
238  {
239  case "SourceAssetRootPath":
240  SourceAssetRootPath = xmlNode.InnerText;
241  break;
242 
243  case "SourceAssetSearchPaths":
244  m_SourceAssetSearchRelativePaths.Clear();
245  XmlNodeList xmlNodeListInner = xmlNode.ChildNodes;
246  XmlNode xmlNodeInner = null;
247  for (int j = 0; j < xmlNodeListInner.Count; j++)
248  {
249  xmlNodeInner = xmlNodeListInner.Item(j);
250  if (xmlNodeInner.Name != "SourceAssetSearchPath")
251  {
252  continue;
253  }
254 
255  m_SourceAssetSearchRelativePaths.Add(xmlNodeInner.Attributes.GetNamedItem("RelativePath").Value);
256  }
257  break;
258 
259  case "SourceAssetUnionTypeFilter":
260  SourceAssetUnionTypeFilter = xmlNode.InnerText;
261  break;
262 
263  case "SourceAssetUnionLabelFilter":
264  SourceAssetUnionLabelFilter = xmlNode.InnerText;
265  break;
266 
267  case "SourceAssetExceptTypeFilter":
268  SourceAssetExceptTypeFilter = xmlNode.InnerText;
269  break;
270 
271  case "SourceAssetExceptLabelFilter":
272  SourceAssetExceptLabelFilter = xmlNode.InnerText;
273  break;
274 
275  case "AssetSorter":
276  AssetSorter = (AssetSorterType)Enum.Parse(typeof(AssetSorterType), xmlNode.InnerText);
277  break;
278  }
279  }
280 
281  RefreshSourceAssetSearchPaths();
282  }
283  catch
284  {
285  File.Delete(m_ConfigurationPath);
286  return false;
287  }
288 
289  ScanSourceAssets();
290 
291  m_ResourceCollection.Load();
292 
293  return true;

542  public void ScanSourceAssets()
543  {
544  m_SourceAssets.Clear();
545  m_SourceAssetRoot.Clear();
546 
547  string[] sourceAssetSearchPaths = m_SourceAssetSearchPaths.ToArray();
548  HashSet<string> tempGuids = new HashSet<string>();
549  tempGuids.UnionWith(AssetDatabase.FindAssets(SourceAssetUnionTypeFilter, sourceAssetSearchPaths));
550  tempGuids.UnionWith(AssetDatabase.FindAssets(SourceAssetUnionLabelFilter, sourceAssetSearchPaths));
551  tempGuids.ExceptWith(AssetDatabase.FindAssets(SourceAssetExceptTypeFilter, sourceAssetSearchPaths));
552  tempGuids.ExceptWith(AssetDatabase.FindAssets(SourceAssetExceptLabelFilter, sourceAssetSearchPaths));
553 
554  string[] guids = new List<string>(tempGuids).ToArray();
555  foreach (string guid in guids)
556  {
557  string fullPath = AssetDatabase.GUIDToAssetPath(guid);
558  if (AssetDatabase.IsValidFolder(fullPath))
559  {
560  // Skip folder.
561  continue;
562  }
563 
564  string assetPath = fullPath.Substring(SourceAssetRootPath.Length + 1);
565  string[] splitedPath = assetPath.Split('/');
566  SourceFolder folder = m_SourceAssetRoot;
567  for (int i = 0; i < splitedPath.Length - 1; i++)
568  {
569  SourceFolder subFolder = folder.GetFolder(splitedPath[i]);
570  folder = subFolder == null ? folder.AddFolder(splitedPath[i]) : subFolder;
571  }
572 
573  SourceAsset asset = folder.AddAsset(guid, fullPath, splitedPath[splitedPath.Length - 1]);
574  m_SourceAssets.Add(asset.Guid, asset);
575  }
576  }
