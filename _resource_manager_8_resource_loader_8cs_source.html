306  public void LoadAsset(string assetName, Type assetType, int priority, LoadAssetCallbacks loadAssetCallbacks, object userData)
307  {
308  ResourceInfo resourceInfo = null;
309  string[] dependencyAssetNames = null;
310  if (!CheckAsset(assetName, out resourceInfo, out dependencyAssetNames))
311  {
312  string errorMessage = Utility.Text.Format("Can not load asset '{0}'.", assetName);
313  if (loadAssetCallbacks.LoadAssetFailureCallback != null)
314  {
315  loadAssetCallbacks.LoadAssetFailureCallback(assetName, resourceInfo != null && !resourceInfo.Ready ? LoadResourceStatus.NotReady : LoadResourceStatus.NotExist, errorMessage, userData);
316  return;
317  }
318 
319  throw new GameFrameworkException(errorMessage);
320  }
321 
322  if (resourceInfo.IsLoadFromBinary)
323  {
324  string errorMessage = Utility.Text.Format("Can not load asset '{0}' which is a binary asset.", assetName);
325  if (loadAssetCallbacks.LoadAssetFailureCallback != null)
326  {
327  loadAssetCallbacks.LoadAssetFailureCallback(assetName, LoadResourceStatus.TypeError, errorMessage, userData);
328  return;
329  }
330 
331  throw new GameFrameworkException(errorMessage);
332  }
333 
334  LoadAssetTask mainTask = LoadAssetTask.Create(assetName, assetType, priority, resourceInfo, dependencyAssetNames, loadAssetCallbacks, userData);
335  foreach (string dependencyAssetName in dependencyAssetNames)
336  {
337  if (!LoadDependencyAsset(dependencyAssetName, priority, mainTask, userData))
338  {
339  string errorMessage = Utility.Text.Format("Can not load dependency asset '{0}' when load asset '{1}'.", dependencyAssetName, assetName);
340  if (loadAssetCallbacks.LoadAssetFailureCallback != null)
341  {
342  loadAssetCallbacks.LoadAssetFailureCallback(assetName, LoadResourceStatus.DependencyError, errorMessage, userData);
343  return;
344  }
345 
346  throw new GameFrameworkException(errorMessage);
347  }
348  }
349 
350  m_TaskPool.AddTask(mainTask);
351  if (!resourceInfo.Ready)
352  {
353  m_ResourceManager.UpdateResource(resourceInfo.ResourceName);
354  }
355  }

572  return;

574 
575  throw new GameFrameworkException(errorMessage);
576  }
577 
578  if (!resourceInfo.Ready)
579  {
580  string errorMessage = Utility.Text.Format("Can not load binary '{0}' which is not ready.", binaryAssetName);
581  if (loadBinaryCallbacks.LoadBinaryFailureCallback != null)
582  {
583  loadBinaryCallbacks.LoadBinaryFailureCallback(binaryAssetName, LoadResourceStatus.NotReady, errorMessage, userData);
584  return;
585  }
586 
587  throw new GameFrameworkException(errorMessage);
588  }
589 
590  if (!resourceInfo.IsLoadFromBinary)
591  {
592  string errorMessage = Utility.Text.Format("Can not load binary '{0}' which is not a binary asset.", binaryAssetName);
593  if (loadBinaryCallbacks.LoadBinaryFailureCallback != null)
594  {
595  loadBinaryCallbacks.LoadBinaryFailureCallback(binaryAssetName, LoadResourceStatus.TypeError, errorMessage, userData);
596  return;
597  }
598 
599  throw new GameFrameworkException(errorMessage);
600  }
601 
602  if (resourceInfo.UseFileSystem)
603  {
604  loadBinaryCallbacks.LoadBinarySuccessCallback(binaryAssetName, LoadBinaryFromFileSystem(binaryAssetName), 0f, userData);
605  }
606  else
607  {
608  string path = Utility.Path.GetRemotePath(Path.Combine(resourceInfo.StorageInReadOnly ? m_ResourceManager.m_ReadOnlyPath : m_ResourceManager.m_ReadWritePath, resourceInfo.ResourceName.FullName));
609  m_ResourceManager.m_ResourceHelper.LoadBytes(path, m_LoadBytesCallbacks, LoadBinaryInfo.Create(binaryAssetName, resourceInfo, loadBinaryCallbacks, userData));
610  }

806  private bool LoadDependencyAsset(string assetName, int priority, LoadResourceTaskBase mainTask, object userData)
807  {

821  {
822  return false;
