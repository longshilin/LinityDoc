81  public CheckVersionListResult CheckVersionList(int latestInternalResourceVersion)
82  {
83  if (string.IsNullOrEmpty(m_ResourceManager.m_ReadWritePath))
84  {
85  throw new GameFrameworkException("Read-write path is invalid.");
86  }
87 
88  string versionListFileName = Utility.Path.GetRegularPath(Path.Combine(m_ResourceManager.m_ReadWritePath, RemoteVersionListFileName));
89  if (!File.Exists(versionListFileName))
90  {
91  return CheckVersionListResult.NeedUpdate;
92  }
93 
94  int internalResourceVersion = 0;
95  FileStream fileStream = null;
96  try
97  {
98  fileStream = new FileStream(versionListFileName, FileMode.Open, FileAccess.Read);
99  object internalResourceVersionObject = null;
100  if (!m_ResourceManager.m_UpdatableVersionListSerializer.TryGetValue(fileStream, "InternalResourceVersion", out internalResourceVersionObject))
101  {
102  return CheckVersionListResult.NeedUpdate;
103  }
104 
105  internalResourceVersion = (int)internalResourceVersionObject;
106  }
107  catch
108  {
109  return CheckVersionListResult.NeedUpdate;
110  }
111  finally
112  {
113  if (fileStream != null)
114  {
115  fileStream.Dispose();
116  fileStream = null;
117  }
118  }
119 
120  if (internalResourceVersion != latestInternalResourceVersion)
121  {
122  return CheckVersionListResult.NeedUpdate;
123  }
124 
125  return CheckVersionListResult.Updated;
126  }

152  private void OnDownloadSuccess(object sender, DownloadSuccessEventArgs e)
153  {
154  VersionListProcessor versionListProcessor = e.UserData as VersionListProcessor;
155  if (versionListProcessor == null || versionListProcessor != this)
156  {
157  return;
158  }
159 
160  try
161  {
162  using (FileStream fileStream = new FileStream(e.DownloadPath, FileMode.Open, FileAccess.ReadWrite))
163  {
164  int length = (int)fileStream.Length;
165  if (length != m_VersionListCompressedLength)
166  {
167  fileStream.Close();
168  string errorMessage = Utility.Text.Format("Latest version list compressed length error, need '{0}', downloaded '{1}'.", m_VersionListCompressedLength.ToString(), length.ToString());
169  DownloadFailureEventArgs downloadFailureEventArgs = DownloadFailureEventArgs.Create(e.SerialId, e.DownloadPath, e.DownloadUri, errorMessage, e.UserData);
170  OnDownloadFailure(this, downloadFailureEventArgs);
171  ReferencePool.Release(downloadFailureEventArgs);
172  return;
173  }
174 
175  fileStream.Position = 0L;
176  int hashCode = Utility.Verifier.GetCrc32(fileStream);
177  if (hashCode != m_VersionListCompressedHashCode)
178  {
179  fileStream.Close();
180  string errorMessage = Utility.Text.Format("Latest version list compressed hash code error, need '{0}', downloaded '{1}'.", m_VersionListCompressedHashCode.ToString(), hashCode.ToString());
181  DownloadFailureEventArgs downloadFailureEventArgs = DownloadFailureEventArgs.Create(e.SerialId, e.DownloadPath, e.DownloadUri, errorMessage, e.UserData);
182  OnDownloadFailure(this, downloadFailureEventArgs);
183  ReferencePool.Release(downloadFailureEventArgs);
184  return;
185  }
186 
187  if (m_ResourceManager.m_DecompressCachedStream == null)
188  {
189  m_ResourceManager.m_DecompressCachedStream = new MemoryStream();
190  }
191 
192  fileStream.Position = 0L;
193  m_ResourceManager.m_DecompressCachedStream.Position = 0L;
194  m_ResourceManager.m_DecompressCachedStream.SetLength(0L);
195  if (!Utility.Compression.Decompress(fileStream, m_ResourceManager.m_DecompressCachedStream))
196  {
197  fileStream.Close();
198  string errorMessage = Utility.Text.Format("Unable to decompress latest version list '{0}'.", e.DownloadPath);
199  DownloadFailureEventArgs downloadFailureEventArgs = DownloadFailureEventArgs.Create(e.SerialId, e.DownloadPath, e.DownloadUri, errorMessage, e.UserData);
200  OnDownloadFailure(this, downloadFailureEventArgs);
201  ReferencePool.Release(downloadFailureEventArgs);
202  return;
203  }
204 
205  if (m_ResourceManager.m_DecompressCachedStream.Length != m_VersionListLength)
206  {
207  fileStream.Close();
208  string errorMessage = Utility.Text.Format("Latest version list length error, need '{0}', downloaded '{1}'.", m_VersionListLength.ToString(), m_ResourceManager.m_DecompressCachedStream.Length.ToString());
209  DownloadFailureEventArgs downloadFailureEventArgs = DownloadFailureEventArgs.Create(e.SerialId, e.DownloadPath, e.DownloadUri, errorMessage, e.UserData);
210  OnDownloadFailure(this, downloadFailureEventArgs);
211  ReferencePool.Release(downloadFailureEventArgs);
212  return;
213  }
214 
215  fileStream.Position = 0L;
216  fileStream.SetLength(0L);
217  fileStream.Write(m_ResourceManager.m_DecompressCachedStream.GetBuffer(), 0, (int)m_ResourceManager.m_DecompressCachedStream.Length);
218  }
219 
220  if (VersionListUpdateSuccess != null)
221  {
222  VersionListUpdateSuccess(e.DownloadPath, e.DownloadUri);
223  }
224  }
225  catch (Exception exception)
226  {
227  string errorMessage = Utility.Text.Format("Update latest version list '{0}' with error message '{1}'.", e.DownloadPath, exception.ToString());
228  DownloadFailureEventArgs downloadFailureEventArgs = DownloadFailureEventArgs.Create(e.SerialId, e.DownloadPath, e.DownloadUri, errorMessage, e.UserData);
229  OnDownloadFailure(this, downloadFailureEventArgs);
230  ReferencePool.Release(downloadFailureEventArgs);
231  }
232  finally
233  {
234  if (m_ResourceManager.m_DecompressCachedStream != null)
235  {
236  m_ResourceManager.m_DecompressCachedStream.Position = 0L;
237  m_ResourceManager.m_DecompressCachedStream.SetLength(0L);
238  }
239  }
240  }
