338  private void AnalysisResourceFilters()
339  {
340  m_ResourceCollection = new ResourceCollection();
341  List<string> signedAssetBundleList = new List<string>();
342 
343  foreach (ResourceRule resourceRule in m_Configuration.rules)
344  {
345  if (resourceRule.variant == "")
346  resourceRule.variant = null;
347 
348  if (resourceRule.valid)
349  {
350  switch (resourceRule.filterType)
351  {
352  case ResourceFilterType.Root:
353  {
354  if (string.IsNullOrEmpty(resourceRule.name))
355  {
356  string relativeDirectoryName =
357  resourceRule.assetsDirectoryPath.Replace("Assets/", "");
358  ApplyResourceFilter(ref signedAssetBundleList, resourceRule,
359  GameFramework.Utility.Path.GetRegularPath(relativeDirectoryName));
360  }
361  else
362  {
363  ApplyResourceFilter(ref signedAssetBundleList, resourceRule,
364  resourceRule.name);
365  }
366  }
367  break;
368 
369  case ResourceFilterType.Children:
370  {
371  string[] patterns = resourceRule.searchPatterns.Split(';', ',', '|');
372  for (int i = 0; i < patterns.Length; i++)
373  {
374  FileInfo[] assetFiles =
375  new DirectoryInfo(resourceRule.assetsDirectoryPath).GetFiles(patterns[i],
376  SearchOption.AllDirectories);
377  foreach (FileInfo file in assetFiles)
378  {
379  if (file.Extension.Contains("meta"))
380  continue;
381 
382  string relativeAssetName = file.FullName.Substring(Application.dataPath.Length + 1);
383  string relativeAssetNameWithoutExtension =
384  GameFramework.Utility.Path.GetRegularPath(
385  relativeAssetName.Substring(0, relativeAssetName.IndexOf('.')));
386 
387  string assetName = Path.Combine("Assets", relativeAssetName);
388  string assetGUID = AssetDatabase.AssetPathToGUID(assetName);
389 
390  if (!m_SourceAssetExceptTypeFilterGUIDArray.Contains(assetGUID) && !m_SourceAssetExceptLabelFilterGUIDArray.Contains(assetGUID))
391  {
392  ApplyResourceFilter(ref signedAssetBundleList, resourceRule,
393  relativeAssetNameWithoutExtension, assetGUID);
394  }
395  }
396  }
397  }
398  break;
399 
400  case ResourceFilterType.ChildrenFoldersOnly:
401  {
402  DirectoryInfo[] assetDirectories =
403  new DirectoryInfo(resourceRule.assetsDirectoryPath).GetDirectories();
404  foreach (DirectoryInfo directory in assetDirectories)
405  {
406  string relativeDirectoryName =
407  directory.FullName.Substring(Application.dataPath.Length + 1);
408 
409  ApplyResourceFilter(ref signedAssetBundleList, resourceRule,
410  GameFramework.Utility.Path.GetRegularPath(relativeDirectoryName), string.Empty,
411  directory.FullName);
412  }
413  }
414  break;
415

457  private void ApplyResourceFilter(ref List<string> signedResourceList, ResourceRule resourceRule,
458  string resourceName, string singleAssetGUID = "", string childDirectoryPath = "")
459  {
460  if (!signedResourceList.Contains(Path.Combine(resourceRule.assetsDirectoryPath, resourceName)))
461  {
462  signedResourceList.Add(Path.Combine(resourceRule.assetsDirectoryPath, resourceName));
463 
464  foreach (GFResource oldResource in GetResources())
465  {
466  if (oldResource.Name == resourceName)
467  {
468  RenameResource(oldResource.Name, oldResource.Variant,
469  resourceName, resourceRule.variant);
470  break;
471  }
472  }
473 
474  if (!HasResource(resourceName, resourceRule.variant))
475  {
476 
477  if (string.IsNullOrEmpty(resourceRule.fileSystem))
478  {
479  resourceRule.fileSystem = null;
480  }
481 
482  AddResource(resourceName, resourceRule.variant,resourceRule.fileSystem,
483  resourceRule.loadType, resourceRule.packed,
484  resourceRule.groups.Split(';', ',', '|'));
485  }
486 
487  switch (resourceRule.filterType)
488  {
489  case ResourceFilterType.Root:
490  case ResourceFilterType.ChildrenFoldersOnly:
491  string[] patterns = resourceRule.searchPatterns.Split(';', ',', '|');
492  if (childDirectoryPath == "")
493  {
494  childDirectoryPath = resourceRule.assetsDirectoryPath;
495  }
496 
497  for (int i = 0; i < patterns.Length; i++)
498  {
499  FileInfo[] assetFiles =
500  new DirectoryInfo(childDirectoryPath).GetFiles(patterns[i],
501  SearchOption.AllDirectories);
502  foreach (FileInfo file in assetFiles)
503  {
504  if (file.Extension.Contains("meta"))
505  continue;
506 
507  string assetName = Path.Combine("Assets",
508  file.FullName.Substring(Application.dataPath.Length + 1));
509 
510  string assetGUID = AssetDatabase.AssetPathToGUID(assetName);
511 
512  if (!m_SourceAssetExceptTypeFilterGUIDArray.Contains(assetGUID) && !m_SourceAssetExceptLabelFilterGUIDArray.Contains(assetGUID))
513  {
514  AssignAsset(assetGUID, resourceName,
515  resourceRule.variant);
516  }
517  }
518  }
519 
520  break;
521 
522  case ResourceFilterType.Children:
523  case ResourceFilterType.ChildrenFilesOnly:
524  {
525  AssignAsset(singleAssetGUID, resourceName,
526  resourceRule.variant);
527  }
528  break;
529  }
