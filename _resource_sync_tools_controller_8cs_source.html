68  HashSet<string> allAssetNames = new HashSet<string>();
69  string[] assetBundleNames = GetUsedAssetBundleNames();
70  foreach (string assetBundleName in assetBundleNames)
71  {
72  string[] assetNames = GetAssetPathsFromAssetBundle(assetBundleName);

83  AssetImporter assetImporter = AssetImporter.GetAtPath(assetName);

114  public bool SyncToProject()
115  {
116  ResourceCollection resourceCollection = new ResourceCollection();
117 
118  resourceCollection.OnLoadingResource += delegate (int index, int count)
119  {
120  if (OnLoadingResource != null)
121  {
122  OnLoadingResource(index, count);
123  }
124  };
125 
126  resourceCollection.OnLoadingAsset += delegate (int index, int count)
127  {
128  if (OnLoadingAsset != null)
129  {
130  OnLoadingAsset(index, count);
131  }
132  };
133 
134  resourceCollection.OnLoadCompleted += delegate ()
135  {
136  if (OnCompleted != null)
137  {
138  OnCompleted();
139  }
140  };
141 
142  if (!resourceCollection.Load())
143  {
144  return false;
145  }
146 
147  int assetIndex = 0;
148  int assetCount = resourceCollection.AssetCount;
149  Resource[] resources = resourceCollection.GetResources();
150  foreach (Resource resource in resources)
151  {
152  if (resource.IsLoadFromBinary)
153  {
154  continue;
155  }
156 
157  Asset[] assets = resource.GetAssets();
158  foreach (Asset asset in assets)
159  {
160  AssetImporter assetImporter = AssetImporter.GetAtPath(asset.Name);
161  if (assetImporter == null)
162  {
163  if (OnCompleted != null)
164  {
165  OnCompleted();
166  }
167 
168  return false;
169  }
170 
171  assetImporter.assetBundleName = resource.Name;
172  assetImporter.assetBundleVariant = resource.Variant;
173  assetImporter.SaveAndReimport();
174 
175  if (OnResourceDataChanged != null)
176  {
177  OnResourceDataChanged(++assetIndex, assetCount, asset.Name);
178  }
179  }
180  }
181 
182  if (OnCompleted != null)
183  {
184  OnCompleted();
185  }
186 
187  return true;
188  }

190  public bool SyncFromProject()
191  {
192  ResourceCollection resourceCollection = new ResourceCollection();
193  string[] assetBundleNames = GetUsedAssetBundleNames();
194  foreach (string assetBundleName in assetBundleNames)
195  {
196  string name = assetBundleName;
197  string variant = null;
198  int dotPosition = assetBundleName.LastIndexOf('.');
199  if (dotPosition > 0 && dotPosition < assetBundleName.Length - 1)
200  {
201  name = assetBundleName.Substring(0, dotPosition);
202  variant = assetBundleName.Substring(dotPosition + 1);
203  }
204 
205  if (!resourceCollection.AddResource(name, variant, null, LoadType.LoadFromFile, false))
206  {
207  return false;
208  }
209 
210  string[] assetNames = GetAssetPathsFromAssetBundle(assetBundleName);
211  foreach (string assetName in assetNames)
212  {
213  string guid = AssetDatabase.AssetPathToGUID(assetName);
214  if (string.IsNullOrEmpty(guid))
215  {
216  return false;
217  }
218 
219  if (!resourceCollection.AssignAsset(guid, name, variant))
220  {
221  return false;
222  }
223  }
224  }
225 
226  return resourceCollection.Save();
227  }
