144  void SubmitLines (RetainedGizmos gizmos, ulong hash) {
145  // Unity only supports 65535 vertices per mesh. 65532 used because MaxLineEndPointsPerBatch needs to be even.
146  const int MaxLineEndPointsPerBatch = 65532/2;
147  int batches = (lines.Count + MaxLineEndPointsPerBatch - 1)/MaxLineEndPointsPerBatch;
148 
149  for (int batch = 0; batch < batches; batch++) {
150  int startIndex = MaxLineEndPointsPerBatch * batch;
151  int endIndex = Mathf.Min(startIndex + MaxLineEndPointsPerBatch, lines.Count);
152  int lineEndPointCount = endIndex - startIndex;
153  UnityEngine.Assertions.Assert.IsTrue(lineEndPointCount % 2 == 0);
154 
155  // Use pooled lists to avoid excessive allocations
156  var vertices = ListPool<Vector3>.Claim(lineEndPointCount*2);
157  var colors = ListPool<Color32>.Claim(lineEndPointCount*2);
158  var normals = ListPool<Vector3>.Claim(lineEndPointCount*2);
159  var uv = ListPool<Vector2>.Claim(lineEndPointCount*2);
160  var tris = ListPool<int>.Claim(lineEndPointCount*3);
161  // Loop through each endpoint of the lines
162  // and add 2 vertices for each
163  for (int j = startIndex; j < endIndex; j++) {
164  var vertex = (Vector3)lines[j];
165  vertices.Add(vertex);
166  vertices.Add(vertex);
167 
168  var color = (Color32)lineColors[j];
169  colors.Add(color);
170  colors.Add(color);
171  uv.Add(new Vector2(0, 0));
172  uv.Add(new Vector2(1, 0));
173  }
174 
175  // Loop through each line and add
176  // one normal for each vertex
177  for (int j = startIndex; j < endIndex; j += 2) {
178  var lineDir = (Vector3)(lines[j+1] - lines[j]);
179  // Store the line direction in the normals.
180  // A line consists of 4 vertices. The line direction will be used to
181  // offset the vertices to create a line with a fixed pixel thickness
182  normals.Add(lineDir);
183  normals.Add(lineDir);
184  normals.Add(lineDir);
185  normals.Add(lineDir);
186  }
187 
188  // Setup triangle indices
189  // A triangle consists of 3 indices
190  // A line (4 vertices) consists of 2 triangles, so 6 triangle indices
191  for (int j = 0, v = 0; j < lineEndPointCount*3; j += 6, v += 4) {
192  // First triangle
193  tris.Add(v+0);
194  tris.Add(v+1);
195  tris.Add(v+2);
196 
197  // Second triangle
198  tris.Add(v+1);
199  tris.Add(v+3);
200  tris.Add(v+2);
201  }
202 
203  var mesh = gizmos.GetMesh();
204 
205  // Set all data on the mesh
206  mesh.SetVertices(vertices);
207  mesh.SetTriangles(tris, 0);
208  mesh.SetColors(colors);
209  mesh.SetNormals(normals);
210  mesh.SetUVs(0, uv);
211 
212  // Upload all data and mark the mesh as unreadable
213  mesh.UploadMeshData(true);
214 
215  // Release the lists back to the pool
216  ListPool<Vector3>.Release(ref vertices);
217  ListPool<Color32>.Release(ref colors);
218  ListPool<Vector3>.Release(ref normals);
219  ListPool<Vector2>.Release(ref uv);
220  ListPool<int>.Release(ref tris);
221 
222  gizmos.meshes.Add(new MeshWithHash { hash = hash, mesh = mesh, lines = true });
223  gizmos.existingHashes.Add(hash);
224  }

352  void RemoveUnusedMeshes (List<MeshWithHash> meshList) {
353  // Walk the array with two pointers
354  // i pointing to the entry that should be filled with something
355  // and j pointing to the entry that is a potential candidate for
356  // filling the entry at i.
357  // When j reaches the end of the list it will be reduced in size
358  for (int i = 0, j = 0; i < meshList.Count; ) {
359  if (j == meshList.Count) {
360  j--;
361  meshList.RemoveAt(j);
362  } else if (usedHashes.Contains(meshList[j].hash)) {
363  meshList[i] = meshList[j];
364  i++;
365  j++;
366  } else {
367  PoolMesh(meshList[j].mesh);
368  existingHashes.Remove(meshList[j].hash);
369  j++;
370  }
371  }
372  }
