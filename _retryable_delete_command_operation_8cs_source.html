103  protected override IEnumerable<Type1CommandMessageSection> CreateCommandPayloads(IChannelHandle channel, int attempt)
104  {
105  BatchableSource<DeleteRequest> deletes;
106  if (attempt == 1)
107  {
108  deletes = _deletes;
109  }
110  else
111  {
112  deletes = new BatchableSource<DeleteRequest>(_deletes.Items, _deletes.Offset, _deletes.ProcessedCount, canBeSplit: false);
113  }
114  var maxBatchCount = Math.Min(MaxBatchCount ?? int.MaxValue, channel.ConnectionDescription.MaxBatchCount);
115  var maxDocumentSize = channel.ConnectionDescription.MaxWireDocumentSize;
116  var payload = new Type1CommandMessageSection<DeleteRequest>("deletes", deletes, DeleteRequestSerializer.Instance, NoOpElementNameValidator.Instance, maxBatchCount, maxDocumentSize);
117  return new Type1CommandMessageSection[] { payload };
