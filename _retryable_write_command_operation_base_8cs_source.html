242  private Action<IMessageEncoderPostProcessor> GetPostWriteAction(List<Type1CommandMessageSection> commandPayloads)
243  {
244  if (!_writeConcern.IsAcknowledged && _isOrdered)
245  {
246  return encoder =>
247  {
248  var requestsPayload = commandPayloads.Single();
249  if (!requestsPayload.Documents.AllItemsWereProcessed)
250  {
251  encoder.ChangeWriteConcernFromW0ToW1();
252  }
253  };
254  }
255  else
256  {
257  return null;
258  }
259  }
260
