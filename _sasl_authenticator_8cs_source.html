64  {

66  Ensure.IsNotNull(description, nameof(description));
67 
68  using (var conversation = new SaslConversation(description.ConnectionId))
69  {
70  var currentStep = _mechanism.Initialize( connection, conversation, description);
71 
72  var command = CreateStartCommand(currentStep);
73  while (true)
74  {

77  {

79  result = protocol.Execute(connection, cancellationToken);
80  }

174  private ISaslStep Transition(SaslConversation conversation, ISaslStep currentStep, BsonDocument result)
175  {
176  // we might be done here if the client is not expecting a reply from the server
177  if (result.GetValue("done", false).ToBoolean() && currentStep.IsComplete)
178  {
179  return null;
180  }
181 
182  currentStep = currentStep.Transition(conversation, result["payload"].AsByteArray);
183 
184  // we might be done here if the client had some final verification it needed to do
185  if (result.GetValue("done", false).ToBoolean() && currentStep.IsComplete)
186  {
187  return null;
188  }
189 
190  return currentStep;
191  }
