119  public static string GetSceneName(string sceneAssetName)
120  {
121  if (string.IsNullOrEmpty(sceneAssetName))
122  {
123  Log.Error("Scene asset name is invalid.");
124  return null;
125  }
126 
127  int sceneNamePosition = sceneAssetName.LastIndexOf('/');
128  if (sceneNamePosition + 1 >= sceneAssetName.Length)
129  {
130  Log.Error("Scene asset name '{0}' is invalid.", sceneAssetName);
131  return null;
132  }
133 
134  string sceneName = sceneAssetName.Substring(sceneNamePosition + 1);
135  sceneNamePosition = sceneName.LastIndexOf(".unity");
136  if (sceneNamePosition > 0)
137  {
138  sceneName = sceneName.Substring(0, sceneNamePosition);
139  }
140 
141  return sceneName;
142  }

339  public void SetSceneOrder(string sceneAssetName, int sceneOrder)
340  {
341  if (string.IsNullOrEmpty(sceneAssetName))
342  {
343  Log.Error("Scene asset name is invalid.");
344  return;
345  }
346 
347  if (!sceneAssetName.StartsWith("Assets/", StringComparison.Ordinal) || !sceneAssetName.EndsWith(".unity", StringComparison.Ordinal))

377  private void RefreshSceneOrder()
378  {
379  if (m_SceneOrder.Count > 0)
380  {
381  string maxSceneName = null;
382  int maxSceneOrder = 0;
383  foreach (KeyValuePair<string, int> sceneOrder in m_SceneOrder)
384  {
385  if (SceneIsLoading(sceneOrder.Key))
386  {
387  continue;
388  }
389 
390  if (maxSceneName == null)
391  {
392  maxSceneName = sceneOrder.Key;
393  maxSceneOrder = sceneOrder.Value;
394  continue;
395  }
396 
397  if (sceneOrder.Value > maxSceneOrder)
398  {
399  maxSceneName = sceneOrder.Key;
400  maxSceneOrder = sceneOrder.Value;
401  }
402  }
403 
404  if (maxSceneName == null)
405  {
406  SetActiveScene(m_GameFrameworkScene);
407  return;
408  }
409 
410  Scene scene = SceneManager.GetSceneByName(GetSceneName(maxSceneName));
411  if (!scene.IsValid())
412  {
413  Log.Error("Active scene '{0}' is invalid.", maxSceneName);
414  return;
415  }
416 
417  SetActiveScene(scene);
418  }
419  else
420  {
421  SetActiveScene(m_GameFrameworkScene);
422  }
423  }
