100  public void Free()
101  {
102  if (BufferPtr != IntPtr.Zero)
103  {
104  if (NumBuffers == 1)
105  {
106 #if NET452
107  var buffer = (SecurityBuffer)Marshal.PtrToStructure(BufferPtr, typeof(SecurityBuffer));
108 #else
109  var buffer = Marshal.PtrToStructure<SecurityBuffer>(BufferPtr);
110 #endif
111  buffer.Free();
112  }
113  else
114  {
115  // Since we aren't sending any messages using the kerberos encrypt/decrypt.
116  // The 1st buffer is going to be empty. We can skip it.
117  for (int i = 1; i < NumBuffers; i++)
118  {
119 #if NET452
120  var currentOffset = i * Marshal.SizeOf(typeof(SecurityBuffer));
121  var totalLength = currentOffset + Marshal.SizeOf(typeof(int)) + Marshal.SizeOf(typeof(int));
122 #else
123  var currentOffset = i * Marshal.SizeOf<SecurityBuffer>();
124  var totalLength = currentOffset + Marshal.SizeOf<int>() + Marshal.SizeOf<int>();
125 #endif
126  var buffer = Marshal.ReadIntPtr(BufferPtr, totalLength);
127  Marshal.FreeHGlobal(buffer);
128  }
129  }
130
