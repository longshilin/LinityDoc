235  public void RunModifiers (ModifierPass pass, Path path) {
236  if (pass == ModifierPass.PreProcess) {
237  if (preProcessPath != null) preProcessPath(path);
238  for (int i = 0; i < modifiers.Count; i++) modifiers[i].PreProcess(path);

418  } else {

422  p.enabledTags = traversableTags;

425  // Cancel a previously requested path is it has not been processed yet and also make sure that it has not been recycled and used somewhere else
426  if (path != null && path.PipelineState <= PathState.Processing && path.CompleteState != PathCompleteState.Error && lastPathID == path.pathID) {
427  path.FailWithError("Canceled path because a new one was requested.\n"+

432  // No callback will be sent for the canceled path
