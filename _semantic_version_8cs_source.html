202  public static bool TryParse(string value, out SemanticVersion result)
203  {
204  if (!string.IsNullOrEmpty(value))
205  {
206  var pattern = @"(?<major>\d+)\.(?<minor>\d+)(\.(?<patch>\d+)(-(?<preRelease>.*))?)?";
207  var match = Regex.Match((string)value, pattern);
208  if (match.Success)
209  {
210  var major = int.Parse(match.Groups["major"].Value);
211  var minor = int.Parse(match.Groups["minor"].Value);
212  var patch = match.Groups["patch"].Success ? int.Parse(match.Groups["patch"].Value) : 0;
213  var preRelease = match.Groups["preRelease"].Success ? match.Groups["preRelease"].Value : null;
214 
215  result = new SemanticVersion(major, minor, patch, preRelease);
216  return true;
217  }
218  }
219 
220  result = null;
221  return false;
222  }
