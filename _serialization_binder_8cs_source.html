47  public override Expression Visit(Expression node)
48  {
49  if (node == null)
50  {
51  return null;
52  }
53 
54  Expression replacement;
55  if (_bindingContext.TryGetExpressionMapping(node, out replacement))
56  {
57  return replacement;
58  }
59 
60  return base.Visit(node);
61  }

63  protected override Expression VisitBinary(BinaryExpression node)
64  {
65  var newNode = base.VisitBinary(node);
66  var binaryExpression = newNode as BinaryExpression;
67  if (binaryExpression != null && binaryExpression.NodeType == ExpressionType.ArrayIndex)
68  {
69  var serializationExpression = binaryExpression.Left as ISerializationExpression;
70  if (serializationExpression != null)
71  {
72  var arraySerializer = serializationExpression.Serializer as IBsonArraySerializer;
73  BsonSerializationInfo itemSerializationInfo;
74  if (arraySerializer != null &&
75  arraySerializer.TryGetItemSerializationInfo(out itemSerializationInfo))
76  {
77  newNode = new ArrayIndexExpression(
78  binaryExpression.Left,
79  binaryExpression.Right,
80  itemSerializationInfo.Serializer,
81  binaryExpression);
82  }
83  }
84  }
85 
86  return newNode;
87  }
88 
89  protected override Expression VisitConstant(ConstantExpression node)
90  {
91  if (node.Type.GetTypeInfo().IsGenericType &&
92  node.Type.GetGenericTypeDefinition() == typeof(IMongoQueryable<>))
93  {
94  var queryable = (IMongoQueryable)node.Value;
95  var provider = (IMongoQueryProvider)queryable.Provider;
96  return new CollectionExpression(
97  provider.CollectionNamespace,
98  provider.CollectionDocumentSerializer);
99  }
100 
101  return base.VisitConstant(node);
102  }
103 
104  protected override Expression VisitLambda<T>(Expression<T> node)
105  {
106  // Don't visit the parameters. We cannot replace a parameter expression
107  // with a document and we don't have a new parameter type to use because
108  // we don't know why we are binding yet.
109  return node.Update(
110  Visit(node.Body),
111  node.Parameters);
112  }

114  protected override Expression VisitMember(MemberExpression node)
115  {
116  Expression newNode;
117  if (_bindingContext.TryGetMemberMapping(node.Member, out newNode))
118  {
119  if (newNode is ISerializationExpression)
120  {
121  return newNode;
122  }
123 
124  var message = string.Format("Could not determine serialization information for member {0} in the expression tree {1}.",
125  node.Member,
126  node.ToString());
127  throw new MongoInternalException(message);
128  }
129 
130  newNode = base.VisitMember(node);
131  var mex = newNode as MemberExpression;
132  if (mex != null)
133  {
134  var serializationExpression = mex.Expression as ISerializationExpression;
135  if (serializationExpression != null)
136  {
137  var documentSerializer = serializationExpression.Serializer as IBsonDocumentSerializer;
138  BsonSerializationInfo memberSerializationInfo;
139  if (documentSerializer != null && documentSerializer.TryGetMemberSerializationInfo(node.Member.Name, out memberSerializationInfo))
140  {
141  if (memberSerializationInfo.ElementName == null)
142  {
143  newNode = new DocumentExpression(memberSerializationInfo.Serializer);
144  }
145  else
146  {
147  newNode = new FieldExpression(
148  mex.Expression,
149  memberSerializationInfo.ElementName,
150  memberSerializationInfo.Serializer,
151  mex);
152  }
153  }
154  }
155  }
156 
157  return newNode;
158  }

160  protected override Expression VisitMethodCall(MethodCallExpression node)
161  {
162  if (!_isClientSideProjection && EmbeddedPipelineBinder.SupportsNode(node))
163  {
164  return BindEmbeddedPipeline(node);
165  }
166 
167  switch (node.Method.Name)
168  {
169  case "ElementAt":
170  return BindElementAt(node);
171  case "get_Item":
172  return BindGetItem(node);
173  case "Inject":
174  return BindInject(node);
175  }
176 
177  // Select and SelectMany are the only supported client-side projection operators
178  if (_isClientSideProjection &&
179  ExpressionHelper.IsLinqMethod(node, "Select", "SelectMany") &&
180  node.Arguments.Count == 2)
181  {
182  return BindClientSideProjector(node);
183  }
184 
185  return base.VisitMethodCall(node);
186  }
187 
188  protected override Expression VisitUnary(UnaryExpression node)
189  {
190  var newNode = (UnaryExpression)base.VisitUnary(node);
191  if (newNode.NodeType == ExpressionType.Convert || newNode.NodeType == ExpressionType.ConvertChecked || newNode.NodeType == ExpressionType.TypeAs)
192  {
193  // handle unnecessary convert expressions
194  if (newNode.Method == null && !newNode.IsLiftedToNull && newNode.Type.GetTypeInfo().IsAssignableFrom(newNode.Operand.Type))
195  {
196  return newNode.Operand;
197  }
198 
199  // handle downcast convert expressions
200  if (newNode.Operand.Type.GetTypeInfo().IsAssignableFrom(newNode.Type))
201  {
202  var serializationExpression = newNode.Operand as SerializationExpression;
203  if (serializationExpression != null)
204  {
205  var serializer = _bindingContext.GetSerializer(newNode.Type, newNode);
206  switch (serializationExpression.ExtensionType)
207  {
208  case ExtensionExpressionType.ArrayIndex:
209  return new ArrayIndexExpression(
210  ((ArrayIndexExpression)serializationExpression).Array,
211  ((ArrayIndexExpression)serializationExpression).Index,
212  serializer);
213  case ExtensionExpressionType.Collection:
214  return new CollectionExpression(
215  ((CollectionExpression)serializationExpression).CollectionNamespace,
216  serializer);
217  case ExtensionExpressionType.Document:
218  return new DocumentExpression(serializer);
219  case ExtensionExpressionType.Field:
220  return ((FieldExpression)serializationExpression).WithSerializer(serializer);
221  case ExtensionExpressionType.FieldAsDocument:
222  return new FieldAsDocumentExpression(
223  ((FieldAsDocumentExpression)serializationExpression).Expression,
224  ((FieldAsDocumentExpression)serializationExpression).FieldName,
225  serializer);
226  }
227  }
228  }
229  }
230 
231  return newNode;
232  }

235  private Expression BindElementAt(MethodCallExpression node)
236  {
237  if (!ExpressionHelper.IsLinqMethod(node))
238  {
239  return base.VisitMethodCall(node);
240  }
241 
242  var newNode = base.VisitMethodCall(node);
243  var methodCallExpression = newNode as MethodCallExpression;
244  if (node != newNode && methodCallExpression != null)
245  {
246  var serializationExpression = methodCallExpression.Arguments[0] as ISerializationExpression;
247  if (serializationExpression != null)
248  {
249  var arraySerializer = serializationExpression.Serializer as IBsonArraySerializer;
250  BsonSerializationInfo itemSerializationInfo;
251  if (arraySerializer != null && arraySerializer.TryGetItemSerializationInfo(out itemSerializationInfo))
252  {
253  newNode = new ArrayIndexExpression(
254  methodCallExpression.Arguments[0],
255  methodCallExpression.Arguments[1],
256  itemSerializationInfo.Serializer,
257  methodCallExpression);
258  }
259  }
260  }
261 
262  return newNode;
263  }

265  private Expression BindEmbeddedPipeline(MethodCallExpression node)
266  {
267  var oldIsInEmbeddedPipeline = _isInEmbeddedPipeline;
268  _isInEmbeddedPipeline = true;
269 
270  node = node.Update(
271  Visit(node.Object),
272  Visit(node.Arguments));
273 
274  _isInEmbeddedPipeline = oldIsInEmbeddedPipeline;
275  if (_isInEmbeddedPipeline)
276  {
277  return node;
278  }
279 
280  // we need to discover if this is rooted at an IMongoQueryable... If so, it
281  // gets processed as a top-level pipeline...
282 
283  return EmbeddedPipelineBinder.Bind(node, _bindingContext);
284  }

286  private Expression BindGetItem(MethodCallExpression node)
287  {
288  var newNode = base.VisitMethodCall(node);
289  var methodCallExpression = newNode as MethodCallExpression;
290  if (node == newNode ||
291  methodCallExpression == null ||
292  node.Object == null ||
293  node.Arguments.Count != 1)
294  {
295  return newNode;
296  }
297 
298  var serializationExpression = methodCallExpression.Object as ISerializationExpression;
299  if (serializationExpression == null)
300  {
301  return newNode;
302  }
303 
304  var indexExpression = methodCallExpression.Arguments[0];
305  switch (Type.GetTypeCode(indexExpression.Type))
306  {
307  case TypeCode.Int16:
308  case TypeCode.Int32:
309  case TypeCode.Int64:
310  case TypeCode.UInt16:
311  case TypeCode.UInt32:
312  case TypeCode.UInt64:
313  var arraySerializer = serializationExpression.Serializer as IBsonArraySerializer;
314  BsonSerializationInfo itemSerializationInfo;
315  if (arraySerializer != null && arraySerializer.TryGetItemSerializationInfo(out itemSerializationInfo))
316  {
317  return new ArrayIndexExpression(
318  methodCallExpression.Object,
319  indexExpression,
320  itemSerializationInfo.Serializer,
321  methodCallExpression);
322  }
323  break;
324  case TypeCode.String:
325  var index = indexExpression as ConstantExpression;
326  if (index != null)
327  {
328  var documentSerializer = serializationExpression.Serializer as IBsonDocumentSerializer;
329  BsonSerializationInfo memberSerializationInfo;
330  if (documentSerializer != null && documentSerializer.TryGetMemberSerializationInfo(index.Value.ToString(), out memberSerializationInfo))
331  {
332  return new FieldExpression(
333  methodCallExpression.Object,
334  memberSerializationInfo.ElementName,
335  memberSerializationInfo.Serializer,
336  methodCallExpression);
337  }
338  }
339  break;
340  }
341 
342  return node; // return the original node because we can't translate this expression.
343  }

346  private Expression BindInject(MethodCallExpression node)
347  {
348  if (node.Method.DeclaringType == typeof(LinqExtensions))
349  {
350  var arg = node.Arguments[0] as ConstantExpression;
351  if (arg != null)
352  {
353  var docType = node.Method.GetGenericArguments()[0];
354  var serializer = _bindingContext.GetSerializer(node.Method.GetGenericArguments()[0], arg);
355  var renderedFilter = (BsonDocument)typeof(FilterDefinition<>).MakeGenericType(docType)
356  .GetTypeInfo()
357  .GetMethod("Render")
358  .Invoke(arg.Value, new object[] { serializer, _bindingContext.SerializerRegistry });
359 
360  return new InjectedFilterExpression(renderedFilter);
361  }
362  }
363 
364  return node;
365  }

367  private Expression BindClientSideProjector(MethodCallExpression node)
368  {
369  var arguments = new List<Expression>();
370  arguments.Add(Visit(node.Arguments[0]));
371 
372  // we need to make sure that the serialization info for the parameter
373  // is the item serialization from the parent IBsonArraySerializer
374  var fieldExpression = arguments[0] as IFieldExpression;
375  if (fieldExpression != null)
376  {
377  var arraySerializer = fieldExpression.Serializer as IBsonArraySerializer;
378  BsonSerializationInfo itemSerializationInfo;
379  if (arraySerializer != null && arraySerializer.TryGetItemSerializationInfo(out itemSerializationInfo))
380  {
381  var lambda = ExpressionHelper.GetLambda(node.Arguments[1]);
382  _bindingContext.AddExpressionMapping(
383  lambda.Parameters[0],
384  new FieldExpression(
385  fieldExpression.FieldName,
386  itemSerializationInfo.Serializer,
387  lambda.Parameters[0]));
388  }
389  }
390 
391  arguments.Add(Visit(node.Arguments[1]));
392 
393  if (node.Arguments[0] != arguments[0] || node.Arguments[1] != arguments[1])
394  {
395  node = Expression.Call(node.Method, arguments.ToArray());
396  }
397 
398  return node;
399  }
