13  /* The argument is an array of matcher strings each of which determine
14  whether a property with a certain path needs to be decorated with a
15  comment.
16 
17  A path is a number of path elements concatenated by '/'. The last path
18  element is either:
19  the key if we're referring to a dict key
20  the value if we're referring to a value in an array
21  the value if we're referring to a value in a dict
22  All other path elements are either:
23  the key if the container is dict
24  '*' if the container is array
25 
26  Path matcher has the same structure as a path, except that any of the
27  elements may be '*'. Matcher matches a path if both have the same number
28  of elements and for each pair matcher element is the same as path element
29  or is '*'.
30 
31  a/b/c matches a/b/c but not a/b nor a/b/c/d
32  a/* /c matches a/d/c but not a/b nor a/b/c/d
33  * /* /* matches any path from three elements
34  */

38  m_All = false;
39  m_Props = props;

49  public PropertyCommentChecker(IEnumerable<string> props)
50  {
51  m_Level = 0;
52  m_All = false;
53  m_Props = new List<List<string>>();
54  foreach (var prop in props)
55  {
56  m_Props.Add(new List<string>(prop.Split('/')));
57  }
58  }
59 
60  bool CheckContained(string prop)
61  {
62  if (m_All)
63  return true;
64  foreach (var list in m_Props)
65  {
66  if (list.Count == m_Level+1)
67  {
68  if (list[m_Level] == prop)
69  return true;
70  if (list[m_Level] == "*")
71  {
72  m_All = true; // short-circuit all at this level
73  return true;
74  }
75  }
76  }
77  return false;
78  }
79 
80  public bool CheckStringValueInArray(string value) { return CheckContained(value); }
81  public bool CheckKeyInDict(string key) { return CheckContained(key); }
82 
83  public bool CheckStringValueInDict(string key, string value)
84  {
85  foreach (var list in m_Props)
86  {
87  if (list.Count == m_Level + 2)
88  {
89  if ((list[m_Level] == "*" || list[m_Level] == key) &&
90  list[m_Level+1] == "*" || list[m_Level+1] == value)
91  return true;
92  }
93  }
94  return false;
95  }
96 
97  public PropertyCommentChecker NextLevel(string prop)
98  {
99  var newList = new List<List<string>>();
100  foreach (var list in m_Props)
101  {
102  if (list.Count <= m_Level+1)
103  continue;
104  if (list[m_Level] == "*" || list[m_Level] == prop)
105  newList.Add(list);
106  }
107  return new PropertyCommentChecker(m_Level + 1, newList);
108  }
109  }

146  public static PBXElementString ParseIdentifierAST(IdentifierAST ast, TokenList tokens, string text)
147  {
148  Token tok = tokens[ast.value];
149  string value;
150  switch (tok.type)
151  {
152  case TokenType.String:
153  value = text.Substring(tok.begin, tok.end - tok.begin);
154  return new PBXElementString(value);
155  case TokenType.QuotedString:
156  value = text.Substring(tok.begin, tok.end - tok.begin);
157  value = PBXStream.UnquoteString(value);
158  return new PBXElementString(value);
159  default:
160  throw new Exception("Internal parser error");
161  }
162  }
