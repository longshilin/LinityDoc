42  public IBsonSerializer Build(Expression node)
43  {
44  if (node is ISerializationExpression)
45  {
46  return ((ISerializationExpression)node).Serializer;
47  }
48 
49  IBsonSerializer serializer = null;
50  switch (node.NodeType)
51  {
52  case ExpressionType.MemberInit:
53  serializer = BuildMemberInit((MemberInitExpression)node);
54  break;
55  case ExpressionType.New:
56  if (!typeof(IEnumerable).GetTypeInfo().IsAssignableFrom(node.Type))
57  {
58  serializer = BuildNew((NewExpression)node);
59  }
60  break;
61  }
62 
63  if (serializer == null && !PreviouslyUsedSerializerFinder.TryFindSerializer(node, node.Type, out serializer))
64  {
65  serializer = _serializerRegistry.GetSerializer(node.Type);
66  var childConfigurable = serializer as IChildSerializerConfigurable;
67  if (childConfigurable != null)
68  {
69  var arraySerializer = serializer as IBsonArraySerializer;
70  BsonSerializationInfo itemSerializationInfo;
71  if (arraySerializer != null && arraySerializer.TryGetItemSerializationInfo(out itemSerializationInfo))
72  {
73  IBsonSerializer itemSerializer;
74  if (PreviouslyUsedSerializerFinder.TryFindSerializer(node, itemSerializationInfo.Serializer.ValueType, out itemSerializer))
75  {
76  serializer = SerializerHelper.RecursiveConfigureChildSerializer(childConfigurable, itemSerializer);
77  }
78  }
79  else
80  {
81  IBsonSerializer childSerializer;
82  if (PreviouslyUsedSerializerFinder.TryFindSerializer(node, childConfigurable.ChildSerializer.ValueType, out childSerializer))
83  {
84  serializer = childConfigurable.WithChildSerializer(childSerializer);
85  }
86  }
87  }
88  }
89 
90  return serializer;
91  }

110  private IBsonSerializer BuildProjectedSerializer(ProjectionMapping mapping)
111  {
112  // We are building a serializer specifically for a projected type based
113  // on serialization information collected from other serializers.
114  // We cannot cache this in the serializer registry because the compiler reuses
115  // the same anonymous type definition in different contexts as long as they
116  // are structurally equatable. As such, it might be that two different queries
117  // projecting the same shape might need to be deserialized differently.
118  var classMap = BuildClassMap(mapping.Expression.Type, mapping);
119 
120  var mappedParameters = mapping.Members
121  .Where(x => x.Parameter != null)
122  .OrderBy(x => x.Parameter.Position)
123  .Select(x => x.Member)
124  .ToList();
125 
126  if (mappedParameters.Count > 0)
127  {
128  classMap.MapConstructor(mapping.Constructor)
129  .SetArguments(mappedParameters);
130  }
131 
132  var serializerType = typeof(BsonClassMapSerializer<>).MakeGenericType(mapping.Expression.Type);
133  return (IBsonSerializer)Activator.CreateInstance(serializerType, classMap.Freeze());
134  }
135 
136  private BsonClassMap BuildClassMap(Type type, ProjectionMapping mapping)
137  {
138  if (type == null || type == typeof(object))
139  {
140  return null;
141  }
142 
143  var baseClassMap = BuildClassMap(type.GetTypeInfo().BaseType, mapping);
144  if (baseClassMap != null)
145  {
146  baseClassMap.Freeze();
147  }
148  var classMap = new BsonClassMap(type, baseClassMap);
149 
150  foreach (var memberMapping in mapping.Members.Where(x => x.Member.DeclaringType == type))
151  {
152  var serializationExpression = memberMapping.Expression as SerializationExpression;
153  if (serializationExpression == null)
154  {
155  var serializer = Build(memberMapping.Expression);
156  serializationExpression = new FieldExpression(
157  memberMapping.Member.Name,
158  serializer,
159  memberMapping.Expression);
160  }
161 
162  var memberMap = classMap.MapMember(memberMapping.Member)
163  .SetSerializer(serializationExpression.Serializer)
164  .SetElementName(memberMapping.Member.Name);
165 
166  if (classMap.IdMemberMap == null && serializationExpression is GroupingKeyExpression)
167  {
168  classMap.SetIdMember(memberMap);
169  }
170  }
171 
172  return classMap;
173  }

175  private static Type GetMemberType(MemberInfo memberInfo)
176  {
177  FieldInfo fieldInfo;
178  if ((fieldInfo = memberInfo as FieldInfo) != null)
179  {
180  return fieldInfo.FieldType;
181  }
182 
183  PropertyInfo propertyInfo;
184  if ((propertyInfo = memberInfo as PropertyInfo) != null)
185  {
186  return propertyInfo.PropertyType;
187  }
188 
189  throw new MongoInternalException("Can't get member type.");
190  }
191  }
