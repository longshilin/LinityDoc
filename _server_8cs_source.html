135  public IChannelHandle GetChannel(CancellationToken cancellationToken)
136  {
137  ThrowIfNotOpen();
138 
139  var connection = _connectionPool.AcquireConnection(cancellationToken);
140  try
141  {
142  // ignoring the user's cancellation token here because we don't
143  // want to throw this connection away simply because the user
144  // wanted to cancel their operation. It will be better for the
145  // collective to complete opening the connection than the throw
146  // it away.
147  connection.Open(CancellationToken.None);
148  return new ServerChannel(this, connection);
149  }
150  catch (Exception ex)
151  {
152  if (ShouldClearConnectionPool(ex))
153  {
154  try
155  {
156  _connectionPool.Clear();
157  }
158  catch
159  {
160  // ignore exceptions
161  }
162 
163  }
164  connection.Dispose();
165  throw;
166  }
167  }

237  private void OnDescriptionChanged(object sender, ServerDescriptionChangedEventArgs e)
238  {
239  if (e.NewServerDescription.HeartbeatException != null)
240  {
241  _connectionPool.Clear();
242  }
243 
244  if (_descriptionChangedEventHandler != null)
245  {
246  _descriptionChangedEventHandler(new ServerDescriptionChangedEvent(e.OldServerDescription, e.NewServerDescription));
247  }
248 
249  var handler = DescriptionChanged;
250  if (handler != null)
251  {
252  try { handler(this, e); }
253  catch { } // ignore exceptions
254  }

257  private void HandleChannelException(IConnection connection, Exception ex)
258  {
259  if (_state.Value != State.Open)
260  {
261  return;
262  }
263 
264  var aggregateException = ex as AggregateException;
265  if (aggregateException != null && aggregateException.InnerExceptions.Count == 1)
266  {
267  ex = aggregateException.InnerException;
268  }
269 
270  // For most connection exceptions, we are going to immediately
271  // invalidate the server. However, we aren't going to invalidate
272  // because of OperationCanceledExceptions. We trust that the
273  // implementations of connection don't leave themselves in a state
274  // where they can't be used based on user cancellation.
275  if (ex.GetType() == typeof(OperationCanceledException))
276  {
277  return;
278  }
279 
280  if (ShouldInvalidateServer(ex))
281  {
282  Invalidate();
283  }
284  else
285  {
286  RequestHeartbeat();
287  }
288  }

290  private bool IsNotMaster(ServerErrorCode code, string message)
291  {
292  switch (code)
293  {
294  case ServerErrorCode.NotMaster: // 10107
295  case ServerErrorCode.NotMasterNoSlaveOk: // 13435
296  return true;
297  }
298 
299  if (message != null)
300  {
301  if (message.IndexOf("not master", StringComparison.OrdinalIgnoreCase) != -1 &&
302  message.IndexOf("not master or secondary", StringComparison.OrdinalIgnoreCase) == -1)
303  {
304  return true;
305  }
306  }
307 
308  return false;
309  }
310 
311  private bool IsNotMasterOrRecovering(ServerErrorCode code, string message)
312  {
313  return IsNotMaster(code, message) || IsRecovering(code, message);
314  }
315 
316  private bool IsRecovering(ServerErrorCode code, string message)
317  {
318  switch (code)
319  {
320  case ServerErrorCode.InterruptedAtShutdown: // 11600
321  case ServerErrorCode.InterruptedDueToReplStateChange: // 11602
322  case ServerErrorCode.NotMasterOrSecondary: // 13436
323  case ServerErrorCode.PrimarySteppedDown: // 189
324  case ServerErrorCode.ShutdownInProgress: // 91
325  return true;
326  }
327 
328  if (message != null)
329  {
330  if (message.IndexOf("not master or secondary", StringComparison.OrdinalIgnoreCase) != -1 ||
331  message.IndexOf("node is recovering", StringComparison.OrdinalIgnoreCase) != -1)
332  {
333  return true;
334  }
335  }
336 
337  return false;
338  }

340  private bool ShouldClearConnectionPool(Exception ex)
341  {
342  return ex is MongoAuthenticationException;
343  }

345  private bool ShouldInvalidateServer(Exception exception)
346  {
347  if (__invalidatingExceptions.Contains(exception.GetType()))
348  {
349  return true;
350  }
351 
352  var commandException = exception as MongoCommandException;
353  if (commandException != null)
354  {
355  var code = (ServerErrorCode)commandException.Code;
356  var message = commandException.ErrorMessage;
357 
358  if (IsNotMasterOrRecovering(code, message))
359  {
360  return true;
361  }
362 
363  if (commandException.GetType() == typeof(MongoWriteConcernException))
364  {
365  var writeConcernException = (MongoWriteConcernException)commandException;
366  var writeConcernResult = writeConcernException.WriteConcernResult;
367  var response = writeConcernResult.Response;
368  var writeConcernError = response["writeConcernError"].AsBsonDocument;
369  if (writeConcernError != null)
370  {
371  code = (ServerErrorCode)writeConcernError.GetValue("code", -1).ToInt32();
372  message = writeConcernError.GetValue("errmsg", null)?.AsString;
373 
374  if (IsNotMasterOrRecovering(code, message))
375  {
376  return true;
377  }
378  }
379  }
380  }
381 
382  return false;
383  }
