110  private async Task MonitorServerAsync()
111  {
112  var metronome = new Metronome(_heartbeatInterval);
113  var heartbeatCancellationToken = _cancellationTokenSource.Token;
114  while (!heartbeatCancellationToken.IsCancellationRequested)
115  {
116  try
117  {
118  try
119  {
120  await HeartbeatAsync(heartbeatCancellationToken).ConfigureAwait(false);
121  }
122  catch (OperationCanceledException) when (heartbeatCancellationToken.IsCancellationRequested)
123  {
124  // ignore OperationCanceledException when heartbeat cancellation is requested
125  }
126  catch (Exception unexpectedException)
127  {
128  // if we catch an exception here it's because of a bug in the driver (but we need to defend ourselves against that)
129 
130  var handler = _sdamInformationEventHandler;
131  if (handler != null)
132  {
133  try
134  {
135  handler.Invoke(new SdamInformationEvent(() =>
136  string.Format(
137  "Unexpected exception in ServerMonitor.MonitorServerAsync: {0}",
138  unexpectedException.ToString())));
139  }
140  catch
141  {
142  // ignore any exceptions thrown by the handler (note: event handlers aren't supposed to throw exceptions)
143  }
144  }
145 
146  // since an unexpected exception was thrown set the server description to Unknown (with the unexpected exception)
147  try
148  {
149  // keep this code as simple as possible to keep the surface area with any remaining possible bugs as small as possible
150  var newDescription = _baseDescription.WithHeartbeatException(unexpectedException); // not With in case the bug is in With
151  SetDescription(newDescription); // not SetDescriptionIfChanged in case the bug is in SetDescriptionIfChanged
152  }
153  catch
154  {
155  // if even the simple code in the try throws just give up (at least we've raised the unexpected exception via an SdamInformationEvent)
156  }
157  }
158 
159  var newHeartbeatDelay = new HeartbeatDelay(metronome.GetNextTickDelay(), __minHeartbeatInterval);
160  var oldHeartbeatDelay = Interlocked.Exchange(ref _heartbeatDelay, newHeartbeatDelay);
161  if (oldHeartbeatDelay != null)
162  {
163  oldHeartbeatDelay.Dispose();
164  }
165  await newHeartbeatDelay.Task.ConfigureAwait(false);
166  }
167  catch
168  {
169  // ignore these exceptions
170  }
171  }
172  }

174  private async Task<bool> HeartbeatAsync(CancellationToken cancellationToken)
175  {
176  const int maxRetryCount = 2;
177  HeartbeatInfo heartbeatInfo = null;
178  Exception heartbeatException = null;
179  for (var attempt = 1; attempt <= maxRetryCount; attempt++)
180  {
181  var connection = _connection;
182  try
183  {
184  if (connection == null)
185  {
186  connection = _connectionFactory.CreateConnection(_serverId, _endPoint);
187  // if we are cancelling, it's because the server has
188  // been shut down and we really don't need to wait.
189  await connection.OpenAsync(cancellationToken).ConfigureAwait(false);
190  }
191 
192  heartbeatInfo = await GetHeartbeatInfoAsync(connection, cancellationToken).ConfigureAwait(false);
193  heartbeatException = null;
194 
195  _connection = connection;
196  break;
197  }
198  catch (Exception ex)
199  {
200  heartbeatException = ex;
201  _connection = null;
202  if (connection != null)
203  {
204  connection.Dispose();
205  }
206  }
207  }
208 
209  ServerDescription newDescription;
210  if (heartbeatInfo != null)
211  {
212  var averageRoundTripTime = _averageRoundTripTimeCalculator.AddSample(heartbeatInfo.RoundTripTime);
213  var averageRoundTripTimeRounded = TimeSpan.FromMilliseconds(Math.Round(averageRoundTripTime.TotalMilliseconds));
214  var isMasterResult = heartbeatInfo.IsMasterResult;
215  var buildInfoResult = heartbeatInfo.BuildInfoResult;
216 
217  newDescription = _baseDescription.With(
218  averageRoundTripTime: averageRoundTripTimeRounded,
219  canonicalEndPoint: isMasterResult.Me,
220  electionId: isMasterResult.ElectionId,
221  lastWriteTimestamp: isMasterResult.LastWriteTimestamp,
222  logicalSessionTimeout: isMasterResult.LogicalSessionTimeout,
223  maxBatchCount: isMasterResult.MaxBatchCount,
224  maxDocumentSize: isMasterResult.MaxDocumentSize,
225  maxMessageSize: isMasterResult.MaxMessageSize,
226  replicaSetConfig: isMasterResult.GetReplicaSetConfig(),
227  state: ServerState.Connected,
228  tags: isMasterResult.Tags,
229  type: isMasterResult.ServerType,
230  version: buildInfoResult.ServerVersion,
231  wireVersionRange: new Range<int>(isMasterResult.MinWireVersion, isMasterResult.MaxWireVersion));
232  }
233  else
234  {
235  newDescription = _baseDescription;
236  }
237 
238  if (heartbeatException != null)
239  {
240  newDescription = newDescription.With(heartbeatException: heartbeatException);
241  }
242 
243  SetDescriptionIfChanged(newDescription);
244 
245  return true;
246  }

248  private async Task<HeartbeatInfo> GetHeartbeatInfoAsync(IConnection connection, CancellationToken cancellationToken)
249  {
250  cancellationToken.ThrowIfCancellationRequested();
251  if (_heartbeatStartedEventHandler != null)
252  {
253  _heartbeatStartedEventHandler(new ServerHeartbeatStartedEvent(connection.ConnectionId));
254  }
255 
256  try
257  {
258  var isMasterCommand = new CommandWireProtocol<BsonDocument>(
259  DatabaseNamespace.Admin,
260  new BsonDocument("isMaster", 1),
261  true,
262  BsonDocumentSerializer.Instance,
263  null);
264 
265  var stopwatch = Stopwatch.StartNew();
266  var isMasterResultDocument = await isMasterCommand.ExecuteAsync(connection, cancellationToken).ConfigureAwait(false);
267  stopwatch.Stop();
268  var isMasterResult = new IsMasterResult(isMasterResultDocument);
269 
270  var buildInfoCommand = new CommandWireProtocol<BsonDocument>(
271  DatabaseNamespace.Admin,
272  new BsonDocument("buildInfo", 1),
273  true,
274  BsonDocumentSerializer.Instance,
275  null);
276 
277  var buildInfoResultRocument = await buildInfoCommand.ExecuteAsync(connection, cancellationToken).ConfigureAwait(false);
278  var buildInfoResult = new BuildInfoResult(buildInfoResultRocument);
279 
280  if (_heartbeatSucceededEventHandler != null)
281  {
282  _heartbeatSucceededEventHandler(new ServerHeartbeatSucceededEvent(connection.ConnectionId, stopwatch.Elapsed));
283  }
284 
285  return new HeartbeatInfo
286  {
287  RoundTripTime = stopwatch.Elapsed,
288  IsMasterResult = isMasterResult,
289  BuildInfoResult = buildInfoResult
290  };
291  }
292  catch (Exception ex)
293  {
294  if (_heartbeatFailedEventHandler != null)
295  {
296  _heartbeatFailedEventHandler(new ServerHeartbeatFailedEvent(connection.ConnectionId, ex));
297  }
298  throw;
299  }
300  }
