119  public List<Vector3> CurvedNonuniform (List<Vector3> path) {
120  if (maxSegmentLength <= 0) {
121  Debug.LogWarning("Max Segment Length is <= 0 which would cause DivByZero-exception or other nasty errors (avoid this)");
122  return path;
123  }

125  int pointCounter = 0;

127  //pointCounter += Mathf.FloorToInt ((path[i]-path[i+1]).magnitude / maxSegmentLength)+1;
128

130  //In order to avoid floating point errors as much as possible, and in lack of a better solution
131  //loop through it EXACTLY as the other code further down will
132  for (float t = 0; t <= dist; t += maxSegmentLength) {
133  pointCounter++;
134  }
135  }

137  List<Vector3> subdivided = ListPool<Vector3>.Claim(pointCounter);
138 
139  // Set first velocity
140  Vector3 preEndVel = (path[1]-path[0]).normalized;
141 
142  for (int i = 0; i < path.Count-1; i++) {
143  float dist = (path[i]-path[i+1]).magnitude;
144 
145  Vector3 startVel1 = preEndVel;
146  Vector3 endVel1 = i < path.Count-2 ? ((path[i+2]-path[i+1]).normalized - (path[i]-path[i+1]).normalized).normalized : (path[i+1]-path[i]).normalized;
147 
148  Vector3 startVel = startVel1 * dist * factor;
149  Vector3 endVel = endVel1 * dist * factor;
150 
151  Vector3 start = path[i];
152  Vector3 end = path[i+1];
153 
154  float onedivdist = 1F / dist;
155 
156  for (float t = 0; t <= dist; t += maxSegmentLength) {
157  float t2 = t * onedivdist;
158

160  }

170  public static Vector3 GetPointOnCubic (Vector3 a, Vector3 b, Vector3 tan1, Vector3 tan2, float t) {
171  float t2 = t*t, t3 = t2*t;
172 
173  float h1 = 2*t3 - 3*t2 + 1; // calculate basis function 1
174  float h2 = -2*t3 + 3*t2; // calculate basis function 2
175  float h3 = t3 - 2*t2 + t; // calculate basis function 3
176  float h4 = t3 - t2; // calculate basis function 4
177 
178  return h1*a + // multiply and sum all funtions
179  h2*b + // together to build the interpolated
180  h3*tan1 + // point along the curve.
181  h4*tan2;
182  }

184  public List<Vector3> SmoothOffsetSimple (List<Vector3> path) {
185  if (path.Count <= 2 || iterations <= 0) {
186  return path;
187  }
188 
189  if (iterations > 12) {
190  Debug.LogWarning("A very high iteration count was passed, won't let this one through");
191  return path;
192  }
193 
194  int maxLength = (path.Count-2)*(int)Mathf.Pow(2, iterations)+2;
195 
196  List<Vector3> subdivided = ListPool<Vector3>.Claim(maxLength);
197  List<Vector3> subdivided2 = ListPool<Vector3>.Claim(maxLength);
198 
199  for (int i = 0; i < maxLength; i++) { subdivided.Add(Vector3.zero); subdivided2.Add(Vector3.zero); }
200 
201  for (int i = 0; i < path.Count; i++) {
202  subdivided[i] = path[i];
203  }
204 
205  for (int iteration = 0; iteration < iterations; iteration++) {
206  int currentPathLength = (path.Count-2)*(int)Mathf.Pow(2, iteration)+2;
207 
208  //Switch the arrays
209  List<Vector3> tmp = subdivided;
210  subdivided = subdivided2;
211  subdivided2 = tmp;
212 
213  const float nextMultiplier = 1F;
214 
215  for (int i = 0; i < currentPathLength-1; i++) {
216  Vector3 current = subdivided2[i];
217  Vector3 next = subdivided2[i+1];
218 
219  Vector3 normal = Vector3.Cross(next-current, Vector3.up);
220  normal = normal.normalized;
221 
222  bool firstRight = false;
223  bool secondRight = false;
224  bool setFirst = false;
225  bool setSecond = false;
226  if (i != 0 && !VectorMath.IsColinearXZ(current, next, subdivided2[i-1])) {
227  setFirst = true;
228  firstRight = VectorMath.RightOrColinearXZ(current, next, subdivided2[i-1]);
229  }
230  if (i < currentPathLength-1 && !VectorMath.IsColinearXZ(current, next, subdivided2[i+2])) {
231  setSecond = true;
232  secondRight = VectorMath.RightOrColinearXZ(current, next, subdivided2[i+2]);
233  }
234 
235  if (setFirst) {
236  subdivided[i*2] = current + (firstRight ? normal*offset*nextMultiplier : -normal*offset*nextMultiplier);
237  } else {
238  subdivided[i*2] = current;
239  }
240 
241  if (setSecond) {
242  subdivided[i*2+1] = next + (secondRight ? normal*offset*nextMultiplier : -normal*offset*nextMultiplier);
243  } else {
244  subdivided[i*2+1] = next;
245  }
246  }
247 
248  subdivided[(path.Count-2)*(int)Mathf.Pow(2, iteration+1)+2-1] = subdivided2[currentPathLength-1];
249  }
250

252

256  public List<Vector3> SmoothSimple (List<Vector3> path) {
257  if (path.Count < 2) return path;
258 
259  List<Vector3> subdivided;
260 
261  if (uniformLength) {
262  // Clamp to a small value to avoid the path being divided into a huge number of segments
263  maxSegmentLength = Mathf.Max(maxSegmentLength, 0.005f);
264 
265  float pathLength = 0;
266  for (int i = 0; i < path.Count-1; i++) {
267  pathLength += Vector3.Distance(path[i], path[i+1]);
268  }
269 
270  int estimatedNumberOfSegments = Mathf.FloorToInt(pathLength / maxSegmentLength);
271  // Get a list with an initial capacity high enough so that we can add all points
272  subdivided = ListPool<Vector3>.Claim(estimatedNumberOfSegments+2);
273 
274  float distanceAlong = 0;
275 
276  // Sample points every [maxSegmentLength] world units along the path
277  for (int i = 0; i < path.Count-1; i++) {
278  var start = path[i];
279  var end = path[i+1];
280 
281  float length = Vector3.Distance(start, end);
282 
283  while (distanceAlong < length) {
284  subdivided.Add(Vector3.Lerp(start, end, distanceAlong / length));
285  distanceAlong += maxSegmentLength;
286  }
287 
288  distanceAlong -= length;
289  }
290 
291  // Make sure we get the exact position of the last point
292  subdivided.Add(path[path.Count-1]);
293  } else {
294  subdivisions = Mathf.Max(subdivisions, 0);
295 
296  if (subdivisions > 10) {
297  Debug.LogWarning("Very large number of subdivisions. Cowardly refusing to subdivide every segment into more than " + (1 << subdivisions) + " subsegments");
298  subdivisions = 10;
299  }
300 
301  int steps = 1 << subdivisions;
302  subdivided = ListPool<Vector3>.Claim((path.Count-1)*steps + 1);
303  Polygon.Subdivide(path, subdivided, steps);
304  }
305 
306  if (strength > 0) {
307  for (int it = 0; it < iterations; it++) {
308  Vector3 prev = subdivided[0];
309 
310  for (int i = 1; i < subdivided.Count-1; i++) {
311  Vector3 tmp = subdivided[i];
312 
313  // prev is at this point set to the value that subdivided[i-1] had before this loop started
314  // Move the point closer to the average of the adjacent points
315  subdivided[i] = Vector3.Lerp(tmp, (prev+subdivided[i+1].ToUnityV3())/2F, strength);
316 
317  prev = tmp;
318  }
319  }
320  }
321 
322  return subdivided;
323  }

325  public List<Vector3> SmoothBezier (List<Vector3> path) {
326  if (subdivisions < 0) subdivisions = 0;
327 
328  int subMult = 1 << subdivisions;
329  List<Vector3> subdivided = ListPool<Vector3>.Claim();
330 
331  for (int i = 0; i < path.Count-1; i++) {
332  Vector3 tangent1;
333  Vector3 tangent2;
334  if (i == 0) {
335  tangent1 = path[i+1]-path[i];
336  } else {
337  tangent1 = path[i+1]-path[i-1];
338  }
339 
340  if (i == path.Count-2) {
341  tangent2 = path[i]-path[i+1];
342  } else {
343  tangent2 = path[i]-path[i+2];
344  }
345

347  tangent2 *= bezierTangentLength;
348

350  Vector3 v2 = v1+tangent1;

352  Vector3 v3 = v4+tangent2;
353

355  subdivided.Add(AstarSplines.CubicBezier(v1, v2, v3, v4, (float)j/subMult));
356  }
357  }
