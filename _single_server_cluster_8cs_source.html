141  private bool IsServerValidForCluster(ClusterType clusterType, ClusterConnectionMode connectionMode, ServerType serverType)
142  {
143  switch (clusterType)

145  case ClusterType.ReplicaSet:
146  return serverType.IsReplicaSetMember();
147 
148  case ClusterType.Sharded:

150 
151  case ClusterType.Standalone:
152  return serverType == ServerType.Standalone;
153 
154  case ClusterType.Unknown:
155  switch (connectionMode)
156  {
157  case ClusterConnectionMode.Automatic:
158  case ClusterConnectionMode.Direct:
159  return true;

163  }
164

175  private void ServerDescriptionChanged(object sender, ServerDescriptionChangedEventArgs args)
176  {
177  var newServerDescription = args.NewServerDescription;
178  var newClusterDescription = Description;
179 
180  if (newServerDescription.State == ServerState.Disconnected)
181  {
182  newClusterDescription = newClusterDescription.WithServerDescription(newServerDescription);
183  }
184  else
185  {
186  if (IsServerValidForCluster(newClusterDescription.Type, Settings.ConnectionMode, newServerDescription.Type))
187  {
188  if (newClusterDescription.Type == ClusterType.Unknown)
189  {
190  newClusterDescription = newClusterDescription.WithType(newServerDescription.Type.ToClusterType());
191  }
192 
193  newClusterDescription = newClusterDescription.WithServerDescription(newServerDescription);
194  }
195  else

198  }
