187  internal sealed class CombinedSortDefinition<TDocument> : SortDefinition<TDocument>
188  {
189  private readonly List<SortDefinition<TDocument>> _sorts;
190 
191  public CombinedSortDefinition(IEnumerable<SortDefinition<TDocument>> sorts)
192  {
193  _sorts = Ensure.IsNotNull(sorts, nameof(sorts)).ToList();
194  }
195 
196  public override BsonDocument Render(IBsonSerializer<TDocument> documentSerializer, IBsonSerializerRegistry serializerRegistry)
197  {
198  var document = new BsonDocument();
199 
200  foreach (var sort in _sorts)
201  {
202  var renderedSort = sort.Render(documentSerializer, serializerRegistry);
203 
204  foreach (var element in renderedSort.Elements)
205  {
206  // the last sort always wins, and we need to make sure that order is preserved.
207  document.Remove(element.Name);
208  document.Add(element);
209  }
210  }
211 
212  return document;
213  }
214  }

217  {

220 
221  public DirectionalSortDefinition(FieldDefinition<TDocument> field, SortDirection direction)
222  {

224  _direction = direction;
225  }

227  public override BsonDocument Render(IBsonSerializer<TDocument> documentSerializer, IBsonSerializerRegistry serializerRegistry)
228  {
229  var renderedField = _field.Render(documentSerializer, serializerRegistry);
230 
231  BsonValue value;
232  switch (_direction)
233  {
234  case SortDirection.Ascending:
235  value = 1;
236  break;
237  case SortDirection.Descending:
238  value = -1;
239  break;
240  default:
241  throw new InvalidOperationException("Unknown value for " + typeof(SortDirection) + ".");
242  }
243 
244  return new BsonDocument(renderedField.FieldName, value);
245  }
246  }
