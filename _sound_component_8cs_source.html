258  public bool AddSoundGroup(string soundGroupName, bool soundGroupAvoidBeingReplacedBySamePriority, bool soundGroupMute, float soundGroupVolume, int soundAgentHelperCount)
259  {
260  if (m_SoundManager.HasSoundGroup(soundGroupName))
261  {
262  return false;
263  }
264 
265  SoundGroupHelperBase soundGroupHelper = Helper.CreateHelper(m_SoundGroupHelperTypeName, m_CustomSoundGroupHelper, SoundGroupCount);
266  if (soundGroupHelper == null)
267  {
268  Log.Error("Can not create sound group helper.");
269  return false;
270  }
271 
272  soundGroupHelper.name = Utility.Text.Format("Sound Group - {0}", soundGroupName);
273  Transform transform = soundGroupHelper.transform;
274  transform.SetParent(m_InstanceRoot);
275  transform.localScale = Vector3.one;
276 
277  if (m_AudioMixer != null)
278  {
279  AudioMixerGroup[] audioMixerGroups = m_AudioMixer.FindMatchingGroups(Utility.Text.Format("Master/{0}", soundGroupName));
280  if (audioMixerGroups.Length > 0)
281  {
282  soundGroupHelper.AudioMixerGroup = audioMixerGroups[0];
283  }
284  else
285  {
286  soundGroupHelper.AudioMixerGroup = m_AudioMixer.FindMatchingGroups("Master")[0];
287  }
288  }
289 
290  if (!m_SoundManager.AddSoundGroup(soundGroupName, soundGroupAvoidBeingReplacedBySamePriority, soundGroupMute, soundGroupVolume, soundGroupHelper))
291  {
292  return false;
293  }
294 
295  for (int i = 0; i < soundAgentHelperCount; i++)
296  {
297  if (!AddSoundAgentHelper(soundGroupName, soundGroupHelper, i))
298  {
299  return false;
300  }
301  }
302 
303  return true;
304  }
