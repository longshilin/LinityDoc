437  public int PlaySound(string soundAssetName, string soundGroupName, int priority, PlaySoundParams playSoundParams, object userData)
438  {
439  if (m_ResourceManager == null)
440  {
441  throw new GameFrameworkException("You must set resource manager first.");
442  }
443 
444  if (m_SoundHelper == null)
445  {
446  throw new GameFrameworkException("You must set sound helper first.");
447  }
448 
449  if (playSoundParams == null)
450  {
451  playSoundParams = PlaySoundParams.Create();
452  }
453 
454  int serialId = ++m_Serial;
455  PlaySoundErrorCode? errorCode = null;
456  string errorMessage = null;
457  SoundGroup soundGroup = (SoundGroup)GetSoundGroup(soundGroupName);
458  if (soundGroup == null)
459  {
460  errorCode = PlaySoundErrorCode.SoundGroupNotExist;
461  errorMessage = Utility.Text.Format("Sound group '{0}' is not exist.", soundGroupName);
462  }
463  else if (soundGroup.SoundAgentCount <= 0)
464  {
465  errorCode = PlaySoundErrorCode.SoundGroupHasNoAgent;
466  errorMessage = Utility.Text.Format("Sound group '{0}' is have no sound agent.", soundGroupName);
467  }
468 
469  if (errorCode.HasValue)
470  {
471  if (m_PlaySoundFailureEventHandler != null)
472  {
473  PlaySoundFailureEventArgs playSoundFailureEventArgs = PlaySoundFailureEventArgs.Create(serialId, soundAssetName, soundGroupName, playSoundParams, errorCode.Value, errorMessage, userData);
474  m_PlaySoundFailureEventHandler(this, playSoundFailureEventArgs);
475  ReferencePool.Release(playSoundFailureEventArgs);
476 
477  if (playSoundParams.Referenced)
478  {
479  ReferencePool.Release(playSoundParams);
480  }
481 
482  return serialId;
483  }
484 
485  throw new GameFrameworkException(errorMessage);
486  }
487 
488  m_SoundsBeingLoaded.Add(serialId);
489  m_ResourceManager.LoadAsset(soundAssetName, priority, m_LoadAssetCallbacks, PlaySoundInfo.Create(serialId, soundGroup, playSoundParams, userData));
490  return serialId;
491  }

614  private void LoadAssetSuccessCallback(string soundAssetName, object soundAsset, float duration, object userData)
615  {
616  PlaySoundInfo playSoundInfo = (PlaySoundInfo)userData;
617  if (playSoundInfo == null)
618  {
619  throw new GameFrameworkException("Play sound info is invalid.");
620  }
621 
622  if (m_SoundsToReleaseOnLoad.Contains(playSoundInfo.SerialId))
623  {
624  m_SoundsToReleaseOnLoad.Remove(playSoundInfo.SerialId);
625  if (playSoundInfo.PlaySoundParams.Referenced)
626  {
627  ReferencePool.Release(playSoundInfo.PlaySoundParams);
628  }
629 
630  ReferencePool.Release(playSoundInfo);
631  m_SoundHelper.ReleaseSoundAsset(soundAsset);
632  return;
633  }
634 
635  m_SoundsBeingLoaded.Remove(playSoundInfo.SerialId);
636 
637  PlaySoundErrorCode? errorCode = null;
638  ISoundAgent soundAgent = playSoundInfo.SoundGroup.PlaySound(playSoundInfo.SerialId, soundAsset, playSoundInfo.PlaySoundParams, out errorCode);
639  if (soundAgent != null)
640  {
641  if (m_PlaySoundSuccessEventHandler != null)
642  {
643  PlaySoundSuccessEventArgs playSoundSuccessEventArgs = PlaySoundSuccessEventArgs.Create(playSoundInfo.SerialId, soundAssetName, soundAgent, duration, playSoundInfo.UserData);
644  m_PlaySoundSuccessEventHandler(this, playSoundSuccessEventArgs);
645  ReferencePool.Release(playSoundSuccessEventArgs);
646  }
647 
648  if (playSoundInfo.PlaySoundParams.Referenced)
649  {
650  ReferencePool.Release(playSoundInfo.PlaySoundParams);
651  }
652 
653  ReferencePool.Release(playSoundInfo);
654  return;
655  }
656 
657  m_SoundsToReleaseOnLoad.Remove(playSoundInfo.SerialId);
658  m_SoundHelper.ReleaseSoundAsset(soundAsset);
659  string errorMessage = Utility.Text.Format("Sound group '{0}' play sound '{1}' failure.", playSoundInfo.SoundGroup.Name, soundAssetName);
660  if (m_PlaySoundFailureEventHandler != null)
661  {
662  PlaySoundFailureEventArgs playSoundFailureEventArgs = PlaySoundFailureEventArgs.Create(playSoundInfo.SerialId, soundAssetName, playSoundInfo.SoundGroup.Name, playSoundInfo.PlaySoundParams, errorCode.Value, errorMessage, playSoundInfo.UserData);
663  m_PlaySoundFailureEventHandler(this, playSoundFailureEventArgs);
664  ReferencePool.Release(playSoundFailureEventArgs);
665 
666  if (playSoundInfo.PlaySoundParams.Referenced)
667  {
668  ReferencePool.Release(playSoundInfo.PlaySoundParams);
669  }
670 
671  ReferencePool.Release(playSoundInfo);
672  return;
673  }
674 
675  if (playSoundInfo.PlaySoundParams.Referenced)
676  {
677  ReferencePool.Release(playSoundInfo.PlaySoundParams);
678  }
679 
680  ReferencePool.Release(playSoundInfo);
681  throw new GameFrameworkException(errorMessage);
682  }
