147  Vector3 Snap (ABPath path, Exactness mode, bool start, out bool forceAddPoint) {
148  var index = start ? 0 : path.path.Count - 1;
149  var node = path.path[index];
150  var nodePos = (Vector3)node.position;
151 
152  forceAddPoint = false;
153 
154  switch (mode) {
155  case Exactness.ClosestOnNode:
156  return start ? path.startPoint : path.endPoint;
157  case Exactness.SnapToNode:
158  return nodePos;
159  case Exactness.Original:
160  case Exactness.Interpolate:
161  case Exactness.NodeConnection:
162  Vector3 relevantPoint;
163  if (start) {
164  relevantPoint = adjustStartPoint != null ? adjustStartPoint().ToPFV3() : path.originalStartPoint;
165  } else {
166  relevantPoint = path.originalEndPoint;
167  }
168 
169  switch (mode) {
170  case Exactness.Original:
171  return GetClampedPoint(nodePos, relevantPoint, node);
172  case Exactness.Interpolate:
173  // Adjacent node to either the start node or the end node in the path
174  var adjacentNode = path.path[Mathf.Clamp(index + (start ? 1 : -1), 0, path.path.Count-1)];
175  return VectorMath.ClosestPointOnSegment(nodePos, (Vector3)adjacentNode.position, relevantPoint);
176  case Exactness.NodeConnection:
177  // This code uses some tricks to avoid allocations
178  // even though it uses delegates heavily
179  // The connectionBufferAddDelegate delegate simply adds whatever node
180  // it is called with to the connectionBuffer
181  connectionBuffer = connectionBuffer ?? new List<GraphNode>();
182  connectionBufferAddDelegate = connectionBufferAddDelegate ?? (System.Action<GraphNode>)connectionBuffer.Add;
183 
184  // Adjacent node to either the start node or the end node in the path
185  adjacentNode = path.path[Mathf.Clamp(index + (start ? 1 : -1), 0, path.path.Count-1)];
186 
187  // Add all neighbours of #node to the connectionBuffer
188  node.GetConnections(connectionBufferAddDelegate);
189  var bestPos = nodePos;
190  var bestDist = float.PositiveInfinity;
191 
192  // Loop through all neighbours
193  // Do it in reverse order because the length of the connectionBuffer
194  // will change during iteration
195  for (int i = connectionBuffer.Count - 1; i >= 0; i--) {
196  var neighbour = connectionBuffer[i];
197 
198  // Find the closest point on the connection between the nodes
199  // and check if the distance to that point is lower than the previous best
200  var closest = VectorMath.ClosestPointOnSegment(nodePos, (Vector3)neighbour.position, relevantPoint);
201 
202  var dist = (closest.ToUnityV3() - relevantPoint).sqrMagnitude;
203  if (dist < bestDist) {
204  bestPos = closest;
205  bestDist = dist;
206 
207  // If this node is not the adjacent node
208  // then the path should go through the start node as well
209  forceAddPoint = neighbour != adjacentNode;
210  }
211  }
212 
213  connectionBuffer.Clear();
214  return bestPos;
215  default:
216  throw new System.ArgumentException("Cannot reach this point, but the compiler is not smart enough to realize that.");
217  }
218  default:
219  throw new System.ArgumentException("Invalid mode");
220  }
