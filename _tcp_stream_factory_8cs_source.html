125  private void Connect(Socket socket, EndPoint endPoint, CancellationToken cancellationToken)
126  {
127  var connected = false;
128  var cancelled = false;
129  var timedOut = false;
130 
131  using (var registration = cancellationToken.Register(() => { if (!connected) { cancelled = true; try { socket.Dispose(); } catch { } } }))
132  using (var timer = new Timer(_ => { if (!connected) { timedOut = true; try { socket.Dispose(); } catch { } } }, null, _settings.ConnectTimeout, Timeout.InfiniteTimeSpan))
133  {
134  try
135  {
136  var dnsEndPoint = endPoint as DnsEndPoint;
137  if (dnsEndPoint != null)
138  {
139  // mono doesn't support DnsEndPoint in its BeginConnect method.
140  socket.Connect(dnsEndPoint.Host, dnsEndPoint.Port);
141  }
142  else
143  {
144  socket.Connect(endPoint);
145  }
146  connected = true;
147  return;
148  }
149  catch
150  {
151  if (!cancelled && !timedOut)
152  {
153  try { socket.Dispose(); } catch { }
154  throw;
155  }
156  }
157  }
158 
159  try { socket.Dispose(); } catch { }
160 
161  cancellationToken.ThrowIfCancellationRequested();
162  if (timedOut)
163  {
164  var message = string.Format("Timed out connecting to {0}. Timeout was {1}.", endPoint, _settings.ConnectTimeout);
165  throw new TimeoutException(message);
166  }
167  }

244  private Socket CreateSocket(EndPoint endPoint)
245  {
246  var addressFamily = endPoint.AddressFamily;
247  if (addressFamily == AddressFamily.Unspecified || addressFamily == AddressFamily.Unknown)
248  {
249  addressFamily = _settings.AddressFamily;
250  }
251 
252  var socket = new Socket(addressFamily, SocketType.Stream, ProtocolType.Tcp);
253 
254  // not all platforms support IOControl
255  try
256  {
257  var keepAliveValues = new KeepAliveValues
258  {
259  OnOff = 1,
260  KeepAliveTime = 300000, // 300 seconds in milliseconds
261  KeepAliveInterval = 10000 // 10 seconds in milliseconds
262  };
263  socket.IOControl(IOControlCode.KeepAliveValues, keepAliveValues.ToBytes(), null);
264  }
265  catch (PlatformNotSupportedException)
266  {
267  // most platforms should support this call to SetSocketOption, but just in case call it in a try/catch also
268  try
269  {
270  socket.SetSocketOption(SocketOptionLevel.Socket, SocketOptionName.KeepAlive, true);
271  }
272  catch (PlatformNotSupportedException)
273  {
274  // ignore PlatformNotSupportedException
275  }
276  }
277 
278  return socket;
279  }

281  private async Task<EndPoint[]> ResolveEndPointsAsync(EndPoint initial)
282  {
283  var dnsInitial = initial as DnsEndPoint;
284  if (dnsInitial == null)
285  {
286  return new[] { initial };
287  }
288 
289  IPAddress address;
290  if (IPAddress.TryParse(dnsInitial.Host, out address))
291  {
292  return new[] { new IPEndPoint(address, dnsInitial.Port) };
293  }
294 
295  var preferred = initial.AddressFamily;
296  if (preferred == AddressFamily.Unspecified || preferred == AddressFamily.Unknown)
297  {
298  preferred = _settings.AddressFamily;
299  }
300 
301  return (await Dns.GetHostAddressesAsync(dnsInitial.Host).ConfigureAwait(false))
302  .Select(x => new IPEndPoint(x, dnsInitial.Port))
303  .OrderBy(x => x, new PreferredAddressFamilyComparer(preferred))
304  .ToArray();
305  }

307  private class PreferredAddressFamilyComparer : IComparer<EndPoint>

309  private readonly AddressFamily _preferred;

311  public PreferredAddressFamilyComparer(AddressFamily preferred)
312  {
313  _preferred = preferred;
314  }
315 
316  public int Compare(EndPoint x, EndPoint y)
317  {
318  if (x.AddressFamily == y.AddressFamily)
319  {
320  return 0;
321  }
322  if (x.AddressFamily == _preferred)
323  {
324  return -1;
325  }
326  else if (y.AddressFamily == _preferred)
327  {
328  return 1;
329  }
330 
331  return 0;
332  }
333  }
