135  internal Int3 ClosestPointOnNodeXZInGraphSpace (Vector3 p) {
136  // Get the vertices that make up the triangle
137  Int3 a, b, c;
138 
139  GetVerticesInGraphSpace(out a, out b, out c);
140 
141  // Convert p to graph space
142  p = GetNavmeshHolder(GraphIndex).transform.InverseTransform(p);
143 
144  // Find the closest point on the triangle to p when looking at the triangle from above (relative to the graph)
145  var closest = Polygon.ClosestPointOnTriangleXZ((Vector3)a, (Vector3)b, (Vector3)c, p);
146 
147  // Make sure the point is actually inside the node
148  var i3closest = (Int3)closest;
149  if (ContainsPointInGraphSpace(i3closest)) {
150  // Common case
151  return i3closest;
152  } else {
153  // Annoying...
154  // The closest point when converted from floating point coordinates to integer coordinates
155  // is not actually inside the node. It needs to be inside the node for some methods
156  // (like for example Linecast) to work properly.
157 
158  // Try the 8 integer coordinates around the closest point
159  // and check if any one of them are completely inside the node.
160  // This will most likely succeed as it should be very close.
161  for (int dx = -1; dx <= 1; dx++) {
162  for (int dz = -1; dz <= 1; dz++) {
163  if ((dx != 0 || dz != 0)) {
164  var candidate = new Int3(i3closest.x + dx, i3closest.y, i3closest.z + dz);
165  if (ContainsPointInGraphSpace(candidate)) return candidate;
166  }
167  }
168  }
169 
170  // Happens veery rarely.
171  // Pick the closest vertex of the triangle.
172  // The vertex is guaranteed to be inside the triangle.
173  var da = (a - i3closest).sqrMagnitudeLong;
174  var db = (b - i3closest).sqrMagnitudeLong;
175  var dc = (c - i3closest).sqrMagnitudeLong;
176  return da < db ? (da < dc ? a : c) : (db < dc ? b : c);
177  }
178  }

225  public override void Open (Path path, PathNode pathNode, PathHandler handler) {
226  if (connections == null) return;
227 
228  // Flag2 indicates if this node needs special treatment
229  // with regard to connection costs
230  bool flag2 = pathNode.flag2;
231 
232  // Loop through all connections
233  for (int i = connections.Length-1; i >= 0; i--) {
234  var conn = connections[i];
235  var other = conn.node;
236 
237  // Make sure we can traverse the neighbour
238  if (path.CanTraverse(conn.node)) {
239  PathNode pathOther = handler.GetPathNode(conn.node);
240 
241  // Fast path out, worth it for triangle mesh nodes since they usually have degree 2 or 3
242  if (pathOther == pathNode.parent) {
243  continue;
244  }
245 
246  uint cost = conn.cost;
247 
248  if (flag2 || pathOther.flag2) {
249  // Get special connection cost from the path
250  // This is used by the start and end nodes
251  cost = path.GetConnectionSpecialCost(this, conn.node, cost);
252  }
253 
254  // Test if we have seen the other node before
255  if (pathOther.pathID != handler.PathID) {
256  // We have not seen the other node before
257  // So the path from the start through this node to the other node
258  // must be the shortest one so far
259 
260  // Might not be assigned
261  pathOther.node = conn.node;
262 
263  pathOther.parent = pathNode;
264  pathOther.pathID = handler.PathID;
265 
266  pathOther.cost = cost;
267 
268  pathOther.H = path.CalculateHScore(other);
269  pathOther.UpdateG(path);
270 
271  handler.heap.Add(pathOther);
272  } else {
273  // If not we can test if the path from this node to the other one is a better one than the one already used
274  if (pathNode.G + cost + path.GetTraversalCost(other) < pathOther.G) {
275  pathOther.cost = cost;
276  pathOther.parent = pathNode;
277 
278  other.UpdateRecursiveG(path, pathOther, handler);
279  }
280  }

315  aIndex = -1;
316  bIndex = -1;
317 
318  //If the nodes are in different graphs, this function has no idea on how to find a shared edge.
319  if (backwards || toNode.GraphIndex != GraphIndex) return false;
320 
321  // Since the nodes are in the same graph, they are both TriangleMeshNodes
322  // So we don't need to care about other types of nodes
323  var toTriNode = toNode as TriangleMeshNode;
324  var edge = SharedEdge(toTriNode);
325 
326  // A connection was found, but it specifically didn't use an edge
327  if (edge == 0xFF) return false;
328 
329  // No connection was found between the nodes
330  // Check if there is a node link that connects them
331  if (edge == -1) {
332  return false;
333  }
334 
335  aIndex = edge;
336  bIndex = (edge + 1) % GetVertexCount();
337 
338  // Get the vertices of the shared edge for the first node
339  Int3 v1a = GetVertex(edge);
340  Int3 v1b = GetVertex((edge+1) % GetVertexCount());
341 
342  // Get tile indices
343  int tileIndex1 = (GetVertexIndex(0) >> NavmeshBase.TileIndexOffset) & NavmeshBase.TileIndexMask;
344  int tileIndex2 = (toTriNode.GetVertexIndex(0) >> NavmeshBase.TileIndexOffset) & NavmeshBase.TileIndexMask;
345 
346  if (tileIndex1 != tileIndex2) {
347  // When the nodes are in different tiles, the edges might not be completely identical
348  // so another technique is needed.
349 
350  // Get the tile coordinates, from them we can figure out which edge is going to be shared
351  int x1, x2, z1, z2, coord;
352  INavmeshHolder nm = GetNavmeshHolder(GraphIndex);
353  nm.GetTileCoordinates(tileIndex1, out x1, out z1);
354  nm.GetTileCoordinates(tileIndex2, out x2, out z2);
355 
356  if (System.Math.Abs(x1-x2) == 1) coord = 2;
357  else if (System.Math.Abs(z1-z2) == 1) coord = 0;
358  else return false; // Tiles are not adjacent. This is likely a custom connection between two nodes.
359 
360  var otherEdge = toTriNode.SharedEdge(this);
361 
362  // A connection was found, but it specifically didn't use an edge. This is odd since the connection in the other direction did use an edge
363  if (otherEdge == 0xFF) throw new System.Exception("Connection used edge in one direction, but not in the other direction. Has the wrong overload of AddConnection been used?");
364 
365  // If it is -1 then it must be a one-way connection. Fall back to using the whole edge
366  if (otherEdge != -1) {
367  // When the nodes are in different tiles, they might not share exactly the same edge
368  // so we clamp the portal to the segment of the edges which they both have.
369  int mincoord = System.Math.Min(v1a[coord], v1b[coord]);
370  int maxcoord = System.Math.Max(v1a[coord], v1b[coord]);
371 
372  // Get the vertices of the shared edge for the second node
373  Int3 v2a = toTriNode.GetVertex(otherEdge);
374  Int3 v2b = toTriNode.GetVertex((otherEdge+1) % toTriNode.GetVertexCount());
375 
376  mincoord = System.Math.Max(mincoord, System.Math.Min(v2a[coord], v2b[coord]));
377  maxcoord = System.Math.Min(maxcoord, System.Math.Max(v2a[coord], v2b[coord]));
378 
379  if (v1a[coord] < v1b[coord]) {
380  v1a[coord] = mincoord;
381  v1b[coord] = maxcoord;
382  } else {
383  v1a[coord] = maxcoord;
384  v1b[coord] = mincoord;
385  }
386  }
387  }
388 
389  if (left != null) {
390  // All triangles should be laid out in clockwise order so v1b is the rightmost vertex (seen from this node)
391  left.Add((Vector3)v1a);
392  right.Add((Vector3)v1b);
393  }
394  return true;
