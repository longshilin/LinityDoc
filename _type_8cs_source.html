59  internal static string GetConfigurationPath<T>() where T : ConfigPathAttribute
60  {
61  foreach (System.Type type in Utility.Assembly.GetTypes())
62  {
63  if (!type.IsAbstract || !type.IsSealed)
64  {
65  continue;
66  }
67 
68  foreach (FieldInfo fieldInfo in type.GetFields(BindingFlags.Static | BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.DeclaredOnly))
69  {
70  if (fieldInfo.FieldType == typeof(string) && fieldInfo.IsDefined(typeof(T), false))
71  {
72  return (string)fieldInfo.GetValue(null);
73  }
74  }
75 
76  foreach (PropertyInfo propertyInfo in type.GetProperties(BindingFlags.Static | BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.DeclaredOnly))
77  {
78  if (propertyInfo.PropertyType == typeof(string) && propertyInfo.IsDefined(typeof(T), false))
79  {
80  return (string)propertyInfo.GetValue(null, null);

108  private static string[] GetTypeNames(System.Type typeBase, string[] assemblyNames)
109  {
110  List<string> typeNames = new List<string>();
111  foreach (string assemblyName in assemblyNames)
112  {
113  Assembly assembly = null;
114  try
115  {
116  assembly = Assembly.Load(assemblyName);
117  }
118  catch
119  {
120  continue;
121  }
122 
123  if (assembly == null)
124  {
125  continue;
126  }
127 
128  System.Type[] types = assembly.GetTypes();
129  foreach (System.Type type in types)
130  {
131  if (type.IsClass && !type.IsAbstract && typeBase.IsAssignableFrom(type))
132  {
133  typeNames.Add(type.FullName);
134  }
135  }
136  }
137 
138  typeNames.Sort();
139  return typeNames.ToArray();
140  }
