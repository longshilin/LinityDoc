184  private void Start()
185  {
186  BaseComponent baseComponent = GameEntry.GetComponent<BaseComponent>();
187  if (baseComponent == null)
188  {
189  Log.Fatal("Base component is invalid.");
190  return;
191  }
192 
193  m_EventComponent = GameEntry.GetComponent<EventComponent>();
194  if (m_EventComponent == null)
195  {
196  Log.Fatal("Event component is invalid.");
197  return;
198  }
199 
200  if (baseComponent.EditorResourceMode)
201  {
202  m_UIManager.SetResourceManager(baseComponent.EditorResourceHelper);
203  }
204  else
205  {
206  m_UIManager.SetResourceManager(GameFrameworkEntry.GetModule<IResourceManager>());
207  }
208 
209  m_UIManager.SetObjectPoolManager(GameFrameworkEntry.GetModule<IObjectPoolManager>());
210  m_UIManager.InstanceAutoReleaseInterval = m_InstanceAutoReleaseInterval;
211  m_UIManager.InstanceCapacity = m_InstanceCapacity;
212  m_UIManager.InstanceExpireTime = m_InstanceExpireTime;
213  m_UIManager.InstancePriority = m_InstancePriority;
214 
215  UIFormHelperBase uiFormHelper = Helper.CreateHelper(m_UIFormHelperTypeName, m_CustomUIFormHelper);
216  if (uiFormHelper == null)
217  {
218  Log.Error("Can not create UI form helper.");
219  return;
220  }
221 
222  uiFormHelper.name = "UI Form Helper";
223  Transform transform = uiFormHelper.transform;
224  transform.SetParent(this.transform);
225  transform.localScale = Vector3.one;
226 
227  m_UIManager.SetUIFormHelper(uiFormHelper);
228 
229  if (m_InstanceRoot == null)
230  {
231  m_InstanceRoot = new GameObject("UI Form Instances").transform;
232  m_InstanceRoot.SetParent(gameObject.transform);
233  m_InstanceRoot.localScale = Vector3.one;
234  }
235 
236  m_InstanceRoot.gameObject.layer = LayerMask.NameToLayer("UI");
237 
238  for (int i = 0; i < m_UIGroups.Length; i++)
239  {
240  if (!AddUIGroup(m_UIGroups[i].Name, m_UIGroups[i].Depth))
241  {
242  Log.Warning("Add UI group '{0}' failure.", m_UIGroups[i].Name);
243  continue;
244  }
245  }
246  }
