212  {
213  while (m_RecycleQueue.Count > 0)
214  {
215  IUIForm uiForm = m_RecycleQueue.Dequeue();

221  {
222  uiGroup.Value.Update(elapseSeconds, realElapseSeconds);

723  public int OpenUIForm(string uiFormAssetName, string uiGroupName, int priority, bool pauseCoveredUIForm, object userData)
724  {
725  if (m_ResourceManager == null)
726  {
727  throw new GameFrameworkException("You must set resource manager first.");
728  }
729 
730  if (m_UIFormHelper == null)
731  {
732  throw new GameFrameworkException("You must set UI form helper first.");
733  }
734 
735  if (string.IsNullOrEmpty(uiFormAssetName))
736  {
737  throw new GameFrameworkException("UI form asset name is invalid.");
738  }
739 
740  if (string.IsNullOrEmpty(uiGroupName))
741  {
742  throw new GameFrameworkException("UI group name is invalid.");
743  }
744 
745  UIGroup uiGroup = (UIGroup)GetUIGroup(uiGroupName);
746  if (uiGroup == null)
747  {
748  throw new GameFrameworkException(Utility.Text.Format("UI group '{0}' is not exist.", uiGroupName));
749  }
750 
751  int serialId = ++m_Serial;
752  UIFormInstanceObject uiFormInstanceObject = m_InstancePool.Spawn(uiFormAssetName);
753  if (uiFormInstanceObject == null)
754  {
755  m_UIFormsBeingLoaded.Add(serialId, uiFormAssetName);
756  m_ResourceManager.LoadAsset(uiFormAssetName, priority, m_LoadAssetCallbacks, OpenUIFormInfo.Create(serialId, uiGroup, pauseCoveredUIForm, userData));
757  }
758  else
759  {
760  InternalOpenUIForm(serialId, uiFormAssetName, uiGroup, uiFormInstanceObject.Target, pauseCoveredUIForm, false, 0f, userData);
761  }
762 
763  return serialId;
764  }

940  private void InternalOpenUIForm(int serialId, string uiFormAssetName, UIGroup uiGroup, object uiFormInstance, bool pauseCoveredUIForm, bool isNewInstance, float duration, object userData)
941  {
942  try
943  {
944  IUIForm uiForm = m_UIFormHelper.CreateUIForm(uiFormInstance, uiGroup, userData);
945  if (uiForm == null)
946  {
947  throw new GameFrameworkException("Can not create UI form in UI form helper.");
948  }
949 
950  uiForm.OnInit(serialId, uiFormAssetName, uiGroup, pauseCoveredUIForm, isNewInstance, userData);
951  uiGroup.AddUIForm(uiForm);
952  uiForm.OnOpen(userData);
953  uiGroup.Refresh();
954 
955  if (m_OpenUIFormSuccessEventHandler != null)
956  {
957  OpenUIFormSuccessEventArgs openUIFormSuccessEventArgs = OpenUIFormSuccessEventArgs.Create(uiForm, duration, userData);
958  m_OpenUIFormSuccessEventHandler(this, openUIFormSuccessEventArgs);
959  ReferencePool.Release(openUIFormSuccessEventArgs);
960  }
961  }
962  catch (Exception exception)
963  {
964  if (m_OpenUIFormFailureEventHandler != null)
965  {
966  OpenUIFormFailureEventArgs openUIFormFailureEventArgs = OpenUIFormFailureEventArgs.Create(serialId, uiFormAssetName, uiGroup.Name, pauseCoveredUIForm, exception.ToString(), userData);
967  m_OpenUIFormFailureEventHandler(this, openUIFormFailureEventArgs);
968  ReferencePool.Release(openUIFormFailureEventArgs);
969  return;
970  }
971 
972  throw;
973  }
974  }

976  private void LoadAssetSuccessCallback(string uiFormAssetName, object uiFormAsset, float duration, object userData)
977  {
978  OpenUIFormInfo openUIFormInfo = (OpenUIFormInfo)userData;
979  if (openUIFormInfo == null)
980  {
981  throw new GameFrameworkException("Open UI form info is invalid.");
982  }
983 
984  if (m_UIFormsToReleaseOnLoad.Contains(openUIFormInfo.SerialId))
985  {
986  m_UIFormsToReleaseOnLoad.Remove(openUIFormInfo.SerialId);
987  ReferencePool.Release(openUIFormInfo);
988  m_UIFormHelper.ReleaseUIForm(uiFormAsset, null);
989  return;
990  }
991 
992  m_UIFormsBeingLoaded.Remove(openUIFormInfo.SerialId);
993  UIFormInstanceObject uiFormInstanceObject = UIFormInstanceObject.Create(uiFormAssetName, uiFormAsset, m_UIFormHelper.InstantiateUIForm(uiFormAsset), m_UIFormHelper);
994  m_InstancePool.Register(uiFormInstanceObject, true);
995 
996  InternalOpenUIForm(openUIFormInfo.SerialId, uiFormAssetName, openUIFormInfo.UIGroup, uiFormInstanceObject.Target, openUIFormInfo.PauseCoveredUIForm, true, duration, openUIFormInfo.UserData);
997  ReferencePool.Release(openUIFormInfo);
998  }

1000  private void LoadAssetFailureCallback(string uiFormAssetName, LoadResourceStatus status, string errorMessage, object userData)
1001  {
1002  OpenUIFormInfo openUIFormInfo = (OpenUIFormInfo)userData;
1003  if (openUIFormInfo == null)
1004  {
1005  throw new GameFrameworkException("Open UI form info is invalid.");
1006  }
1007 
1008  if (m_UIFormsToReleaseOnLoad.Contains(openUIFormInfo.SerialId))
1009  {
1010  m_UIFormsToReleaseOnLoad.Remove(openUIFormInfo.SerialId);
1011  return;
1012  }
1013 
1014  m_UIFormsBeingLoaded.Remove(openUIFormInfo.SerialId);
1015  string appendErrorMessage = Utility.Text.Format("Load UI form failure, asset name '{0}', status '{1}', error message '{2}'.", uiFormAssetName, status.ToString(), errorMessage);
