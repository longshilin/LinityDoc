199  public static void CreateNavmeshOutlineVisualization (NavmeshTile tile, GraphGizmoHelper helper) {
200  var sharedEdges = new bool[3];
201 
202  for (int j = 0; j < tile.nodes.Length; j++) {
203  sharedEdges[0] = sharedEdges[1] = sharedEdges[2] = false;
204 
205  var node = tile.nodes[j];
206  for (int c = 0; c < node.connections.Length; c++) {
207  var other = node.connections[c].node as TriangleMeshNode;
208 
209  // Loop through neighbours to figure out which edges are shared
210  if (other != null && other.GraphIndex == node.GraphIndex) {
211  for (int v = 0; v < 3; v++) {
212  for (int v2 = 0; v2 < 3; v2++) {
213  if (node.GetVertexIndex(v) == other.GetVertexIndex((v2+1)%3) && node.GetVertexIndex((v+1)%3) == other.GetVertexIndex(v2)) {
214  // Found a shared edge with the other node
215  sharedEdges[v] = true;
216  v = 3;
217  break;
218  }
219  }
220  }
221  }
222  }
223 
224  var color = helper.NodeColor(node);
225  for (int v = 0; v < 3; v++) {
226  if (!sharedEdges[v]) {
227  helper.builder.DrawLine((Vector3)node.GetVertex(v), (Vector3)node.GetVertex((v+1)%3), color);

241  public static IEnumerable<Progress> ScanAllTiles (this RecastGraph self) {
242  self.transform = self.CalculateTransform();
243  self.InitializeTileInfo();
244 
245  // If this is true, just fill the graph with empty tiles
246  if (self.scanEmptyGraph) {
247  self.FillWithEmptyTiles();
248  yield break;
249  }
250 
251  // A walkableClimb higher than walkableHeight can cause issues when generating the navmesh since then it can in some cases
252  // Both be valid for a character to walk under an obstacle and climb up on top of it (and that cannot be handled with navmesh without links)
253  // The editor scripts also enforce this but we enforce it here too just to be sure
254  self.walkableClimb = Mathf.Min(self.walkableClimb, self.walkableHeight);
255 
256  yield return new Progress(0, "Finding Meshes");
257  var bounds = self.transform.Transform(new Bounds(self.forcedBoundsSize*0.5f, self.forcedBoundsSize));
258  var meshes = self.CollectMeshes(bounds);
259  var buckets = self.PutMeshesIntoTileBuckets(meshes);
260 
261  Queue<Int2> tileQueue = new Queue<Int2>();
262 
263  // Put all tiles in the queue
264  for (int z = 0; z < self.tileZCount; z++) {
265  for (int x = 0; x < self.tileXCount; x++) {
266  tileQueue.Enqueue(new Int2(x, z));
267  }
268  }
269 
270  var workQueue = new ParallelWorkQueue<Int2>(tileQueue);
271  // Create the voxelizers and set all settings (one for each thread)
272  var voxelizers = new Voxelize[workQueue.threadCount];
273  for (int i = 0; i < voxelizers.Length; i++) voxelizers[i] = new Voxelize(self.CellHeight, self.cellSize, self.walkableClimb, self.walkableHeight, self.maxSlope, self.maxEdgeLength);
274  workQueue.action = (tile, threadIndex) => {
275  voxelizers[threadIndex].inputMeshes = buckets[tile.x + tile.y*self.tileXCount];
276  self.tiles[tile.x + tile.y*self.tileXCount] = self.BuildTileMesh(voxelizers[threadIndex], tile.x, tile.y, threadIndex);
277  };
278 
279  // Prioritize responsiveness while playing
280  // but when not playing prioritize throughput
281  // (the Unity progress bar is also pretty slow to update)
282  int timeoutMillis = Application.isPlaying ? 1 : 200;
283 
284  // Scan all tiles in parallel
285  foreach (var done in workQueue.Run(timeoutMillis)) {
286  yield return new Progress(Mathf.Lerp(0.1f, 0.9f, done / (float)self.tiles.Length), "Calculated Tiles: " + done + "/" + self.tiles.Length);
287  }
288 
289  yield return new Progress(0.9f, "Assigning Graph Indices");
290 
291  // Assign graph index to nodes
292  uint graphIndex = (uint)AstarPath.active.data.GetGraphIndex(self);
293 
294  self.GetNodes(node => node.GraphIndex = graphIndex);
295 
296  // First connect all tiles with an EVEN coordinate sum
297  // This would be the white squares on a chess board.
298  // Then connect all tiles with an ODD coordinate sum (which would be all black squares on a chess board).
299  // This will prevent the different threads that do all
300  // this in parallel from conflicting with each other.
301  // The directions are also done separately
302  // first they are connected along the X direction and then along the Z direction.
303  // Looping over 0 and then 1
304  for (int coordinateSum = 0; coordinateSum <= 1; coordinateSum++) {
305  for (int direction = 0; direction <= 1; direction++) {
306  for (int i = 0; i < self.tiles.Length; i++) {
307  if ((self.tiles[i].x + self.tiles[i].z) % 2 == coordinateSum) {
308  tileQueue.Enqueue(new Int2(self.tiles[i].x, self.tiles[i].z));
309  }
310  }
311 
312  workQueue = new ParallelWorkQueue<Int2>(tileQueue);
313  workQueue.action = (tile, threadIndex) => {
314  // Connect with tile at (x+1,z) and (x,z+1)
315  if (direction == 0 && tile.x < self.tileXCount - 1)
316  self.ConnectTiles(self.tiles[tile.x + tile.y * self.tileXCount], self.tiles[tile.x + 1 + tile.y * self.tileXCount]);
317  if (direction == 1 && tile.y < self.tileZCount - 1)
318  self.ConnectTiles(self.tiles[tile.x + tile.y * self.tileXCount], self.tiles[tile.x + (tile.y + 1) * self.tileXCount]);
319  };
320 
321  var numTilesInQueue = tileQueue.Count;
322  // Connect all tiles in parallel
323  foreach (var done in workQueue.Run(timeoutMillis)) {
324  yield return new Progress(0.95f, "Connected Tiles " + (numTilesInQueue - done) + "/" + numTilesInQueue + " (Phase " + (direction + 1 + 2*coordinateSum) + " of 4)");
325  }
326  }
327  }
328 
329  for (int i = 0; i < meshes.Count; i++) meshes[i].Pool();
330  ListPool<RasterizationMesh>.Release(ref meshes);
331 
332  // This may be used by the TileHandlerHelper script to update the tiles
333  // while taking NavmeshCuts into account after the graph has been completely recalculated.
334  if (self.OnRecalculatedTiles != null) {
335  self.OnRecalculatedTiles(self.tiles.Clone() as NavmeshTile[]);
336  }
337  }

341  var result = new List<RasterizationMesh>[self.tiles.Length];
342  var borderExpansion = new Vector3(1, 0, 1)*self.TileBorderSizeInWorldUnits*2;
343 
344  for (int i = 0; i < result.Length; i++) {
345  result[i] = ListPool<RasterizationMesh>.Claim();
346  }

356  for (int x = rect.xmin; x <= rect.xmax; x++) {

417  AstarProfiler.StartProfile("Build Tile");
418  AstarProfiler.StartProfile("Init");
419 
420  vox.borderSize = self.TileBorderSizeInVoxels;
421  vox.forcedBounds = self.CalculateTileBoundsWithBorder(x, z);
422  vox.width = self.tileSizeX + vox.borderSize*2;
423  vox.depth = self.tileSizeZ + vox.borderSize*2;
424 
425  if (!self.useTiles && self.relevantGraphSurfaceMode == RecastGraph.RelevantGraphSurfaceMode.OnlyForCompletelyInsideTile) {
426  // This best reflects what the user would actually want
427  vox.relevantGraphSurfaceMode = RecastGraph.RelevantGraphSurfaceMode.RequireForAll;
428  } else {
429  vox.relevantGraphSurfaceMode = self.relevantGraphSurfaceMode;
430  }
431

433 
434  AstarProfiler.EndProfile("Init");
435 
436 
437  // Init voxelizer
438  vox.Init();
439  vox.VoxelizeInput(self.transform, self.CalculateTileBoundsWithBorder(x, z));
440 
441  AstarProfiler.StartProfile("Filter Ledges");
442 
443

445 
446  AstarProfiler.EndProfile("Filter Ledges");
447 
448  AstarProfiler.StartProfile("Filter Low Height Spans");
449  vox.FilterLowHeightSpans(vox.voxelWalkableHeight, vox.cellSize, vox.cellHeight);
450  AstarProfiler.EndProfile("Filter Low Height Spans");

467  for (int i = 0; i < mesh.verts.Length; i++) {
468  mesh.verts[i] *= Int3.Precision;
469  }

471

473 
474  AstarProfiler.EndProfile("Build Nodes");

483  public static NavmeshTile CreateTile (this RecastGraph self, Voxelize vox, VoxelMesh mesh, int x, int z, int threadIndex) {
484  if (mesh.tris == null) throw new System.ArgumentNullException("mesh.tris");
485  if (mesh.verts == null) throw new System.ArgumentNullException("mesh.verts");
486  if (mesh.tris.Length % 3 != 0) throw new System.ArgumentException("Indices array's length must be a multiple of 3 (mesh.tris)");
487  if (mesh.verts.Length >= NavmeshBase.VertexIndexMask) {
488  if (self.tileXCount*self.tileZCount == 1) {
489  throw new System.ArgumentException("Too many vertices per tile (more than " + NavmeshBase.VertexIndexMask + ")." +
490  "\n<b>Try enabling tiling in the recast graph settings.</b>\n");
491  } else {
492  throw new System.ArgumentException("Too many vertices per tile (more than " + NavmeshBase.VertexIndexMask + ")." +
493  "\n<b>Try reducing tile size or enabling ASTAR_RECAST_LARGER_TILES under the 'Optimizations' tab in the A* Inspector</b>");
494  }
495  }
496 
497  // Create a new navmesh tile and assign its settings
498  var tile = new NavmeshTile {
499  x = x,
500  z = z,
501  w = 1,
502  d = 1,
503  tris = mesh.tris,
504  bbTree = new BBTree(),
505  graph = self,
506  };
507 
508  tile.vertsInGraphSpace = Utility.RemoveDuplicateVertices(mesh.verts, tile.tris);
509  tile.verts = (Int3[])tile.vertsInGraphSpace.Clone();
510  self.transform.Transform(tile.verts);
511 
512  // Here we are faking a new graph
513  // The tile is not added to any graphs yet, but to get the position queries from the nodes
514  // to work correctly (not throw exceptions because the tile is not calculated) we fake a new graph
515  // and direct the position queries directly to the tile
516  // The thread index is added to make sure that if multiple threads are calculating tiles at the same time
517  // they will not use the same temporary graph index
518  uint temporaryGraphIndex = (uint)(AstarPath.active.data.graphs.Length + threadIndex);
519 
520  if (temporaryGraphIndex > GraphNode.MaxGraphIndex) {
521  // Multithreaded tile calculations use fake graph indices, see above.
522  throw new System.Exception("Graph limit reached. Multithreaded recast calculations cannot be done because a few scratch graph indices are required.");
523  }
524 
525  TriangleMeshNode.SetNavmeshHolder((int)temporaryGraphIndex, tile);
526  // We need to lock here because creating nodes is not thread safe
527  // and we may be doing this from multiple threads at the same time
528  tile.nodes = new TriangleMeshNode[tile.tris.Length/3];
529  lock (AstarPath.active) {
530  self.CreateNodes(tile.nodes, tile.tris, x + z*self.tileXCount, temporaryGraphIndex);
531  }
532 
533  tile.bbTree.RebuildFrom(tile.nodes);
534  NavmeshBase.CreateNodeConnections(tile.nodes);
535  // Remove the fake graph
536  TriangleMeshNode.SetNavmeshHolder((int)temporaryGraphIndex, null);
537 
538  return tile;
539  }

575  public static Bounds Transform (this GraphTransform self, Bounds bounds) {
576  if (self.onlyTranslational) return new Bounds(bounds.center + self.translation.ToUnityV3(), bounds.size);
577 
578  var corners = ArrayPool<Vector3>.Claim(8);
579  var extents = bounds.extents;
580  corners[0] = self.Transform(bounds.center + new Vector3(extents.x, extents.y, extents.z));
581  corners[1] = self.Transform(bounds.center + new Vector3(extents.x, extents.y, -extents.z));
582  corners[2] = self.Transform(bounds.center + new Vector3(extents.x, -extents.y, extents.z));
583  corners[3] = self.Transform(bounds.center + new Vector3(extents.x, -extents.y, -extents.z));
584  corners[4] = self.Transform(bounds.center + new Vector3(-extents.x, extents.y, extents.z));
585  corners[5] = self.Transform(bounds.center + new Vector3(-extents.x, extents.y, -extents.z));
586  corners[6] = self.Transform(bounds.center + new Vector3(-extents.x, -extents.y, extents.z));
587  corners[7] = self.Transform(bounds.center + new Vector3(-extents.x, -extents.y, -extents.z));
588 
589  var min = corners[0];
590  var max = corners[0];
591  for (int i = 1; i < 8; i++) {
592  min = Vector3.Min(min, corners[i]);
593  max = Vector3.Max(max, corners[i]);
594  }
