923  public UpdateDefinition<TDocument> CurrentDate(FieldDefinition<TDocument> field, UpdateDefinitionCurrentDateType? type = null)
924  {
925  BsonValue value;
926  if (type.HasValue)
927  {
928  switch (type.Value)
929  {
930  case UpdateDefinitionCurrentDateType.Date:
931  value = new BsonDocument("$type", "date");
932  break;
933  case UpdateDefinitionCurrentDateType.Timestamp:
934  value = new BsonDocument("$type", "timestamp");
935  break;
936  default:
937  throw new InvalidOperationException("Unknown value for " + typeof(UpdateDefinitionCurrentDateType));
938  }
939  }
940  else
941  {
942  value = true;
943  }
944 
945  return new OperatorUpdateDefinition<TDocument>("$currentDate", field, value);
946  }
