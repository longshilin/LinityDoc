122  public string GetInsecurePassword()
123  {
124  if (_password.Length == 0)
125  {
126  return "";
127  }
128  else
129  {
130 #if NET452
131  var passwordIntPtr = Marshal.SecureStringToGlobalAllocUnicode(_password);
132 #else
133  var passwordIntPtr = SecureStringMarshal.SecureStringToGlobalAllocUnicode(_password);
134 #endif
135  try
136  {
137  return Marshal.PtrToStringUni(passwordIntPtr, _password.Length);
138  }
139  finally
140  {
141  Marshal.ZeroFreeGlobalAllocUnicode(passwordIntPtr);
142  }
143  }
