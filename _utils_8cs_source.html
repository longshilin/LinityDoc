84  static bool DontNeedQuotes(string src)
85  {
86  // using a regex instead of explicit matching slows down common cases by 40%
87  if (src.Length == 0)
88  return false;
89 
90  bool hasSlash = false;
91  for (int i = 0; i < src.Length; ++i)
92  {
93  char c = src[i];
94  if (Char.IsLetterOrDigit(c) || c == '.' || c == '*' || c == '_')
95  continue;
96  if (c == '/')
97  {
98  hasSlash = true;
99  continue;
100  }
101  return false;
102  }
103  if (hasSlash)
104  {
105  if (src.Contains("//") || src.Contains("/*") || src.Contains("*/"))
106  return false;
107  }
108  return true;
109  }
110 
111  // Quotes the given string if it contains special characters. Note: if the string already
112  // contains quotes, then they are escaped and the entire string quoted again
113  public static string QuoteStringIfNeeded(string src)
114  {
115  if (DontNeedQuotes(src))
116  return src;
117  return "\"" + src.Replace("\\", "\\\\").Replace("\"", "\\\"").Replace("\n", "\\n") + "\"";
118  }
119 
120  // If the given string is quoted, removes the quotes and unescapes any quotes within the string
121  public static string UnquoteString(string src)
122  {
123  if (!src.StartsWith("\"") || !src.EndsWith("\""))
124  return src;
125  return src.Substring(1, src.Length - 2).Replace("\\\\", "\u569f").Replace("\\\"", "\"")
126  .Replace("\\n", "\n").Replace("\u569f", "\\"); // U+569f is a rarely used Chinese character
127  }
128  }
