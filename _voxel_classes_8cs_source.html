182  public void AddLinkedSpan (int index, uint bottom, uint top, int area, int voxelWalkableClimb) {
183 #if ASTAR_RECAST_CLASS_BASED_LINKED_LIST
184  cells[index].AddSpan(bottom, top, area, voxelWalkableClimb);
185 #else
186  // linkedSpans[index] is the span with the lowest y-coordinate at the position x,z such that index=x+z*width
187  // i.e linkedSpans is a 2D array laid out in a 1D array (for performance and simplicity)
188 
189  // Check if there is a root span, otherwise we can just add a new (valid) span and exit
190  if (linkedSpans[index].bottom == InvalidSpanValue) {
191  linkedSpans[index] = new LinkedVoxelSpan(bottom, top, area);
192  return;
193  }
194 
195  int prev = -1;
196 
197  // Original index, the first span we visited
198  int oindex = index;
199 
200  while (index != -1) {
201  if (linkedSpans[index].bottom > top) {
202  // If the current span's bottom higher up than the span we want to insert's top, then they do not intersect
203  // and we should just insert a new span here
204  break;
205  } else if (linkedSpans[index].top < bottom) {
206  // The current span and the span we want to insert do not intersect
207  // so just skip to the next span (it might intersect)
208  prev = index;
209  index = linkedSpans[index].next;
210  } else {
211  // Intersection! Merge the spans
212 
213  // Find the new bottom and top for the merged span
214  bottom = System.Math.Min(linkedSpans[index].bottom, bottom);
215  top = System.Math.Max(linkedSpans[index].top, top);
216 
217  // voxelWalkableClimb is flagMergeDistance, when a walkable flag is favored before an unwalkable one
218  // So if a walkable span intersects an unwalkable span, the walkable span can be up to voxelWalkableClimb
219  // below the unwalkable span and the merged span will still be walkable
220  if (System.Math.Abs((int)top - (int)linkedSpans[index].top) <= voxelWalkableClimb) {
221  // linkedSpans[index] is the lowest span, but we might use that span's area anyway if it is walkable
222  area = System.Math.Max(area, linkedSpans[index].area);
223  }
224 
225  // Find the next span in the linked list
226  int next = linkedSpans[index].next;
227  if (prev != -1) {
228  // There is a previous span
229  // Remove this span from the linked list
230  linkedSpans[prev].next = next;
231 
232  // Add this span index to a list for recycling
233  PushToSpanRemovedStack(index);
234 
235  // Move to the next span in the list
236  index = next;
237  } else if (next != -1) {
238  // This was the root span and there is a span left in the linked list
239  // Remove this span from the linked list by assigning the next span as the root span
240  linkedSpans[oindex] = linkedSpans[next];
241 
242  // Recycle the old span index
243  PushToSpanRemovedStack(next);
244 
245  // Move to the next span in the list
246  // NOP since we just removed the current span, the next span
247  // we want to visit will have the same index as we are on now (i.e oindex)
248  } else {
249  // This was the root span and there are no other spans in the linked list
250  // Just replace the root span with the merged span and exit
251  linkedSpans[oindex] = new LinkedVoxelSpan(bottom, top, area);
252  return;
253  }
254  }
255  }
256 
257  // We now have a merged span that needs to be inserted
258  // and connected with the existing spans
259 
260  // The new merged span will be inserted right after 'prev' (if it exists, otherwise before index)
261 
262  // Make sure that we have enough room in our array
263  if (linkedSpanCount >= linkedSpans.Length) {
264  LinkedVoxelSpan[] tmp = linkedSpans;
265  int count = linkedSpanCount;
266  int popped = removedStackCount;
267  linkedSpans = new LinkedVoxelSpan[linkedSpans.Length*2];
268  ResetLinkedVoxelSpans();
269  linkedSpanCount = count;
270  removedStackCount = popped;
271  for (int i = 0; i < linkedSpanCount; i++) {
272  linkedSpans[i] = tmp[i];
273  }
274  Debug.Log("Layer estimate too low, doubling size of buffer.\nThis message is harmless.");
275  }
276 
277  // Take a node from the recycling stack if possible
278  // Otherwise create a new node (well, just a new index really)
279  int nextIndex;
280  if (removedStackCount > 0) {
281  removedStackCount--;
282  nextIndex = removedStack[removedStackCount];
283  } else {
284  nextIndex = linkedSpanCount;
285  linkedSpanCount++;
286  }
287 
288  if (prev != -1) {
289  //span.next = prev.next;
290  //prev.next = span;
291 
292  linkedSpans[nextIndex] = new LinkedVoxelSpan(bottom, top, area, linkedSpans[prev].next);
293  linkedSpans[prev].next = nextIndex;
294  } else {
295  //span.next = firstSpan;
296  //firstSpan = span;
297 
298  linkedSpans[nextIndex] = linkedSpans[oindex];
299  linkedSpans[oindex] = new LinkedVoxelSpan(bottom, top, area, nextIndex);
300  }
301 #endif
302  }
