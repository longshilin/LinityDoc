363  hasConnections = true;
364  break;
365  }
366  }
367 
368  if (hasConnections) {
369  // The contour has some portals to other regions.
370  // Add a new point to every location where the region changes.
371  for (int i = 0, ni = verts.Count/4; i < ni; i++) {
372  int ii = (i+1) % ni;
373  bool differentRegs = (verts[i*4+3] & ContourRegMask) != (verts[ii*4+3] & ContourRegMask);
374  bool areaBorders = (verts[i*4+3] & RC_AREA_BORDER) != (verts[ii*4+3] & RC_AREA_BORDER);
375

377  simplified.Add(verts[i*4+0]);
378  simplified.Add(verts[i*4+1]);
379  simplified.Add(verts[i*4+2]);
380  simplified.Add(i);
381  }
382  }

384 
385 
386  if (simplified.Count == 0) {
387  // If there is no connections at all,
388  // create some initial points for the simplification process.
389  // Find lower-left and upper-right vertices of the contour.
390  int llx = verts[0];
391  int lly = verts[1];
392  int llz = verts[2];
393  int lli = 0;
394  int urx = verts[0];
395  int ury = verts[1];
396  int urz = verts[2];
397  int uri = 0;

402  int z = verts[i+2];

404  llx = x;

406  llz = z;
407  lli = i/4;
408  }
409  if (x > urx || (x == urx && z > urz)) {

411  ury = y;
412  urz = z;

429  // error tolerance to the simplified shape.
430  int pn = verts.Count/4;
431 
432  //Use the max squared error instead
433  maxError *= maxError;
434

437

444  int bi = simplified[ii*4+3];

761  public int GetCornerHeight (int x, int z, int i, int dir, ref bool isBorderVertex) {
762  CompactVoxelSpan s = voxelArea.compactSpans[i];
763 
764  int ch = (int)s.y;
765 
766  //dir + clockwise direction
767  int dirp = (dir+1) & 0x3;
768 
769  //int dirp = (dir+3) & 0x3;
770 
771  uint[] regs = new uint[4];
772 
773  regs[0] = (uint)voxelArea.compactSpans[i].reg | ((uint)voxelArea.areaTypes[i] << 16);
774 
775  if (s.GetConnection(dir) != NotConnected) {
776  int nx = x + voxelArea.DirectionX[dir];
777  int nz = z + voxelArea.DirectionZ[dir];
778  int ni = (int)voxelArea.compactCells[nx+nz].index + s.GetConnection(dir);
779 
780  CompactVoxelSpan ns = voxelArea.compactSpans[ni];
781 
782  ch = System.Math.Max(ch, (int)ns.y);
783  regs[1] = (uint)ns.reg | ((uint)voxelArea.areaTypes[ni] << 16);
784 
785  if (ns.GetConnection(dirp) != NotConnected) {
786  int nx2 = nx + voxelArea.DirectionX[dirp];
787  int nz2 = nz + voxelArea.DirectionZ[dirp];
788  int ni2 = (int)voxelArea.compactCells[nx2+nz2].index + ns.GetConnection(dirp);
789 
790  CompactVoxelSpan ns2 = voxelArea.compactSpans[ni2];
791 
792  ch = System.Math.Max(ch, (int)ns2.y);
793  regs[2] = (uint)ns2.reg | ((uint)voxelArea.areaTypes[ni2] << 16);
794  }
795  }
796 
797  if (s.GetConnection(dirp) != NotConnected) {
798  int nx = x + voxelArea.DirectionX[dirp];
799  int nz = z + voxelArea.DirectionZ[dirp];
800  int ni = (int)voxelArea.compactCells[nx+nz].index + s.GetConnection(dirp);
801 
802  CompactVoxelSpan ns = voxelArea.compactSpans[ni];
803 
804  ch = System.Math.Max(ch, (int)ns.y);
805  regs[3] = (uint)ns.reg | ((uint)voxelArea.areaTypes[ni] << 16);
806 
807  if (ns.GetConnection(dir) != NotConnected) {
808  int nx2 = nx + voxelArea.DirectionX[dir];
809  int nz2 = nz + voxelArea.DirectionZ[dir];
810  int ni2 = (int)voxelArea.compactCells[nx2+nz2].index + ns.GetConnection(dir);
811 
812  CompactVoxelSpan ns2 = voxelArea.compactSpans[ni2];
813 
814  ch = System.Math.Max(ch, (int)ns2.y);
815  regs[2] = (uint)ns2.reg | ((uint)voxelArea.areaTypes[ni2] << 16);
816  }
817  }
818 
819  // Check if the vertex is special edge vertex, these vertices will be removed later.
820  for (int j = 0; j < 4; ++j) {
821  int a = j;
822  int b = (j+1) & 0x3;
823  int c = (j+2) & 0x3;
824  int d = (j+3) & 0x3;
825 
826  // The vertex is a border vertex there are two same exterior cells in a row,
827  // followed by two interior cells and none of the regions are out of bounds.
828  bool twoSameExts = (regs[a] & regs[b] & BorderReg) != 0 && regs[a] == regs[b];
829  bool twoInts = ((regs[c] | regs[d]) & BorderReg) == 0;
830  bool intsSameArea = (regs[c]>>16) == (regs[d]>>16);
831  bool noZeros = regs[a] != 0 && regs[b] != 0 && regs[c] != 0 && regs[d] != 0;
832  if (twoSameExts && twoInts && intsSameArea && noZeros) {
833  isBorderVertex = true;
834  break;
835  }
836  }
837 
838  return ch;
839  }
