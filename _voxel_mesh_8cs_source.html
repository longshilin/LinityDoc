15  public static bool InCone (int i, int j, int n, int[] verts, int[] indices) {
16  int pi = (indices[i] & 0x0fffffff) * 4;
17  int pj = (indices[j] & 0x0fffffff) * 4;
18  int pi1 = (indices[Next(i, n)] & 0x0fffffff) * 4;
19  int pin1 = (indices[Prev(i, n)] & 0x0fffffff) * 4;
20 
21  // If P[i] is a convex vertex [ i+1 left or on (i-1,i) ].
22  if (LeftOn(pin1, pi, pi1, verts))
23  return Left(pi, pj, pin1, verts) && Left(pj, pi, pi1, verts);
24  // Assume (i-1,i,i+1) not collinear.
25  // else P[i] is reflex.
26  return !(LeftOn(pi, pj, pi1, verts) && LeftOn(pj, pi, pin1, verts));

92  // Returns true iff ab properly intersects cd: they share
93  // a point interior to both segments. The properness of the
94  // intersection is ensured by using strict leftness.
95  public static bool IntersectProp (int a, int b, int c, int d, int[] verts) {
96  // Eliminate improper cases.
97  if (Collinear(a, b, c, verts) || Collinear(a, b, d, verts) ||
98  Collinear(c, d, a, verts) || Collinear(c, d, b, verts))
99  return false;
100 
101  return Xorb(Left(a, b, c, verts), Left(a, b, d, verts)) && Xorb(Left(c, d, a, verts), Left(c, d, b, verts));
102  }

104  // Returns T iff (a,b,c) are collinear and point c lies
105  // on the closed segement ab.
106  static bool Between (int a, int b, int c, int[] verts) {
107  if (!Collinear(a, b, c, verts))
108  return false;
109  // If ab not vertical, check betweenness on x; else on y.
110  if (verts[a+0] != verts[b+0])
111  return ((verts[a+0] <= verts[c+0]) && (verts[c+0] <= verts[b+0])) || ((verts[a+0] >= verts[c+0]) && (verts[c+0] >= verts[b+0]));
112  else
113  return ((verts[a+2] <= verts[c+2]) && (verts[c+2] <= verts[b+2])) || ((verts[a+2] >= verts[c+2]) && (verts[c+2] >= verts[b+2]));
114  }

116  // Returns true iff segments ab and cd intersect, properly or improperly.
117  static bool Intersect (int a, int b, int c, int d, int[] verts) {
118  if (IntersectProp(a, b, c, d, verts))
119  return true;
120  else if (Between(a, b, c, verts) || Between(a, b, d, verts) ||
121  Between(c, d, a, verts) || Between(c, d, b, verts))
122  return true;
123  else
124  return false;
125  }
