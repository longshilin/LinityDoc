149 
150  AstarProfiler.StartProfile("Voxelizing - Step 1");
151 
152  // Transform from voxel space to graph space.
153  // then scale from voxel space (one unit equals one voxel)
154  // Finally add min
155  Matrix4x4 voxelMatrix = Matrix4x4.TRS(graphSpaceBounds.min, Quaternion.identity, Vector3.one) * Matrix4x4.Scale(new Vector3(cellSize, cellHeight, cellSize));
156  transformVoxel2Graph = new GraphTransform(voxelMatrix);
157 
158  // Transform from voxel space to world space
159  // add half a voxel to fix rounding
160  transform = graphTransform * voxelMatrix * Matrix4x4.TRS(new Vector3(0.5f, 0, 0.5f), Quaternion.identity, Vector3.one);
161 
162  int maximumVoxelYCoord = (int)(graphSpaceBounds.size.y / cellHeight);
163 
164  AstarProfiler.EndProfile("Voxelizing - Step 1");
165 
166  AstarProfiler.StartProfile("Voxelizing - Step 2 - Init");
167 
168  // Cosine of the slope limit in voxel space (some tweaks are needed because the voxel space might be stretched out along the y axis)
169  float slopeLimit = Mathf.Cos(Mathf.Atan(Mathf.Tan(maxSlope*Mathf.Deg2Rad)*(cellSize/cellHeight)));
170 
171  // Temporary arrays used for rasterization
172  float[] vTris = new float[3*3];
173  float[] vOut = new float[7*3];
174  float[] vRow = new float[7*3];
175  float[] vCellOut = new float[7*3];
176  float[] vCell = new float[7*3];
177 
178  if (inputMeshes == null) throw new System.NullReferenceException("inputMeshes not set");
179 
180  // Find the largest lengths of vertex arrays and check for meshes which can be skipped
181  int maxVerts = 0;
182  for (int m = 0; m < inputMeshes.Count; m++) {
183  maxVerts = System.Math.Max(inputMeshes[m].vertices.Length, maxVerts);
184  }
185 
186  // Create buffer, here vertices will be stored multiplied with the local-to-voxel-space matrix
187  var verts = new Vector3[maxVerts];
188 
189  AstarProfiler.EndProfile("Voxelizing - Step 2 - Init");
190 
191  AstarProfiler.StartProfile("Voxelizing - Step 2");
192 
193  // This loop is the hottest place in the whole rasterization process
194  // it usually accounts for around 50% of the time
195  for (int m = 0; m < inputMeshes.Count; m++) {
196  RasterizationMesh mesh = inputMeshes[m];
197  var meshMatrix = mesh.matrix;
198 
199  // Flip the orientation of all faces if the mesh is scaled in such a way
200  // that the face orientations would change
201  // This happens for example if a mesh has a negative scale along an odd number of axes
202  // e.g it happens for the scale (-1, 1, 1) but not for (-1, -1, 1) or (1,1,1)
203  var flipOrientation = UnityHelper.ReversesFaceOrientations(meshMatrix);
204 
205  Vector3[] vs = mesh.vertices;
206  int[] tris = mesh.triangles;
207  int trisLength = mesh.numTriangles;
208 
209  // Transform vertices first to world space and then to voxel space
210  for (int i = 0; i < vs.Length; i++) verts[i] = transform.InverseTransform(meshMatrix.MultiplyPoint3x4(vs[i]));
211 
212  int mesharea = mesh.area;
213 
214  for (int i = 0; i < trisLength; i += 3) {
215  Vector3 p1 = verts[tris[i]];
216  Vector3 p2 = verts[tris[i+1]];
217  Vector3 p3 = verts[tris[i+2]];
218 
219  if (flipOrientation) {
220  var tmp = p1;
221  p1 = p3;
222  p3 = tmp;
223  }
224 
225  int minX = (int)(Utility.Min(p1.x, p2.x, p3.x));
226  int minZ = (int)(Utility.Min(p1.z, p2.z, p3.z));
227 
228  int maxX = (int)System.Math.Ceiling(Utility.Max(p1.x, p2.x, p3.x));
229  int maxZ = (int)System.Math.Ceiling(Utility.Max(p1.z, p2.z, p3.z));
230 
231  minX = Mathf.Clamp(minX, 0, voxelArea.width-1);
232  maxX = Mathf.Clamp(maxX, 0, voxelArea.width-1);
233  minZ = Mathf.Clamp(minZ, 0, voxelArea.depth-1);
234  maxZ = Mathf.Clamp(maxZ, 0, voxelArea.depth-1);
235 
236  // Check if the mesh is completely out of bounds
237  if (minX >= voxelArea.width || minZ >= voxelArea.depth || maxX <= 0 || maxZ <= 0) continue;
238 
239  Vector3 normal;
240 
241  int area;
242 
243  //AstarProfiler.StartProfile ("Rasterize...");
244 
245  normal = Vector3.Cross(p2-p1, p3-p1);
246 
247  float cosSlopeAngle = Vector3.Dot(normal.normalized, Vector3.up);
248 
249  if (cosSlopeAngle < slopeLimit) {
250  area = UnwalkableArea;
251  } else {
252  area = 1 + mesharea;
253  }
254 
255  Utility.CopyVector(vTris, 0, p1);
256  Utility.CopyVector(vTris, 3, p2);
257  Utility.CopyVector(vTris, 6, p3);
258

261 
262  if (nrow < 3) {
263  continue;
264  }
265 
266  nrow = clipper.ClipPolygon(vOut, nrow, vRow, -1F, x+0.5F, 0);
267 
268  if (nrow < 3) {
269  continue;
270  }
271

273  float clampZ2 = vRow[2];
274  for (int q = 1; q < nrow; q++) {
275  float val = vRow[q*3+2];
276  clampZ1 = System.Math.Min(clampZ1, val);
277  clampZ2 = System.Math.Max(clampZ2, val);
278  }
279 
280  int clampZ1I = Mathf.Clamp((int)System.Math.Round(clampZ1), 0, voxelArea.depth-1);
281  int clampZ2I = Mathf.Clamp((int)System.Math.Round(clampZ2), 0, voxelArea.depth-1);
282 
283 
284  for (int z = clampZ1I; z <= clampZ2I; z++) {
285  //AstarProfiler.StartFastProfile(1);
286  int ncell = clipper.ClipPolygon(vRow, nrow, vCellOut, 1F, -z+0.5F, 2);
287 
288  if (ncell < 3) {
289  //AstarProfiler.EndFastProfile(1);
290  continue;
291  }
292 
293  ncell = clipper.ClipPolygonY(vCellOut, ncell, vCell, -1F, z+0.5F, 2);
294 
295  if (ncell < 3) {
296  //AstarProfiler.EndFastProfile(1);
297  continue;
298  }
299 
300  //AstarProfiler.EndFastProfile(1);
301  //AstarProfiler.StartFastProfile(2);
302  float sMin = vCell[1];
303  float sMax = vCell[1];
304  for (int q = 1; q < ncell; q++) {
305  float val = vCell[q*3+1];
306  sMin = System.Math.Min(sMin, val);
307  sMax = System.Math.Max(sMax, val);
308  }
309 
310  //AstarProfiler.EndFastProfile(2);
311  int maxi = (int)System.Math.Ceiling(sMax);
312 
313  // Skip span if below or above the bounding box
314  if (maxi >= 0 && sMin <= maximumVoxelYCoord) {
315  // Make sure mini >= 0
316  int mini = System.Math.Max(0, (int)sMin);
317 
318  // Make sure the span is at least 1 voxel high
319  maxi = System.Math.Max(mini+1, maxi);
320 
321  voxelArea.AddLinkedSpan(z*voxelArea.width+x, (uint)mini, (uint)maxi, area, voxelWalkableClimb);
322  }
323  }
324  }
325  }
326  //AstarProfiler.EndFastProfile(0);
327  //AstarProfiler.EndProfile ("Rasterize...");
328  }
329  AstarProfiler.EndProfile("Voxelizing - Step 2");

833  }
