404  public int AddWebRequest(string webRequestUri, byte[] postData, string tag, int priority, object userData)
405  {
406  if (string.IsNullOrEmpty(webRequestUri))
407  {
408  throw new GameFrameworkException("Web request uri is invalid.");
409  }
410 
411  if (TotalAgentCount <= 0)
412  {
413  throw new GameFrameworkException("You must add web request agent first.");
414  }
415 
416  WebRequestTask webRequestTask = WebRequestTask.Create(webRequestUri, postData, tag, priority, m_Timeout, userData);
417  m_TaskPool.AddTask(webRequestTask);
418  return webRequestTask.SerialId;
419  }
