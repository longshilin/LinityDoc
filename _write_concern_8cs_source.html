110  public static WriteConcern FromBsonDocument(BsonDocument document)
111  {
112  var writeConcern = WriteConcern.Acknowledged;
113 
114  BsonValue w;
115  if (document.TryGetValue("w", out w))
116  {
117  writeConcern = writeConcern.With(w: WriteConcern.WValue.Parse(w.ToString()));
118  }
119 
120  BsonValue wTimeout;
121  if (document.TryGetValue("wtimeout", out wTimeout))
122  {
123  writeConcern = writeConcern.With(wTimeout: TimeSpan.FromMilliseconds(wTimeout.ToDouble()));
124  }
125 
126  BsonValue fsync;
127  if (document.TryGetValue("fsync", out fsync))
128  {
129  writeConcern = writeConcern.With(fsync: fsync.ToBoolean());
130  }
131 
132  BsonValue j;
133  if (document.TryGetValue("j", out j))
134  {
135  writeConcern = writeConcern.With(journal: j.ToBoolean());
136  }
137 
138  return writeConcern;
139  }
