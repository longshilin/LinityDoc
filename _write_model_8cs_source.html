31  public abstract class WriteModel<TDocument>
32  {
33  // static methods
34  // These static methods are only called from the Legacy
35  // API, so there is type safety in how they got allowed
36  // into the system, meaning that even though
37  // some things below seem unsafe, they are in a roundabout
38  // way. In addition, we know that there will always
39  // be one level of BsonDocumentWrapper for everything, even
40  // when the type is already a BsonDocument :(.
41  internal static WriteModel<TDocument> FromCore(WriteRequest request)
42  {
43  switch (request.RequestType)
44  {
45  case WriteRequestType.Delete:
46  return ConvertDeleteRequest((DeleteRequest)request);
47  case WriteRequestType.Insert:
48  return ConvertInsertRequest((InsertRequest)request);
49  case WriteRequestType.Update:
50  return ConvertUpdateRequest((UpdateRequest)request);
51  default:
52  throw new NotSupportedException();
53  }
54  }
55 
56  private static WriteModel<TDocument> ConvertDeleteRequest(DeleteRequest request)
57  {
58  if (request.Limit == 1)
59  {
60  return new DeleteOneModel<TDocument>(UnwrapFilter(request.Filter))
61  {
62  Collation = request.Collation
63  };
64  }
65 
66  return new DeleteManyModel<TDocument>(UnwrapFilter(request.Filter))
67  {
68  Collation = request.Collation
69  };
70  }
71 
72  private static WriteModel<TDocument> ConvertInsertRequest(InsertRequest request)
73  {
74  var document = (TDocument)Unwrap(request.Document);
75  return new InsertOneModel<TDocument>(document);
76  }
77 
78  private static WriteModel<TDocument> ConvertUpdateRequest(UpdateRequest request)
79  {
80  if (request.IsMulti)
81  {
82  return new UpdateManyModel<TDocument>(UnwrapFilter(request.Filter), UnwrapUpdate(request.Update))
83  {
84  ArrayFilters = request.ArrayFilters == null ? null : new List<ArrayFilterDefinition>(request.ArrayFilters.Select(f => new BsonDocumentArrayFilterDefinition<BsonValue>(f))),
85  Collation = request.Collation,
86  IsUpsert = request.IsUpsert
87  };
88  }
89 
90  var firstElement = request.Update.GetElement(0).Name;
91  if (firstElement.StartsWith("$"))
92  {
93  return new UpdateOneModel<TDocument>(UnwrapFilter(request.Filter), UnwrapUpdate(request.Update))
94  {
95  ArrayFilters = request.ArrayFilters == null ? null : new List<ArrayFilterDefinition>(request.ArrayFilters.Select(f => new BsonDocumentArrayFilterDefinition<BsonValue>(f))),
96  Collation = request.Collation,
97  IsUpsert = request.IsUpsert
98  };
99  }
100 
101  return ConvertToReplaceOne(request);
102  }
103 
104  private static WriteModel<TDocument> ConvertToReplaceOne(UpdateRequest request)
105  {
106  var document = (TDocument)Unwrap(request.Update);
107  if (request.ArrayFilters != null)
108  {
109  throw new ArgumentException("ReplaceOne does not support arrayFilters.", nameof(request));
110  }
111 
112  return new ReplaceOneModel<TDocument>(UnwrapFilter(request.Filter), document)
113  {
114  Collation = request.Collation,
115  IsUpsert = request.IsUpsert
116  };
117  }
118 
119  private static FilterDefinition<TDocument> UnwrapFilter(BsonDocument filter)
120  {
121  var wrapper = filter as BsonDocumentWrapper;
122  if (wrapper != null)
123  {
124  if (wrapper.Wrapped is BsonDocument)
125  {
126  return new BsonDocumentFilterDefinition<TDocument>((BsonDocument)wrapper.Wrapped);
127  }
128  return new ObjectFilterDefinition<TDocument>(wrapper.Wrapped);
129  }
130 
131  return new BsonDocumentFilterDefinition<TDocument>(filter);
132  }
133 
134  private static UpdateDefinition<TDocument> UnwrapUpdate(BsonDocument update)
135  {
136  var wrapper = update as BsonDocumentWrapper;
137  if (wrapper != null)
138  {
139  if (wrapper.Wrapped is BsonDocument)
140  {
141  return new BsonDocumentUpdateDefinition<TDocument>((BsonDocument)wrapper.Wrapped);
142  }
143  return new ObjectUpdateDefinition<TDocument>(wrapper.Wrapped);
144  }
145 
146  return new BsonDocumentUpdateDefinition<TDocument>(update);
147  }
148 
149  private static object Unwrap(BsonDocument wrapper)
150  {
151  return ((BsonDocumentWrapper)wrapper).Wrapped;
152  }
