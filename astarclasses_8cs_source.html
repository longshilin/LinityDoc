246  public virtual void RevertFromBackup () {
247  if (trackChangedNodes) {
248  if (changedNodes == null) return;
249 
250  int counter = 0;
251  for (int i = 0; i < changedNodes.Count; i++) {
252  changedNodes[i].Penalty = backupData[counter];
253  counter++;
254  changedNodes[i].Flags = backupData[counter];
255  counter++;
256  changedNodes[i].position = backupPositionData[i];
257  }
258 
259  ListPool<GraphNode>.Release(ref changedNodes);
260  ListPool<uint>.Release(ref backupData);
261  ListPool<Int3>.Release(ref backupPositionData);
262  } else {
263  throw new System.InvalidOperationException("Changed nodes have not been tracked, cannot revert from backup. Please set trackChangedNodes to true before applying the update.");
264  }
